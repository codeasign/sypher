# Reorder List — Solution

## Approach: Find Middle, Reverse, Merge

**Idea:** Three-step process: find the middle, reverse the second half, then interleave the two halves.

**Algorithm:**
1. Find middle using slow/fast pointers.
2. Reverse the second half.
3. Merge the two halves by alternating nodes.

**Complexity:** O(n) time, O(1) space.

```python
def reorder_list(head: list[int]) -> list[int]:
    if not head: return []
    # Find middle
    mid = (len(head) + 1) // 2
    first = head[:mid]
    second = head[mid:][::-1]
    # Interleave
    result = []
    i, j = 0, 0
    while i < len(first) and j < len(second):
        result.append(first[i]); i += 1
        if j < len(second): result.append(second[j]); j += 1
    while i < len(first): result.append(first[i]); i += 1
    return result
```

```java
class Solution {
    public void reorderList(ListNode head) {
        if (head == null) return;
        // Find middle
        ListNode slow = head, fast = head;
        while (fast.next != null && fast.next.next != null) {
            slow = slow.next;
            fast = fast.next.next;
        }
        // Reverse second half
        ListNode prev = null, curr = slow.next;
        while (curr != null) {
            ListNode next = curr.next;
            curr.next = prev;
            prev = curr;
            curr = next;
        }
        slow.next = null;
        // Merge two halves
        ListNode first = head, second = prev;
        while (second != null) {
            ListNode tmp1 = first.next, tmp2 = second.next;
            first.next = second;
            second.next = tmp1;
            first = tmp1;
            second = tmp2;
        }
    }
}
```
