# Word Ladder II — Solution

## Problem Overview

Find all shortest transformation sequences from beginWord to endWord, where each step changes exactly one letter and every intermediate word must be in the dictionary.

## Approach: BFS + Backtracking

**Idea:** Use BFS from beginWord to find the shortest distance to endWord, while building a graph of neighbor relationships. Then use DFS to reconstruct all paths that follow the shortest distance.

**Algorithm:**
1. Add beginWord to the word set (if not already present)
2. Build a mapping from each word to its neighbors using BFS:
   - For each word, try changing each character to 'a'..'z'
   - If the new word is in the word set, it is a valid neighbor
3. BFS from beginWord to find shortest distance to endWord:
   - Track distance from beginWord to each word
   - Build adjacency list of valid transformations
4. Use DFS/backtracking from beginWord to endWord, following only edges that move toward the endWord (distance increases by 1)
5. Collect all paths that reach endWord

**Complexity Analysis:**
- **Time:** O(L * N * 26) where L is word length and N is the number of words -- for each word we generate all possible single-letter variations
- **Space:** O(N * L) -- storing the graph, distances, and all paths

import java.util.*;

class Solution {
    public List<List<String>> findLadders(String beginWord, String endWord, List<String> wordList) {
        Set<String> wordSet = new HashSet<>(wordList);
        List<List<String>> result = new ArrayList<>();
        if (!wordSet.contains(endWord)) return result;

        // Build neighbor graph and distances using BFS
        Map<String, List<String>> neighbors = new HashMap<>();
        Map<String, Integer> distance = new HashMap<>();
        for (String word : wordSet) {
            neighbors.put(word, new ArrayList<>());
        }
        neighbors.put(beginWord, new ArrayList<>());
        distance.put(beginWord, 0);

        Queue<String> queue = new LinkedList<>();
        queue.offer(beginWord);
        boolean found = false;

        while (!queue.isEmpty() && !found) {
            int size = queue.size();
            for (int i = 0; i < size; i++) {
                String curr = queue.poll();
                int currDist = distance.get(curr);
                // Generate all possible one-letter variations
                for (String next : getNeighbors(curr, wordSet)) {
                    neighbors.get(curr).add(next);
                    if (!distance.containsKey(next)) {
                        distance.put(next, currDist + 1);
                        if (next.equals(endWord)) found = true;
                        else queue.offer(next);
                    }
                }
            }
        }

        // DFS to reconstruct all shortest paths
        List<String> path = new ArrayList<>();
        path.add(beginWord);
        dfs(beginWord, endWord, neighbors, distance, path, result);
        return result;
    }

    private List<String> getNeighbors(String word, Set<String> wordSet) {
        List<String> result = new ArrayList<>();
        char[] chars = word.toCharArray();
        for (int i = 0; i < chars.length; i++) {
            char original = chars[i];
            for (char c = 'a'; c <= 'z'; c++) {
                if (c == original) continue;
                chars[i] = c;
                String newWord = new String(chars);
                if (wordSet.contains(newWord)) result.add(newWord);
            }
            chars[i] = original;
        }
        return result;
    }

    private void dfs(String word, String endWord, Map<String, List<String>> neighbors,
                     Map<String, Integer> distance, List<String> path, List<List<String>> result) {
        if (word.equals(endWord)) {
            result.add(new ArrayList<>(path));
            return;
        }
        for (String next : neighbors.get(word)) {
            if (distance.get(next) == distance.get(word) + 1) {
                path.add(next);
                dfs(next, endWord, neighbors, distance, path, result);
                path.remove(path.size() - 1);
            }
        }
    }
}

function findLadders(beginWord: string, endWord: string, wordList: string[]): string[][] {
    const wordSet = new Set(wordList);
    const result: string[][] = [];
    if (!wordSet.has(endWord)) return result;

    // BFS: build neighbor graph and distances
    const neighbors = new Map<string, string[]>();
    const distance = new Map<string, number>();
    wordSet.forEach(w => neighbors.set(w, []));
    neighbors.set(beginWord, []);
    distance.set(beginWord, 0);

    const queue: string[] = [beginWord];
    let found = false;

    while (queue.length > 0 && !found) {
        const size = queue.length;
        for (let i = 0; i < size; i++) {
            const curr = queue.shift()!;
            const currDist = distance.get(curr)!;

                neighbors.get(curr)!.push(next);
                if (!distance.has(next)) {
                    distance.set(next, currDist + 1);
                    if (next === endWord) found = true;
                    else queue.push(next);
                }
            }
        }
    }

    // DFS: reconstruct all shortest paths
    const path: string[] = [beginWord];
    dfs(beginWord, endWord, neighbors, distance, path, result);
    return result;
}

function getNeighbors(word: string, wordSet: Set<string>): string[] {
    const result: string[] = [];

    for (let i = 0; i < chars.length; i++) {
        const original = chars[i];
        for (let c = 'a'.charCodeAt(0); c <= 'z'.charCodeAt(0); c++) {
            const ch = String.fromCharCode(c);
            if (ch === original) continue;
            chars[i] = ch;
            const newWord = chars.join('');
            if (wordSet.has(newWord)) result.push(newWord);
        }
        chars[i] = original;
    }
    return result;
}

            distance: Map<string, number>, path: string[], result: string[][]): void {

            path.push(next);
            dfs(next, endWord, neighbors, distance, path, result);
            path.pop();
        }
    }

from collections import deque, defaultdict

def find_ladders(begin_word: str, end_word: str, word_list: list[str]) -> list[list[str]]:
    word_set = set(word_list)
    result = []
    if end_word not in word_set:
        return result

    # BFS: build neighbor graph and compute distances from beginWord
    neighbors = defaultdict(list)
    distance = {begin_word: 0}
    q = deque([begin_word])
    found = False

    while q and not found:
        for _ in range(len(q)):
            curr = q.popleft()
            curr_dist = distance[curr]
            for next_word in get_neighbors(curr, word_set):
                neighbors[curr].append(next_word)
                if next_word not in distance:
                    distance[next_word] = curr_dist + 1
                    if next_word == end_word:
                        found = True
                    else:
                        q.append(next_word)

    # DFS: reconstruct all shortest paths
    def dfs(word: str, path: list[str]) -> None:
        if word == end_word:
            result.append(path[:])
            return
        for next_word in neighbors[word]:
            if distance[next_word] == distance[word] + 1:
                path.append(next_word)
                dfs(next_word, path)
                path.pop()

    dfs(begin_word, [begin_word])
    return result

def get_neighbors(word: str, word_set: set) -> list[str]:
    result = []
    chars = list(word)
    for i in range(len(chars)):
        original = chars[i]
        for c in 'abcdefghijklmnopqrstuvwxyz':
            if c == original:
                continue
            chars[i] = c
            new_word = ''.join(chars)
            if new_word in word_set:
                result.append(new_word)
        chars[i] = original
    return result

use std::collections::{HashMap, HashSet, VecDeque};

fn find_ladders(begin_word: String, end_word: String, word_list: Vec<String>) -> Vec<Vec<String>> {
    let word_set: HashSet<String> = word_list.into_iter().collect();
    let mut result = Vec::new();
    if !word_set.contains(&end_word) { return result; }

    // BFS: build neighbor graph and distances
    let mut neighbors: HashMap<String, Vec<String>> = HashMap::new();
    let mut distance: HashMap<String, i32> = HashMap::new();
    for w in &word_set { neighbors.insert(w.clone(), vec![]); }
    neighbors.insert(begin_word.clone(), vec![]);
    distance.insert(begin_word.clone(), 0);

    let mut q = VecDeque::new();
    q.push_back(begin_word.clone());
    let mut found = false;

    while !q.is_empty() && !found {
        let size = q.len();
        for _ in 0..size {
            let curr = q.pop_front().unwrap();
            let curr_dist = distance[&curr];
            for next in get_neighbors(&curr, &word_set) {
                neighbors.get_mut(&curr).unwrap().push(next.clone());
                if !distance.contains_key(&next) {
                    distance.insert(next.clone(), curr_dist + 1);
                    if next == end_word { found = true; }
                    else { q.push_back(next); }
                }
            }
        }
    }

    // DFS: reconstruct all shortest paths
    let mut path = vec![begin_word.clone()];
    dfs(&begin_word, &end_word, &neighbors, &distance, &mut path, &mut result);
    result
}

fn get_neighbors(word: &str, word_set: &HashSet<String>) -> Vec<String> {
    let mut result = Vec::new();
    let mut chars: Vec<char> = word.chars().collect();
    for i in 0..chars.len() {
        let original = chars[i];
        for c in 'a'..='z' {
            if c == original { continue; }
            chars[i] = c;
            let new_word: String = chars.iter().collect();
            if word_set.contains(&new_word) { result.push(new_word); }
        }
        chars[i] = original;
    }
    result
}

fn dfs(word: &str, end_word: &str, neighbors: &HashMap<String, Vec<String>>,
       distance: &HashMap<String, i32>, path: &mut Vec<String>, result: &mut Vec<Vec<String>>) {
    if word == end_word {
        result.push(path.clone());
        return;
    }
    if let Some(nexts) = neighbors.get(word) {
        for next in nexts {
            if distance.get(next) == Some(&(distance[word] + 1)) {
                path.push(next.clone());
                dfs(next, end_word, neighbors, distance, path, result);
                path.pop();
            }
        }
    }
}

class Solution {
    public IList<IList<string>> FindLadders(string beginWord, string endWord, IList<string> wordList) {
        var wordSet = new HashSet<string>(wordList);
        var result = new List<IList<string>>();
        if (!wordSet.Contains(endWord)) return result;

        // BFS: build neighbor graph and distances
        var neighbors = new Dictionary<string, List<string>>();
        var distance = new Dictionary<string, int>();
        foreach (var w in wordSet) neighbors[w] = new List<string>();
        neighbors[beginWord] = new List<string>();
        distance[beginWord] = 0;

        var queue = new Queue<string>();
        queue.Enqueue(beginWord);
        bool found = false;

        while (queue.Count > 0 && !found) {
            int size = queue.Count;
            for (int i = 0; i < size; i++) {
                var curr = queue.Dequeue();
                int currDist = distance[curr];
                foreach (var next in GetNeighbors(curr, wordSet)) {
                    neighbors[curr].Add(next);
                    if (!distance.ContainsKey(next)) {
                        distance[next] = currDist + 1;
                        if (next == endWord) found = true;
                        else queue.Enqueue(next);
                    }
                }
            }
        }

        // DFS: reconstruct all shortest paths
        var path = new List<string> { beginWord };
        Dfs(beginWord, endWord, neighbors, distance, path, result);
        return result;
    }

    private List<string> GetNeighbors(string word, HashSet<string> wordSet) {
        var result = new List<string>();
        var chars = word.ToCharArray();
        for (int i = 0; i < chars.Length; i++) {
            char original = chars[i];
            for (char c = 'a'; c <= 'z'; c++) {
                if (c == original) continue;
                chars[i] = c;
                var newWord = new string(chars);
                if (wordSet.Contains(newWord)) result.Add(newWord);
            }
            chars[i] = original;
        }
        return result;
    }

    private void Dfs(string word, string endWord, Dictionary<string, List<string>> neighbors,
                     Dictionary<string, int> distance, List<string> path, IList<IList<string>> result) {
        if (word == endWord) {
            result.Add(new List<string>(path));
            return;
        }
        foreach (var next in neighbors[word]) {
            if (distance[next] == distance[word] + 1) {
                path.Add(next);
                Dfs(next, endWord, neighbors, distance, path, result);
                path.RemoveAt(path.Count - 1);
            }
        }
    }

func findLadders(beginWord string, endWord string, wordList []string) [][]string {
    wordSet := make(map[string]bool)
    for _, w := range wordList { wordSet[w] = true }
    result := [][]string{}
    if !wordSet[endWord] { return result }

    // BFS: build neighbor graph and distances
    neighbors := make(map[string][]string)
    distance := make(map[string]int)
    for w := range wordSet { neighbors[w] = []string{} }
    neighbors[beginWord] = []string{}
    distance[beginWord] = 0

    queue := []string{beginWord}
    found := false

    for len(queue) > 0 && !found {
        size := len(queue)
        for i := 0; i < size; i++ {
            curr := queue[0]
            queue = queue[1:]
            currDist := distance[curr]
            for _, next := range getNeighbors(curr, wordSet) {
                neighbors[curr] = append(neighbors[curr], next)
                if _, ok := distance[next]; !ok {
                    distance[next] = currDist + 1
                    if next == endWord { found = true } else { queue = append(queue, next) }
                }
            }
        }
    }

    // DFS: reconstruct all shortest paths
    path := []string{beginWord}
    dfs(beginWord, endWord, neighbors, distance, path, &result)
    return result
}

func getNeighbors(word string, wordSet map[string]bool) []string {
    result := []string{}
    chars := []byte(word)
    for i := 0; i < len(chars); i++ {
        original := chars[i]
        for c := byte('a'); c <= 'z'; c++ {
            if c == original { continue }
            chars[i] = c
            if wordSet[string(chars)] { result = append(result, string(chars)) }
        }
        chars[i] = original
    }
    return result
}

func dfs(word, endWord string, neighbors map[string][]string,
         distance map[string]int, path []string, result *[][]string) {
    if word == endWord {
        seq := make([]string, len(path))
        copy(seq, path)
        *result = append(*result, seq)
        return
    }
    for _, next := range neighbors[word] {
        if distance[next] == distance[word]+1 {
            path = append(path, next)
            dfs(next, endWord, neighbors, distance, path, result)
            path = path[:len(path)-1]
        }
    }
}

import java.util.*

class Solution {
    fun findLadders(beginWord: String, endWord: String, wordList: List<String>): List<List<String>> {
        val wordSet = wordList.toHashSet()
        val result = mutableListOf<List<String>>()
        if (endWord !in wordSet) return result

        // BFS: build neighbor graph and distances
        val neighbors = mutableMapOf<String, MutableList<String>>()
        val distance = mutableMapOf<String, Int>()
        wordSet.forEach { neighbors[it] = mutableListOf() }
        neighbors[beginWord] = mutableListOf()
        distance[beginWord] = 0

        val queue: Queue<String> = LinkedList()
        queue.offer(beginWord)
        var found = false

        while (queue.isNotEmpty() && !found) {
            val size = queue.size
            for (i in 0 until size) {
                val curr = queue.poll()
                val currDist = distance[curr]!!
                for (next in getNeighbors(curr, wordSet)) {
                    neighbors[curr]!!.add(next)
                    if (next !in distance) {
                        distance[next] = currDist + 1
                        if (next == endWord) found = true
                        else queue.offer(next)
                    }
                }
            }
        }

        // DFS: reconstruct all shortest paths
        val path = mutableListOf(beginWord)
        dfs(beginWord, endWord, neighbors, distance, path, result)
        return result
    }

    private fun getNeighbors(word: String, wordSet: Set<String>): List<String> {
        val result = mutableListOf<String>()
        val chars = word.toCharArray()
        for (i in chars.indices) {
            val original = chars[i]
            for (c in 'a'..'z') {
                if (c == original) continue
                chars[i] = c
                val newWord = String(chars)
                if (newWord in wordSet) result.add(newWord)
            }
            chars[i] = original
        }
        return result
    }

    private fun dfs(word: String, endWord: String, neighbors: Map<String, List<String>>,
                    distance: Map<String, Int>, path: MutableList<String>, result: MutableList<List<String>>) {
        if (word == endWord) {
            result.add(ArrayList(path))
            return
        }
        for (next in neighbors[word].orEmpty()) {
            if (distance[next] == distance[word]!! + 1) {
                path.add(next)
                dfs(next, endWord, neighbors, distance, path, result)
                path.removeAt(path.lastIndex)
            }
        }
    }

## Key Takeaways

- **BFS finds the shortest distance** -- it ensures we only reconstruct paths of minimal length
- **DFS reconstructs all paths** -- backtracking explores every valid sequence
- **Neighbor generation** by changing each character to 'a'..'z' is O(26 * L) per word
- **Distance constraint** during DFS ensures we only follow shortest paths
- This is a classic **BFS + DFS** two-phase algorithm
