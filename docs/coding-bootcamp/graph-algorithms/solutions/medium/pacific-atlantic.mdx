# Pacific Atlantic Water Flow — Solution

## Problem Overview

Find all cells where water can flow to both the Pacific Ocean (top/left) and Atlantic Ocean (bottom/right).

## Approach: Reverse BFS/DFS

**Idea:** Instead of checking each cell (expensive), work backwards from the ocean borders. Water flows downhill, so if we start from the ocean and move to equal or higher cells, we find all cells that can reach that ocean.

**Algorithm:**
1. Create two boolean matrices: `pacific` and `atlantic`
2. Run DFS/BFS from all Pacific border cells (top row + left column):
   - Mark reachable cells in `pacific`
   - Move to a neighbor if its height >= current cell's height
3. Run DFS/BFS from all Atlantic border cells (bottom row + right column):
   - Mark reachable cells in `atlantic`
   - Same height condition
4. Collect all cells where both `pacific[r][c]` and `atlantic[r][c]` are true

**Complexity Analysis:**
- **Time:** O(m * n) — each cell visited at most twice (once per ocean)
- **Space:** O(m * n) — two boolean matrices

import java.util.*;

class Solution {
    private int m, n;
    private static final int[][] DIRS = {{1,0},{-1,0},{0,1},{0,-1}};

    public List<List<Integer>> pacificAtlantic(int[][] heights) {
        m = heights.length;
        n = heights[0].length;
        boolean[][] pacific = new boolean[m][n];
        boolean[][] atlantic = new boolean[m][n];

        // DFS from Pacific borders (top row, left column)
        for (int i = 0; i < m; i++) dfs(heights, i, 0, pacific);
        for (int j = 0; j < n; j++) dfs(heights, 0, j, pacific);

        // DFS from Atlantic borders (bottom row, right column)
        for (int i = 0; i < m; i++) dfs(heights, i, n - 1, atlantic);
        for (int j = 0; j < n; j++) dfs(heights, m - 1, j, atlantic);

        // Intersection: cells reachable by both oceans
        List<List<Integer>> result = new ArrayList<>();
        for (int i = 0; i < m; i++) {
            for (int j = 0; j < n; j++) {
                if (pacific[i][j] && atlantic[i][j]) {
                    result.add(Arrays.asList(i, j));
                }
            }
        }
        return result;
    }

    private void dfs(int[][] heights, int i, int j, boolean[][] ocean) {
        ocean[i][j] = true;
        for (int[] d : DIRS) {
            int ni = i + d[0], nj = j + d[1];
            if (ni >= 0 && ni < m && nj >= 0 && nj < n
                && !ocean[ni][nj] && heights[ni][nj] >= heights[i][j]) {
                dfs(heights, ni, nj, ocean);
            }
        }
    }
}

fn pacific_atlantic(heights: Vec<Vec<i32>>) -> Vec<Vec<i32>> {
    let m = heights.len();
    let n = heights[0].len();
    let mut pacific = vec![vec![false; n]; m];
    let mut atlantic = vec![vec![false; n]; m];
    let dirs = [(1, 0), (-1, 0), (0, 1), (0, -1)];

    fn dfs(heights: &Vec<Vec<i32>>, i: i32, j: i32, ocean: &mut Vec<Vec<bool>>, m: i32, n: i32, dirs: &[(i32, i32)]) {
        ocean[i as usize][j as usize] = true;
        for &(di, dj) in dirs {
            let ni = i + di;
            let nj = j + dj;
            if ni >= 0 && ni < m && nj >= 0 && nj < n
                && !ocean[ni as usize][nj as usize]
                && heights[ni as usize][nj as usize] >= heights[i as usize][j as usize] {
                dfs(heights, ni, nj, ocean, m, n, dirs);
            }
        }
    }

    let (mi, ni) = (m as i32, n as i32);
    // DFS from Pacific borders
    for i in 0..m { dfs(&heights, i as i32, 0, &mut pacific, mi, ni, &dirs); }
    for j in 0..n { dfs(&heights, 0, j as i32, &mut pacific, mi, ni, &dirs); }
    // DFS from Atlantic borders
    for i in 0..m { dfs(&heights, i as i32, ni - 1, &mut atlantic, mi, ni, &dirs); }
    for j in 0..n { dfs(&heights, mi - 1, j as i32, &mut atlantic, mi, ni, &dirs); }

    // Intersection
    let mut result = Vec::new();
    for i in 0..m {
        for j in 0..n {
            if pacific[i][j] && atlantic[i][j] {
                result.push(vec![i as i32, j as i32]);
            }
        }
    }
    result
}

int** pacificAtlantic(int** heights, int heightsSize, int* heightsColSize, int* returnSize, int** returnColumnSizes) {
    int m = heightsSize, n = heightsColSize[0];
    bool** pacific = malloc(m * sizeof(bool*));
    bool** atlantic = malloc(m * sizeof(bool*));
    for (int i = 0; i < m; i++) {
        pacific[i] = calloc(n, sizeof(bool));
        atlantic[i] = calloc(n, sizeof(bool));
    }

    // DFS from Pacific borders
    for (int i = 0; i < m; i++) dfs(heights, i, 0, m, n, pacific);
    for (int j = 0; j < n; j++) dfs(heights, 0, j, m, n, pacific);
    // DFS from Atlantic borders
    for (int i = 0; i < m; i++) dfs(heights, i, n - 1, m, n, atlantic);
    for (int j = 0; j < n; j++) dfs(heights, m - 1, j, m, n, atlantic);

    // Count intersection cells
    int count = 0;
    for (int i = 0; i < m; i++)
        for (int j = 0; j < n; j++)
            if (pacific[i][j] && atlantic[i][j]) count++;

    *returnSize = count;
    *returnColumnSizes = malloc(count * sizeof(int));
    int** result = malloc(count * sizeof(int*));
    int idx = 0;
    for (int i = 0; i < m; i++)
        for (int j = 0; j < n; j++)
            if (pacific[i][j] && atlantic[i][j]) {
                result[idx] = malloc(2 * sizeof(int));
                result[idx][0] = i;
                result[idx][1] = j;
                (*returnColumnSizes)[idx] = 2;
                idx++;
            }

    for (int i = 0; i < m; i++) { free(pacific[i]); free(atlantic[i]); }
    free(pacific); free(atlantic);
    return result;
}

## Key Takeaways

- **Work backwards from the ocean** — much more efficient than checking every cell
- **Two-pass DFS/BFS** — once from each ocean's borders
- **Height condition is reversed** — moving to cells with equal or greater height (going uphill)
- The answer is the **intersection** of the two reachable sets
