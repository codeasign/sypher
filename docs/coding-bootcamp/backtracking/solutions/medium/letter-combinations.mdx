# Letter Combinations of a Phone Number — Solution

## Problem Overview

Given a string containing digits from `2-9` inclusive, return all possible letter combinations that the number could represent (like on a telephone keypad). Return the answer in **lexicographic order** (same order as iterating digits left to right, mapping each digit to its letters in order).

## Approach: Backtracking

**Idea:** Build letter combinations incrementally by mapping each digit to its corresponding letters on a phone keypad, and using recursion to try each possible letter for each digit.

**Algorithm:**
1. Define a mapping from digit to letters: `2 → abc`, `3 → def`, `4 → ghi`, `5 → jkl`, `6 → mno`, `7 → pqrs`, `8 → tuv`, `9 → wxyz`
2. If the input is empty, return an empty list
3. Use a backtracking helper that builds the current combination:
   - **Base case:** If the current combination has the same length as the input, add it to the result
   - **Recursive case:** For the current digit, try each of its corresponding letters, add it to the path, and recurse to the next digit

**Why This Works:** The backtracking approach explores all possible choices at each position. By processing digits left to right, and letters in their natural order, the results naturally come out in lexicographic order.

**Complexity Analysis:**
- **Time:** O(4^n) where n is the number of digits — each digit maps to at most 4 letters
- **Space:** O(n) for the recursion stack, plus O(4^n * n) for the output

def letter_combinations(digits: str) -> list[str]:
    if not digits:
        return []

    # Mapping from digit to corresponding letters
    mapping = {
        '2': 'abc', '3': 'def', '4': 'ghi', '5': 'jkl',
        '6': 'mno', '7': 'pqrs', '8': 'tuv', '9': 'wxyz',
    }

    result = []

    def backtrack(index: int, path: list[str]) -> None:
        # Base case: built a complete combination
        if index == len(digits):
            result.append(''.join(path))
            return
        # Try each letter for the current digit
        digit = digits[index]
        for letter in mapping[digit]:
            path.append(letter)
            backtrack(index + 1, path)
            path.pop()

    backtrack(0, [])
    return result

import java.util.ArrayList;
import java.util.List;
import java.util.Map;

class Solution {
    private static final Map<Character, String> MAPPING = Map.of(
        '2', "abc", '3', "def", '4', "ghi", '5', "jkl",
        '6', "mno", '7', "pqrs", '8', "tuv", '9', "wxyz"
    );

    public List<String> letterCombinations(String digits) {
        List<String> result = new ArrayList<>();
        if (digits == null || digits.isEmpty()) {
            return result;
        }
        backtrack(0, new StringBuilder(), digits, result);
        return result;
    }

    private void backtrack(int index, StringBuilder path, String digits, List<String> result) {
        // Base case: built a complete combination
        if (index == digits.length()) {
            result.add(path.toString());
            return;
        }
        // Try each letter for the current digit
        String letters = MAPPING.get(digits.charAt(index));
        for (char letter : letters.toCharArray()) {
            path.append(letter);
            backtrack(index + 1, path, digits, result);
            path.deleteCharAt(path.length() - 1);
        }
    }
}

use std::collections::HashMap;

fn letter_combinations(digits: &str) -> Vec<String> {
    if digits.is_empty() {
        return Vec::new();
    }

    let mapping: HashMap<char, &str> = [
        ('2', "abc"), ('3', "def"), ('4', "ghi"), ('5', "jkl"),
        ('6', "mno"), ('7', "pqrs"), ('8', "tuv"), ('9', "wxyz"),
    ].iter().cloned().collect();

    let mut result = Vec::new();
    let mut path = String::new();
    backtrack(0, digits, &mapping, &mut path, &mut result);
    result
}

fn backtrack(
    index: usize,
    digits: &str,
    mapping: &HashMap<char, &str>,
    path: &mut String,
    result: &mut Vec<String>,
) {
    // Base case: built a complete combination
    if index == digits.len() {
        result.push(path.clone());
        return;
    }
    // Try each letter for the current digit
    let digit = digits.chars().nth(index).unwrap();
    if let Some(&letters) = mapping.get(&digit) {
        for letter in letters.chars() {
            path.push(letter);
            backtrack(index + 1, digits, mapping, path, result);
            path.pop();
        }
    }
}

#include <stdio.h>
#include <stdlib.h>
#include <string.h>

void backtrack(int index, char* digits, char* path, char** result, int* returnSize) {
    static const char* mapping[] = {
        "", "", "abc", "def", "ghi", "jkl", "mno", "pqrs", "tuv", "wxyz"
    };

    // Base case: built a complete combination
    if (digits[index] == '\0') {
        path[index] = '\0';
        result[*returnSize] = strdup(path);
        (*returnSize)++;
        return;
    }

    // Try each letter for the current digit
    const char* letters = mapping[digits[index] - '0'];
    for (int i = 0; letters[i] != '\0'; i++) {
        path[index] = letters[i];
        backtrack(index + 1, digits, path, result, returnSize);
    }
}

char** letterCombinations(char* digits, int* returnSize) {
    *returnSize = 0;
    if (digits == NULL || digits[0] == '\0') {
        return NULL;
    }

    int len = strlen(digits);
    int capacity = 1;
    for (int i = 0; i < len; i++) {
        char d = digits[i];
        if (d == '7' || d == '9') capacity *= 4;
        else capacity *= 3;
    }

    char** result = (char**)malloc(capacity * sizeof(char*));
    char* path = (char*)malloc((len + 1) * sizeof(char));
    backtrack(0, digits, path, result, returnSize);
    free(path);
    return result;
}

using System.Collections.Generic;

class Solution {
    private static readonly Dictionary<char, string> Mapping = new() {
        {'2', "abc"}, {'3', "def"}, {'4', "ghi"}, {'5', "jkl"},
        {'6', "mno"}, {'7', "pqrs"}, {'8', "tuv"}, {'9', "wxyz"}
    };

    public IList<string> LetterCombinations(string digits) {
        var result = new List<string>();
        if (string.IsNullOrEmpty(digits)) {
            return result;
        }
        Backtrack(0, new List<char>(), digits, result);
        return result;
    }

    private void Backtrack(int index, List<char> path, string digits, IList<string> result) {
        // Base case: built a complete combination
        if (index == digits.Length) {
            result.Add(new string(path.ToArray()));
            return;
        }
        // Try each letter for the current digit
        string letters = Mapping[digits[index]];
        foreach (char letter in letters) {
            path.Add(letter);
            Backtrack(index + 1, path, digits, result);
            path.RemoveAt(path.Count - 1);
        }
    }
}

func letterCombinations(digits string) []string {
    if len(digits) == 0 {
        return []string{}
    }

    mapping := map[byte]string{
        '2': "abc", '3': "def", '4': "ghi", '5': "jkl",
        '6': "mno", '7': "pqrs", '8': "tuv", '9': "wxyz",
    }

    result := []string{}
    var backtrack func(index int, path []byte)
    backtrack = func(index int, path []byte) {
        // Base case: built a complete combination
        if index == len(digits) {
            result = append(result, string(path))
            return
        }
        // Try each letter for the current digit
        letters := mapping[digits[index]]
        for i := 0; i < len(letters); i++ {
            path = append(path, letters[i])
            backtrack(index+1, path)
            path = path[:len(path)-1]
        }
    }

    backtrack(0, []byte{})
    return result
}

## Key Takeaways

- **Backtracking** is the natural approach for generating all combinations
- The **digit-to-letter mapping** follows a standard telephone keypad layout
- Processing digits **left to right** and letters **in order** produces lexicographic output automatically
- Base case handles empty input early for cleaner code
