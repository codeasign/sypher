## Build It

### Step 1: Database and Models

```python
# app/database.py
import sqlite3, os
from datetime import datetime

DB_PATH = os.environ.get("DB_PATH", "./data/learnai.db")

def get_conn():
    os.makedirs(os.path.dirname(DB_PATH) or ".", exist_ok=True)
    conn = sqlite3.connect(DB_PATH)
    conn.row_factory = sqlite3.Row
    return conn

def init_db():
    conn = get_conn()
    conn.executescript("""
        CREATE TABLE IF NOT EXISTS courses (
            id TEXT PRIMARY KEY,
            title TEXT NOT NULL,
            topic TEXT NOT NULL,
            description TEXT,
            lesson_count INTEGER DEFAULT 0,
            created_at TEXT NOT NULL
        );
        CREATE TABLE IF NOT EXISTS lessons (
            id TEXT PRIMARY KEY,
            course_id TEXT NOT NULL,
            title TEXT NOT NULL,
            content TEXT NOT NULL,
            order_index INTEGER NOT NULL,
            FOREIGN KEY (course_id) REFERENCES courses(id)
        );
        CREATE TABLE IF NOT EXISTS quizzes (
            id TEXT PRIMARY KEY,
            lesson_id TEXT NOT NULL,
            question TEXT NOT NULL,
            options TEXT,
            correct_answer TEXT,
            difficulty TEXT DEFAULT 'medium',
            FOREIGN KEY (lesson_id) REFERENCES lessons(id)
        );
        CREATE TABLE IF NOT EXISTS progress (
            user_id TEXT NOT NULL,
            lesson_id TEXT NOT NULL,
            completed INTEGER DEFAULT 0,
            score REAL DEFAULT 0,
            attempts INTEGER DEFAULT 0,
            PRIMARY KEY (user_id, lesson_id)
        );
    """)
    conn.commit()
    conn.close()
```

### Step 2: Course Generator

```python
# app/generator.py
from anthropic import Anthropic
from app.config import settings
import json, uuid
from app.database import get_conn

client = Anthropic(api_key=settings.ANTHROPIC_API_KEY)

def generate_course(topic: str, difficulty: str = "beginner") -> dict:
    prompt = f"""Generate a structured course outline for "{topic}" at {difficulty} level.

Return JSON:
{{
    "title": "Course title",
    "description": "2-3 sentence description",
    "lessons": [
        {{
            "title": "Lesson title",
            "content": "Full lesson content (500-1000 words with examples)",
            "order_index": 1,
            "quizzes": [
                {{
                    "question": "Quiz question",
                    "options": ["A", "B", "C", "D"],
                    "correct_answer": "A",
                    "difficulty": "easy|medium|hard"
                }}
            ]
        }}
    ]
}}

Generate 5-8 lessons with 2-3 quizzes per lesson."""

    response = client.messages.create(
        model=settings.LLM_MODEL,
        max_tokens=8000,
        system=f"You are an expert educator creating a {difficulty}-level course. Be thorough and accurate.",
        messages=[{"role": "user", "content": prompt}]
    )

    try:
        course_data = json.loads(response.content[0].text)
    except json.JSONDecodeError:
        raise ValueError("Failed to parse course generation output")

    # Save to database
    conn = get_conn()
    course_id = uuid.uuid4().hex[:12]
    conn.execute(
        "INSERT INTO courses VALUES (?, ?, ?, ?, ?, ?)",
        (course_id, course_data["title"], topic, course_data["description"],
         len(course_data["lessons"]), datetime.utcnow().isoformat())
    )

    for lesson in course_data["lessons"]:
        lesson_id = uuid.uuid4().hex[:12]
        conn.execute(
            "INSERT INTO lessons VALUES (?, ?, ?, ?, ?)",
            (lesson_id, course_id, lesson["title"], lesson["content"], lesson["order_index"])
        )
        for quiz in lesson.get("quizzes", []):
            conn.execute(
                "INSERT INTO quizzes (id, lesson_id, question, options, correct_answer, difficulty) VALUES (?, ?, ?, ?, ?, ?)",
                (uuid.uuid4().hex[:12], lesson_id, quiz["question"],
                 json.dumps(quiz["options"]), quiz["correct_answer"], quiz.get("difficulty", "medium"))
            )

    conn.commit()
    conn.close()
    return {"course_id": course_id, "title": course_data["title"], "lessons": len(course_data["lessons"])}
```

### Step 3: Quiz Engine

```python
# app/quiz.py
from anthropic import Anthropic
from app.config import settings
from app.database import get_conn
import json, random

client = Anthropic(api_key=settings.ANTHROPIC_API_KEY)

def get_quiz(lesson_id: str, user_id: str) -> dict:
    conn = get_conn()
    quizzes = conn.execute(
        "SELECT * FROM quizzes WHERE lesson_id = ? ORDER BY RANDOM() LIMIT 3",
        (lesson_id,)
    ).fetchall()

    progress = conn.execute(
        "SELECT score FROM progress WHERE user_id = ? AND lesson_id = ?",
        (user_id, lesson_id)
    ).fetchone()
    conn.close()

    difficulty = "easy"
    if progress and progress["score"] > 0.8:
        difficulty = "hard"
    elif progress and progress["score"] > 0.5:
        difficulty = "medium"

    filtered = [q for q in quizzes if q["difficulty"] == difficulty] or quizzes

    return {
        "quizzes": [{
            "id": q["id"],
            "question": q["question"],
            "options": json.loads(q["options"]),
            "difficulty": q["difficulty"]
        } for q in filtered[:3]]
    }

def evaluate_answer(user_id: str, lesson_id: str, quiz_id: str, answer: str) -> dict:
    conn = get_conn()
    quiz = conn.execute("SELECT * FROM quizzes WHERE id = ?", (quiz_id,)).fetchone()

    is_correct = answer.strip().upper() == quiz["correct_answer"].strip().upper()

    # Update progress
    existing = conn.execute(
        "SELECT * FROM progress WHERE user_id = ? AND lesson_id = ?",
        (user_id, lesson_id)
    ).fetchone()

    if existing:
        new_attempts = existing["attempts"] + 1
        new_score = (existing["score"] * existing["attempts"] + (1.0 if is_correct else 0.0)) / new_attempts
        conn.execute(
            "UPDATE progress SET score = ?, attempts = ?, completed = ? WHERE user_id = ? AND lesson_id = ?",
            (new_score, new_attempts, 1 if new_score > 0.6 else 0, user_id, lesson_id)
        )
    else:
        conn.execute(
            "INSERT INTO progress VALUES (?, ?, ?, ?, ?)",
            (user_id, lesson_id, 1 if is_correct else 0, 1.0 if is_correct else 0.0, 1)
        )

    conn.commit()
    conn.close()

    return {
        "correct": is_correct,
        "correct_answer": quiz["correct_answer"],
        "passed": is_correct
    }
```

### Step 4: Progress Tracking

```python
# app/progress.py
from app.database import get_conn

def get_course_progress(user_id: str, course_id: str) -> dict:
    conn = get_conn()
    total = conn.execute(
        "SELECT COUNT(*) as c FROM lessons WHERE course_id = ?", (course_id,)
    ).fetchone()["c"]
    completed = conn.execute(
        """SELECT COUNT(*) as c FROM progress p
           JOIN lessons l ON p.lesson_id = l.id
           WHERE p.user_id = ? AND l.course_id = ? AND p.completed = 1""",
        (user_id, course_id)
    ).fetchone()["c"]
    conn.close()
    return {"total": total, "completed": completed, "percent": round(completed / total * 100) if total > 0 else 0}

def recommend_review(user_id: str) -> list:
    conn = get_conn()
    weak = conn.execute(
        """SELECT l.title, p.score FROM progress p
           JOIN lessons l ON p.lesson_id = l.id
           WHERE p.user_id = ? AND p.score < 0.6
           ORDER BY p.score ASC LIMIT 5""",
        (user_id,)
    ).fetchall()
    conn.close()
    return [{"lesson": r["title"], "score": r["score"]} for r in weak]
```

### Step 5: Flashcard Generator

```python
# app/flashcards.py
from anthropic import Anthropic
from app.config import settings
import json

client = Anthropic(api_key=settings.ANTHROPIC_API_KEY)

def generate_flashcards(lesson_content: str, count: int = 10) -> list[dict]:
    response = client.messages.create(
        model=settings.LLM_MODEL,
        max_tokens=2000,
        system="Generate flashcards from lesson content. Return JSON array: [{\"front\": \"question\", \"back\": \"answer\"}]",
        messages=[{"role": "user", "content": f"Lesson content:\n{lesson_content[:3000]}\n\nGenerate {count} flashcards."}]
    )
    try:
        return json.loads(response.content[0].text)
    except json.JSONDecodeError:
        return [{"front": "Error", "back": "Failed to generate flashcards"}]
```

### Step 6: API Routes

```python
# app/api.py
from fastapi import FastAPI, HTTPException
from pydantic import BaseModel
from app.generator import generate_course
from app.quiz import get_quiz, evaluate_answer
from app.progress import get_course_progress, recommend_review
from app.database import get_conn, init_db
from app.flashcards import generate_flashcards
import uuid

app = FastAPI(title="LearnAI API")

class GenerateRequest(BaseModel):
    topic: str
    difficulty: str = "beginner"

class QuizAnswer(BaseModel):
    user_id: str
    lesson_id: str
    quiz_id: str
    answer: str

@app.on_event("startup")
async def startup():
    init_db()

@app.post("/api/courses/generate")
def create_course(req: GenerateRequest):
    result = generate_course(req.topic, req.difficulty)
    return result

@app.get("/api/courses")
def list_courses():
    conn = get_conn()
    courses = conn.execute("SELECT id, title, topic, lesson_count, created_at FROM courses ORDER BY created_at DESC").fetchall()
    conn.close()
    return [dict(c) for c in courses]

@app.get("/api/courses/{course_id}")
def get_course(course_id: str):
    conn = get_conn()
    course = conn.execute("SELECT * FROM courses WHERE id = ?", (course_id,)).fetchone()
    if not course:
        raise HTTPException(404, "Course not found")
    lessons = conn.execute("SELECT id, title, order_index FROM lessons WHERE course_id = ? ORDER BY order_index", (course_id,)).fetchall()
    conn.close()
    return {**dict(course), "lessons": [dict(l) for l in lessons]}

@app.get("/api/lessons/{lesson_id}")
def get_lesson(lesson_id: str):
    conn = get_conn()
    lesson = conn.execute("SELECT * FROM lessons WHERE id = ?", (lesson_id,)).fetchone()
    conn.close()
    if not lesson:
        raise HTTPException(404, "Lesson not found")
    return dict(lesson)

@app.get("/api/quiz/{lesson_id}")
def quiz(lesson_id: str, user_id: str = "default"):
    return get_quiz(lesson_id, user_id)

@app.post("/api/quiz/answer")
def answer(data: QuizAnswer):
    return evaluate_answer(data.user_id, data.lesson_id, data.quiz_id, data.answer)

@app.get("/api/progress/{user_id}/{course_id}")
def progress(user_id: str, course_id: str):
    return get_course_progress(user_id, course_id)

@app.get("/api/recommendations/{user_id}")
def recommendations(user_id: str):
    return recommend_review(user_id)

@app.post("/api/flashcards/{lesson_id}")
def flashcards(lesson_id: str):
    conn = get_conn()
    lesson = conn.execute("SELECT content FROM lessons WHERE id = ?", (lesson_id,)).fetchone()
    conn.close()
    if not lesson:
        raise HTTPException(404, "Lesson not found")
    return {"flashcards": generate_flashcards(lesson["content"])}
```

## Understand What Happened

- **Course generation** creates 5-8 lessons with quizzes, each with content written for the specified difficulty level
- **Adaptive quizzes** track performance per user and adjust question difficulty accordingly
- **Progress tracking** computes completion percentage and identifies weak areas for review
- **Flashcards** are generated on-demand from lesson content — perfect for spaced repetition review

## Common Errors

### Error 1: Course Generation Fails JSON Validation

```
json.JSONDecodeError: Expecting value
```

**Root cause:** The LLM returned malformed JSON or exceeded the token limit.

**Fix:** Increase `max_tokens` to 8000+ and add retry logic:
```python
for attempt in range(3):
    try:
        return json.loads(response.content[0].text)
    except json.JSONDecodeError:
        continue
```

### Error 2: Quiz Options Are Not Valid JSON

**Root cause:** The quiz options field is stored as a JSON string and fails to parse.

**Fix:** Validate quiz options before insertion:
```python
options = json.dumps(quiz.get("options", ["A", "B", "C", "D"]))
```
