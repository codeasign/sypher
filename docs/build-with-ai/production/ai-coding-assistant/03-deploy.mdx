## Production Hardening

### 1. Safe File Operations

```python
# app/safety.py
from pathlib import Path
import difflib

def safe_write(filepath: Path, content: str, dry_run: bool = True):
    """Preview or apply changes with rollback support."""
    if dry_run:
        if filepath.exists():
            original = filepath.read_text(encoding="utf-8")
            diff = difflib.unified_diff(
                original.splitlines(keepends=True),
                content.splitlines(keepends=True),
                fromfile=str(filepath),
                tofile=str(filepath)
            )
            print("".join(diff))
        else:
            print(f"[New file] {filepath}\n{content[:200]}...")
        return

    backup = None
    if filepath.exists():
        backup = filepath.read_bytes()
        filepath.rename(filepath.with_suffix(filepath.suffix + ".bak"))

    try:
        filepath.write_text(content, encoding="utf-8")
    except Exception:
        if backup:
            filepath.write_bytes(backup)
        raise
```

### 2. Review Result Caching

```python
# app/cache.py
import sqlite3, json, hashlib
from app.config import settings

def get_review_cache():
    conn = sqlite3.connect(settings.CACHE_PATH)
    conn.execute("CREATE TABLE IF NOT EXISTS review_cache (content_hash TEXT PRIMARY KEY, findings TEXT, created_at TEXT)")
    return conn

def cached_review(content: str, review_fn):
    content_hash = hashlib.sha256(content.encode()).hexdigest()[:16]
    conn = get_review_cache()
    row = conn.execute("SELECT findings FROM review_cache WHERE content_hash = ?", (content_hash,)).fetchone()
    if row:
        conn.close()
        return json.loads(row[0])
    findings = review_fn(content)
    conn.execute("INSERT OR REPLACE INTO review_cache VALUES (?, ?, datetime('now'))", (content_hash, json.dumps(findings)))
    conn.commit()
    conn.close()
    return findings
```

## Deploy It

### PyPI Package

Package CodeMate as a pip-installable CLI tool:

```python
# setup.py
from setuptools import setup, find_packages

setup(
    name="codemate",
    version="0.1.0",
    packages=find_packages(),
    install_requires=[
        "anthropic>=0.49.0",
        "click>=8.1.0",
        "chromadb>=0.5.0",
        "pydantic>=2.0.0",
        "pygments>=2.18.0",
    ],
    entry_points={
        "console_scripts": ["codemate=app.cli:cli"],
    },
)
```

```bash
pip install -e .
codemate --help
# => Usage: codemate [OPTIONS] COMMAND...
```
