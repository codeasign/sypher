## Build It

### Step 1: Project Scanner

```python
# app/scanner.py
from pathlib import Path
import hashlib
from typing import List
import os

def is_text_file(filepath: Path) -> bool:
    ext = filepath.suffix.lower()
    text_extensions = {
        ".py", ".js", ".ts", ".jsx", ".tsx", ".java", ".go", ".rs", ".rb",
        ".c", ".cpp", ".h", ".hpp", ".cs", ".swift", ".kt", ".scala",
        ".php", ".sh", ".bash", ".zsh", ".yaml", ".yml", ".json", ".xml",
        ".html", ".css", ".scss", ".sql", ".r", ".m", ".mm"
    }
    return ext in text_extensions

def scan_project(root: Path) -> List[dict]:
    files = []
    ignore_dirs = {".git", "__pycache__", "node_modules", ".venv", "venv", "dist", "build", ".next"}

    for filepath in root.rglob("*"):
        if any(part.startswith(".") and part != "." for part in filepath.parts):
            if filepath.name.startswith("."):
                continue
        if filepath.is_dir() and filepath.name in ignore_dirs:
            continue
        if filepath.is_file() and is_text_file(filepath):
            rel_path = filepath.relative_to(root)
            content = filepath.read_text(encoding="utf-8", errors="replace")
            files.append({
                "path": str(rel_path),
                "language": filepath.suffix.lstrip("."),
                "size": len(content),
                "checksum": hashlib.sha256(content.encode()).hexdigest()[:12],
                "content": content
            })
    return files
```

### Step 2: Symbol Extraction

```python
# app/parser.py
import re
from typing import List
from app.models import CodeSymbol

def extract_symbols(file_path: str, content: str, language: str) -> List[CodeSymbol]:
    symbols = []

    if language == "py":
        # Functions
        for match in re.finditer(r'^def\s+(\w+)\s*\(', content, re.MULTILINE):
            line = content[:match.start()].count("\n") + 1
            symbols.append(CodeSymbol(
                name=match.group(1), kind="function",
                file_path=file_path, start_line=line, end_line=line
            ))
        # Classes
        for match in re.finditer(r'^class\s+(\w+)', content, re.MULTILINE):
            line = content[:match.start()].count("\n") + 1
            symbols.append(CodeSymbol(
                name=match.group(1), kind="class",
                file_path=file_path, start_line=line, end_line=line
            ))

    elif language in ("js", "ts", "jsx", "tsx"):
        # Functions
        for match in re.finditer(r'(?:function|const)\s+(\w+)\s*(?:=|\(|:)', content):
            line = content[:match.start()].count("\n") + 1
            symbols.append(CodeSymbol(
                name=match.group(1), kind="function",
                file_path=file_path, start_line=line, end_line=line
            ))

    return symbols
```

### Step 3: Code Indexer

```python
# app/indexer.py
import chromadb
from chromadb.config import Settings
from app.config import settings
from app.scanner import scan_project
from app.parser import extract_symbols
from pathlib import Path
import json

class CodeIndexer:
    def __init__(self):
        self.client = chromadb.PersistentClient(
            path=settings.INDEX_PATH,
            settings=Settings(anonymized_telemetry=False)
        )
        self.collection = self.client.get_or_create_collection("code_index")

    def index_project(self, root: str = "."):
        files = scan_project(Path(root))
        documents = []
        metadatas = []
        ids = []

        for f in files:
            symbols = extract_symbols(f["path"], f["content"], f["language"])
            symbols_json = json.dumps([s.model_dump() for s in symbols])

            documents.append(f"{f['path']}\n\n{f['content'][:2000]}")
            metadatas.append({
                "file_path": f["path"],
                "language": f["language"],
                "size": f["size"],
                "symbols": symbols_json,
                "checksum": f["checksum"]
            })
            ids.append(f["path"])

        self.collection.upsert(ids=ids, documents=documents, metadatas=metadatas)
        print(f"Indexed {len(files)} files")

    def search(self, query: str, top_k: int = 5):
        results = self.collection.query(query_texts=[query], n_results=top_k)
        return [{
            "file_path": m["file_path"],
            "language": m["language"],
            "symbols": json.loads(m.get("symbols", "[]"))
        } for m in results["metadatas"][0]]
```

```bash
for r in results:

# =>   app/parser.py (py)
"
```

### Step 4: Code Explanation

```python
# app/commands/explain.py
from anthropic import Anthropic
from app.config import settings
from app.indexer import CodeIndexer
from pathlib import Path

client = Anthropic(api_key=settings.ANTHROPIC_API_KEY)

def explain(file_path: str, function: str = None) -> str:
    content = Path(file_path).read_text(encoding="utf-8", errors="replace")
    lines = content.split("\n")

    target = ""
    if function:
        for i, line in enumerate(lines):
            if function in line:
                start = max(0, i - 2)
                end = min(len(lines), i + 30)
                target = "\n".join(lines[start:end])
                break
    else:
        target = content[:3000]

    response = client.messages.create(
        model=settings.LLM_MODEL,
        max_tokens=1000,
        system="You are an expert programmer. Explain the provided code in detail: what it does, how it works, edge cases, and any potential issues.",
        messages=[{
            "role": "user",
            "content": f"```{Path(file_path).suffix.lstrip('.')}\n{target}\n```\n\nExplain what this code does."
        }]
    )
    return response.content[0].text
```

### Step 5: Code Review

```python
# app/commands/review.py
from anthropic import Anthropic
from app.config import settings
from pathlib import Path
import json

client = Anthropic(api_key=settings.ANTHROPIC_API_KEY)

def review(file_path: str, diff: str = None) -> list[dict]:
    content = Path(file_path).read_text(encoding="utf-8", errors="replace")
    target = diff if diff else content[:5000]

    response = client.messages.create(
        model="claude-sonnet-4-20250514",
        max_tokens=2000,
        system="You are a senior code reviewer. Find bugs, security issues, performance problems, and design flaws. Return findings as a JSON array with severity, line, message, suggestion, and category.",
        messages=[{
            "role": "user",
            "content": f"Review this code and return a JSON array of findings:\n\n```\n{target}\n```"
        }]
    )
    try:
        return json.loads(response.content[0].text)
    except json.JSONDecodeError:
        return [{"severity": "info", "message": "Could not parse review results", "category": "other"}]
```

### Step 6: Code Generation

```python
# app/commands/generate.py
from anthropic import Anthropic
from app.config import settings
from pathlib import Path

client = Anthropic(api_key=settings.ANTHROPIC_API_KEY)

def generate(description: str, language: str = "python", output_file: str = None) -> str:
    response = client.messages.create(
        model=settings.LLM_MODEL,
        max_tokens=2000,
        system=f"You are an expert {language} developer. Generate production-quality code based on the description. Include type hints, docstrings, and error handling.",
        messages=[{
            "role": "user",
            "content": f"Generate {language} code:\n\n{description}"
        }]
    )

    code = response.content[0].text

    if output_file:
        Path(output_file).write_text(code, encoding="utf-8")
        print(f"Generated: {output_file}")

    return code
```

### Step 7: CLI Router

```python
# app/cli.py
import click
from pathlib import Path

@click.group()
def cli():
    """CodeMate — AI-powered coding assistant"""
    pass

@cli.command()
@click.argument("path")
@click.option("--function", "-f", help="Specific function to explain")
def explain(path, function):
    """Explain code in natural language."""
    from app.commands.explain import explain as explain_code
    result = explain_code(path, function)
    click.echo(result)

@cli.command()
@click.argument("path")
@click.option("--diff", "-d", help="Diff text to review (optional)")
def review(path, diff):
    """Review code for bugs and issues."""
    from app.commands.review import review as review_code
    findings = review_code(path, diff)
    for f in findings:
        severity_colors = {"error": "red", "warning": "yellow", "info": "blue"}
        click.secho(f"[{f['severity'].upper()}] {f['message']}", fg=severity_colors.get(f['severity'], 'white'))
        if 'suggestion' in f:
            click.echo(f"  Suggestion: {f['suggestion']}")

@cli.command()
@click.argument("description")
@click.option("--language", "-l", default="python", help="Target language")
@click.option("--output", "-o", help="Output file path")
def generate(description, language, output):
    """Generate code from a description."""
    from app.commands.generate import generate as gen_code
    result = gen_code(description, language, output)
    click.echo(result)

@cli.command()
def index():
    """Index the current project for searching."""
    from app.indexer import CodeIndexer
    idx = CodeIndexer()
    idx.index_project(".")

@cli.command()
@click.argument("query")
@click.option("--top-k", default=5, help="Number of results")
def search(query, top_k):
    """Search indexed codebase."""
    from app.indexer import CodeIndexer
    idx = CodeIndexer()
    results = idx.search(query, top_k)
    for r in results:
        click.echo(f"{r['file_path']} ({len(r['symbols'])} symbols)")

if __name__ == "__main__":
    cli()
```

## Understand What Happened

- **Project scanner** respects `.gitignore`-like patterns and only indexes text files you actually want
- **Symbol extraction** uses regex for speed — tree-sitter would be more accurate but 10x slower for initial indexing
- **ChromaDB code index** lets you search semantically: "where do we handle auth?" instead of grepping for specific terms
- **Sonnet 4** for code review catches subtle bugs that simpler models miss — worth the extra cost per review

## Common Errors

### Error 1: Tree-Sitter Language Not Found

```
tree_sitter.language.LanguageError: No language for extension .xyz
```

**Root cause:** You're using a language tree-sitter doesn't support natively.

**Fix:** The scanner already filters to known extensions. Add your language's parser:
```bash
pip install tree-sitter-<language>
```

### Error 2: ChromaDB Embedding Rate Limit

```
openai.RateLimitError
```

**Root cause:** Indexing a large project hits OpenAI embedding rate limits.

**Fix:** Index in batches with delays:
```python
import time
for batch in batched(files, 20):
    collection.upsert(...)
    time.sleep(1)
```
