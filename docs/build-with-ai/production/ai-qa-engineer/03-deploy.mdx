## Production Hardening

### 1. Timeout Protection

```python
# app/executor.py (add timeout)
import signal

class TimeoutError(Exception):
    pass

def timeout_handler(seconds: int):
    def decorator(func):
        def wrapper(*args, **kwargs):
            signal.alarm(seconds)
            try:
                return func(*args, **kwargs)
            except signal.SIGALRM:
                raise TimeoutError(f"Test execution exceeded {seconds}s")
            finally:
                signal.alarm(0)
        return wrapper
    return decorator
```

### 2. Test Sandboxing

```python
# app/sandbox.py
import tempfile, os, sys, shutil
from pathlib import Path

class TestSandbox:
    def __init__(self):
        self.temp_dir = Path(tempfile.mkdtemp(prefix="testbot_"))

    def __enter__(self):
        return self

    def __exit__(self, *args):
        shutil.rmtree(self.temp_dir, ignore_errors=True)

    def prepare(self, source_file: Path, test_content: str) -> Path:
        shutil.copy2(source_file, self.temp_dir / source_file.name)
        test_file = self.temp_dir / f"test_{source_file.name}"
        test_file.write_text(test_content, encoding="utf-8")
        return test_file

    def run(self, test_file: Path):
        old_cwd = os.getcwd()
        os.chdir(self.temp_dir)
        try:
            result = subprocess.run(
                [sys.executable, "-m", "pytest", str(test_file), "-v", "--tb=short"],
                capture_output=True, text=True, timeout=60,
                env={**os.environ, "PYTHONPATH": str(self.temp_dir)}
            )
            return result
        finally:
            os.chdir(old_cwd)
```

## Deploy It

### Pre-commit Hook

Integrate TestBot into CI as a pre-commit hook:

```yaml
# .pre-commit-config.yaml
repos:
  - repo: local
    hooks:
      - id: testbot
        name: TestBot AI QA
        entry: testbot full-pipeline
        language: python
        files: \.py$
```

### GitHub Actions

```yaml
# .github/workflows/testbot.yml
name: TestBot AI QA
on: [pull_request]
jobs:
  testbot:
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v4
      - uses: actions/setup-python@v5
        with:
          python-version: "3.11"
      - name: Install TestBot
        run: pip install testbot-ai
      - name: Run AI Tests
        run: testbot full-pipeline --changed-only
        env:
          ANTHROPIC_API_KEY: ${{ secrets.ANTHROPIC_API_KEY }}
```
