## Build It

### Step 1: Source Analyzer

```python
# app/analyzer.py
import ast
from pathlib import Path
from typing import List, Dict

class FunctionInfo:
    def __init__(self, name: str, node: ast.FunctionDef, source: str):
        self.name = name
        self.lineno = node.lineno
        self.end_lineno = node.end_lineno
        self.args = [a.arg for a in node.args.args]
        self.decorators = [ast.unparse(d) for d in node.decorator_list]
        self.docstring = ast.get_docstring(node)
        self.returns = ast.unparse(node.returns) if node.returns else None
        self.source = source
        self.complexity = self._compute_complexity(node)

    def _compute_complexity(self, node: ast.AST) -> int:
        complexity = 1
        for child in ast.walk(node):
            if isinstance(child, (ast.If, ast.While, ast.For, ast.ExceptHandler, ast.AsyncFor, ast.AsyncWith)):
                complexity += 1
            elif isinstance(child, ast.BoolOp):
                complexity += len(child.values) - 1
        return complexity

def analyze_file(filepath: Path) -> List[FunctionInfo]:
    source = filepath.read_text(encoding="utf-8")
    tree = ast.parse(source)
    functions = []

    for node in ast.walk(tree):
        if isinstance(node, ast.FunctionDef):
            func_info = FunctionInfo(node.name, node, source)
            functions.append(func_info)

    return functions
```

```bash
python -c "
from app.analyzer import analyze_file
from pathlib import Path
# Create a test file
Path('/tmp/calc.py').write_text('''def add(a, b):\\n    return a + b\\n\\ndef divide(a, b):\\n    if b == 0:\\n        raise ValueError(\\\"Cannot divide by zero\\\")\\n    return a / b\\n''')
funcs = analyze_file(Path('/tmp/calc.py'))
for f in funcs:
    print(f'{f.name}({f.args}) complexity={f.complexity}')
# => add(['a', 'b']) complexity=1
# => divide(['a', 'b']) complexity=2
"
```

**Milestone 1**: source analyzer extracts function signatures, args, docstrings, and cyclomatic complexity.

### Step 2: Test Generator

```python
# app/generator.py
from anthropic import Anthropic
from app.config import settings
from app.analyzer import FunctionInfo
import json

client = Anthropic(api_key=settings.ANTHROPIC_API_KEY)

def generate_tests(func: FunctionInfo, module_name: str) -> str:
    prompt = f"""Generate pytest tests for this function:

Function: {func.name}
Arguments: {func.args}
Returns: {func.returns}
Docstring: {func.docstring}
Complexity: {func.complexity}
Source: {func.source}

Requirements:
- Use pytest.raises for error cases
- Use parametrize for multiple inputs
- Include edge cases (empty, None, boundary values)
- Test both happy path and error path
- Use descriptive test names
- Return ONLY the test code, no explanations

```python
import pytest
from {module_name} import {func.name}

"""

    response = client.messages.create(
        model=settings.LLM_MODEL,
        max_tokens=2000,
        system="You are a senior QA engineer. Generate comprehensive pytest test cases.",
        messages=[{"role": "user", "content": prompt}]
    )

    code = response.content[0].text
    # Extract code from markdown if present
    if "```" in code:
        code = code.split("```")[1]
        if code.startswith("python"):
            code = code[6:]
    return code.strip()
```

### Step 3: Test Executor

```python
# app/executor.py
import subprocess
import sys
from pathlib import Path
from typing import Dict
import json

def run_tests(test_file: Path, source_dir: Path = None) -> Dict:
    cmd = [
        sys.executable, "-m", "pytest",
        str(test_file),
        "-v", "--tb=short",
        "--json-report"
    ]
    if source_dir:
        cmd.extend(["--rootdir", str(source_dir)])

    result = subprocess.run(
        cmd, capture_output=True, text=True, timeout=60
    )

    return {
        "returncode": result.returncode,
        "passed": "passed" in result.stdout and result.returncode == 0,
        "stdout": result.stdout,
        "stderr": result.stderr,
        "output": result.stdout + result.stderr
    }

def run_with_coverage(test_file: Path, source_file: Path) -> Dict:
    cmd = [
        sys.executable, "-m", "pytest",
        str(test_file),
        f"--cov={source_file.parent}",
        f"--cov-report=json",
        "-v", "--tb=short"
    ]
    result = subprocess.run(cmd, capture_output=True, text=True, timeout=60)

    coverage_data = {}
    cov_json = Path("coverage.json")
    if cov_json.exists():
        coverage_data = json.loads(cov_json.read_text())
        cov_json.unlink()

    return {
        "returncode": result.returncode,
        "coverage": coverage_data,
        "output": result.stdout + result.stderr
    }
```

### Step 4: Failure Analysis

```python
# app/failure_analyzer.py
from anthropic import Anthropic
from app.config import settings

client = Anthropic(api_key=settings.ANTHROPIC_API_KEY)

def analyze_failure(source_code: str, test_code: str, test_output: str) -> dict:
    response = client.messages.create(
        model=settings.LLM_MODEL,
        max_tokens=1000,
        system="You are a debugging expert. Determine if test failures are caused by bugs in the source code or incorrect test assumptions. Return JSON with: root_cause, is_bug, fix_suggestion, confidence.",
        messages=[{
            "role": "user",
            "content": f"Source code:\n```\n{source_code[:1500]}\n```\n\nTest code:\n```\n{test_code[:1500]}\n```\n\nTest output:\n```\n{test_output[:1000]}\n```\n\nIs this a bug in the source or a bad test?"
        }]
    )
    import json
    try:
        return json.loads(response.content[0].text)
    except json.JSONDecodeError:
        return {"root_cause": "unknown", "is_bug": False, "confidence": 0.0}
```

### Step 5: Coverage Reporter

```python
# app/reporter.py
from rich.console import Console
from rich.table import Table
from rich.panel import Panel
from typing import Dict, List

console = Console()

def print_report(results: Dict):
    table = Table(title="Test Results")
    table.add_column("Function", style="cyan")
    table.add_column("Tests", justify="right")
    table.add_column("Passed", justify="right")
    table.add_column("Failed", justify="right")
    table.add_column("Coverage", justify="right")

    for func, data in results.items():
        passed = data.get("passed", 0)
        failed = data.get("failed", 0)
        total = passed + failed
        coverage = data.get("coverage", 0)
        status = "green" if failed == 0 else "red"

        table.add_row(
            func, str(total), str(passed), str(failed),
            f"{coverage:.0f}%", style=status
        )

    console.print(table)
    console.print(Panel(f"Total: {sum(r['passed']+r['failed'] for r in results.values())} tests"))
```

### Step 6: CLI

```python
# app/cli.py
import click
from pathlib import Path
from app.analyzer import analyze_file
from app.generator import generate_tests
from app.executor import run_tests, run_with_coverage
from app.failure_analyzer import analyze_failure
from app.reporter import print_report
import tempfile

@click.group()
def cli():
    """TestBot — AI-powered QA Engineer"""
    pass

@cli.command()
@click.argument("source_file")
@click.option("--output", "-o", help="Output test file")
def generate(source_file, output):
    """Generate tests for a Python file."""
    source_path = Path(source_file)
    functions = analyze_file(source_path)

    all_tests = []
    module_name = source_path.stem
    for func in functions:
        click.echo(f"Generating tests for {func.name}...")
        tests = generate_tests(func, module_name)
        all_tests.append(tests)

    output_path = Path(output or f"test_{source_path.name}")
    output_path.write_text("\n\n\n".join(all_tests), encoding="utf-8")
    click.echo(f"Tests written to {output_path}")

@cli.command()
@click.argument("test_file")
@click.option("--source", help="Source file for coverage")
def run(test_file, source):
    """Run generated tests."""
    test_path = Path(test_file)
    if source:
        source_path = Path(source)
        result = run_with_coverage(test_path, source_path)
        if result["coverage"]:
            print_report({test_file: {"passed": 0, "failed": 0, "coverage": result["coverage"].get("totals", {}).get("percent_covered", 0)}})
    else:
        result = run_tests(test_path)

    click.echo(result["output"])

@cli.command()
@click.argument("source_file")
def full_pipeline(source_file):
    """Generate, run, and analyze tests for a file."""
    source_path = Path(source_file)
    functions = analyze_file(source_path)

    with tempfile.TemporaryDirectory() as tmpdir:
        test_file = Path(tmpdir) / f"test_{source_path.name}"
        module_name = source_path.stem

        all_tests = []
        for func in functions:
            click.echo(f"Generating tests for {func.name}...")
            tests = generate_tests(func, module_name)
            all_tests.append(tests)

        test_file.write_text("\n\n\n".join(all_tests), encoding="utf-8")
        click.echo(f"\nRunning {len(functions)} test suites...")

        result = run_tests(test_file)
        if result["passed"]:
            click.secho("All tests passed!", fg="green", bold=True)
        else:
            click.secho("Some tests failed. Analyzing...", fg="yellow")
            analysis = analyze_failure(
                source_path.read_text(),
                test_file.read_text(),
                result["output"]
            )
            click.echo(f"Root cause: {analysis.get('root_cause', 'unknown')}")
            if analysis.get("is_bug"):
                click.secho("BUG FOUND in source code!", fg="red", bold=True)

        click.echo(result["output"])

if __name__ == "__main__":
    cli()
```

# => Generating tests for add...

# => Running 2 test suites...
# => All tests passed!

```

**Milestone 4**: end-to-end pipeline generates, runs, and analyzes tests in one command.

## Understand What Happened

- **AST analysis** extracts function signatures without executing code — safe for any input
- **LLM test generation** produces edge-case-aware tests that humans often miss
- **Subprocess execution** keeps the test environment isolated from the main process
- **Failure analysis** distinguishes real bugs from bad test assumptions using a second LLM call

## Common Errors

### Error 1: Syntax Error in Generated Test

```
SyntaxError: invalid syntax
```

**Root cause:** LLM sometimes generates code with syntax errors or references missing imports.

**Fix:** Run `pytest --collect-only` to validate syntax before execution:
```python
subprocess.run([sys.executable, "-m", "pytest", "--collect-only", test_file])
```

### Error 2: Module Not Found in Subprocess

```
ModuleNotFoundError: No module named 'calc'
```

**Root cause:** The subprocess doesn't have the source directory in PYTHONPATH.

**Fix:** Add `PYTHONPATH` to the subprocess environment:
```python
env = {**os.environ, "PYTHONPATH": str(source_dir)}
result = subprocess.run(cmd, capture_output=True, text=True, env=env)
```
