```bash
pip install \
  mcp==1.0.0 \
  httpx==0.28.0 \
  pydantic==2.9.0 \
```

```bash

```

## Build It

### Step 1: GitHub API Client

```python
# app/github_client.py
import httpx
import os
from typing import Optional, Dict, Any, AsyncIterator

class GitHubClient:
    def __init__(self, token: str = None):
        self.token = token or os.environ["GITHUB_TOKEN"]
        self.rest_base = "https://api.github.com"
        self.graphql_base = "https://api.github.com/graphql"
        self.headers = {
            "Authorization": f"Bearer {self.token}",
            "Accept": "application/vnd.github.v3+json",
            "User-Agent": "github-mcp-server/1.0"
        }

    async def _get(self, path: str, params: dict = None) -> dict:
        async with httpx.AsyncClient() as client:
            response = await client.get(
                f"{self.rest_base}{path}",
                headers=self.headers,
                params=params,
                timeout=10
            )
            response.raise_for_status()
            return response.json()

    async def _post(self, path: str, data: dict = None) -> dict:
        async with httpx.AsyncClient() as client:
            response = await client.post(
                f"{self.rest_base}{path}",
                headers=self.headers,
                json=data,
                timeout=10
            )
            response.raise_for_status()
            return response.json()

    async def _patch(self, path: str, data: dict = None) -> dict:
        async with httpx.AsyncClient() as client:
            response = await client.patch(
                f"{self.rest_base}{path}",
                headers=self.headers,
                json=data,
                timeout=10
            )
            response.raise_for_status()
            return response.json()

    async def search_repositories(self, query: str, limit: int = 10) -> list[dict]:
        data = await self._get("/search/repositories", {"q": query, "per_page": limit})
        return [{
            "name": r["full_name"],
            "description": r.get("description", ""),
            "url": r["html_url"],
            "stars": r["stargazers_count"],
            "language": r.get("language", ""),
            "updated": r["updated_at"]
        } for r in data.get("items", [])[:limit]]

    async def get_repository(self, owner: str, repo: str) -> dict:
        return await self._get(f"/repos/{owner}/{repo}")

    async def list_files(self, owner: str, repo: str, path: str = "", ref: str = "main") -> list[dict]:
        data = await self._get(f"/repos/{owner}/{repo}/contents/{path}", {"ref": ref})
        if isinstance(data, list):
            return [{"name": f["name"], "type": f["type"], "path": f["path"], "size": f["size"]} for f in data]
        return [{"name": data["name"], "type": "file", "path": data["path"], "size": data["size"]}]

    async def get_file_content(self, owner: str, repo: str, path: str, ref: str = "main") -> str:
        data = await self._get(f"/repos/{owner}/{repo}/contents/{path}", {"ref": ref})
        import base64
        content = base64.b64decode(data["content"]).decode("utf-8")
        return content

    async def search_code(self, query: str, limit: int = 10) -> list[dict]:
        data = await self._get("/search/code", {"q": query, "per_page": limit})
        return [{
            "name": r["name"],
            "path": r["path"],
            "repository": r["repository"]["full_name"],
            "url": r["html_url"]
        } for r in data.get("items", [])[:limit]]

    async def list_issues(self, owner: str, repo: str, state: str = "open", labels: str = None) -> list[dict]:
        params = {"state": state, "per_page": 20}
        if labels:
            params["labels"] = labels
        data = await self._get(f"/repos/{owner}/{repo}/issues", params)
        return [{"number": i["number"], "title": i["title"], "state": i["state"],
                 "user": i["user"]["login"], "labels": [l["name"] for l in i["labels"]],
                 "created_at": i["created_at"], "comments": i["comments"]} for i in data]

    async def create_issue(self, owner: str, repo: str, title: str, body: str = "", labels: list[str] = None) -> dict:
        data = {"title": title, "body": body}
        if labels:
            data["labels"] = labels
        return await self._post(f"/repos/{owner}/{repo}/issues", data)

    async def create_pull_request(self, owner: str, repo: str, title: str, head: str, base: str, body: str = "") -> dict:
        return await self._post(f"/repos/{owner}/{repo}/pulls", {
            "title": title, "head": head, "base": base, "body": body
        })

    async def get_pull_request(self, owner: str, repo: str, pull_number: int) -> dict:
        return await self._get(f"/repos/{owner}/{repo}/pulls/{pull_number}")

    async def list_workflow_runs(self, owner: str, repo: str, branch: str = None, status: str = None) -> list[dict]:
        params = {"per_page": 10}
        if branch:
            params["branch"] = branch
        if status:
            params["status"] = status
        data = await self._get(f"/repos/{owner}/{repo}/actions/runs", params)
        return [{"id": r["id"], "name": r["name"], "status": r["status"],
                 "conclusion": r["conclusion"], "branch": r["head_branch"],
                 "created_at": r["created_at"], "url": r["html_url"]}
                for r in data.get("workflow_runs", [])]
```

### Step 2: MCP Server with Tool Definitions

```python
# app/server.py
from mcp.server import Server
from mcp.types import Tool, TextContent
from app.github_client import GitHubClient
import json

server = Server("github-mcp")
client = GitHubClient()

@server.list_tools()
async def list_tools() -> list[Tool]:
    return [
        Tool(name="search_repositories", description="Search GitHub repositories", inputSchema={
            "type": "object",
            "properties": {
                "query": {"type": "string", "description": "Search query"},
                "limit": {"type": "integer", "default": 10}
            },
            "required": ["query"]
        }),
        Tool(name="get_repository", description="Get repository details", inputSchema={
            "type": "object",
            "properties": {
                "owner": {"type": "string"}, "repo": {"type": "string"}
            },
            "required": ["owner", "repo"]
        }),
        Tool(name="list_files", description="List files in a repository path", inputSchema={
            "type": "object",
            "properties": {
                "owner": {"type": "string"}, "repo": {"type": "string"},
                "path": {"type": "string", "default": ""},
                "ref": {"type": "string", "default": "main"}
            },
            "required": ["owner", "repo"]
        }),
        Tool(name="get_file_content", description="Get contents of a file", inputSchema={
            "type": "object",
            "properties": {
                "owner": {"type": "string"}, "repo": {"type": "string"},
                "path": {"type": "string"}, "ref": {"type": "string", "default": "main"}
            },
            "required": ["owner", "repo", "path"]
        }),
        Tool(name="search_code", description="Search code on GitHub", inputSchema={
            "type": "object",
            "properties": {
                "query": {"type": "string"},
                "limit": {"type": "integer", "default": 10}
            },
            "required": ["query"]
        }),
        Tool(name="list_issues", description="List repository issues", inputSchema={
            "type": "object",
            "properties": {
                "owner": {"type": "string"}, "repo": {"type": "string"},
                "state": {"type": "string", "default": "open"},
                "labels": {"type": "string"}
            },
            "required": ["owner", "repo"]
        }),
        Tool(name="create_issue", description="Create a new issue", inputSchema={
            "type": "object",
            "properties": {
                "owner": {"type": "string"}, "repo": {"type": "string"},
                "title": {"type": "string"}, "body": {"type": "string"},
                "labels": {"type": "array", "items": {"type": "string"}}

            "required": ["owner", "repo", "title"]

        Tool(name="create_pull_request", description="Create a pull request", inputSchema={
            "type": "object",
            "properties": {
                "owner": {"type": "string"}, "repo": {"type": "string"},
                "title": {"type": "string"}, "head": {"type": "string"},
                "base": {"type": "string"}, "body": {"type": "string"}
            },
            "required": ["owner", "repo", "title", "head", "base"]
        }),
        Tool(name="list_workflow_runs", description="List GitHub Actions workflow runs", inputSchema={
            "type": "object",
            "properties": {
                "owner": {"type": "string"}, "repo": {"type": "string"},
                "branch": {"type": "string"}, "status": {"type": "string"}
            },
            "required": ["owner", "repo"]
        }),
    ]
```

### Step 3: Tool Execution

```python
@server.call_tool()
async def call_tool(name: str, arguments: dict) -> list[TextContent]:
    try:
        result = await _execute_tool(name, arguments)
        return [TextContent(type="text", text=json.dumps(result, indent=2, default=str))]
    except Exception as e:
        return [TextContent(type="text", text=f"Error: {str(e)}")]

async def _execute_tool(name: str, args: dict):
    tool_map = {
        "search_repositories": lambda: client.search_repositories(args["query"], args.get("limit", 10)),
        "get_repository": lambda: client.get_repository(args["owner"], args["repo"]),
        "list_files": lambda: client.list_files(args["owner"], args["repo"], args.get("path", ""), args.get("ref", "main")),
        "get_file_content": lambda: client.get_file_content(args["owner"], args["repo"], args["path"], args.get("ref", "main")),
        "search_code": lambda: client.search_code(args["query"], args.get("limit", 10)),
        "list_issues": lambda: client.list_issues(args["owner"], args["repo"], args.get("state", "open"), args.get("labels")),
        "create_issue": lambda: client.create_issue(args["owner"], args["repo"], args["title"], args.get("body", ""), args.get("labels")),
        "create_pull_request": lambda: client.create_pull_request(args["owner"], args["repo"], args["title"], args["head"], args["base"], args.get("body", "")),
        "get_pull_request": lambda: client.get_pull_request(args["owner"], args["repo"], args["pull_number"]),
        "list_workflow_runs": lambda: client.list_workflow_runs(args["owner"], args["repo"], args.get("branch"), args.get("status")),
    }
    func = tool_map.get(name)
    if not func:
        raise ValueError(f"Unknown tool: {name}")
    return await func()
```

### Step 4: Main Entrypoint

```python
# app/main.py
from app.server import server
from mcp.server.stdio import stdio_server
import anyio

async def main():
    async with stdio_server() as (read_stream, write_stream):
        await server.run(
            read_stream, write_stream,
            server.create_initialization_options()
        )

if __name__ == "__main__":
    anyio.run(main())
```

### Step 5: Add Resource Templates

```python
@server.list_resource_templates()
async def list_resource_templates():
    return [
        ResourceTemplate(
            uriTemplate="github://{owner}/{repo}",
            name="GitHub Repository",
            mimeType="application/json"
        ),
        ResourceTemplate(
            uriTemplate="github://{owner}/{repo}/contents/{path}",
            name="File Contents",
            mimeType="text/plain"
        ),
    ]

@server.read_resource()
async def read_resource(uri: str) -> ReadResourceResult:
    import re
    match = re.match(r"github://([^/]+)/([^/]+)(?:/contents/(.+))?", uri)
    if not match:
        raise ValueError(f"Invalid URI: {uri}")
    owner, repo = match.group(1), match.group(2)
    path = match.group(3)

    if path:
        content = await client.get_file_content(owner, repo, path)
        return ReadResourceResult(contents=[TextContent(type="text", text=content, mimeType="text/plain")])
    else:
        data = await client.get_repository(owner, repo)
        return ReadResourceResult(contents=[TextContent(type="text", text=json.dumps(data, default=str), mimeType="application/json")])
```

## Understand What Happened

- **GitHub REST API v3** is used for all operations. It's well-documented and has predictable rate limits
- **MCP tool definitions** mirror the GitHub API surface. Each tool corresponds to one or more API calls
- **Resource templates** let Claude read repo metadata and file contents as if they were local files

## Common Errors

### Error 1: Rate Limit Exceeded

```
{"message": "API rate limit exceeded"}
```

**Root cause:** Unauthenticated requests are limited to 60/hour. Authenticated get 5,000/hour.

**Fix:** Make sure `GITHUB_TOKEN` is set correctly.

### Error 2: Not Found

```
Error: 404 - Not Found
```

**Root cause:** The repo doesn't exist, or your token doesn't have access.

**Fix:** Verify the repo is public or your token has `repo` scope.
