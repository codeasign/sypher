### 1. Error Handling and Retries

Add structured error handling to the LLM client and embedding calls.

```python
# app/llm/client.py (additions)
import time
import random
from functools import wraps

def retry(max_attempts=3, base_delay=1.0):
    def decorator(func):
        @wraps(func)
        def wrapper(*args, **kwargs):
            last_error = None
            for attempt in range(max_attempts):
                try:
                    return func(*args, **kwargs)
                except Exception as e:
                    last_error = e
                    if attempt < max_attempts - 1:
                        delay = base_delay * (2 ** attempt) + random.uniform(0, 0.5)
                        print(f"Retrying in {delay:.1f}s (attempt {attempt + 1}/{max_attempts})")
                        time.sleep(delay)
            raise last_error
        return wrapper
    return decorator

class LLMClient:
    @retry(max_attempts=3)
    def query(self, query: str, context: list, stream: bool = False):
        # existing implementation
        pass
```

### 2. Input Validation and Sanitization

Prevent injection attacks through document content.

```python
# app/ingest/loader.py (additions)
import re

def sanitize_content(text: str) -> str:
    """Remove potential prompt injection from ingested content."""
    # Remove null bytes
    text = text.replace("\x00", "")
    # Limit line length to prevent overflow
    lines = []
    for line in text.split("\n"):
        if len(line) > 5000:
            line = line[:5000] + "... [truncated]"
        lines.append(line)
    text = "\n".join(lines)
    # Remove control characters (except newlines and tabs)
    text = re.sub(r"[\x00-\x08\x0b\x0c\x0e-\x1f]", "", text)
    return text.strip()

def load_file(filepath: Path) -> tuple[str, str]:
    content, content_type = _load_raw(filepath)
    content = sanitize_content(content)
    return content, content_type
```

### 3. Logging and Monitoring

Add structured JSON logging for production observability.

```python
# app/audit/logger.py
import json
import logging
import sys
from datetime import datetime

class AuditLogger:
    def __init__(self):
        self.logger = logging.getLogger("ragflow")
        self.logger.setLevel(logging.INFO)
        handler = logging.StreamHandler(sys.stdout)
        handler.setFormatter(logging.Formatter(
            "%(asctime)s | %(levelname)s | %(message)s"
        ))
        self.logger.addHandler(handler)

    def log_query(self, user: str, query: str, tokens: int, latency: int, sources: list):
        record = {
            "event": "query",
            "user": user,
            "query_preview": query[:100],
            "tokens_used": tokens,
            "latency_ms": latency,
            "sources_count": len(sources),
            "timestamp": datetime.utcnow().isoformat()
        }
        self.logger.info(json.dumps(record))

    def log_ingestion(self, user: str, doc_id: str, filename: str, chunks: int):
        record = {
            "event": "ingestion",
            "user": user,
            "document_id": doc_id,
            "filename": filename,
            "chunks_created": chunks,
            "timestamp": datetime.utcnow().isoformat()
        }
        self.logger.info(json.dumps(record))

    def log_error(self, operation: str, error: str, details: dict = None):
        record = {
            "event": "error",
            "operation": operation,
            "error": error,
            "details": details or {},
            "timestamp": datetime.utcnow().isoformat()
        }
        self.logger.error(json.dumps(record))

audit = AuditLogger()
```

### 4. Cost Controls — Token Budget and Rate Limiting

Enforce per-user and global rate limits.

```python
# app/middleware/rate_limit.py
import time
from collections import defaultdict
from fastapi import Request, HTTPException

class RateLimiter:
    def __init__(self, requests_per_minute: int = 60):
        self.requests_per_minute = requests_per_minute
        self.window = 60.0
        self.requests: dict[str, list[float]] = defaultdict(list)

    async def __call__(self, request: Request):
        user = request.headers.get("authorization", "unknown")
        now = time.time()

        # Clean old entries
        self.requests[user] = [
            t for t in self.requests[user]
            if now - t < self.window
        ]

        if len(self.requests[user]) >= self.requests_per_minute:
            raise HTTPException(
                status_code=429,
                detail=f"Rate limit exceeded. Max {self.requests_per_minute} requests/minute."
            )

        self.requests[user].append(now)
```

### 5. Caching Strategy

Cache embeddings for common queries to reduce API costs.

```python
# app/search/cache.py
import hashlib
import json
import sqlite3
from datetime import datetime
from app.config import settings

class QueryCache:
    def __init__(self, ttl_seconds: int = 3600):
        self.ttl = ttl_seconds
        self._init_cache_db()

    def _init_cache_db(self):
        conn = sqlite3.connect(settings.SQLITE_PATH)
        conn.execute("""
            CREATE TABLE IF NOT EXISTS query_cache (
                query_hash TEXT PRIMARY KEY,
                response TEXT NOT NULL,
                created_at TEXT NOT NULL DEFAULT (datetime('now'))
            )
        """)
        conn.commit()
        conn.close()

    def get(self, query: str) -> dict | None:
        q_hash = hashlib.sha256(query.encode()).hexdigest()
        conn = sqlite3.connect(settings.SQLITE_PATH)
        row = conn.execute(
            "SELECT response, created_at FROM query_cache WHERE query_hash = ?",
            (q_hash,)
        ).fetchone()
        conn.close()

        if row:
            created = datetime.fromisoformat(row["created_at"])
            if (datetime.utcnow() - created).seconds < self.ttl:
                return json.loads(row["response"])
        return None

    def set(self, query: str, response: dict):
        q_hash = hashlib.sha256(query.encode()).hexdigest()
        conn = sqlite3.connect(settings.SQLITE_PATH)
        conn.execute(
            "INSERT OR REPLACE INTO query_cache (query_hash, response, created_at) VALUES (?, ?, ?)",
            (q_hash, json.dumps(response), datetime.utcnow().isoformat())
        )
        conn.commit()
        conn.close()
```

## Scale It

### What Breaks First at 10x Load

At 100 concurrent users, the system shows these failure modes:

| Load Level | Failure Point | Symptom |
|---|---|---|
| 10 concurrent | ChromaDB query latency | Queries take 500ms+ instead of 50ms |
| 50 concurrent | SQLite write lock | Ingestion fails with `SQLITE_BUSY` |
| 100 concurrent | Embedding API rate limit | Embedding requests fail with HTTP 429 |
| 500 concurrent | Single-process GIL | Uvicorn can't parallelize Python CPU work |

### Architectural Changes Required

1. **Replace SQLite with PostgreSQL** — handles concurrent writes natively
2. **Replace ChromaDB with Qdrant or Pinecone** — cloud-native vector DB with horizontal scaling
3. **Add Redis cache** — cache frequent queries to reduce LLM calls by 40-60%
4. **Replace single FastAPI process with Gunicorn + multiple workers** — scales to CPU cores
5. **Add message queue (Redis Streams / SQS)** — async ingestion that doesn't block the API
