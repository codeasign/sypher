## Build It

### Step 1: Database Schema

```python
# app/database.py
import sqlite3, os
from datetime import datetime

DB_PATH = os.environ.get("DB_PATH", "./data/chatclone.db")

def get_conn():
    os.makedirs(os.path.dirname(DB_PATH) or ".", exist_ok=True)
    conn = sqlite3.connect(DB_PATH)
    conn.row_factory = sqlite3.Row
    return conn

def init_db():
    conn = get_conn()
    conn.executescript("""
        CREATE TABLE IF NOT EXISTS conversations (
            id TEXT PRIMARY KEY,
            title TEXT NOT NULL,
            model TEXT NOT NULL DEFAULT 'claude-haiku',
            system_prompt TEXT DEFAULT 'You are a helpful assistant.',
            created_at TEXT NOT NULL,
            updated_at TEXT NOT NULL
        );
        CREATE TABLE IF NOT EXISTS messages (
            id TEXT PRIMARY KEY,
            conversation_id TEXT NOT NULL,
            role TEXT NOT NULL,
            content TEXT NOT NULL,
            tokens INTEGER DEFAULT 0,
            created_at TEXT NOT NULL,
            FOREIGN KEY (conversation_id) REFERENCES conversations(id)
        );
        CREATE TABLE IF NOT EXISTS settings (
            key TEXT PRIMARY KEY,
            value TEXT NOT NULL
        );
    """)
    conn.commit()
    conn.close()
```

### Step 2: Backend API — Conversations

```python
# app/api.py
from fastapi import FastAPI, HTTPException
from pydantic import BaseModel
from app.database import get_conn, init_db
from datetime import datetime
import uuid
from typing import Optional

app = FastAPI(title="ChatClone API")

class ConversationCreate(BaseModel):
    title: str = "New Chat"
    model: str = "claude-haiku-3-5"
    system_prompt: str = "You are a helpful assistant."

class MessageCreate(BaseModel):
    role: str
    content: str
    tokens: int = 0

class ChatRequest(BaseModel):
    conversation_id: str
    message: str

@app.on_event("startup")
async def startup():
    init_db()

@app.get("/api/conversations")
def list_conversations():
    conn = get_conn()
    rows = conn.execute(
        "SELECT id, title, model, created_at, updated_at FROM conversations ORDER BY updated_at DESC"
    ).fetchall()
    conn.close()
    return [dict(r) for r in rows]

@app.post("/api/conversations")
def create_conversation(req: ConversationCreate):
    conn = get_conn()
    conv_id = uuid.uuid4().hex[:12]
    now = datetime.utcnow().isoformat()
    conn.execute(
        "INSERT INTO conversations VALUES (?, ?, ?, ?, ?, ?)",
        (conv_id, req.title, req.model, req.system_prompt, now, now)
    )
    conn.commit()
    conn.close()
    return {"id": conv_id, "title": req.title}

@app.delete("/api/conversations/{conv_id}")
def delete_conversation(conv_id: str):
    conn = get_conn()
    conn.execute("DELETE FROM messages WHERE conversation_id = ?", (conv_id,))
    conn.execute("DELETE FROM conversations WHERE id = ?", (conv_id,))
    conn.commit()
    conn.close()
    return {"deleted": True}

@app.get("/api/conversations/{conv_id}/messages")
def get_messages(conv_id: str):
    conn = get_conn()
    rows = conn.execute(
        "SELECT id, role, content, tokens, created_at FROM messages WHERE conversation_id = ? ORDER BY created_at",
        (conv_id,)
    ).fetchall()
    conn.close()
    return [dict(r) for r in rows]
```

### Step 3: Streaming Chat Endpoint

```python
# app/stream.py
from sse_starlette.sse import EventSourceResponse
from app.database import get_conn
from app.providers import get_provider
from datetime import datetime
import uuid, json, asyncio

async def stream_chat(conversation_id: str, message: str):
    conn = get_conn()
    conv = conn.execute("SELECT * FROM conversations WHERE id = ?", (conversation_id,)).fetchone()
    if not conv:
        raise HTTPException(404, "Conversation not found")

    # Save user message
    msg_id = uuid.uuid4().hex[:12]
    conn.execute(
        "INSERT INTO messages VALUES (?, ?, ?, ?, ?, ?)",
        (msg_id, conversation_id, "user", message, 0, datetime.utcnow().isoformat())
    )
    conn.commit()

    # Get conversation history
    messages = conn.execute(
        "SELECT role, content FROM messages WHERE conversation_id = ? ORDER BY created_at",
        (conversation_id,)
    ).fetchall()
    conn.close()

    # Get provider
    provider = get_provider(conv["model"])

    async def event_generator():
        full_response = ""
        async for chunk in provider.stream(conv["system_prompt"], [dict(m) for m in messages]):
            full_response += chunk
            yield {"event": "token", "data": json.dumps({"token": chunk})}

        # Save assistant message
        conn = get_conn()
        conn.execute(
            "INSERT INTO messages VALUES (?, ?, ?, ?, ?, ?)",
            (uuid.uuid4().hex[:12], conversation_id, "assistant", full_response, 0, datetime.utcnow().isoformat())
        )
        conn.execute("UPDATE conversations SET updated_at = ? WHERE id = ?",
                     (datetime.utcnow().isoformat(), conversation_id))
        conn.commit()
        conn.close()

        yield {"event": "done", "data": json.dumps({"full": full_response})}

    return EventSourceResponse(event_generator())
```

### Step 4: LLM Providers

```python
# app/providers.py
from anthropic import Anthropic
from openai import OpenAI
import os

class ClaudeProvider:
    def __init__(self, model="claude-3-haiku-20240307"):
        self.client = Anthropic(api_key=os.environ["ANTHROPIC_API_KEY"])
        self.model = model

    async def stream(self, system_prompt: str, messages: list[dict]):
        response = self.client.messages.create(
            model=self.model,
            max_tokens=4096,
            system=system_prompt,
            messages=messages,
            stream=True
        )
        for event in response:
            if event.type == "content_block_delta":
                yield event.delta.text

class OpenAIProvider:
    def __init__(self, model="gpt-4o-mini"):
        self.client = OpenAI(api_key=os.environ["OPENAI_API_KEY"])
        self.model = model

    async def stream(self, system_prompt: str, messages: list[dict]):
        formatted = [{"role": "system", "content": system_prompt}] + messages
        stream = self.client.chat.completions.create(
            model=self.model,
            messages=formatted,
            stream=True
        )
        for chunk in stream:
            if chunk.choices[0].delta.content:
                yield chunk.choices[0].delta.content

class OllamaProvider:
    def __init__(self, model="llama3.2"):
        from openai import OpenAI as OpenAILike
        self.client = OpenAILike(base_url="http://localhost:11434/v1", api_key="ollama")
        self.model = model

    async def stream(self, system_prompt: str, messages: list[dict]):
        formatted = [{"role": "system", "content": system_prompt}] + messages
        stream = self.client.chat.completions.create(
            model=self.model,
            messages=formatted,
            stream=True
        )
        for chunk in stream:
            if chunk.choices[0].delta.content:
                yield chunk.choices[0].delta.content

PROVIDERS = {
    "claude-haiku-3-5": lambda: ClaudeProvider("claude-3-haiku-20240307"),
    "claude-sonnet-4": lambda: ClaudeProvider("claude-sonnet-4-20250514"),
    "gpt-4o-mini": lambda: OpenAIProvider("gpt-4o-mini"),
    "ollama": lambda: OllamaProvider(),
}

def get_provider(model_id: str):
    factory = PROVIDERS.get(model_id)
    if not factory:
        raise ValueError(f"Unknown model: {model_id}")
    return factory()
```

### Step 5: Frontend — Chat Component

```tsx
// frontend/src/components/Chat.tsx
import { useState, useEffect, useRef } from 'react'
import ReactMarkdown from 'react-markdown'
import { Prism as SyntaxHighlighter } from 'react-syntax-highlighter'
import { oneDark } from 'react-syntax-highlighter/dist/esm/styles/prism'

interface Message {
  role: 'user' | 'assistant'
  content: string
}

export function Chat({ conversationId }: { conversationId: string }) {
  const [messages, setMessages] = useState<Message[]>([])
  const [input, setInput] = useState('')
  const [streaming, setStreaming] = useState(false)
  const [streamContent, setStreamContent] = useState('')
  const chatRef = useRef<HTMLDivElement>(null)

  useEffect(() => {
    fetch(`/api/conversations/${conversationId}/messages`)
      .then(r => r.json())
      .then(data => setMessages(data.map((m: any) => ({ role: m.role, content: m.content }))))
  }, [conversationId])

  useEffect(() => {
    chatRef.current?.scrollTo(0, chatRef.current.scrollHeight)
  }, [messages, streamContent])

  async function sendMessage() {
    if (!input.trim() || streaming) return

    const userMsg: Message = { role: 'user', content: input }
    setMessages(prev => [...prev, userMsg])
    setInput('')
    setStreaming(true)
    setStreamContent('')

    const response = await fetch(`/api/chat/stream`, {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify({ conversation_id: conversationId, message: input })
    })

    const reader = response.body?.getReader()
    const decoder = new TextDecoder()

    while (reader) {
      const { done, value } = await reader.read()
      if (done) break

      const text = decoder.decode(value)
      const lines = text.split('\n')

      for (const line of lines) {
        if (line.startsWith('data: ')) {
          try {
            const data = JSON.parse(line.slice(6))
            if (data.token) {
              setStreamContent(prev => prev + data.token)
            }
            if (data.full) {
              setMessages(prev => [...prev, { role: 'assistant', content: data.full }])
              setStreamContent('')
              setStreaming(false)
            }
          } catch {}
        }
      }
    }
  }

  return (
    <div className="flex flex-col h-screen">
      <div ref={chatRef} className="flex-1 overflow-y-auto p-4 space-y-4">
        {messages.map((msg, i) => (
          <div key={i} className={`flex ${msg.role === 'user' ? 'justify-end' : 'justify-start'}`}>
            <div className={`max-w-3/4 p-3 rounded-lg ${
              msg.role === 'user' ? 'bg-blue-600 text-white' : 'bg-gray-100 dark:bg-gray-800'
            }`}>
              <ReactMarkdown
                components={{
                  code({ node, inline, className, children, ...props }) {
                    const match = /language-(\w+)/.exec(className || '')
                    return !inline && match ? (
                      <SyntaxHighlighter style={oneDark} language={match[1]} PreTag="div">
                        {String(children).replace(/\n$/, '')}
                      </SyntaxHighlighter>
                    ) : (
                      <code className={className} {...props}>{children}</code>
                    )
                  }
                }}
              >
                {msg.content}
              </ReactMarkdown>
            </div>
          </div>
        ))}
        {streaming && streamContent && (
          <div className="flex justify-start">
            <div className="max-w-3/4 p-3 rounded-lg bg-gray-100 dark:bg-gray-800">
              <ReactMarkdown>{streamContent}</ReactMarkdown>
            </div>
          </div>
        )}
      </div>

      <div className="border-t p-4">
        <div className="flex gap-2">
          <input
            className="flex-1 p-2 border rounded-lg dark:bg-gray-800 dark:border-gray-700"
            value={input}
            onChange={e => setInput(e.target.value)}
            onKeyDown={e => e.key === 'Enter' && !e.shiftKey && sendMessage()}
            placeholder="Type a message..."
            disabled={streaming}
          />
          <button
            className="px-4 py-2 bg-blue-600 text-white rounded-lg disabled:opacity-50"
            onClick={sendMessage}
            disabled={streaming || !input.trim()}
          >
            Send
          </button>
        </div>
      </div>
    </div>
  )
}
```

## Understand What Happened

- **SSE streaming** sends tokens to the browser as the LLM generates them, providing real-time feedback
- **Provider abstraction** means adding a new LLM is one class + one config entry
- **SQLite persistence** saves every message so conversations survive server restarts
- **React Markdown** renders LLM output safely — no dangerous innerHTML

## Common Errors

### Error 1: CORS Errors

```
Access to fetch at 'http://localhost:8000' from origin 'http://localhost:5173' has been blocked by CORS
```

**Root cause:** Backend and frontend are on different ports.

**Fix:** Add CORS middleware:
```python
from fastapi.middleware.cors import CORSMiddleware
app.add_middleware(CORSMiddleware, allow_origins=["http://localhost:5173"], allow_methods=["*"], allow_headers=["*"])
```

### Error 2: SSE Not Streaming

**Root cause:** Using `uvicorn` with only 1 worker. SSE needs at least 1 worker per stream.

**Fix:** Nothing — single worker with `asyncio` handles concurrent streams fine. Don't use `gunicorn` with sync workers.
