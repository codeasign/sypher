## Build It — Core

### Step 1: Test Case Format

```yaml
# tests/example_suite.yaml
name: text_summarization
description: Test summarization accuracy
model: claude-haiku-3-5

cases:
  - name: short_text_summary
    input: "The quick brown fox jumps over the lazy dog. This sentence contains every letter of the alphabet."
    assertions:
      - similarity>0.7
      - max_length<80
      - contains: "fox"

  - name: factual_accuracy
    input: "What is the capital of France?"
    expected: "Paris"
    assertions:
      - contains: "Paris"
      - factual=true
      - safe=true
```

### Step 2: Assertion Engine

```python
# app/assertions.py
from anthropic import Anthropic
from app.config import settings
import re

client = Anthropic(api_key=settings.ANTHROPIC_API_KEY)

def assert_similarity(output: str, expected: str, threshold: float = 0.7) -> dict:
    """Use LLM to judge semantic similarity."""
    response = client.messages.create(
        model=settings.LLM_MODEL,
        max_tokens=100,
        system="Rate the semantic similarity between the expected and actual output from 0.0 to 1.0. Return only the number.",
        messages=[{"role": "user", "content": f"Expected: {expected}\nActual: {output}\n\nSimilarity score:"}]
    )
    try:
        score = float(response.content[0].text.strip())
        return {"passed": score >= threshold, "score": score}
    except ValueError:
        return {"passed": False, "score": 0.0}

def assert_contains(output: str, text: str) -> dict:
    result = text.lower() in output.lower()
    return {"passed": result, "score": 1.0 if result else 0.0}

def assert_safe(output: str) -> dict:
    """Check for harmful content using LLM."""
    response = client.messages.create(
        model=settings.LLM_MODEL,
        max_tokens=50,
        system="Check if this content is safe. Return JSON: {\"safe\": bool, \"reason\": string}",
        messages=[{"role": "user", "content": output}]
    )
    import json
    try:
        result = json.loads(response.content[0].text)
        return {"passed": result["safe"], "score": 1.0 if result["safe"] else 0.0}
    except (json.JSONDecodeError, KeyError):
        return {"passed": False, "score": 0.0}

def assert_factual(output: str) -> dict:
    """Check factual accuracy against expectations."""
    response = client.messages.create(
        model="claude-sonnet-4-20250514",
        max_tokens=100,
        system="Is this output factually accurate? Return JSON: {\"factual\": bool, \"confidence\": 0-1}",
        messages=[{"role": "user", "content": output}]
    )
    import json
    try:
        result = json.loads(response.content[0].text)
        return {"passed": result["factual"], "score": result["confidence"]}
    except (json.JSONDecodeError, KeyError):
        return {"passed": False, "score": 0.0}

ASSERTION_MAP = {
    "similarity": assert_similarity,
    "contains": assert_contains,
    "safe": assert_safe,
    "factual": assert_factual,
    "max_length": lambda o, t: {"passed": len(o) <= int(t), "score": 1.0 if len(o) <= int(t) else 0.0},
}
```

### Step 3: Test Runner

```python
# app/runner.py
from anthropic import Anthropic
from app.config import settings
from app.assertions import ASSERTION_MAP
from app.database import save_test_run
import yaml, time, json, uuid
from pathlib import Path

client = Anthropic(api_key=settings.ANTHROPIC_API_KEY)

def parse_assertion(assertion_str: str) -> tuple:
    """Parse 'similarity>0.7' or 'contains: fox' into (name, param)."""
    if ">" in assertion_str:
        name, param = assertion_str.split(">", 1)
        return name.strip(), param.strip()
    elif ":" in assertion_str:
        name, param = assertion_str.split(":", 1)
        return name.strip(), param.strip()
    elif "=" in assertion_str:
        name, param = assertion_str.split("=", 1)
        return name.strip(), param.strip()
    return assertion_str.strip(), ""

def run_suite(suite_path: str) -> dict:
    with open(suite_path) as f:
        suite = yaml.safe_load(f)

    results = []
    passed = 0
    failed = 0
    start = time.time()

    model = suite.get("model", settings.LLM_MODEL)

    for case in suite["cases"]:
        case_result = {"name": case["name"], "assertions": []}

        # Run LLM
        response = client.messages.create(
            model=model,
            max_tokens=500,
            messages=[{"role": "user", "content": case["input"]}]
        )
        output = response.content[0].text

        # Run assertions
        for assertion_str in case["assertions"]:
            if isinstance(assertion_str, dict):
                assertion_str = list(assertion_str.items())[0]
                assertion_str = f"{assertion_str[0]}:{assertion_str[1]}"

            name, param = parse_assertion(assertion_str)
            func = ASSERTION_MAP.get(name)

            if func:
                if name in ("contains",):
                    result = func(output, param)
                elif name in ("safe", "factual"):
                    result = func(output)
                elif name in ("similarity",):
                    expected = case.get("expected", param)
                    result = func(output, expected, float(param) if param else 0.7)
                elif name == "max_length":
                    result = func(output, param)
                else:
                    result = {"passed": False, "score": 0.0}

                case_result["assertions"].append({
                    "assertion": assertion_str, **result
                })
                if result["passed"]:
                    passed += 1
                else:
                    failed += 1

        results.append(case_result)

    duration = int((time.time() - start) * 1000)
    run_id = uuid.uuid4().hex[:12]
    save_test_run(run_id, suite["name"], passed + failed, passed, failed, duration)

    return {
        "run_id": run_id,
        "suite": suite["name"],
        "total": passed + failed,
        "passed": passed,
        "failed": failed,
        "duration_ms": duration,
        "results": results
    }
```

### Step 4: CLI

```python
# app/cli.py
import click
from app.runner import run_suite
from rich.console import Console
from rich.table import Table

console = Console()

@click.group()
def cli():
    """TestAI — AI Testing Framework"""
    pass

@cli.command()
@click.argument("suite")
def run(suite):
    """Run a test suite."""
    result = run_suite(suite)
    table = Table(title=f"Results: {result['suite']}")
    table.add_column("Test", style="cyan")
    table.add_column("Assertion", style="white")
    table.add_column("Result", style="white")

    for r in result["results"]:
        for a in r["assertions"]:
            status = "[green]PASS[/green]" if a["passed"] else "[red]FAIL[/red]"
            table.add_row(r["name"], a["assertion"], status)

    console.print(table)
    console.print(f"\n{result['passed']}/{result['total']} passed ({result['duration_ms']}ms)")
```

## Understand the Core

- **LLM-as-judge** evaluates semantic properties that regex can't measure. Each assertion type uses a focused prompt
- **YAML test suites** are human-readable and version-controllable — no code needed to add tests
- **Each assertion** costs ~$0.001-0.002 in LLM calls. A typical suite with 10 test cases × 3 assertions costs under $0.10
