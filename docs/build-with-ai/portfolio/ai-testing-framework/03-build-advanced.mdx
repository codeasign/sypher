## Advanced Feature 1: Regression Comparison

Compare test results between two runs to detect regressions.

```python
# app/regression.py
from app.database import get_conn
import json

def compare_runs(run_a: str, run_b: str) -> dict:
    conn = get_conn()
    a = conn.execute("SELECT * FROM test_runs WHERE id = ?", (run_a,)).fetchone()
    b = conn.execute("SELECT * FROM test_runs WHERE id = ?", (run_b,)).fetchone()
    conn.close()

    return {
        "previous": dict(a) if a else None,
        "current": dict(b) if b else None,
        "regression": (a and b and b["passed"] < a["passed"]),
        "improvement": (a and b and b["passed"] > a["passed"]),
    }
```

## Advanced Feature 2: Benchmark Suite

```python
# app/benchmark.py
import time
from anthropic import Anthropic
from app.config import settings
from app.database import save_benchmark

client = Anthropic(api_key=settings.ANTHROPIC_API_KEY)

def benchmark_model(model: str, iterations: int = 5) -> dict:
    prompts = [
        "Explain quantum computing in simple terms",
        "Write a Python function to sort a list",
        "What are the benefits of REST APIs?",
    ]

    latencies = []
    output_lengths = []

    for i in range(iterations):
        for prompt in prompts:
            start = time.time()
            response = client.messages.create(
                model=model,
                max_tokens=500,
                messages=[{"role": "user", "content": prompt}]
            )
            latency = (time.time() - start) * 1000
            latencies.append(latency)
            output_lengths.append(len(response.content[0].text))

    return {
        "model": model,
        "avg_latency_ms": sum(latencies) / len(latencies),
        "p95_latency_ms": sorted(latencies)[int(len(latencies) * 0.95)],
        "avg_output_length": sum(output_lengths) / len(output_lengths),
        "total_cost": (sum(latencies) / 1000) * 0.00001,
    }
```

## Advanced Feature 3: JUnit XML Output for CI

```python
# app/junit.py
import xml.etree.ElementTree as ET
from datetime import datetime

def to_junit_xml(result: dict) -> str:
    testsuite = ET.Element("testsuite", {
        "name": result["suite"],
        "tests": str(result["total"]),
        "failures": str(result["failed"]),
        "time": str(result["duration_ms"] / 1000)
    })

    for r in result["results"]:
        for a in r["assertions"]:
            testcase = ET.SubElement(testsuite, "testcase", {
                "name": f"{r['name']} - {a['assertion']}",
                "time": "0"
            })
            if not a["passed"]:
                ET.SubElement(testcase, "failure", {
                    "message": f"Score: {a['score']}"
                })

    return ET.tostring(testsuite, encoding="unicode")
```

## Testing

```python
# tests/test_assertions.py
from app.assertions import assert_contains, assert_similarity

def test_assert_contains():
    assert assert_contains("Hello World", "world")["passed"]
    assert not assert_contains("Hello World", "Python")["passed"]

def test_assert_similarity_high():
    result = assert_similarity("Paris is the capital of France.", "The capital of France is Paris.", 0.7)
    assert result["passed"]
    assert result["score"] >= 0.7
```
