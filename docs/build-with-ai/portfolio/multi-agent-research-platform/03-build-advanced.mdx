## Advanced Feature 1: Fact-Checker Agent

The fact-checker cross-references claims across multiple sources and flags contradictions.

```python
# app/researchers/fact_checker.py
from anthropic import Anthropic
from app.config import settings
import json

client = Anthropic(api_key=settings.ANTHROPIC_API_KEY)

def fact_check(report: str, findings: list[dict]) -> dict:
    findings_text = "\n\n".join([f["content"] for f in findings])

    response = client.messages.create(
        model=settings.LLM_MODEL,
        max_tokens=2000,
        system="You are a fact-checker. Identify claims that are unsupported, contradictory, or potentially incorrect. Return JSON.",
        messages=[{
            "role": "user",
            "content": f"Report:\n{report}\n\nSource findings:\n{findings_text}\n\nIdentify: unsupported claims, contradictions, and verification status."
        }]
    )

    try:
        return json.loads(response.content[0].text)
    except json.JSONDecodeError:
        return {"unsupported_claims": [], "contradictions": [], "overall_confidence": 0.5}
```

## Advanced Feature 2: Critic Agent with Iterative Refinement

The critic reviews the report, and if it finds gaps, triggers another research cycle.

```python
# app/critic.py
from anthropic import Anthropic
from app.config import settings
import json

client = Anthropic(api_key=settings.ANTHROPIC_API_KEY)

def critique_report(report: str, query: str) -> dict:
    response = client.messages.create(
        model="claude-sonnet-4-20250514",
        max_tokens=2000,
        system="You are a critical reviewer. Identify gaps, missing perspectives, weak arguments, and unsupported claims in research reports. Be constructive but thorough.",
        messages=[{
            "role": "user",
            "content": f"Research Query: {query}\n\nReport:\n{report}\n\nCritique this report. Identify gaps and suggest improvements."
        }]
    )

    try:
        return json.loads(response.content[0].text)
    except json.JSONDecodeError:
        return {"gaps": ["Unable to parse critique"], "score": 0.5, "requires_revision": False}

# Add iterative refinement to the graph
def build_research_graph_with_critic():
    from app.graph import build_research_graph
    from langgraph.graph import StateGraph, END
    from app.state import ResearchState

    workflow = StateGraph(ResearchState)

    workflow.add_node("decompose", decompose_node)
    workflow.add_node("research", research_node)
    workflow.add_node("synthesize", synthesize_node)
    workflow.add_node("critique", critique_node)
    workflow.add_node("refine", refine_node)

    workflow.set_entry_point("decompose")
    workflow.add_edge("decompose", "research")
    workflow.add_edge("research", "synthesize")
    workflow.add_edge("synthesize", "critique")
    workflow.add_conditional_edges("critique", should_refine, {True: "refine", False: END})
    workflow.add_edge("refine", "research")
    return workflow.compile()

def should_refine(state: ResearchState) -> bool:
    return state["iteration"] < state["max_iterations"] and state.get("requires_revision", False)
```

## Advanced Feature 3: Web UI with Agent Activity Stream

```tsx
// frontend/src/ResearchView.tsx
import { useState, useEffect } from 'react'

interface AgentMessage {
  agent: string
  content: string
  type: 'finding' | 'status' | 'error'
}

export function ResearchView() {
  const [query, setQuery] = useState('')
  const [messages, setMessages] = useState<AgentMessage[]>([])
  const [report, setReport] = useState('')
  const [researching, setResearching] = useState(false)

  async function startResearch() {
    setResearching(true)
    setMessages([])
    setReport('')

    const response = await fetch('/api/research', {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify({ query })
    })

    const reader = response.body!.getReader()
    const decoder = new TextDecoder()

    while (true) {
      const { done, value } = await reader.read()
      if (done) break

      const text = decoder.decode(value)
      for (const line of text.split('\n')) {
        if (line.startsWith('data: ')) {
          try {
            const data = JSON.parse(line.slice(6))
            if (data.message) {
              setMessages(prev => [...prev, { agent: 'system', content: data.message, type: 'status' }])
            }
            if (data.report) {
              setReport(data.report)
              setResearching(false)
            }
          } catch {}
        }
      }
    }
  }

  return (
    <div className="max-w-4xl mx-auto p-4">
      <h1 className="text-2xl font-bold mb-4">ResearchNet</h1>
      <div className="flex gap-2 mb-6">
        <input
          className="flex-1 p-2 border rounded"
          value={query}
          onChange={e => setQuery(e.target.value)}
          placeholder="Enter a research question..."
          disabled={researching}
        />
        <button
          className="px-4 py-2 bg-blue-600 text-white rounded disabled:opacity-50"
          onClick={startResearch}
          disabled={researching || !query.trim()}
        >
          {researching ? 'Researching...' : 'Research'}
        </button>
      </div>

      <div className="grid grid-cols-2 gap-4">
        <div className="border rounded p-4">
          <h2 className="font-semibold mb-2">Agent Activity</h2>
          <div className="space-y-2">
            {messages.map((msg, i) => (
              <div key={i} className={`p-2 rounded text-sm ${
                msg.type === 'error' ? 'bg-red-100' :
                msg.type === 'status' ? 'bg-blue-50' : 'bg-gray-50'
              }`}>
                <span className="font-medium">{msg.agent}:</span> {msg.content}
              </div>
            ))}
          </div>
        </div>

        <div className="border rounded p-4">
          <h2 className="font-semibold mb-2">Report</h2>
          <div className="prose prose-sm max-w-none">
            {report ? report.split('\n').map((line, i) => (
              <p key={i}>{line}</p>
            )) : <p className="text-gray-400">Waiting for results...</p>}
          </div>
        </div>
      </div>
    </div>
  )
}
```
