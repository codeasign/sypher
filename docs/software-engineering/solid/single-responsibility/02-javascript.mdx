// CORRECT IMPLEMENTATION
// Each class has exactly one responsibility
// ============================================================

class InvoiceCalculator {
  /** Responsibility: calculate invoice totals */
  calculate(orderItems) {
    const subtotal = orderItems.reduce((sum, item) => sum + item.price * item.qty, 0);
    const tax = subtotal * 0.08;
    const discount = subtotal > 100 ? subtotal * 0.1 : 0;
    const total = subtotal + tax - discount;
    return { subtotal, tax, discount, total };
  }
}

class InvoiceRepository {
  /** Responsibility: persist and retrieve invoices */
  constructor() {
    this.invoices = [];
  }

  save(invoice) {
    this.invoices.push(invoice);
    console.log(`Saved invoice ${invoice.id} to database`);
    return invoice;
  }

  findById(id) {
    return this.invoices.find(inv => inv.id === id);
  }
}

class InvoicePrinter {
  /** Responsibility: format and print invoices */
  print(invoice) {
    console.log("=== INVOICE ===");
    console.log(`Invoice #${invoice.id}`);
    invoice.items.forEach(item => {
      console.log(`${item.name.padEnd(20)} x${item.qty}  $${(item.price * item.qty).toFixed(2)}`);
    });
    console.log("-".repeat(40));
    console.log(`Subtotal:$${invoice.subtotal.toFixed(10)}`.padStart(38));
    console.log(`Discount: -$${invoice.discount.toFixed(2)}`.padStart(38));
    console.log(`Tax:      $${invoice.tax.toFixed(2)}`.padStart(38));
    console.log(`Total:    $${invoice.total.toFixed(2)}`.padStart(38));
    console.log("=".repeat(40));
  }
}

class InvoiceEmailer {
  /** Responsibility: send invoice emails */
  sendConfirmation(invoice, to) {
    console.log(`Sending invoice #${invoice.id} to ${to}...`);
    console.log(`Email sent to ${to}`);
  }
}

// --- Main driver ---
function main() {
  const orderItems = [
    { name: "Widget A", price: 29.99, qty: 2 },
    { name: "Gadget B", price: 49.99, qty: 1 },
    { name: "Accessory C", price: 9.99, qty: 3 }
  ];
  const customerEmail = "customer@example.com";

  // Compose SRP components
  const calculator = new InvoiceCalculator();
  const repository = new InvoiceRepository();
  const printer = new InvoicePrinter();
  const emailer = new InvoiceEmailer();

  // Process
  const { subtotal, tax, discount, total } = calculator.calculate(orderItems);
  const invoice = {
    id: Date.now(),
    items: orderItems,
    subtotal,
    tax,
    discount,
    total,
    customerEmail,
    createdAt: new Date().toISOString()
  };

  repository.save(invoice);
  printer.print(invoice);
  emailer.sendConfirmation(invoice, customerEmail);

  console.log(`\nInvoice ${invoice.id} processed. Total: $${total.toFixed(2)}`);
}

main();

## Key Language-Specific Notes

- **ES6 class syntax** provides clean constructor injection вҖ” each SRP class receives its dependencies through the constructor, making them testable in isolation.
- **No private fields needed** вҖ” SRP classes naturally have few methods, so convention-based `_` prefix or closure-based privacy suffices.
- **Arrow functions and array methods** (`reduce`, `forEach`) keep each method short and focused on its single responsibility.
- **Module pattern** (`import`/`export`) would let each class live in its own file, enforcing SRP at the file system level вҖ” `InvoiceCalculator` in `invoice-calculator.js`, `InvoiceRepository` in `invoice-repository.js`, etc.
- **The orchestration layer** (`main()` here; a Controller class in production) is the only place that composes SRP components, keeping coupling visible and intentional.
