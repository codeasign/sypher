## Implementation

A C# 12 / .NET 8 implementation of SRP using a payment processing system. The violation shows a `PaymentProcessor` class handling validation, amount calculation, gateway communication, receipt generation, and logging. The correct version splits each concern.

```csharp
// SRP — C# 12 / .NET 8 implementation
// Domain: Payment processing

// ============================================================
// VIOLATION: PaymentProcessor handles validation, calculation,
// gateway communication, receipt generation, and logging
// ============================================================
//
// class PaymentProcessor {
//     public void ProcessPayment(string cardNumber, decimal amount, string currency) {
//         // Validation
//         if (string.IsNullOrEmpty(cardNumber)) throw new ArgumentException("Card required");
//         if (amount <= 0) throw new ArgumentException("Invalid amount");
//
//         // Calculation
//         decimal tax = amount * 0.08m;
//         decimal fee = amount * 0.02m;
//         decimal total = amount + tax + fee;
//
//         // Gateway
//         Console.WriteLine($"Charging {total:C} to card {cardNumber[^4..]}...");
//
//         // Receipt
//         Console.WriteLine($"Receipt: Payment of {total:C} in {currency}");
//
//         // Logging
//         File.AppendAllText("payments.log",
//             $"{DateTime.UtcNow:O} | {cardNumber[^4..]} | {total:C}\n");
//     }
// }
//
// Problem: Five different concerns change for different reasons.

// ============================================================
// CORRECT IMPLEMENTATION
// Each class has exactly one responsibility
// ============================================================

// --- Responsibility 1: Payment validation ---

interface IPaymentValidator
{
    void Validate(string? cardNumber, decimal amount);
}

sealed class PaymentValidator : IPaymentValidator
{
    public void Validate(string? cardNumber, decimal amount)
    {
        ArgumentException.ThrowIfNullOrWhiteSpace(cardNumber);
        if (amount <= 0)
            throw new ArgumentException("Amount must be positive", nameof(amount));
        if (cardNumber.Length < 13 || cardNumber.Length > 19)
            throw new ArgumentException("Invalid card number length");
    }
}

// --- Responsibility 2: Payment calculation ---

sealed class PaymentCalculator
{
    public decimal TaxRate { get; } = 0.08m;
    public decimal FeeRate { get; } = 0.02m;

    public (decimal Tax, decimal Fee, decimal Total) Calculate(decimal amount)
    {
        decimal tax = amount * TaxRate;
        decimal fee = amount * FeeRate;
        decimal total = amount + tax + fee;
        return (tax, fee, total);
    }
}

// --- Responsibility 3: Payment gateway ---

interface IPaymentGateway
{
    void Charge(string cardNumber, decimal amount, string currency);
}

sealed class SimulatedPaymentGateway : IPaymentGateway
{
    public void Charge(string cardNumber, decimal amount, string currency)
    {
        string masked = $"****{cardNumber[^4..]}";
        Console.WriteLine($"Charging {amount:C} to card {masked} in {currency}");
        Console.WriteLine($"Transaction approved: REF-{Random.Shared.Next(100000, 999999)}");
    }
}

// --- Responsibility 4: Receipt generation ---

interface IReceiptGenerator
{
    string Generate(string cardNumber, decimal amount, string currency, string reference);
}

sealed class ConsoleReceiptGenerator : IReceiptGenerator
{
    public string Generate(string cardNumber, decimal amount, string currency, string reference)
    {
        string receipt = $"""
            ╔══════════════════════════════════╗
            ║           PAYMENT RECEIPT        ║
            ║  Reference: {reference,-15} ║
            ║  Card: ****{cardNumber[^4..],-19} ║
            ║  Amount: {amount,15:C} {currency} ║
            ╚══════════════════════════════════╝
            """;
        Console.WriteLine(receipt);
        return receipt;
    }
}

// --- Responsibility 5: Audit logging ---

interface IAuditLogger
{
    void Log(string message);
}

sealed class ConsoleAuditLogger : IAuditLogger
{
    public void Log(string message)
    {
        string entry = $"[{DateTime.UtcNow:O}] {message}";
        Console.WriteLine(entry);
    }
}

// --- Orchestrator ---

sealed class PaymentOrchestrator
{
    private readonly IPaymentValidator _validator;
    private readonly PaymentCalculator _calculator;
    private readonly IPaymentGateway _gateway;
    private readonly IReceiptGenerator _receiptGen;
    private readonly IAuditLogger _logger;

    public PaymentOrchestrator(
        IPaymentValidator validator,
        PaymentCalculator calculator,
        IPaymentGateway gateway,
        IReceiptGenerator receiptGen,
        IAuditLogger logger)
    {
        _validator = validator;
        _calculator = calculator;
        _gateway = gateway;
        _receiptGen = receiptGen;
        _logger = logger;
    }

    public void ProcessPayment(string cardNumber, decimal amount, string currency)
    {
        _validator.Validate(cardNumber, amount);
        var (tax, fee, total) = _calculator.Calculate(amount);
        _gateway.Charge(cardNumber, total, currency);
        string reference = $"REF-{Random.Shared.Next(100000, 999999)}";
        _receiptGen.Generate(cardNumber, total, currency, reference);
        _logger.Log($"Payment {reference}: {total:C} {currency}");
    }
}

// --- Main driver ---

var orchestrator = new PaymentOrchestrator(
    new PaymentValidator(),
    new PaymentCalculator(),
    new SimulatedPaymentGateway(),
    new ConsoleReceiptGenerator(),
    new ConsoleAuditLogger()
);

orchestrator.ProcessPayment("4111111111111111", 149.99m, "USD");
```

## Key Language-Specific Notes

- **Top-level statements** (C# 9+) let the driver code at the bottom stand alone without a `Program` class wrapper, keeping the focus on the SRP component structure.
- **`record` types and tuples** — C# 12 tuples (`(decimal Tax, decimal Fee, decimal Total)`) provide lightweight multi-return without creating a dedicated class for each calculation result, a pragmatic balance with SRP.
- **Interfaces with single methods** — Each interface (`IPaymentValidator`, `IPaymentGateway`, etc.) has exactly one method. This is the interface-level application of SRP: one method, one concern.
- **Primary constructors** (C# 12) can reduce boilerplate in simple SRP classes, though explicit constructor injection is used here for clarity.
- **Sealed classes** — `sealed` prevents unintended inheritance, reinforcing the SRP intent that each class does exactly one job and is not meant to be extended.
