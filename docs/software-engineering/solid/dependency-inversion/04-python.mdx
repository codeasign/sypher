## DIP in Python

Python's duck typing means any object with the right methods satisfies an interface without explicit declaration. However, using the `abc` module to define Abstract Base Classes (ABCs) makes the contract explicit and encourages DIP compliance, especially in larger projects.

This example builds a **weather reporting system**. A high-level `WeatherReporter` fetches weather data and presents it. Without the DIP, it would be coupled to a specific weather API provider.

## The Violation

In the violating version, `WeatherReporter` directly creates an `OpenWeatherMapApi` instance and calls its methods. Switching to a different provider (WeatherStack, WeatherAPI.com) requires editing `WeatherReporter`.

```python
# VIOLATION: WeatherReporter depends on concrete OpenWeatherMapApi.
# Changing the weather provider requires modifying WeatherReporter.
# Testing requires a real API key and network access.

import json
import urllib.request
from datetime import datetime


class OpenWeatherMapApi:
    """Concrete implementation for OpenWeatherMap."""

    def __init__(self, api_key: str) -> None:
        self.api_key = api_key
        self.base_url = "https://api.openweathermap.org/data/2.5"

    def get_current(self, city: str) -> dict:
        url = f"{self.base_url}/weather?q={city}&appid={self.api_key}&units=metric"
        with urllib.request.urlopen(url) as response:
            data = json.loads(response.read().decode())
            return {
                "temperature": data["main"]["temp"],
                "humidity": data["main"]["humidity"],
                "description": data["weather"][0]["description"],
                "city": city,
                "timestamp": datetime.now().isoformat(),
            }

    def get_forecast(self, city: str, days: int = 5) -> list[dict]:
        url = f"{self.base_url}/forecast?q={city}&appid={self.api_key}&units=metric"
        with urllib.request.urlopen(url) as response:
            data = json.loads(response.read().decode())
            forecasts = []
            for item in data["list"][:days]:
                forecasts.append({
                    "temperature": item["main"]["temp"],
                    "humidity": item["main"]["humidity"],
                    "description": item["weather"][0]["description"],
                    "timestamp": item["dt_txt"],
                })
            return forecasts


class WeatherReporter:
    """Reports weather information."""

    def __init__(self) -> None:
        # VIOLATION: Direct instantiation of concrete implementation.
        self.api = OpenWeatherMapApi(api_key="DEMO_KEY")

    def report_current(self, city: str) -> str:
        data = self.api.get_current(city)
        return (
            f"Weather in {data['city']}: {data['description']}, "
            f"{data['temperature']:.1f}°C, humidity {data['humidity']}%"
        )

    def report_forecast(self, city: str) -> str:
        forecasts = self.api.get_forecast(city, days=3)
        lines = [f"3-day forecast for {city}:"]
        for f in forecasts:
            lines.append(
                f"  {f['timestamp']}: {f['description']}, "
                f"{f['temperature']:.1f}°C"
            )
        return "\n".join(lines)


# Usage
def main() -> None:
    reporter = WeatherReporter()
    print(reporter.report_current("London"))
    print()
    print(reporter.report_forecast("London"))


# PROBLEM: Cannot switch to WeatherStack without editing WeatherReporter.
# PROBLEM: Unit tests would hit the real API endpoint.
```

## The Correct Implementation

We define a `WeatherApi` abstract base class. `OpenWeatherMapApi` and `WeatherStackApi` implement it. `WeatherReporter` receives its dependency through the constructor and works with any implementation that conforms to the ABC.

```python
# CORRECT: WeatherReporter depends on WeatherApi abstraction.
# Multiple API providers can be swapped without changing the reporter.

from abc import ABC, abstractmethod
import json
import urllib.request
from datetime import datetime
from typing import Protocol, runtime_checkable


# ── Abstraction (ABC approach) ──

class WeatherApi(ABC):
    """Abstract interface for weather data providers."""

    @abstractmethod
    def get_current(self, city: str) -> dict:
        """Get current weather for a city."""
        ...

    @abstractmethod
    def get_forecast(self, city: str, days: int = 5) -> list[dict]:
        """Get forecast for a city."""
        ...


# ── Low-level implementation 1: OpenWeatherMap ──

class OpenWeatherMapApi(WeatherApi):
    """OpenWeatherMap implementation."""

    def __init__(self, api_key: str) -> None:
        self.api_key = api_key
        self.base_url = "https://api.openweathermap.org/data/2.5"

    def get_current(self, city: str) -> dict:
        print(f"[OpenWeatherMap] Fetching current weather for {city}")
        # In production, this would make a real HTTP request:
        # url = f"{self.base_url}/weather?q={city}&appid={self.api_key}&units=metric"
        # with urllib.request.urlopen(url) as response:
        #     data = json.loads(response.read().decode())
        # Simulated response for demonstration:
        return {
            "temperature": 18.5,
            "humidity": 72,
            "description": "partly cloudy",
            "city": city,
            "source": "OpenWeatherMap",
            "timestamp": datetime.now().isoformat(),
        }

    def get_forecast(self, city: str, days: int = 5) -> list[dict]:
        print(f"[OpenWeatherMap] Fetching {days}-day forecast for {city}")
        # Simulated forecast data:
        conditions = ["clear sky", "light rain", "overcast clouds", "few clouds", "moderate rain"]
        forecasts = []
        for i in range(days):
            forecasts.append({
                "temperature": 15.0 + i * 1.5,
                "humidity": 65 + i * 3,
                "description": conditions[i % len(conditions)],
                "timestamp": f"2025-06-{15 + i:02d} 12:00:00",
                "source": "OpenWeatherMap",
            })
        return forecasts


# ── Low-level implementation 2: WeatherStack ──

class WeatherStackApi(WeatherApi):
    """WeatherStack implementation."""

    def __init__(self, access_key: str) -> None:
        self.access_key = access_key
        self.base_url = "https://api.weatherstack.com"

    def get_current(self, city: str) -> dict:
        print(f"[WeatherStack] Fetching current weather for {city}")
        # Simulated response:
        return {
            "temperature": 22.0,
            "humidity": 55,
            "description": "sunny",
            "city": city,
            "source": "WeatherStack",
            "timestamp": datetime.now().isoformat(),
        }

    def get_forecast(self, city: str, days: int = 5) -> list[dict]:
        print(f"[WeatherStack] Fetching {days}-day forecast for {city}")
        conditions = ["sunny", "partly cloudy", "rain showers", "thunderstorms", "fair"]
        forecasts = []
        for i in range(days):
            forecasts.append({
                "temperature": 20.0 + i * 2.0,
                "humidity": 50 + i * 5,
                "description": conditions[i % len(conditions)],
                "timestamp": f"2025-06-{15 + i:02d} 12:00:00",
                "source": "WeatherStack",
            })
        return forecasts


# ── Low-level implementation 3: Local mock (for testing) ──

class LocalWeatherApi(WeatherApi):
    """In-memory mock for testing."""

    def __init__(self) -> None:
        self.current_calls: list[str] = []
        self.forecast_calls: list[str] = []

    def get_current(self, city: str) -> dict:
        self.current_calls.append(city)
        return {
            "temperature": 25.0,
            "humidity": 60,
            "description": "test conditions",
            "city": city,
            "source": "LocalWeatherApi",
            "timestamp": datetime.now().isoformat(),
        }

    def get_forecast(self, city: str, days: int = 5) -> list[dict]:
        self.forecast_calls.append(city)
        return [
            {
                "temperature": 25.0 + i,
                "humidity": 60 - i * 2,
                "description": f"test day {i + 1}",
                "timestamp": f"2025-06-{15 + i:02d} 12:00:00",
                "source": "LocalWeatherApi",
            }
            for i in range(days)
        ]


# ── High-level module ──
# CORRECT: Depends only on WeatherApi abstraction.

class WeatherReporter:
    """Reports weather information using any WeatherApi implementation."""

    def __init__(self, api: WeatherApi) -> None:
        # CORRECT: Dependency is injected, not created.
        self.api = api

    def report_current(self, city: str) -> str:
        data = self.api.get_current(city)
        return (
            f"Weather in {data['city']} ({data.get('source', 'unknown')}): "
            f"{data['description']}, {data['temperature']:.1f}°C, "
            f"humidity {data['humidity']}%"
        )

    def report_forecast(self, city: str, days: int = 3) -> str:
        forecasts = self.api.get_forecast(city, days=days)
        lines = [f"{days}-day forecast for {city}:"]
        for f in forecasts:
            lines.append(
                f"  {f['timestamp']}: {f['description']}, "
                f"{f['temperature']:.1f}°C"
            )
        return "\n".join(lines)


# ── Composition root ──

def main() -> None:
    # Choose provider based on configuration:
    provider = "openweathermap"
    # provider = "weatherstack"

    if provider == "openweathermap":
        api: WeatherApi = OpenWeatherMapApi(api_key="demo_key_123")
    elif provider == "weatherstack":
        api = WeatherStackApi(access_key="demo_key_456")
    else:
        api = LocalWeatherApi()

    reporter = WeatherReporter(api)

    # Generate reports
    print("=== Current Weather ===")
    print(reporter.report_current("London"))
    print()

    print("=== Forecast ===")
    print(reporter.report_forecast("London", days=3))
    print()

    # Switch city
    print("=== Another City ===")
    print(reporter.report_current("Tokyo"))


# Run with OpenWeatherMap:
main()

# Run with LocalWeatherApi for testing:
print("\n" + "=" * 50)
print("TESTING MODE")
print("=" * 50)

test_reporter = WeatherReporter(test_api)
print(test_reporter.report_current("Berlin"))
print(f"Cities queried for current: {test_api.current_calls}")
print(f"Cities queried for forecast: {test_api.forecast_calls}")
```

## Key Points

- `WeatherApi` is an ABC that defines the contract. Python's `@abstractmethod` decorator enforces that subclasses implement both methods.
- `WeatherReporter` never imports or instantiates `OpenWeatherMapApi` or `WeatherStackApi`. It works purely through the `WeatherApi` abstraction.
- Adding a new provider (e.g., `WeatherApiComApi`) requires a new class implementing `WeatherApi` and a change only in the composition root.
- `LocalWeatherApi` provides a lightweight test double that records which cities were queried — perfect for unit testing without network access.
- Python also supports `Protocol` from `typing` for structural subtyping (duck typing with static checking), which is an even more flexible form of abstraction than ABCs.
