## Breaking It Down

### Step 1: Basic try/except

```python
# basic_error.py

def divide(a, b):
    """Safely divide two numbers."""
    try:
        result = a / b
    except ZeroDivisionError:
        print("Error: cannot divide by zero")
        result = None
    return result

print(divide(10, 2))  # 5.0
print(divide(10, 0))  # Error: cannot divide by zero → None
```

### Step 2: Catching Multiple Exception Types

```python
# multiple_exceptions.py

def process_value(value):
    """Parse and process a numeric value safely."""
    try:
        # Convert to float — may raise ValueError
        num = float(value)
        # Compute reciprocal — may raise ZeroDivisionError
        result = 1.0 / num
    except ValueError:
        print(f"Cannot convert '{value}' to a number")
        return None
    except ZeroDivisionError:
        print("Cannot compute reciprocal of zero")
        return None
    else:
        print(f"1/{num} = {result}")
        return result
    finally:
        print("  (process_value completed)")

print(process_value("10"))     # 1/10.0 = 0.1
print(process_value("abc"))    # Cannot convert 'abc' to a number
print(process_value("0"))      # Cannot compute reciprocal of zero
```

def read_config(path):
    """Read a JSON config file, handling common errors."""
    try:
        with open(path, "r", encoding="utf-8") as f:
            import json
            return json.load(f)
    except (FileNotFoundError, json.JSONDecodeError) as e:
        print(f"Config error: {e}")
        return {}


# Catching the base exception class
def safe_api_call(url):
    """Call an API with broad error handling."""
    import requests
    try:
        response = requests.get(url, timeout=10)
        response.raise_for_status()  # Raises for 4xx/5xx
        return response.json()
    except requests.Timeout:
        return {"error": "timeout"}
    except requests.ConnectionError:
        return {"error": "connection_failed"}
    except requests.RequestException as e:
        return {"error": str(e)}

### Step 4: Raising Exceptions

```python
# raising.py

def validate_confidence(score):
    """Validate a confidence score, raising on invalid input."""
    if score is None:
        raise ValueError("Confidence score cannot be None")
    if not isinstance(score, (int, float)):
        raise TypeError(f"Expected numeric score, got {type(score).__name__}")
    if score < 0.0 or score > 1.0:
        raise ValueError(f"Score {score} is outside [0.0, 1.0]")
    return score

validate_confidence(0.85)   # OK
# validate_confidence(None)  # ValueError: Confidence score cannot be None
# validate_confidence(1.5)   # ValueError: Score 1.5 is outside [0.0, 1.0]
```

### Step 5: Custom Exception Classes

Create meaningful exception types for your domain:

```python
# custom_exceptions.py

class AIEngineError(Exception):
    """Base exception for AI engine errors."""
    pass

class ModelLoadError(AIEngineError):
    """Raised when a model fails to load."""
    pass

class InferenceError(AIEngineError):
    """Raised when inference fails."""
    def __init__(self, message, model_name=None, input_preview=None):
        self.model_name = model_name
        self.input_preview = input_preview
        super().__init__(message)

class RateLimitError(AIEngineError):
    """Raised when API rate limit is exceeded."""
    def __init__(self, retry_after=None):
        self.retry_after = retry_after
        super().__init__(f"Rate limited. Retry after {retry_after}s" if retry_after else "Rate limited")

# Usage
def run_inference(model, input_text):
    try:
        if model not in ["gpt-4", "claude-3"]:
            raise ModelLoadError(f"Unknown model: {model}")
        if not input_text.strip():
            raise InferenceError("Empty input", model_name=model)
        return f"Result for: {input_text}"
    except AIEngineError:
        # Catch all AI engine errors
        raise  # Re-raise after logging
```

### Step 7: Build Something Real — Resilient API Client

```python
# resilient_client.py
import time
import random

class APIError(Exception):
    """Base API error."""
    pass

class RateLimitError(APIError):
    def __init__(self, retry_after=1.0):
        self.retry_after = retry_after
        super().__init__(f"Rate limited, retry in {retry_after}s")

class ServerError(APIError):
    pass

def call_llm_api(prompt):
    """Simulated LLM API — fails unpredictably."""
    roll = random.random()
    if roll < 0.3:
        raise RateLimitError(retry_after=0.5)
    if roll < 0.5:
        raise ServerError("Internal server error")
    if roll < 0.6:
        raise ConnectionError("Network timeout")
    return {"text": f"Response to: {prompt}", "tokens": 42}

def resilient_invoke(prompt, max_retries=3):
    """Call the API with retry and backoff."""
    last_error = None

    for attempt in range(1, max_retries + 1):
        try:
            return call_llm_api(prompt)
        except RateLimitError as e:
            print(f"  Attempt {attempt}: rate limited, waiting {e.retry_after}s")
            time.sleep(e.retry_after)
            last_error = e
        except ServerError as e:
            print(f"  Attempt {attempt}: server error, retrying...")
            time.sleep(1.0 * attempt)  # Linear backoff
            last_error = e
        except ConnectionError as e:
            print(f"  Attempt {attempt}: network issue, retrying...")
            time.sleep(0.5)
            last_error = e

    raise RuntimeError(f"All {max_retries} attempts failed") from last_error

# Test
for i in range(3):
    print(f"\nRequest {i + 1}:")
    try:
        result = resilient_invoke("Hello!")
        print(f"  Success: {result['text']}")
    except RuntimeError as e:
        print(f"  Failed: {e}")
```
