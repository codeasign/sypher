# CSV — Hands On

⏱ 14 minutes

## Breaking It Down

### Step 1: Reading a CSV File

```python
# read_csv.py
import csv

# Create a sample CSV
with open("models.csv", "w", newline="", encoding="utf-8") as f:
    f.write("model,accuracy,latency_ms\n")
    f.write("gpt-4,0.94,1200\n")
    f.write("claude-3,0.93,1800\n")
    f.write("llama-3,0.89,900\n")

# Read with csv.reader — each row is a list of strings
with open("models.csv", "r", newline="", encoding="utf-8") as f:
    reader = csv.reader(f)
    header = next(reader)  # Read the first row as header
    print(f"Header: {header}")  # ['model', 'accuracy', 'latency_ms']
    for row in reader:
        print(f"Model: {row[0]}, Accuracy: {row[1]}, Latency: {row[2]}")

# Clean up
import os
os.remove("models.csv")
```

Always open CSV files with `newline=""` — this prevents the csv module from double-handling newline characters on some platforms.

### Step 2: Using DictReader and DictWriter

```python
# dict_csv.py
import csv

data = [
    {"model": "gpt-4", "accuracy": 0.94, "latency_ms": 1200},
    {"model": "claude-3", "accuracy": 0.93, "latency_ms": 1800},
    {"model": "llama-3", "accuracy": 0.89, "latency_ms": 900},
]

# Write with DictWriter
with open("models.csv", "w", newline="", encoding="utf-8") as f:
    fieldnames = ["model", "accuracy", "latency_ms"]
    writer = csv.DictWriter(f, fieldnames=fieldnames)
    writer.writeheader()
    writer.writerows(data)

# Read with DictReader — access by column name
with open("models.csv", "r", newline="", encoding="utf-8") as f:
    reader = csv.DictReader(f)
    for row in reader:
        acc = float(row["accuracy"])
        lat = int(row["latency_ms"])
        print(f"{row['model']}: acc={acc:.2f}, lat={lat}ms")

# Clean up
os.remove("models.csv")
```

`DictWriter` uses the `fieldnames` list to determine the column order. `DictReader` uses the header row to determine field names automatically.

### Step 3: Handling Different Delimiters

```python
# delimiters.py
import csv

# TSV — tab-separated values
tsv_data = "model\taccuracy\tlatency_ms\ngpt-4\t0.94\t1200\nclaude-3\t0.93\t1800\n"

with open("models.tsv", "w", newline="", encoding="utf-8") as f:
    f.write(tsv_data)

# Read TSV with delimiter="\t"
with open("models.tsv", "r", newline="", encoding="utf-8") as f:
    reader = csv.DictReader(f, delimiter="\t")
    for row in reader:
        print(f"{row['model']}: {row['accuracy']}")

# Custom delimiter — semicolons (common in European CSV)
csv_data = "model;accuracy;latency_ms\ngpt-4;0.94;1200\n"
reader = csv.DictReader(csv_data.splitlines(), delimiter=";")
for row in reader:
    print(f"{row['model']}: {row['accuracy']}")

import os
os.remove("models.tsv")
```

### Step 4: Filtering and Processing CSV Data

```python
# filter_csv.py
import csv

rows = [
    {"model": "gpt-4", "accuracy": "0.94", "latency_ms": "1200"},
    {"model": "claude-3", "accuracy": "0.93", "latency_ms": "1800"},
    {"model": "llama-3", "accuracy": "0.89", "latency_ms": "900"},
    {"model": "gemini", "accuracy": "0.91", "latency_ms": "1500"},
]

with open("results.csv", "w", newline="", encoding="utf-8") as f:
    writer = csv.DictWriter(f, fieldnames=["model", "accuracy", "latency_ms"])
    writer.writeheader()
    writer.writerows(rows)

# Filter: keep only models with accuracy > 0.9
filtered = []
with open("results.csv", "r", newline="", encoding="utf-8") as f:
    reader = csv.DictReader(f)
    for row in reader:
        if float(row["accuracy"]) > 0.9:
            filtered.append(row)

print(f"Filtered {len(filtered)} rows")  # 3 (gpt-4, claude-3, gemini)

# Write filtered results to a new CSV
with open("high_accuracy.csv", "w", newline="", encoding="utf-8") as f:
    writer = csv.DictWriter(f, fieldnames=["model", "accuracy", "latency_ms"])
    writer.writeheader()
    writer.writerows(filtered)

# Sort by accuracy descending
sorted_rows = sorted(filtered, key=lambda r: float(r["accuracy"]), reverse=True)
for r in sorted_rows:
    print(f"{r['model']}: {r['accuracy']}")
# gpt-4: 0.94
# claude-3: 0.93
# gemini: 0.91

# Clean up
import os
os.remove("results.csv")
os.remove("high_accuracy.csv")
```

### Step 5: Handling Quoting and Embedded Commas

```python
# quoting.py
import csv

# Some fields contain commas or quotes — CSV handles this with quoting
rows = [
    {"model": "gpt-4", "description": "Latest OpenAI model, supports vision", "accuracy": "0.94"},
    {"model": "claude-3", "description": "Strong at reasoning & analysis", "accuracy": "0.93"},
]

with open("models.csv", "w", newline="", encoding="utf-8") as f:
    writer = csv.DictWriter(f, fieldnames=["model", "description", "accuracy"])
    writer.writeheader()
    writer.writerows(rows)

# Read it back to see how csv module handles quoting
with open("models.csv", "r", encoding="utf-8") as f:
    content = f.read()
    print(content)
    # model,description,accuracy
    # gpt-4,"Latest OpenAI model, supports vision",0.94
    # Note: the comma inside the description is quoted

# Reading works transparently
with open("models.csv", "r", newline="", encoding="utf-8") as f:
    reader = csv.DictReader(f)
    for row in reader:
        print(f"{row['model']}: {row['description']}")

import os
os.remove("models.csv")
```

The `csv` module handles quoting automatically. The writer adds quotes around fields that contain the delimiter, newlines, or the quote character. The reader strips them transparently.

### Step 6: Processing Large CSV Files in Chunks

```python
# chunked.py
import csv

# Generate a larger sample (10,000 rows) — in a real scenario this would be a file on disk
import io
buf = io.StringIO()
writer = csv.writer(buf)
writer.writerow(["id", "value", "category"])
for i in range(10000):
    writer.writerow([i, float(i % 100), "A" if i % 2 == 0 else "B"])
buf.seek(0)

# Process in chunks of 1000
chunk_size = 1000
reader = csv.DictReader(buf)
chunk = []
for i, row in enumerate(reader):
    row["value"] = float(row["value"])
    chunk.append(row)
    if (i + 1) % chunk_size == 0:
        # Process chunk (e.g., compute partial stats)
        print(f"Processed rows 0–{i}: sum = {sum(r['value'] for r in chunk):.0f}")
        chunk = []

# Process remaining rows
if chunk:
    print(f"Final partial chunk: {len(chunk)} rows")
```

Processing in chunks is essential when the CSV file is too large for memory. For truly huge files, iterate row by row and write filtered results to a new file incrementally rather than accumulating in memory.

## Build Something Real — CSV Dataset Cleaner

```python
# dataset_cleaner.py
import csv
from pathlib import Path
import re

def clean_csv(input_path, output_path, expected_columns=None):
    """Read a CSV, clean it, and write the cleaned version."""

    def is_valid_row(row):
        """Check if a row is valid."""
        if expected_columns and len(row) != expected_columns:
            return False
        # All values should be non-empty strings
        return all(cell.strip() for cell in row)

    with open(input_path, "r", newline="", encoding="utf-8") as fin:
        reader = csv.reader(fin)
        header = next(reader) if expected_columns is None else None

        valid_rows = []
        skipped = 0

        for row in reader:
            if is_valid_row(row):
                # Strip whitespace from each cell
                cleaned = [cell.strip() for cell in row]
                valid_rows.append(cleaned)
            else:
                skipped += 1

    with open(output_path, "w", newline="", encoding="utf-8") as fout:
        writer = csv.writer(fout)
        if header:
            writer.writerow(header)
        writer.writerows(valid_rows)

    return len(valid_rows), skipped

# Create a dirty CSV to clean
with open("dirty.csv", "w", newline="", encoding="utf-8") as f:
    f.write("name,score,label\n")
    f.write("Alice,0.95,positive\n")
    f.write("Bob,,positive\n")         # missing score
    f.write("Carol,0.88,negative\n")
    f.write("Dave,0.72,\n")             # empty label
    f.write(" Eve , 0.91 , positive\n") # extra whitespace
    f.write("Frank,0.85,positive,extra\n")  # too many columns

valid, skipped = clean_csv("dirty.csv", "clean.csv", expected_columns=3)
print(f"Valid rows: {valid}, Skipped: {skipped}")
# Valid rows: 3 (Alice, Carol, Eve), Skipped: 3 (Bob, Dave, Frank)

# Show cleaned file
with open("clean.csv", "r", encoding="utf-8") as f:
    print(f.read())
# name,score,label
# Alice,0.95,positive
# Carol,0.88,negative
# Eve,0.91,positive

# Clean up
import os
os.remove("dirty.csv")
os.remove("clean.csv")
```

The cleaner strips whitespace from every cell, filters out rows with missing values, and rejects rows with the wrong column count. Always validate raw CSV data before feeding it into analysis or model training pipelines.
