# File Handling — Hands On

⏱ 14 minutes

## Breaking It Down

### Step 1: Reading a Text File

```python
# read_text.py

# Create a sample file first (in a real scenario this would already exist)
with open("sample.txt", "w", encoding="utf-8") as f:
    f.write("Line 1: AI engineering uses Python\n")
    f.write("Line 2: File handling is essential\n")
    f.write("Line 3: Practice makes perfect\n")

# Read the entire file
with open("sample.txt", "r", encoding="utf-8") as f:
    content = f.read()
print(content)

# Read line by line (memory-efficient for large files)
with open("sample.txt", "r", encoding="utf-8") as f:
    for line in f:
        print(f"Read: {line.strip()}")

# Read all lines into a list
with open("sample.txt", "r", encoding="utf-8") as f:
    lines = f.readlines()
print(f"Total lines: {len(lines)}")  # 3

# Clean up
import os
os.remove("sample.txt")
```

### Step 2: Writing to a File

```python
# write_text.py

# Write mode "w" overwrites the file
with open("output.txt", "w", encoding="utf-8") as f:
    f.write("First line\n")
    f.write("Second line\n")

# Append mode "a" adds to the end
with open("output.txt", "a", encoding="utf-8") as f:
    f.write("Third line (appended)\n")

with open("output.txt", "r", encoding="utf-8") as f:
    print(f.read())
# First line
# Second line
# Third line (appended)

os.remove("output.txt")
```

### Step 3: Working with JSON

JSON is the standard data format for AI APIs:

```python
# json_demo.py
import json

# Python dict -> JSON string
data = {
    "model": "gpt-4",
    "parameters": {
        "temperature": 0.7,
        "max_tokens": 2048,
    },
    "tags": ["production", "v2"],
}

# Write JSON to file
with open("config.json", "w", encoding="utf-8") as f:
    json.dump(data, f, indent=2)

# Read JSON from file
with open("config.json", "r", encoding="utf-8") as f:
    loaded = json.load(f)

print(loaded["model"])               # gpt-4
print(loaded["parameters"]["temperature"])  # 0.7

# JSON string in memory
json_str = json.dumps(data, indent=2)
print(json_str)

# Parse JSON string
parsed = json.loads(json_str)
print(parsed["tags"])  # ['production', 'v2']

os.remove("config.json")
```

### Step 4: Working with CSV

CSV is common for datasets and evaluation results:

```python
# csv_demo.py
import csv

# Write CSV
headers = ["model", "accuracy", "f1_score", "latency_ms"]
rows = [
    ["gpt-4", 0.94, 0.92, 1200],
    ["claude-3", 0.93, 0.91, 1800],
    ["llama-3", 0.89, 0.87, 900],
]

with open("results.csv", "w", newline="", encoding="utf-8") as f:
    writer = csv.writer(f)
    writer.writerow(headers)
    writer.writerows(rows)

# Read CSV
with open("results.csv", "r", newline="", encoding="utf-8") as f:
    reader = csv.reader(f)
    for row in reader:
        print(row)

# Using DictReader and DictWriter
with open("results.csv", "r", newline="", encoding="utf-8") as f:
    reader = csv.DictReader(f)
    for row in reader:
        print(f"{row['model']}: acc={row['accuracy']}")

os.remove("results.csv")
```

### Step 5: Working with Paths (pathlib)

`pathlib` provides an object-oriented approach to file paths:

```python
# pathlib_demo.py
from pathlib import Path

# Create a path object
data_dir = Path("data")
logs_dir = data_dir / "logs"  # Uses / operator for path joining

# Create directories
data_dir.mkdir(exist_ok=True)
logs_dir.mkdir(parents=True, exist_ok=True)

# Create a file in the directory
file_path = logs_dir / "experiment_1.log"
file_path.write_text("Experiment started at 2026-06-30\n", encoding="utf-8")

# Read it back
content = file_path.read_text(encoding="utf-8")
print(content)

# Check properties
print(f"Exists: {file_path.exists()}")     # True
print(f"Is file: {file_path.is_file()}")   # True
print(f"Suffix: {file_path.suffix}")       # .log
print(f"Stem: {file_path.stem}")           # experiment_1
print(f"Parent: {file_path.parent}")       # data/logs

# List all .log files in directory
for log_file in logs_dir.glob("*.log"):
    print(f"Found log: {log_file}")

# Clean up
import shutil
shutil.rmtree(data_dir)
```

### Step 6: Build Something Real — Experiment Logger

```python
# experiment_logger.py — Log experiment results to files

import json
from pathlib import Path
from datetime import datetime

class ExperimentLogger:
    """Logs experiment results to structured files."""

    def __init__(self, experiment_dir="experiments"):
        self.base_dir = Path(experiment_dir)
        self.base_dir.mkdir(exist_ok=True)

    def log_run(self, model_name, config, metrics):
        """Log a single experiment run."""
        timestamp = datetime.now().strftime("%Y%m%d_%H%M%S")
        run_name = f"{model_name}_{timestamp}"

        # Create a directory for this run
        run_dir = self.base_dir / run_name
        run_dir.mkdir(parents=True)

        # Save config
        config_path = run_dir / "config.json"
        config_path.write_text(json.dumps(config, indent=2), encoding="utf-8")

        # Save metrics
        metrics_path = run_dir / "metrics.json"
        metrics_path.write_text(json.dumps(metrics, indent=2), encoding="utf-8")

        # Summary line
        summary_path = self.base_dir / "summary.csv"
        header = "run_name,model,accuracy,f1_score\n"
        line = f"{run_name},{model_name},{metrics.get('accuracy', 'N/A')},{metrics.get('f1', 'N/A')}\n"

        if not summary_path.exists():
            summary_path.write_text(header, encoding="utf-8")

        with open(summary_path, "a", encoding="utf-8") as f:
            f.write(line)

        print(f"Logged run: {run_name}")
        return run_name

# Usage
logger = ExperimentLogger()
logger.log_run(
    "gpt-4",
    config={"temperature": 0.7, "max_tokens": 2048},
    metrics={"accuracy": 0.94, "f1": 0.92, "latency_ms": 1200},
)
logger.log_run(
    "claude-3",
    config={"temperature": 0.5, "max_tokens": 4096},
    metrics={"accuracy": 0.93, "f1": 0.91, "latency_ms": 1800},
)

# Read back the summary
summary_path = Path("experiments/summary.csv")
if summary_path.exists():
    print(summary_path.read_text(encoding="utf-8"))

# Clean up
import shutil
shutil.rmtree("experiments")
```

## Security Considerations

- **Path traversal attacks:** If you accept file paths from user input, a malicious user could write `../../../etc/passwd`. Validate paths with `Path.resolve()` and check they are within an expected directory.
- **Encoding:** Always specify `encoding="utf-8"` explicitly. The default encoding varies by OS, and omitting it leads to `UnicodeDecodeError` when moving between machines.
- **Temporary files:** Use `tempfile.NamedTemporaryFile` for sensitive data that should not persist.
- **JSON from untrusted sources:** `json.loads()` is safe (it cannot execute arbitrary code), but be prepared for any structure — validate after loading.
