# Project Structure — Hands On

⏱ 15 minutes

## Breaking It Down

### Step 1: Create the Directory Skeleton

Start from your project root. Every directory you create has a specific role:

```bash
mkdir -p src/my_project/data
mkdir -p src/my_project/models
mkdir -p src/my_project/pipelines
mkdir -p tests
mkdir -p configs
mkdir -p data/raw
mkdir -p data/processed
mkdir -p notebooks
mkdir -p scripts
```

Your working tree now looks like this:

```
my_project/
├── src/
│   └── my_project/
│       ├── __init__.py
│       ├── data/
│       ├── models/
│       └── pipelines/
├── tests/
├── configs/
├── data/
│   ├── raw/
│   └── processed/
├── notebooks/
└── scripts/
```

### Step 2: Configure pyproject.toml

This file makes your project installable and records dependencies:

```toml
# pyproject.toml
[build-system]
requires = ["setuptools>=68.0.0", "wheel"]
build-backend = "setuptools.backends._legacy:_Backend"

[project]
name = "my_project"
version = "0.1.0"
description = "AI pipeline for text classification"
requires-python = ">=3.10"

dependencies = [
    "numpy>=1.24.0",
    "pandas>=2.0.0",
    "openai>=1.0.0",
    "pyyaml>=6.0",
]

[project.optional-dependencies]
dev = [
    "pytest>=7.0.0",
    "pytest-cov>=4.0.0",
    "black>=23.0.0",
    "ruff>=0.1.0",
]

[tool.setuptools.packages.find]
where = ["src"]
```

With `pyproject.toml` in place, install your package in editable mode:

```bash
pip install -e ".[dev]"
```

This lets you run `from my_project.config import load_config` from anywhere without path hacks.

### Step 3: Build the Configuration Module

Keep configuration in data files, not in code:

```yaml
# configs/default.yaml
model:
  name: gpt-4o-mini
  temperature: 0.3
  max_tokens: 1024

data:
  raw_path: data/raw/reviews.csv
  processed_path: data/processed/reviews_cleaned.parquet
  test_split: 0.2

training:
  batch_size: 32
  epochs: 5
  learning_rate: 0.001
```

```python
# src/my_project/config.py
"""Load and validate YAML configuration files."""

from pathlib import Path
from typing import Any, Dict

import yaml


def load_config(path: str = "configs/default.yaml") -> Dict[str, Any]:
    """Load a YAML configuration file and return it as a dictionary.

    Args:
        path: Path to the YAML config file, relative to project root.

    Returns:
        Dictionary of configuration parameters.

    Raises:
        FileNotFoundError: If the config file does not exist.
        yaml.YAMLError: If the YAML is malformed.
    """
    config_path = Path(path)
    if not config_path.exists():
        raise FileNotFoundError(f"Config file not found: {config_path.resolve()}")

    with open(config_path, "r") as f:
        config = yaml.safe_load(f)

    return config
```

### Step 4: Build the Data Module

Data loading and preprocessing live in `src/my_project/data/`:

```python
# src/my_project/data/loader.py
"""Utilities for loading and preprocessing text data."""

from pathlib import Path
from typing import List, Tuple

import pandas as pd


def load_reviews(path: str) -> pd.DataFrame:
    """Load review data from CSV into a DataFrame.

    Expected columns: text, label (positive/negative).
    """
    df = pd.read_csv(path)
    required = {"text", "label"}
    missing = required - set(df.columns)
    if missing:
        raise ValueError(f"CSV missing required columns: {missing}")
    return df


def split_train_test(
    df: pd.DataFrame,
    test_ratio: float = 0.2,
    random_state: int = 42,
) -> Tuple[pd.DataFrame, pd.DataFrame]:
    """Split a DataFrame into training and test sets."""
    test_size = int(len(df) * test_ratio)
    train = df.iloc[test_size:].reset_index(drop=True)
    test = df.iloc[:test_size].reset_index(drop=True)
    return train, test
```

### Step 5: Build the Models Module

Each model type gets its own module:

```python
# src/my_project/models/classifier.py
"""LLM-based text classifier."""

from typing import List

from openai import OpenAI


class LLMClassifier:
    """Classify text sentiment using an LLM API."""

    def __init__(self, model: str = "gpt-4o-mini", temperature: float = 0.0):
        self.model = model
        self.temperature = temperature
        self.client = OpenAI()

    def predict(self, texts: List[str]) -> List[str]:
        """Classify a list of texts as positive or negative."""
        predictions = []
        for text in texts:
            response = self.client.chat.completions.create(
                model=self.model,
                temperature=self.temperature,
                messages=[
                    {
                        "role": "system",
                        "content": "Classify the sentiment as positive or negative. "
                                   "Respond with one word only.",
                    },
                    {"role": "user", "content": text},
                ],
            )
            label = response.choices[0].message.content.strip().lower()
            predictions.append(label)
        return predictions
```

### Step 6: Build the Pipeline Module

Pipelines orchestrate the data, model, and evaluation steps:

```python
# src/my_project/pipelines/train_pipeline.py
"""End-to-end training and evaluation pipeline."""

from my_project.config import load_config
from my_project.data.loader import load_reviews, split_train_test
from my_project.models.classifier import LLMClassifier


def evaluate_accuracy(predictions: list, labels: list) -> float:
    """Calculate the fraction of correct predictions."""
    correct = sum(1 for p, l in zip(predictions, labels) if p == l)
    return correct / len(predictions) if predictions else 0.0


def run_pipeline(config_path: str = "configs/default.yaml") -> dict:
    """Execute the full training and evaluation pipeline."""
    config = load_config(config_path)

    df = load_reviews(config["data"]["raw_path"])
    train_df, test_df = split_train_test(df, config["data"]["test_split"])

    classifier = LLMClassifier(
        model=config["model"]["name"],
        temperature=config["model"]["temperature"],
    )

    predictions = classifier.predict(test_df["text"].tolist())
    labels = test_df["label"].tolist()

    accuracy = evaluate_accuracy(predictions, labels)

    return {
        "accuracy": accuracy,
        "test_samples": len(test_df),
        "model": config["model"]["name"],
    }


if __name__ == "__main__":
    results = run_pipeline()
    print(f"Accuracy: {results['accuracy']:.2%}")
```

### Step 7: Write the Entry Point Script

Keep executable entry points in `scripts/`:

```python
#!/usr/bin/env python3
# scripts/run_pipeline.py
"""CLI entry point for running the classification pipeline."""

import sys
import argparse

from my_project.pipelines.train_pipeline import run_pipeline


def main():
    parser = argparse.ArgumentParser(description="Run classification pipeline")
    parser.add_argument(
        "--config",
        default="configs/default.yaml",
        help="Path to configuration file",
    )
    args = parser.parse_args()

    results = run_pipeline(args.config)
    print(f"Pipeline complete — accuracy: {results['accuracy']:.2%}")
    return 0


if __name__ == "__main__":
    sys.exit(main())
```

## Performance Considerations

A project structure affects development velocity more than runtime performance. The key performance concern is import time: grouping code into packages with clean `__init__.py` files means Python resolves imports quickly, and lazy imports inside a package (e.g., importing a heavy library only in the module that needs it) keep the initial import fast.
