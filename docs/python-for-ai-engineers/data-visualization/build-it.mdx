# Data Visualization — Hands On

⏱ 16 minutes

## Breaking It Down

### Step 1: Basic Line Plot

```python
# line_plot.py
import matplotlib.pyplot as plt
import numpy as np

# Create data
x = np.linspace(0, 10, 100)
y = np.sin(x)

# Object-oriented interface
fig, ax = plt.subplots(figsize=(8, 4))
ax.plot(x, y, label="sin(x)", color="blue", linewidth=2)
ax.set_xlabel("x")
ax.set_ylabel("sin(x)")
ax.set_title("Sine Wave")
ax.legend()
ax.grid(True, alpha=0.3)

plt.show()
# To save instead of showing: fig.savefig("sine.png", dpi=150)
```

`fig` is the overall figure container. `ax` is the axes where data is drawn. All customization goes through the `ax` object.

### Step 2: Multiple Lines and Styles

```python
# multi_line.py
import matplotlib.pyplot as plt
import numpy as np

x = np.linspace(0, 10, 100)

fig, ax = plt.subplots(figsize=(8, 4))
ax.plot(x, np.sin(x), label="sin", linestyle="-", color="blue")
ax.plot(x, np.cos(x), label="cos", linestyle="--", color="red")
ax.plot(x, np.sin(x) * np.cos(x), label="sin*cos", linestyle=":", color="green")
ax.set_xlabel("x")
ax.set_ylabel("value")
ax.set_title("Trigonometric Functions")
ax.legend()
fig.savefig("trig.png", dpi=150, bbox_inches="tight")
```

Line styles: `'-'` solid, `'--'` dashed, `':'` dotted, `'-.'` dash-dot. Color can be a name (`"blue"`), hex (`"#1f77b4"`), or single character (`"b"`).

### Step 3: Bar Charts for Model Comparison

```python
# bar_chart.py
import matplotlib.pyplot as plt
import numpy as np

models = ["gpt-4", "claude-3", "llama-3", "gemini"]
accuracy = [0.94, 0.93, 0.89, 0.91]
latency = [1200, 1800, 900, 1500]

fig, (ax1, ax2) = plt.subplots(1, 2, figsize=(10, 4))

# Accuracy bar chart
colors = ["steelblue", "steelblue", "coral", "steelblue"]
bars = ax1.bar(models, accuracy, color=colors, edgecolor="black")
ax1.set_ylabel("Accuracy")
ax1.set_title("Model Accuracy")
ax1.set_ylim(0, 1.0)

# Add value labels on bars
for bar in bars:
    height = bar.get_height()
    ax1.text(bar.get_x() + bar.get_width()/2., height + 0.01,
             f"{height:.2f}", ha="center", va="bottom")

# Latency bar chart
bars2 = ax2.bar(models, latency, color="lightcoral", edgecolor="black")
ax2.set_ylabel("Latency (ms)")
ax2.set_title("Inference Latency")

for bar in bars2:
    h = bar.get_height()
    ax2.text(bar.get_x() + bar.get_width()/2., h + 20,
             f"{h:.0f}", ha="center", va="bottom")

plt.tight_layout()
fig.savefig("model_comparison.png", dpi=150)
```

`subplots(1, 2)` creates one row with two side-by-side axes. `fig.tight_layout()` prevents overlapping labels.

### Step 4: Histograms for Distributions

```python
# histogram.py
import matplotlib.pyplot as plt
import numpy as np

# Generate sample data — two distributions
rng = np.random.default_rng(42)
gpt_scores = rng.normal(0.85, 0.08, 200)   # mean 0.85, std 0.08
claude_scores = rng.normal(0.88, 0.12, 200)  # mean 0.88, std 0.12

fig, ax = plt.subplots(figsize=(8, 4))

ax.hist(gpt_scores, bins=20, alpha=0.6, label="gpt-4", color="blue", edgecolor="black")
ax.hist(claude_scores, bins=20, alpha=0.6, label="claude-3", color="red", edgecolor="black")
ax.set_xlabel("Accuracy")
ax.set_ylabel("Frequency")
ax.set_title("Accuracy Distribution by Model")
ax.legend()

fig.savefig("histogram.png", dpi=150)
```

`alpha=0.6` makes overlapping bars semi-transparent so both distributions are visible.

### Step 5: Scatter Plots for Relationships

```python
# scatter.py
import matplotlib.pyplot as plt
import numpy as np
import pandas as pd

df = pd.DataFrame({
    "model": np.repeat(["gpt-4", "claude-3", "llama-3"], 30),
    "latency_ms": np.concatenate([
        np.random.normal(1200, 100, 30),
        np.random.normal(1800, 150, 30),
        np.random.normal(900, 80, 30),
    ]),
    "accuracy": np.concatenate([
        np.random.normal(0.94, 0.02, 30),
        np.random.normal(0.93, 0.03, 30),
        np.random.normal(0.89, 0.02, 30),
    ]),
})

fig, ax = plt.subplots(figsize=(8, 5))

colors = {"gpt-4": "blue", "claude-3": "red", "llama-3": "green"}
for model, group in df.groupby("model"):
    ax.scatter(group["latency_ms"], group["accuracy"],
               label=model, color=colors[model], alpha=0.7, edgecolors="black")

ax.set_xlabel("Latency (ms)")
ax.set_ylabel("Accuracy")
ax.set_title("Accuracy vs Latency by Model")
ax.legend()
ax.grid(True, alpha=0.3)

fig.savefig("scatter.png", dpi=150)
```

Scatter plots reveal tradeoffs — in this case, whether higher latency correlates with higher accuracy.

### Step 6: Plotting Pandas DataFrames Directly

```python
# pandas_plot.py
import pandas as pd
import numpy as np
import matplotlib.pyplot as plt

# Create a time series
dates = pd.date_range("2026-01-01", periods=30, freq="D")
df = pd.DataFrame({
    "date": dates,
    "accuracy": 0.8 + np.cumsum(np.random.normal(0, 0.01, 30)),
    "loss": 0.5 - np.cumsum(np.random.normal(0, 0.02, 30)),
})
df = df.set_index("date")

fig, (ax1, ax2) = plt.subplots(2, 1, figsize=(8, 6), sharex=True)

df["accuracy"].plot(ax=ax1, title="Validation Accuracy", color="blue", linewidth=2)
ax1.set_ylabel("Accuracy")
ax1.grid(True, alpha=0.3)

df["loss"].plot(ax=ax2, title="Validation Loss", color="red", linewidth=2)
ax2.set_ylabel("Loss")
ax2.set_xlabel("Date")
ax2.grid(True, alpha=0.3)

plt.tight_layout()
fig.savefig("training_curve.png", dpi=150)
```

The `.plot()` method on a pandas Series is a thin wrapper around `ax.plot()`. It uses the index as the x-axis automatically.

## Build Something Real — Model Evaluation Dashboard

```python
# evaluation_dashboard.py
import matplotlib.pyplot as plt
import numpy as np
import pandas as pd

rng = np.random.default_rng(42)
n = 50

df = pd.DataFrame({
    "model": np.random.choice(["gpt-4", "claude-3", "llama-3", "gemini"], n),
    "accuracy": rng.uniform(0.75, 1.0, n),
    "f1_score": rng.uniform(0.70, 0.98, n),
    "latency_ms": rng.uniform(300, 2000, n),
    "dataset": np.random.choice(["benchmark_a", "benchmark_b"], n),
})

fig = plt.figure(figsize=(12, 8))
fig.suptitle("Model Evaluation Dashboard", fontsize=16, fontweight="bold")

# 1. Accuracy distribution per model
ax1 = fig.add_subplot(2, 2, 1)
for model, group in df.groupby("model"):
    ax1.hist(group["accuracy"], bins=10, alpha=0.5, label=model)
ax1.set_xlabel("Accuracy")
ax1.set_ylabel("Frequency")
ax1.set_title("Accuracy Distribution")
ax1.legend(fontsize=8)

# 2. F1 vs accuracy scatter
ax2 = fig.add_subplot(2, 2, 2)
for model, group in df.groupby("model"):
    ax2.scatter(group["accuracy"], group["f1_score"], label=model, alpha=0.6)
ax2.set_xlabel("Accuracy")
ax2.set_ylabel("F1 Score")
ax2.set_title("F1 vs Accuracy")
ax2.legend(fontsize=8)
ax2.grid(True, alpha=0.3)

# 3. Mean metrics by model
ax3 = fig.add_subplot(2, 2, 3)
summary = df.groupby("model")[["accuracy", "f1_score"]].mean()
x = range(len(summary))
width = 0.35
ax3.bar([i - width/2 for i in x], summary["accuracy"], width, label="Accuracy", color="steelblue")
ax3.bar([i + width/2 for i in x], summary["f1_score"], width, label="F1", color="coral")
ax3.set_xticks(x)
ax3.set_xticklabels(summary.index)
ax3.set_ylabel("Score")
ax3.set_title("Mean Metrics by Model")
ax3.legend()

# 4. Latency box plot
ax4 = fig.add_subplot(2, 2, 4)
df.boxplot(column="latency_ms", by="model", ax=ax4)
ax4.set_title("Latency Distribution")
ax4.set_xlabel("Model")
ax4.set_ylabel("Latency (ms)")

plt.tight_layout()
fig.savefig("dashboard.png", dpi=150)
print("Dashboard saved as dashboard.png")
```

This dashboard combines four plot types on one figure — a quick way to compare models across multiple dimensions at a glance.

## Performance Considerations

- **Save as SVG for documents:** Vector format scales perfectly. Use `fig.savefig("plot.svg")`.
- **Save as PNG for web:** Use `dpi=150` for quality, `dpi=72` for quick previews.
- **Close figures to free memory:** `plt.close(fig)` — matplotlib holds figure references that accumulate in loops.
- **Interactive vs file output:** `plt.show()` blocks execution. For scripts that generate many plots, save directly with `savefig()` and avoid `show()`.
