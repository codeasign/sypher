# Requests — Hands On

⏱ 16 minutes

## Breaking It Down

### Step 1: Your First GET Request

```python
# get_request.py
import requests

response = requests.get("https://httpbin.org/json")
print(response.status_code)          # 200
print(response.headers["content-type"])  # application/json

# Parse JSON response
data = response.json()
print(data["slideshow"]["title"])    # Sample Slide Show

# Text response (for non-JSON)
# print(response.text)
```

The `requests` library handles URL encoding, connection setup, and response parsing. The `.json()` method parses the response body as JSON and returns a Python dict.

### Step 2: GET with Query Parameters

```python
# query_params.py
import requests

# Pass query parameters as a dict
params = {
    "q": "AI engineering",
    "page": 1,
    "limit": 10,
}
response = requests.get("https://httpbin.org/get", params=params)

# Check the URL that was actually requested
print(response.url)
# https://httpbin.org/get?q=AI+engineering&page=1&limit=10

# The response echoes back what was sent
data = response.json()
print(data["args"]["q"])  # AI engineering
```

Passing `params` as a dict lets `requests` handle URL encoding correctly (spaces become `+`, special characters are percent-encoded).

### Step 3: POST with JSON Body

```python
# post_json.py
import requests

# API request to a simulated LLM endpoint
payload = {
    "model": "gpt-4",
    "messages": [
        {"role": "user", "content": "What is the capital of France?"}
    ],
    "temperature": 0.7,
    "max_tokens": 100,
}
headers = {
    "Authorization": "Bearer YOUR_API_KEY",
    "Content-Type": "application/json",
}

response = requests.post(
    "https://httpbin.org/post",
    json=payload,
    headers=headers,
)

print(response.status_code)  # 200

# httpbin echoes the request body
data = response.json()
print(data["json"]["model"])               # gpt-4
print(data["json"]["messages"][0]["content"])  # What is the capital of France?
```

Using `json=payload` (not `data=payload`) tells `requests` to:
1. Serialize the dict to JSON automatically
2. Set `Content-Type: application/json`

### Step 4: Handling Status Codes and Errors

```python
# error_handling.py
import requests

# Simulate various HTTP responses
urls = [
    "https://httpbin.org/status/200",
    "https://httpbin.org/status/404",
    "https://httpbin.org/status/500",
]

for url in urls:
    response = requests.get(url)
    print(f"{url}: {response.status_code}", end=" ")

    if response.ok:  # True for 200–399
        print("OK")
    elif response.status_code == 404:
        print("Not found")
    elif response.status_code >= 500:
        print("Server error")

# raise_for_status() — raise an exception for error codes
try:
    response = requests.get("https://httpbin.org/status/404")
    response.raise_for_status()  # Raises HTTPError for 4xx/5xx
except requests.exceptions.HTTPError as e:
    print(f"HTTP error: {e}")

# Connection error (no network or wrong domain)
try:
    requests.get("https://nonexistent-domain-xyz.com", timeout=5)
except requests.exceptions.ConnectionError as e:
    print(f"Connection error: {e}")

# Timeout
try:
    requests.get("https://httpbin.org/delay/10", timeout=3)
except requests.exceptions.Timeout as e:
    print(f"Timeout: {e}")
```

Always use `timeout=N` — without it, a request can hang indefinitely if the server does not respond.

### Step 5: Sessions for Connection Reuse

```python
# sessions.py
import requests

# Without a session — each request opens a new connection
for _ in range(5):
    requests.get("https://httpbin.org/get")

# With a session — connections are reused (faster)
with requests.Session() as session:
    # Set default headers for all requests through this session
    session.headers.update({
        "Authorization": "Bearer YOUR_API_KEY",
        "Content-Type": "application/json",
    })

    for model in ["gpt-4", "claude-3", "llama-3"]:
        payload = {
            "model": model,
            "messages": [{"role": "user", "content": "Hello"}],
        }
        response = session.post("https://httpbin.org/post", json=payload)
        print(f"{model}: {response.status_code}")
```

A `Session` object:
- Reuses the underlying TCP connection for multiple requests (faster)
- Keeps default headers across requests
- Stores cookies and authentication
- Is the right choice for any API client making multiple calls

### Step 6: Streaming Responses

For models that support streaming (server-sent events), you can process the response incrementally:

```python
# streaming.py
import requests

# Stream a response — useful for large payloads or SSE
with requests.get(
    "https://httpbin.org/stream/5",  # Streams 5 lines
    stream=True,
) as response:
    for line in response.iter_lines():
        if line:
            print(line.decode("utf-8"))
```

Streaming is essential for LLM chat completions where tokens arrive one at a time. You read each line as it arrives instead of waiting for the entire response.

## Build Something Real — AI Provider Client

```python
# llm_client.py
import requests

class LLMClient:
    """A reusable client for LLM API calls."""

    def __init__(self, api_key: str, base_url: str = "https://api.openai.com/v1"):
        self.session = requests.Session()
        self.session.headers.update({
            "Authorization": f"Bearer {api_key}",
            "Content-Type": "application/json",
        })
        self.base_url = base_url.rstrip("/")

    def complete(
        self,
        prompt: str,
        model: str = "gpt-4",
        temperature: float = 0.7,
        max_tokens: int = 256,
        timeout: int = 30,
    ) -> dict:
        """Send a completion request and return the parsed response."""
        payload = {
            "model": model,
            "messages": [{"role": "user", "content": prompt}],
            "temperature": temperature,
            "max_tokens": max_tokens,
        }

        try:
            response = self.session.post(
                f"{self.base_url}/chat/completions",
                json=payload,
                timeout=timeout,
            )
            response.raise_for_status()
            return response.json()
        except requests.exceptions.Timeout:
            return {"error": "Request timed out"}
        except requests.exceptions.HTTPError as e:
            return {"error": f"HTTP {e.response.status_code}: {e.response.text}"}
        except requests.exceptions.ConnectionError:
            return {"error": "Connection failed"}

    def close(self):
        self.session.close()

# Usage
client = LLMClient(api_key="sk-placeholder", base_url="https://httpbin.org/post")
result = client.complete("Explain Python comprehensions")
# httpbin echoes the payload under the "json" key
if "error" not in result:
    print(f"Model used: {result.get('json', {}).get('model')}")
else:
    print(f"Error: {result['error']}")

client.close()
```

This client wraps all the patterns from this lesson: session reuse, auth headers, error handling, timeout management, and clean resource cleanup.

## Security Considerations

- **Never hardcode API keys** in source code. Use environment variables (`os.environ["API_KEY"]`) or a secrets manager.
- **Validate HTTPS:** Always use `https://` URLs. Never send API keys over plain HTTP.
- **Timeout every request:** Without `timeout`, a request can hang forever, leaving connections open and resources unreleased.
- **Rate limiting:** Many APIs return `429 Too Many Requests`. Implement retry with exponential backoff.
- **Logging:** Never log full request/response bodies that may contain API keys or user data.
