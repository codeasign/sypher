## Breaking It Down

### Step 1: Creating DataFrames and Series

```python
# create_df.py
import pandas as pd
import numpy as np

# From a dict of lists
data = {
    "model": ["gpt-4", "claude-3", "llama-3", "gemini"],
    "accuracy": [0.94, 0.93, 0.89, 0.91],
    "latency_ms": [1200, 1800, 900, 1500],
}
df = pd.DataFrame(data)
print(df)
#      model  accuracy  latency_ms
# 0    gpt-4      0.94        1200
# 1  claude-3      0.93        1800
# 2   llama-3      0.89         900
# 3    gemini      0.91        1500

# From a list of dicts (one dict per row)
rows = [
    {"model": "gpt-4", "accuracy": 0.94},
    {"model": "claude-3", "accuracy": 0.93},
]
df2 = pd.DataFrame(rows)

# A Series is a single column
scores = pd.Series([0.94, 0.93, 0.89, 0.91], name="accuracy")
print(scores)
# 0    0.94
# 1    0.93
# 2    0.89
# 3    0.91
# Name: accuracy, dtype: float64

# Basic inspection
print(df.shape)       # (4, 3)
print(df.columns)     # Index(['model', 'accuracy', 'latency_ms'], dtype='object')
print(df.dtypes)      # model: object, accuracy: float64, latency_ms: int64
print(df.describe())  # Summary statistics for numeric columns
```

### Step 2: Selecting Data

```python
# selecting.py
import pandas as pd

df = pd.DataFrame({
    "model": ["gpt-4", "claude-3", "llama-3", "gemini", "gpt-4"],
    "accuracy": [0.94, 0.93, 0.89, 0.91, 0.95],
    "latency_ms": [1200, 1800, 900, 1500, 1100],
})

# Select columns by label
print(df["model"])          # Series
print(df[["model", "accuracy"]])  # DataFrame — note double brackets

# Select rows by position — iloc
print(df.iloc[0])           # First row as Series
print(df.iloc[1:3])         # Rows 1 and 2

# Select rows by label — loc
print(df.loc[0])            # Row with index label 0
print(df.loc[0:2])          # Rows 0 through 2 (inclusive — unlike iloc!)

# Select specific rows and columns with loc
print(df.loc[0:2, ["model", "accuracy"]])
#      model  accuracy
# 0    gpt-4      0.94
# 1  claude-3      0.93
# 2   llama-3      0.89
```

### Step 3: Filtering with Conditions

```python
# filtering.py
import pandas as pd

df = pd.DataFrame({
    "model": ["gpt-4", "claude-3", "llama-3", "gemini", "gpt-4"],
    "accuracy": [0.94, 0.93, 0.89, 0.91, 0.95],
    "latency_ms": [1200, 1800, 900, 1500, 1100],
})

# Boolean filter
high_acc = df[df["accuracy"] > 0.92]
print(high_acc)
#      model  accuracy  latency_ms
# 0    gpt-4      0.94        1200
# 1  claude-3      0.93        1800
# 4    gpt-4      0.95        1100

# Multiple conditions — use & (not and)
filtered = df[(df["accuracy"] > 0.9) & (df["latency_ms"] < 1500)]
print(filtered)
#    model  accuracy  latency_ms
# 0  gpt-4      0.94        1200
# 4  gpt-4      0.95        1100

# isin — filter by value set
gpt = df[df["model"].isin(["gpt-4", "gpt-3.5"])]
print(gpt)
#    model  accuracy  latency_ms
# 0  gpt-4      0.94        1200
# 4  gpt-4      0.95        1100

# Query method — string expression (cleaner for complex filters)
fast_accurate = df.query("accuracy > 0.9 and latency_ms < 1500")
print(fast_accurate)
```

### Step 4: Adding and Modifying Columns

```python
# columns.py
import pandas as pd

df = pd.DataFrame({
    "model": ["gpt-4", "claude-3", "llama-3"],
    "accuracy": [0.94, 0.93, 0.89],
    "latency_ms": [1200, 1800, 900],
})

# Create a new column from existing ones
df["error_rate"] = 1 - df["accuracy"]
print(df)
#      model  accuracy  latency_ms  error_rate
# 0    gpt-4      0.94        1200        0.06
# 1  claude-3      0.93        1800        0.07
# 2   llama-3      0.89         900        0.11

# Column with a condition — np.where
import numpy as np
df["fast"] = np.where(df["latency_ms"] < 1000, "yes", "no")

# Apply a function to a column
df["latency_s"] = df["latency_ms"] / 1000
print(df)
#      model  accuracy  latency_ms  error_rate fast  latency_s
# 0    gpt-4      0.94        1200        0.06   no       1.2
# 1  claude-3      0.93        1800        0.07   no       1.8
# 2   llama-3      0.89         900        0.11  yes       0.9
```

### Step 5: Handling Missing Data

```python
# missing.py
import pandas as pd
import numpy as np

df = pd.DataFrame({
    "model": ["gpt-4", "claude-3", "llama-3", "gemini"],
    "accuracy": [0.94, None, 0.89, 0.91],
    "latency_ms": [1200, 1800, None, 1500],
})

print(df)
#      model  accuracy  latency_ms
# 0    gpt-4      0.94      1200.0
# 1  claude-3       NaN      1800.0
# 2   llama-3      0.89         NaN
# 3    gemini      0.91      1500.0

# Check for missing values
print(df.isna())
print(df.isna().sum())
# accuracy: 1, latency_ms: 1

# Drop rows with any missing values
clean = df.dropna()
print(clean)
#    model  accuracy  latency_ms
# 0  gpt-4      0.94      1200.0
# 3  gemini      0.91      1500.0

# Fill missing values
filled = df.fillna({"accuracy": df["accuracy"].mean(), "latency_ms": 0})
print(filled)
#      model  accuracy  latency_ms
# 0    gpt-4  0.940000      1200.0
# 1  claude-3  0.913333  ← mean of [0.94, 0.89, 0.91] = 0.9133
# 2   llama-3  0.890000         0.0  ← filled with 0
# 3    gemini  0.910000      1500.0
```

The `mean()` call for accuracy: (0.94 + 0.89 + 0.91) / 3 = 0.9133...

### Step 6: Grouping and Aggregation

```python
# groupby.py
import pandas as pd

df = pd.DataFrame({
    "model": ["gpt-4", "gpt-4", "claude-3", "claude-3", "llama-3"],
    "dataset": ["v1", "v2", "v1", "v2", "v1"],
    "accuracy": [0.94, 0.95, 0.93, 0.92, 0.89],
    "latency_ms": [1200, 1100, 1800, 1700, 900],
})

# Group by model and compute mean
avg = df.groupby("model")["accuracy"].mean()
print(avg)
# model
# claude-3    0.925
# gpt-4       0.945
# llama-3     0.890
# Name: accuracy, dtype: float64

# Multiple aggregations
stats = df.groupby("model").agg({
    "accuracy": ["mean", "std"],
    "latency_ms": ["mean", "min", "max"],
})
print(stats)
#           accuracy          latency_ms
#               mean       std    mean  min   max
# model
# claude-3    0.925  0.007071  1750.0 1700  1800
# gpt-4       0.945  0.007071  1150.0 1100  1200
# llama-3     0.890       NaN   900.0  900   900

# Group by two columns
by_model_dataset = df.groupby(["model", "dataset"])["accuracy"].mean()
print(by_model_dataset)
# model     dataset
# claude-3  v1         0.93
#           v2         0.92
# gpt-4     v1         0.94
#           v2         0.95
# llama-3   v1         0.89
```

`claude-3` mean accuracy: (0.93 + 0.92) / 2 = 0.925. `gpt-4` mean: (0.94 + 0.95) / 2 = 0.945.

### Step 7: Merging DataFrames

```python
# merge.py
import pandas as pd

models = pd.DataFrame({
    "model_id": ["A1", "A2", "A3"],
    "model_name": ["gpt-4", "claude-3", "llama-3"],
    "provider": ["OpenAI", "Anthropic", "Meta"],
})

results = pd.DataFrame({
    "model_id": ["A1", "A1", "A2", "A3"],
    "accuracy": [0.94, 0.95, 0.93, 0.89],
})

# Merge — like SQL JOIN
merged = pd.merge(results, models, on="model_id", how="left")
print(merged)
#   model_id  accuracy model_name   provider
# 0       A1      0.94      gpt-4     OpenAI
# 1       A1      0.95      gpt-4     OpenAI
# 2       A2      0.93   claude-3  Anthropic
# 3       A3      0.89    llama-3       Meta

# Concatenate rows
more_results = pd.DataFrame({
    "model_id": ["A4"],
    "accuracy": [0.91],
})
combined = pd.concat([results, more_results], ignore_index=True)
print(combined)
#   model_id  accuracy
# 0       A1      0.94
# 1       A1      0.95
# 2       A2      0.93
# 3       A3      0.89
# 4       A4      0.91
```

## Build Something Real — Experiment Analysis Pipeline

```python
# analyze_experiments.py
import pandas as pd
import numpy as np

# Simulate experiment data
np.random.seed(42)
n = 100

df = pd.DataFrame({
    "model": np.random.choice(["gpt-4", "claude-3", "llama-3"], n),
    "dataset": np.random.choice(["v1", "v2", "v3"], n),
    "accuracy": np.random.uniform(0.7, 1.0, n),
    "latency_ms": np.random.randint(200, 2000, n),
})

# Introduce some missing values
df.loc[df.sample(5).index, "accuracy"] = None

# Clean: drop rows with missing accuracy
clean = df.dropna(subset=["accuracy"])

# Create derived columns
clean["error_rate"] = 1 - clean["accuracy"]
clean["fast"] = clean["latency_ms"] < 500

# Group by model
summary = clean.groupby("model").agg(
    avg_accuracy=("accuracy", "mean"),
    std_accuracy=("accuracy", "std"),
    avg_latency=("latency_ms", "mean"),
    run_count=("accuracy", "count"),
).reset_index()

summary["accuracy_rank"] = summary["avg_accuracy"].rank(ascending=False)
print(summary.sort_values("accuracy_rank"))
#       model  avg_accuracy  std_accuracy  avg_latency  run_count  accuracy_rank
# 0  claude-3      0.869886      0.089026  1065.967742          31            1.0
# 1    gpt-4      0.856939      0.089846  1085.678571          28            2.0
# 2   llama-3      0.828785      0.088077  1110.914286          35            3.0

# Find best run per model
best_per_model = clean.loc[clean.groupby("model")["accuracy"].idxmax()]
print(best_per_model[["model", "accuracy", "latency_ms"]])
```

This pipeline — load, clean, transform, group, rank — is the core pattern for every data analysis task you will do with pandas.

## Performance Considerations

- **Vectorized operations over loops:** Use `df["col"].mean()` instead of iterating rows. Vectorized methods are orders of magnitude faster.
- **`apply()` is slow:** `df.apply(func, axis=1)` applies a Python function row-by-row. For performance-critical code, convert to NumPy or use vectorized pandas methods.
- **`query()` is fast:** For large DataFrames, `df.query()` uses numexpr to evaluate expressions without creating intermediate DataFrames.
- **Chunks for large files:** Use `pd.read_csv("file.csv", chunksize=10000)` to process files larger than memory.
- **Inplace vs copy:** Methods like `dropna()` return a new DataFrame by default. The `inplace=True` parameter modifies the original — use it sparingly as it can be confusing.
