# Tuples — Hands On

⏱ 10 minutes

## Breaking It Down

### Step 1: Creating Tuples

```python
# tuple_creation.py

# Parentheses are optional — the comma creates the tuple
empty = ()
single = (42,)          # Note: trailing comma — without it, it is just an int
pair = (1, 2)           # Two elements
triple = 1, 2, 3        # Parentheses optional
nested = (1, (2, 3))    # Nested tuples
mixed = (42, "hello", True, 3.14)

# From an iterable
chars = tuple("abc")
print(chars)  # ('a', 'b', 'c')

# From a list
converted = tuple([1, 2, 3])
print(converted)  # (1, 2, 3)
```

Single-element tuples require a trailing comma. Without it, `(42)` is just the integer `42`, and `("hello")` is just the string `"hello"`.

### Step 2: Accessing Elements

```python
# tuple_access.py
record = ("gpt-4", 2026, 0.95)

# Indexing — same as lists
print(record[0])    # gpt-4
print(record[-1])   # 0.95 (last element)

# Slicing
print(record[0:2])  # ('gpt-4', 2026)

# Iteration
for item in record:
    print(item)

# Membership
print("gpt-4" in record)  # True
```

### Step 3: Unpacking

Unpacking is where tuples shine — you assign each element to a variable in one line:

```python
# unpacking.py

# Basic unpacking
model, year, score = ("gpt-4", 2026, 0.95)
print(model)   # gpt-4
print(year)    # 2026
print(score)   # 0.95

# Unpacking works with any iterable, including lists and the tuple returned by functions
coordinates = (10, 20)
x, y = coordinates
print(x, y)  # 10 20

# Swapping variables — uses tuple packing and unpacking
a, b = 5, 10
a, b = b, a
print(a, b)  # 10 5

# Ignoring values with _
_, middle, _ = (1, 2, 3)
print(middle)  # 2

# Extended unpacking with * (Python 3+)
first, *rest = (1, 2, 3, 4)
print(first)  # 1
print(rest)   # [2, 3, 4] — note: rest is a list
```

### Step 4: Tuple Methods

Tuples have only two methods — `count()` and `index()` — because there is nothing else you can do to an immutable sequence:

```python
# tuple_methods.py
data = (1, 2, 2, 3, 2, 4)

# Count occurrences
print(data.count(2))  # 3

# Find index of first occurrence
print(data.index(3))  # 3
```

That is it. No `append()`, `extend()`, `insert()`, `remove()`, `pop()`, `sort()`, or `reverse()` — those would require modification.

### Step 5: Tuples as Dictionary Keys

Because tuples are hashable (when all elements are hashable), they can be dictionary keys:

```python
# tuple_keys.py

# Tuple as a composite key
cache = {}
cache[("gpt-4", "en")] = "response in English"
cache[("claude-3", "fr")] = "réponse en français"

print(cache[("gpt-4", "en")])  # response in English

# List as key — fails
# cache[["gpt-4", "en"]] = "error"  # TypeError: unhashable type: 'list'
```

This pattern is useful for caching function results keyed by argument combinations.

### Step 6: Build Something Real — Model Evaluation Record

```python
# evaluation_record.py — Store and process evaluation results

from datetime import datetime

# Each record is a tuple: (model_name, accuracy, f1_score, timestamp)
def evaluate_model(model_name):
    """Simulate evaluating a model and returning results."""
    # In reality, you would run actual evaluation
    accuracy = 0.85 + (hash(model_name) % 15) / 100.0
    f1 = accuracy - 0.03
    return (model_name, accuracy, f1, datetime.now())

# Collect evaluation records
models = ["gpt-4", "claude-3", "llama-3", "gemini-pro"]
results = [evaluate_model(m) for m in models]

# Unpack and print each record
for name, acc, f1, ts in results:
    print(f"{name}: acc={acc:.3f}, f1={f1:.3f}")

# Find the best model by accuracy
best = max(results, key=lambda r: r[1])
print(f"\nBest model: {best[0]} with accuracy {best[1]:.3f}")

# Use tuple as key in a cache
prediction_cache = {}

def cached_predict(model, prompt):
    key = (model, prompt)
    if key not in prediction_cache:
        # Simulate prediction
        prediction_cache[key] = ("positive", 0.95)
    return prediction_cache[key]
```
