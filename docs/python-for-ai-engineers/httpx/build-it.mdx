# HTTPX — Hands On

⏱ 16 minutes

## Breaking It Down

### Step 1: Synchronous Client

The sync httpx client works almost identically to `requests.Session`:

```python
# sync_client.py
import httpx

# Basic GET
response = httpx.get("https://httpbin.org/json")
print(response.status_code)   # 200
print(response.json()["slideshow"]["title"])  # Sample Slide Show

# POST with JSON
payload = {
    "model": "gpt-4",
    "messages": [{"role": "user", "content": "Hello"}],
}
response = httpx.post(
    "https://httpbin.org/post",
    json=payload,
    timeout=10,
)
print(response.status_code)  # 200

# Using a Client for connection reuse
with httpx.Client() as client:
    client.headers.update({"Authorization": "Bearer test-key"})
    for model in ["gpt-4", "claude-3", "llama-3"]:
        r = client.post(
            "https://httpbin.org/post",
            json={"model": model, "prompt": "Hi"},
        )
        print(f"{model}: {r.status_code}")
```

### Step 2: Async Client

The async client is where httpx differentiates itself:

```python
# async_client.py
import httpx
import asyncio

async def main():
    async with httpx.AsyncClient() as client:
        response = await client.get("https://httpbin.org/json")
        print(response.status_code)  # 200
        data = response.json()
        print(data["slideshow"]["title"])

        # POST with JSON
        payload = {
            "model": "gpt-4",
            "messages": [{"role": "user", "content": "Hello"}],
        }
        response = await client.post(
            "https://httpbin.org/post",
            json=payload,
            timeout=10,
        )
        print(f"Posted: {response.status_code}")

asyncio.run(main())
```

### Step 3: Concurrent Requests

This is the key advantage of async — send many requests concurrently:

```python
# concurrent.py
import httpx
import asyncio

async def fetch_model(client, model_name):
    """Send a single model inference request."""
    payload = {
        "model": model_name,
        "messages": [{"role": "user", "content": "Hello world"}],
    }
    response = await client.post(
        "https://httpbin.org/post",
        json=payload,
        timeout=30,
    )
    return model_name, response.status_code

async def main():
    models = ["gpt-4", "claude-3", "llama-3", "gemini", "mistral"]

    async with httpx.AsyncClient() as client:
        # Create concurrent tasks
        tasks = [fetch_model(client, m) for m in models]

        # Run all tasks concurrently
        results = await asyncio.gather(*tasks)

        for model, status in results:
            print(f"{model}: {status}")

asyncio.run(main())
```

With synchronous code, these 5 requests take roughly `5 × latency` time. With async, they complete in roughly the time of the single slowest request.

### Step 4: Timeout Configuration

httpx provides more granular timeout control than requests:

```python
# timeouts.py
import httpx

# Single timeout value (connect + read + write + pool)
with httpx.Client(timeout=10.0) as client:
    response = client.get("https://httpbin.org/delay/1")
    print(f"OK: {response.status_code}")  # 200

# Per-operation timeouts
timeout = httpx.Timeout(
    connect=3.0,    # Time to establish connection
    read=10.0,      # Time to receive response body
    write=5.0,      # Time to send request body
    pool=30.0,      # Time to wait for a connection from pool
)

with httpx.Client(timeout=timeout) as client:
    response = client.get("https://httpbin.org/get")
    print(f"Granular timeout: {response.status_code}")

# Per-request override
response = client.get("https://httpbin.org/delay/5", timeout=30.0)
```

### Step 5: Streaming Responses

httpx supports streaming both response bodies and server-sent events:

```python
# streaming.py
import httpx

# Stream response content (for large downloads)
with httpx.Client() as client:
    with client.stream("GET", "https://httpbin.org/stream/5") as response:
        for line in response.iter_lines():
            if line:
                print(line.decode())

# Async streaming (for LLM token streaming)
import asyncio

async def stream_async():
    async with httpx.AsyncClient() as client:
        async with client.stream(
            "GET",
            "https://httpbin.org/stream/3",
        ) as response:
            async for line in response.aiter_lines():
                if line:
                    print(f"Chunk: {line}")

asyncio.run(stream_async())
```

`stream()` sends the request but does not download the entire response body at once. Each chunk or line is processed as it arrives — essential for LLM token streaming.

### Step 6: HTTP/2 Support

httpx can use HTTP/2, which multiplexes multiple requests over a single connection:

```python
# http2.py
import httpx

# Enable HTTP/2 support
with httpx.Client(http2=True) as client:
    response = client.get("https://httpbin.org/get")
    print(f"HTTP version: {response.http_version}")  # HTTP/2 or HTTP/1.1

# With async
import asyncio

async def http2_async():
    async with httpx.AsyncClient(http2=True) as client:
        response = await client.get("https://httpbin.org/get")
        print(f"Async HTTP version: {response.http_version}")

asyncio.run(http2_async())
```

HTTP/2 requires `pip install httpx[http2]` (installs the `h2` package). When the server supports HTTP/2, multiple concurrent requests share one connection, reducing latency further.

## Build Something Real — Concurrent Batch Inference

```python
# batch_inference.py
import httpx
import asyncio
import time

class AsyncBatchClient:
    """Send batch inference requests concurrently."""

    def __init__(self, base_url: str, api_key: str, max_concurrent: int = 10):
        self.base_url = base_url.rstrip("/")
        self.api_key = api_key
        self.max_concurrent = max_concurrent

    async def infer_one(self, client: httpx.AsyncClient, prompt: str, idx: int) -> dict:
        """Send a single inference request."""
        payload = {
            "model": "gpt-4",
            "messages": [{"role": "user", "content": prompt}],
            "max_tokens": 100,
        }
        try:
            response = await client.post(
                f"{self.base_url}/chat/completions",
                json=payload,
                timeout=30,
            )
            response.raise_for_status()
            data = response.json()
            return {"index": idx, "success": True, "data": data}
        except Exception as e:
            return {"index": idx, "success": False, "error": str(e)}

    async def run_batch(self, prompts: list[str]) -> list[dict]:
        """Run a batch of prompts concurrently."""
        sem = asyncio.Semaphore(self.max_concurrent)

        async def limited(prompt, idx):
            async with sem:
                async with httpx.AsyncClient(timeout=30) as client:
                    client.headers.update({
                        "Authorization": f"Bearer {self.api_key}",
                    })
                    return await self.infer_one(client, prompt, idx)

        tasks = [limited(p, i) for i, p in enumerate(prompts)]
        results = await asyncio.gather(*tasks)
        return results

async def main():
    prompts = [
        "What is Python?",
        "Explain list comprehensions.",
        "What is the difference between a list and a tuple?",
        "How does async/await work?",
        "What is NumPy used for?",
    ]

    client = AsyncBatchClient(
        base_url="https://httpbin.org/post",
        api_key="test-key",
        max_concurrent=5,
    )

    start = time.time()
    results = await client.run_batch(prompts)
    elapsed = time.time() - start

    success_count = sum(1 for r in results if r["success"])
    print(f"Completed {success_count}/{len(prompts)} in {elapsed:.2f}s")
    for r in results:
        status = "OK" if r["success"] else f"FAIL: {r.get('error')}"
        print(f"  [{r['index']}] {status}")

asyncio.run(main())
```

With `max_concurrent=5`, all 5 requests run simultaneously. The total time is roughly the slowest single request, not the sum of all 5.

## Performance Considerations

- **Connection pooling:** httpx reuses connections within a Client/AsyncClient instance. For concurrent requests, this means less TLS overhead per request.
- **HTTP/2:** Enables true multiplexing — multiple concurrent streams over one connection. Without it, concurrent requests may still be limited by connection-per-host limits.
- **Async is not always faster:** For a single request, sync is slightly faster due to less overhead. Async shines with concurrent I/O-bound workloads.
- **Resource limits:** Too many concurrent requests can overwhelm the server, your network, or your local connection pool. Always use a semaphore to limit concurrency.
