# Context Managers — Hands On

⏱ 12 minutes

## Breaking It Down

### Step 1: Class-Based Context Manager

Every context manager needs `__enter__` and `__exit__`:

```python
# class_cm.py

class Timer:
    """Context manager that times the enclosed block."""

    def __enter__(self):
        self.start = time.perf_counter()
        return self  # Return value goes to 'as' variable

    def __exit__(self, exc_type, exc_val, exc_tb):
        self.elapsed = time.perf_counter() - self.start
        print(f"Block took {self.elapsed:.4f}s")
        # Return False (or None) to propagate any exception
        # Return True to suppress the exception

import time

with Timer() as t:
    total = sum(range(10_000_000))

print(f"Elapsed stored: {t.elapsed:.4f}s")
# Block took 0.XXXXs
# Elapsed stored: 0.XXXXs
```

The `__exit__` method receives three arguments: exception type, exception value, and traceback. If the block runs normally, all three are `None`.

### Step 2: Generator-Based Context Manager with `@contextmanager`

The `contextlib.contextmanager` decorator lets you write a context manager as a generator with one `yield`:

```python
# gen_cm.py
from contextlib import contextmanager
import time

@contextmanager
def timer():
    """Time a block of code using generator syntax."""
    start = time.perf_counter()
    try:
        yield  # Block body runs here
    finally:
        elapsed = time.perf_counter() - start
        print(f"Block took {elapsed:.4f}s")

with timer():
    total = sum(range(10_000_000))
```

The `yield` is the boundary between setup and teardown. Code before `yield` runs on enter. Code after `yield` runs on exit — and the `finally` guarantees it runs even on exceptions.

### Step 3: Context Manager with Exception Handling

```python
# exception_cm.py
from contextlib import contextmanager

@contextmanager
def suppress(*exceptions):
    """Suppress specified exceptions within the block."""
    try:
        yield
    except exceptions:
        pass  # Silently ignore

with suppress(ValueError, ZeroDivisionError):
    result = 1 / 0  # Would raise ZeroDivisionError — suppressed
    print("This line never runs")

print("Execution continues after suppressed exception")

# This is how contextlib.suppress works internally
```

### Step 4: Nested Context Managers

```python
# nested_cm.py
from contextlib import contextmanager
import time

@contextmanager
def log_step(name):
    print(f"Starting: {name}")
    yield
    print(f"Finished: {name}")

@contextmanager
def timer(name):
    start = time.perf_counter()
    yield
    elapsed = time.perf_counter() - start
    print(f"  [{name} took {elapsed:.3f}s]")

# Nested with
with log_step("Training pipeline"):
    with timer("data_loading"):
        time.sleep(0.1)
    with timer("model_training"):
        time.sleep(0.2)
    with timer("evaluation"):
        time.sleep(0.05)

# Or use a single line with comma (Python 3.1+)
with log_step("Combined") as step, timer("combined"):
    time.sleep(0.3)
```

### Step 5: Build Something Real — Model Inference Session Manager

```python
# inference_session.py
from contextlib import contextmanager
import time

class InferenceSession:
    """Simulate a managed model inference session."""

    def __init__(self, model_name):
        self.model_name = model_name
        self.total_time = 0.0

    def predict(self, text):
        """Simulate a prediction call."""
        time.sleep(0.05)
        return f"Prediction for: {text}"

    def __enter__(self):
        print(f"Loading model '{self.model_name}'...")
        time.sleep(0.1)  # Simulate model loading
        print("Model ready.")
        return self

    def __exit__(self, exc_type, exc_val, exc_tb):
        print(f"Unloading model '{self.model_name}'...")
        if exc_type:
            print(f"Session ended with error: {exc_val}")
        else:
            print("Session ended normally.")
        # Return False to propagate exceptions (default)

# Usage
with InferenceSession("gpt-4") as session:
    result1 = session.predict("Hello!")
    result2 = session.predict("How are you?")
    print(result1)
    print(result2)
# Output:
# Loading model 'gpt-4'...
# Model ready.
# Prediction for: Hello!
# Prediction for: How are you?
# Unloading model 'gpt-4'...
# Session ended normally.
```
