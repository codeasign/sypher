# Dictionaries — Hands On

⏱ 14 minutes

## Breaking It Down

### Step 1: Creating Dictionaries

```python
# dict_creation.py

# Empty dict
empty = {}
also_empty = dict()

# String keys — most common
config = {
    "model": "gpt-4",
    "temperature": 0.7,
    "max_tokens": 2048,
}

# Integer keys — useful for sparse arrays
number_map = {0: "zero", 1: "one", 2: "two"}

# Mixed types — possible but avoid for readability
mixed = {"key": 42, 100: "hundred", (1, 2): "tuple key"}

# From paired sequences using zip
keys = ["name", "role", "years"]
values = ["Alice", "engineer", 5]
person = dict(zip(keys, values))
print(person)  # {'name': 'Alice', 'role': 'engineer', 'years': 5}
```

### Step 2: Accessing and Modifying

```python
# dict_access.py
config = {"model": "gpt-4", "temperature": 0.7, "max_tokens": 2048}

# Access by key
print(config["model"])  # gpt-4

# Using get() — returns None (or a default) if key is missing
print(config.get("model"))       # gpt-4
print(config.get("top_p"))       # None — key does not exist
print(config.get("top_p", 1.0))  # 1.0 — default value returned

# KeyError with bracket access
# print(config["top_p"])  # KeyError: 'top_p'

# Add or update
config["temperature"] = 0.5   # Update existing key
config["top_p"] = 0.9         # Add new key
print(config)
# {'model': 'gpt-4', 'temperature': 0.5, 'max_tokens': 2048, 'top_p': 0.9}

# Delete a key
del config["top_p"]
print(config)
# {'model': 'gpt-4', 'temperature': 0.5, 'max_tokens': 2048}

# Pop — remove and return the value
value = config.pop("temperature")
print(value)    # 0.5
print(config)   # {'model': 'gpt-4', 'max_tokens': 2048}
```

### Step 3: Dictionary Methods

```python
# dict_methods.py
record = {"name": "Alice", "role": "engineer", "years": 5}

# Get all keys — view object (iterable, not a list)
print(record.keys())     # dict_keys(['name', 'role', 'years'])

# Get all values
print(record.values())   # dict_values(['Alice', 'engineer', 5])

# Get all key-value pairs as tuples
print(record.items())    # dict_items([('name', 'Alice'), ('role', 'engineer'), ('years', 5)])

# Convert to list if needed
keys = list(record.keys())

# Check if a key exists
print("name" in record)      # True
print("salary" in record)    # False

# Merge another dictionary
updates = {"role": "senior engineer", "salary": 120000}
record.update(updates)
print(record)
# {'name': 'Alice', 'role': 'senior engineer', 'years': 5, 'salary': 120000}
```

### Step 4: Iterating Over Dictionaries

```python
# dict_iteration.py
config = {"model": "gpt-4", "temperature": 0.7, "max_tokens": 2048}

# Iterate over keys (default)
for key in config:
    print(key)  # model, temperature, max_tokens

# Iterate over key-value pairs (preferred)
for key, value in config.items():
    print(f"{key}: {value}")

# Iterate over values
for value in config.values():
    print(value)
```

### Step 5: defaultdict — Handling Missing Keys Gracefully

```python
# defaultdict_demo.py
from collections import defaultdict

# Without defaultdict — you must check each time
word_counts = {}
words = ["apple", "banana", "apple", "cherry", "banana", "apple"]
for word in words:
    if word not in word_counts:
        word_counts[word] = 0
    word_counts[word] += 1

# With defaultdict — automatically provides default for missing keys
word_counts = defaultdict(int)  # int() returns 0
for word in words:
    word_counts[word] += 1

print(dict(word_counts))  # {'apple': 3, 'banana': 2, 'cherry': 1}

# Grouping items with defaultdict(list)
from collections import defaultdict

models_by_provider = defaultdict(list)
models_by_provider["OpenAI"].append("gpt-4")
models_by_provider["Anthropic"].append("claude-3")
models_by_provider["OpenAI"].append("gpt-3.5-turbo")

print(dict(models_by_provider))
# {'OpenAI': ['gpt-4', 'gpt-3.5-turbo'], 'Anthropic': ['claude-3']}
```

### Step 6: Build Something Real — Model Configuration and API Response Handler

```python
# model_config.py — Manage model configurations

from collections import defaultdict

# Default configuration for any model
DEFAULT_CONFIG = {
    "temperature": 0.7,
    "max_tokens": 1024,
    "top_p": 1.0,
    "frequency_penalty": 0.0,
    "presence_penalty": 0.0,
    "stop": [],
}

def create_model_config(model_name, overrides=None):
    """Create a model configuration, merging defaults with overrides."""
    config = DEFAULT_CONFIG.copy()
    config["model"] = model_name
    if overrides:
        config.update(overrides)
    return config

# Create specific configurations
gpt4_config = create_model_config("gpt-4", {
    "temperature": 0.3,
    "max_tokens": 4096,
})
claude_config = create_model_config("claude-3-opus", {
    "temperature": 0.5,
    "max_tokens": 8192,
    "top_p": 0.9,
})

print(gpt4_config)
print(claude_config)

# Process API response (like what you would get from OpenAI)
```

## Performance Considerations

- **Lookup by key** is O(1) on average — hash table access is nearly instantaneous.
- **Insert and delete** are also O(1) on average.
- **Memory** is higher than lists — each entry stores a hash, a key pointer, and a value pointer. For millions of entries, this matters.
- **Key choice matters** — string keys are slightly slower to hash than integer keys. For hot loops, consider integer or tuple keys.
