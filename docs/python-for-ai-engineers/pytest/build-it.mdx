# Pytest — Hands On

⏱ 18 minutes

## Breaking It Down

### Step 1: Install and First Test

```bash
pip install pytest pytest-cov pytest-mock
```

Create a file `test_basics.py`:

```python
# test_basics.py
def test_addition():
    result = 2 + 2
    assert result == 4

def test_string_functions():
    assert "hello".upper() == "HELLO"
    assert "HELLO".lower() == "hello"

def test_list_operations():
    items = [1, 2, 3]
    items.append(4)
    assert len(items) == 4
    assert 4 in items
```

Run with:

```bash
pytest test_basics.py -v
```

Output:

```
test_basics.py::test_addition PASSED
test_basics.py::test_string_functions PASSED
test_basics.py::test_list_operations PASSED
```

### Step 2: Testing the LLM Client Wrapper

Test the `LLMClient` from the first lesson. First, refactor it to accept an optional client for injectability:

```python
# llm_client.py (refactored for testability)
import time
from dataclasses import dataclass
from typing import Optional


@dataclass
class LLMResponse:
    text: str
    model: str
    input_tokens: int
    output_tokens: int
    latency_ms: float


class LLMClient:
    def __init__(self, client=None):
        # Allow injecting a mock client for testing
        self._client = client

    def generate(self, messages, **kwargs) -> LLMResponse:
        start = time.perf_counter()
        response = self._call(messages, **kwargs)
        elapsed = time.perf_counter() - start
        return LLMResponse(
            text=self._extract_text(response),
            model=self._extract_model(response),
            input_tokens=self._extract_input_tokens(response),
            output_tokens=self._extract_output_tokens(response),
            latency_ms=round(elapsed * 1000),
        )

    def _call(self, messages, **kwargs):
        return self._client.messages.create(**kwargs)

    def _extract_text(self, response):
        return response.content[0].text

    def _extract_model(self, response):
        return response.model

    def _extract_input_tokens(self, response):
        return response.usage.input_tokens

    def _extract_output_tokens(self, response):
        return response.usage.output_tokens
```

Now write the test:

```python
# test_llm_client.py
from dataclasses import dataclass
from llm_client import LLMClient, LLMResponse


# Create a fake API response matching the SDK shape
@dataclass
class FakeContent:
    text: str

@dataclass
class FakeUsage:
    input_tokens: int
    output_tokens: int

@dataclass
class FakeResponse:
    content: list
    model: str
    usage: FakeUsage


class FakeAnthropicClient:
    """Pretends to be an Anthropic SDK client."""

    class messages:
        @staticmethod
        def create(**kwargs):
            return FakeResponse(
                content=[FakeContent(text="Hello, world!")],
                model="claude-sonnet-4-20250514",
                usage=FakeUsage(input_tokens=10, output_tokens=5),
            )


def test_generate_returns_response():
    client = LLMClient(client=FakeAnthropicClient())
    result = client.generate(
        messages=[{"role": "user", "content": "Hi"}],
        max_tokens=128,
    )

    assert isinstance(result, LLMResponse)
    assert result.text == "Hello, world!"
    assert result.model == "claude-sonnet-4-20250514"
    assert result.input_tokens == 10
    assert result.output_tokens == 5
    assert result.latency_ms >= 0


def test_generate_with_empty_messages():
    client = LLMClient(client=FakeAnthropicClient())
    result = client.generate(messages=[])
    assert result.text == "Hello, world!"
```

### Step 3: Fixtures for Test Data

Fixtures create reusable test data without repeating yourself:

```python
# test_with_fixtures.py
import pytest
from typing import List

from pydantic import BaseModel


class Email(BaseModel):
    id: str
    body: str
    sender: str


@pytest.fixture
def sample_emails() -> List[Email]:
    """Return sample emails for testing."""
    return [
        Email(id="1", body="I need help with my order #123", sender="alice@example.com"),
        Email(id="2", body="What are your hours?", sender="bob@example.com"),
        Email(id="3", body="Win a free iPhone now!!!", sender="spammer@spam.com"),
    ]


@pytest.fixture
def spam_thresholds() -> dict:
    """Return spam classification thresholds."""
    return {"min_confidence": 0.8, "keywords": ["free", "win", "click"]}


def test_email_count(sample_emails):
    assert len(sample_emails) == 3


def test_spam_keywords_configured(spam_thresholds):
    assert "free" in spam_thresholds["keywords"]
    assert spam_thresholds["min_confidence"] >= 0.5


def test_sample_contains_spam(sample_emails):
    spam_keywords = ["free", "win", "click"]
    spam_body = sample_emails[2].body.lower()
    assert any(k in spam_body for k in spam_keywords)
```

Fixtures can use other fixtures. Pytest resolves the dependency chain automatically.

### Step 4: Parametrized Tests

Parametrize runs the same test logic with different inputs:

```python
# test_parametrize.py
import pytest


def classify_sentiment(text: str) -> str:
    """Simple keyword-based classifier for testing."""
    positive = ["good", "great", "amazing", "love"]
    negative = ["bad", "terrible", "awful", "hate"]

    text_lower = text.lower()
    pos_count = sum(1 for w in positive if w in text_lower)
    neg_count = sum(1 for w in negative if w in text_lower)

    if pos_count > neg_count:
        return "POSITIVE"
    elif neg_count > pos_count:
        return "NEGATIVE"
    return "NEUTRAL"


@pytest.mark.parametrize("text,expected", [
    ("I love this product!", "POSITIVE"),
    ("This is terrible", "NEGATIVE"),
    ("The weather is nice", "NEUTRAL"),
    ("Great service and amazing quality", "POSITIVE"),
    ("Bad experience, awful customer support", "NEGATIVE"),
    ("The item is okay", "NEUTRAL"),
])
def test_classify_sentiment(text: str, expected: str):
    assert classify_sentiment(text) == expected
```

Run with `pytest test_parametrize.py -v` and each parametrized case appears as a separate test.

### Step 5: Mocking LLM API Calls

Use `pytest-mock` or `unittest.mock` to replace real HTTP calls with controlled responses:

```python
# test_mock_llm.py
import pytest
from unittest.mock import MagicMock, patch


# Function that calls the LLM
def get_summary(client, text: str) -> str:
    response = client.messages.create(
        model="claude-sonnet-4-20250514",
        max_tokens=256,
        messages=[{"role": "user", "content": f"Summarize: {text}"}],
    )
    return response.content[0].text


def test_get_summary_returns_expected_text():
    # Create a mock client
    mock_client = MagicMock()
    mock_response = MagicMock()
    mock_content = MagicMock()
    mock_content.text = "This is a summary."
    mock_response.content = [mock_content]
    mock_client.messages.create.return_value = mock_response

    result = get_summary(mock_client, "Long article text here...")

    assert result == "This is a summary."
    mock_client.messages.create.assert_called_once()


@patch("llm_client.Anthropic")
def test_llm_client_uses_api_key(mock_anthropic):
    """Verify the client is constructed with the API key."""
    from llm_client import LLMClient

    client = LLMClient()
    mock_anthropic.assert_called_once()
```

### Step 6: Measure Coverage

```bash
pytest --cov=. --cov-report=term-missing
```

This shows which lines are covered and which are not. Aim for 80%+ coverage on business logic. Do not chase 100% on boilerplate.

## Build Something Real — Testing the RAG Pipeline

Write comprehensive tests for the RAG pipeline from the previous lesson:

```python
# test_rag_pipeline.py
import pytest
from unittest.mock import MagicMock, patch


# Simplified RAG pipeline for testing
class RAGPipeline:
    def __init__(self, embed_client=None, llm_client=None):
        self.embed_client = embed_client
        self.llm_client = llm_client

    def generate_answer(self, question: str, context_chunks: list[str]) -> str:
        if not context_chunks:
            return "No relevant documents found."

        context = "\n".join(context_chunks)
        response = self.llm_client.messages.create(
            model="claude-sonnet-4-20250514",
            max_tokens=512,
            messages=[{
                "role": "user",
                "content": f"Context:\n{context}\n\nQuestion: {question}"
            }],
        )
        return response.content[0].text


class TestRAGPipeline:
    """Comprehensive test suite for the RAG pipeline."""

    @pytest.fixture
    def mock_llm(self):
        """Fixture that returns a configured mock LLM client."""
        client = MagicMock()
        response = MagicMock()
        content = MagicMock()
        content.text = "Python was created by Guido van Rossum in 1991."
        response.content = [content]
        client.messages.create.return_value = response
        return client

    def test_answer_with_context(self, mock_llm):
        pipeline = RAGPipeline(llm_client=mock_llm)
        chunks = ["Python was created by Guido van Rossum and first released in 1991."]
        answer = pipeline.generate_answer("Who created Python?", chunks)
        assert "Guido van Rossum" in answer

    def test_answer_without_context(self, mock_llm):
        pipeline = RAGPipeline(llm_client=mock_llm)
        answer = pipeline.generate_answer("Who created Python?", [])
        assert answer == "No relevant documents found."

    def test_answer_with_multiple_chunks(self, mock_llm):
        pipeline = RAGPipeline(llm_client=mock_llm)
        chunks = ["Chunk 1: Python basics.", "Chunk 2: Advanced topics."]
        answer = pipeline.generate_answer("Tell me about Python", chunks)
        assert answer is not None

    @pytest.mark.parametrize("question,expected_in_answer", [
        ("Who created Python?", "Guido van Rossum"),
        ("When was Python created?", "1991"),
    ])
    def test_multiple_questions(self, mock_llm, question, expected_in_answer):
        pipeline = RAGPipeline(llm_client=mock_llm)
        chunks = ["Python was created by Guido van Rossum in 1991."]
        answer = pipeline.generate_answer(question, chunks)
        assert expected_in_answer in answer
```

## Performance Considerations

- **Test speed.** Tests should run in milliseconds. If a test makes an HTTP call, mock it. A slow test suite discourages running tests frequently.
- **Coverage thresholds.** Set a minimum coverage threshold in your CI (e.g., 80%). Use `pytest --cov-fail-under=80` to enforce it.
- **Parallel execution.** `pytest-xdist` runs tests in parallel across CPU cores: `pytest -n auto`.

## Security Considerations

- **Never mock security-critical code.** Do not mock authentication, authorisation, or encryption in tests that verify security properties. Test the real implementation.
- **Test data exposure.** Ensure test data does not contain real credentials or PII. Use factories (e.g., `factory_boy`) to generate synthetic test data.
