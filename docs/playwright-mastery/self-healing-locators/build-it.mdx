# Build Self-Healing Locators

## Step 1: Use Playwright's Built-in Fallback with `.or()`

Playwright provides the `.or()` method to combine two locators. The combined locator resolves to the first element that matches either strategy.

```typescript
import { test, expect } from '@playwright/test';

test('submit with OR fallback', async ({ page }) => {
  await page.goto('https://example.com/order');

  // Try role first, fall back to data-testid
  const submitButton = page.getByRole('button', { name: 'Submit' })
    .or(page.locator('[data-testid="submit-btn"]'));

  await submitButton.click();
  await expect(page.locator('.success-message')).toBeVisible();
});
```

## Step 2: Handle Multiple Active Selectors

When both selectors resolve simultaneously, `.or()` returns the first one found in the DOM.

```typescript
test('OR resolves first match', async ({ page }) => {
  await page.goto('https://example.com/form');

  // Both selectors may match — first found wins
  const cancelButton = page.getByRole('button', { name: 'Cancel' })
    .or(page.locator('.btn-cancel'));

  await cancelButton.click();
});
```

## Step 3: Chain Multiple Locator Strategies

You can chain three or more strategies to cover every UI variant your team might produce.

```typescript
test('chain multiple fallbacks', async ({ page }) => {
  await page.goto('https://example.com/checkout');

  // Try in order:  role → testid → text → CSS class
  const placeOrderBtn = page
    .getByRole('button', { name: /submit|place order/i })
    .or(page.getByTestId('submit-action'))
    .or(page.getByText('Place Order'))
    .or(page.locator('.btn-primary'));

  await placeOrderBtn.click();
  await expect(page.locator('h1')).toContainText('Order Confirmed');
});
```

## Step 4: Build a Custom Selector Engine

For project-wide fallback patterns, register a custom selector engine. This engine tries role, testid, text, and CSS in sequence automatically.

```typescript
import { selectors, test } from '@playwright/test';

// Register before any tests run
test.beforeAll(async () => {
  await selectors.register('heal', {
    query(root, selector) {
      // selector format: "role=button,name=Submit"
      const parts = Object.fromEntries(
        selector.split(',').map(p => p.split('=').map(s => s.trim()))
      );
      const { role, name, testid, text, css } = parts;

      // Try role first
      if (role) {
        const byRole = root.querySelector(`[role="${role}"]`);
        if (byRole && (!name || byRole.textContent?.includes(name))) {
          return byRole;
        }
      }

      // Try data-testid
      if (testid) {
        const byTestId = root.querySelector(`[data-testid="${testid}"]`);
        if (byTestId) return byTestId;
      }

      // Try text content
      if (text) {
        const allElements = root.querySelectorAll('button, a, input, [role="button"]');
        for (const el of allElements) {
          if (el.textContent?.includes(text)) return el;
        }
      }

      // Try CSS selector
      if (css) return root.querySelector(css);

      return null;
    },
    queryAll(root, selector) {
      const result = this.query(root, selector);
      return result ? [result] : [];
    },
  });
});

test('custom heal engine', async ({ page }) => {
  await page.goto('https://example.com/dashboard');
  // The engine runs the fallback chain automatically
  const btn = page.locator('heal=role=button,name=Submit');
  await btn.click();
});
```

## Step 5: Build Something Real

Write a test against a component that renders one of three button variants. Use `.or()` to handle all variants and a custom selector engine as the final safety net.

```typescript
import { test, expect, selectors } from '@playwright/test';

// Register a project-wide healing engine
test.beforeAll(async () => {
  await selectors.register('sypher-heal', {
    query(root, selector) {
      // Try role → data-testid → text → CSS
      const [role, testid, text, css] = selector.split('|').map(s => s.trim());

      if (role) {
        const [r, n] = role.split('::');
        const found = Array.from(root.querySelectorAll(`[role="${r}"]`))
          .find(el => !n || el.textContent?.includes(n));
        if (found) return found;
      }

      if (testid) {
        const found = root.querySelector(`[data-testid="${testid}"]`);
        if (found) return found;
      }

      if (text) {
        const els = root.querySelectorAll('button, a, [role="button"], input[type="submit"]');
        for (const el of els) {
          if (el.textContent?.includes(text)) return el;
        }
      }

      if (css) return root.querySelector(css);
      return null;
    },
    queryAll(root, selector) {
      const result = this.query(root, selector);
      return result ? [result] : [];
    },
  });
});

test.describe('Order button variants', () => {
  const variants = [
    { html: '<button data-testid="order-btn">Submit Order</button>', name: 'data-testid variant' },
    { html: '<button class="btn-primary">Place Order</button>', name: 'text variant' },
    { html: '<i class="icon-cart"></i>', name: 'icon variant' },
  ];

  for (const variant of variants) {
    test(`handles ${variant.name}`, async ({ page }) => {
      await page.setContent(`
        <div id="app">
          <form>
            ${variant.html}
          </form>
        </div>
      `);

      // OR chain handles all 3 variants
      const orderBtn = page
        .getByRole('button', { name: /submit|place/i })
        .or(page.getByTestId('order-btn'))
        .or(page.locator('.icon-cart'));

      await orderBtn.click();
      // Confirm click happened
      await expect(orderBtn).toBeVisible();
    });
  }

  test('custom engine as final fallback', async ({ page }) => {
    await page.setContent(`
      <div id="app">
        <button class="legacy-btn">Proceed</button>
      </div>
    `);

    // Use custom engine when standard strategies fail
    const btn = page.locator('sypher-heal=button::Proceed | proceed-btn | Proceed | .legacy-btn');
    await btn.click();
    await expect(btn).toBeVisible();
  });
});
```
