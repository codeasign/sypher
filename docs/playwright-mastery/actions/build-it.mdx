# Actions — Build It

In this lesson, you will perform every major Playwright action against real elements. You will learn when each action is appropriate and what options are available.

## Setup

Create a test file called `actions.spec.ts`.

```typescript
// actions.spec.ts
import { test, expect } from '@playwright/test';

test.describe('Playwright actions', () => {
  test.beforeEach(async ({ page }) => {
    await page.goto('https://the-internet.herokuapp.com/login');
  });
});
```

## Step 1: Click with Options

The most common action is `click`. It supports several options that control how the click is performed.

```typescript
test('click with force option', async ({ page }) => {
  // Normal click — Playwright checks actionability first
  await page.getByRole('button', { name: /login/i }).click();

  // Force click — skips actionability checks (use sparingly)
  await page.getByRole('button', { name: /login/i }).click({ force: true });

  // Click at a specific position (x, y offset from element center)
  await page.getByRole('button', { name: /login/i }).click({
    position: { x: 10, y: 10 },
  });

  // Click with modifier keys (Ctrl+Click to open in new tab)
  await page.getByRole('link', { name: /elemental selenium/i }).click({
    modifiers: ['Control'],
  });

  // Click with a delay between mousedown and mouseup (milliseconds)
  await page.getByRole('button', { name: /login/i }).click({ delay: 500 });

  // Do not wait for navigation after click
  await page.getByRole('button', { name: /login/i }).click({ noWaitAfter: true });
});
```

## Step 2: Fill vs Type

The `fill` method clears the input field and then types the value as a single operation. The `type` method types characters one by one, as if a real user were typing.

```typescript
test('fill vs type', async ({ page }) => {
  // fill — preferred: clears then sets value in one operation
  await page.getByLabel('Username').fill('tomsmith');
  await page.getByLabel('Password').fill('SuperSecretPassword!');

  // Verify the values
  await expect(page.getByLabel('Username')).toHaveValue('tomsmith');
  await expect(page.getByLabel('Password')).toHaveValue('SuperSecretPassword!');

  // type — slower, character by character
  // Only use when you need to test keyboard events (auto-suggest, character counters)
  await page.getByLabel('Username').clear();
  await page.getByLabel('Username').type('tomsmith', { delay: 50 });
});
```

**When to use `fill`**: Almost always. It is faster and more reliable.

**When to use `type`**: Only when you need to test character-by-character behavior such as auto-complete dropdowns, character limits, or real-time validation.

## Step 3: Select Options

Use `selectOption` on `<select>` elements. You can select by value, label text, or index.

```typescript
test('select options from dropdown', async ({ page }) => {
  // Navigate to a page with a dropdown
  await page.goto('https://the-internet.herokuapp.com/dropdown');

  const dropdown = page.getByLabel('Select an option');

  // Select by value attribute
  await dropdown.selectOption('1');
  await expect(dropdown).toHaveValue('1');

  // Select by visible label text
  await dropdown.selectOption({ label: 'Option 2' });
  await expect(dropdown).toHaveValue('2');

  // Select by index (0-based)
  await dropdown.selectOption({ index: 1 });
  await expect(dropdown).toHaveValue('1');

  // Select multiple options in a multi-select
  // await multiSelect.selectOption(['value1', 'value3']);
});
```

## Step 4: Check and Uncheck

Checkboxes and radio buttons have dedicated `check` and `uncheck` methods.

```typescript
test('check and uncheck', async ({ page }) => {
  await page.goto('https://the-internet.herokuapp.com/checkboxes');

  const checkboxes = page.getByRole('checkbox');

  // Check the first checkbox
  await checkboxes.nth(0).check();
  await expect(checkboxes.nth(0)).toBeChecked();

  // Uncheck the second checkbox
  await checkboxes.nth(1).uncheck();
  await expect(checkboxes.nth(1)).not.toBeChecked();

  // Force check if the checkbox is obscured (e.g., by a custom styled label)
  // await checkboxes.nth(0).check({ force: true });

  // Use setChecked to set a specific state
  await checkboxes.nth(0).setChecked(false);
  await expect(checkboxes.nth(0)).not.toBeChecked();
});
```

## Step 5: Drag and Drop

Use `dragTo` to drag an element onto another element.

```typescript
test('drag and drop', async ({ page }) => {
  await page.goto('https://the-internet.herokuapp.com/drag_and_drop');

  const columnA = page.getByText('A').first();
  const columnB = page.getByText('B').first();

  // Drag column A onto column B
  await columnA.dragTo(columnB);

  // After drag, the columns should have swapped headers
  // Wait — this demo has some quirks, so let's use a more reliable example
});

test('drag and drop with sortable demo', async ({ page }) => {
  // Use a sortable list for a more reliable drag-and-drop test
  await page.goto('https://the-internet.herokuapp.com/drag_and_drop');

  // The-internet drag and drop uses HTML5 drag and drop API
  // Playwright's dragTo works with the HTML5 API
  const source = page.locator('#column-a');
  const target = page.locator('#column-b');

  await source.dragTo(target);

  // Verify the swap
  await expect(page.locator('#column-a header')).toHaveText('B');
  await expect(page.locator('#column-b header')).toHaveText('A');
});
```

## Step 6: Keyboard and Mouse Actions

Playwright provides low-level keyboard and mouse controls for advanced interactions.

```typescript
test('keyboard actions', async ({ page }) => {
  await page.goto('https://the-internet.herokuapp.com/login');

  // Fill username and press Tab to move to password
  await page.getByLabel('Username').fill('tomsmith');
  await page.keyboard.press('Tab');

  // Now password field should be focused
  await expect(page.getByLabel('Password')).toBeFocused();

  // Type the password and press Enter to submit
  await page.keyboard.type('SuperSecretPassword!');
  await page.keyboard.press('Enter');

  // Assert login succeeded
  await expect(page).toHaveURL(/secure/);
});

test('mouse actions', async ({ page }) => {
  await page.goto('https://the-internet.herokuapp.com/login');

  // Scroll down
  await page.mouse.wheel(0, 500);

  // Move mouse to coordinates
  await page.mouse.move(100, 200);

  // Click at current mouse position
  await page.mouse.click(100, 200);
});
```

## Step 7: Build Something Real

Now combine everything into a complete end-to-end test. This test navigates to a form page, fills all fields, selects options, checks boxes, submits, and asserts the result.

```typescript
import { test, expect } from '@playwright/test';

test.describe('Complete action workflow', () => {
  test('fill and submit a complex form', async ({ page }) => {
    // Navigate to the form page
    await page.goto('https://the-internet.herokuapp.com/login');

    // 1. Fill text inputs using getByLabel
    await page.getByLabel('Username').fill('tomsmith');
    await page.getByLabel('Password').fill('SuperSecretPassword!');

    // 2. Verify filled values
    await expect(page.getByLabel('Username')).toHaveValue('tomsmith');
    await expect(page.getByLabel('Password')).toHaveValue('SuperSecretPassword!');

    // 3. Click the login button
    await page.getByRole('button', { name: /login/i }).click();

    // 4. Wait for navigation and assert URL
    await expect(page).toHaveURL(/secure/);

    // 5. Assert success message is visible
    const successBanner = page.getByText('You logged into a secure area');
    await expect(successBanner).toBeVisible();

    // 6. Click the logout button
    await page.getByRole('link', { name: /logout/i }).click();
    await expect(page).toHaveURL(/login/);
  });

  test('keyboard-driven form submission', async ({ page }) => {
    await page.goto('https://the-internet.herokuapp.com/login');

    // Navigate form entirely with keyboard
    await page.keyboard.press('Tab'); // focus moves to username
    await page.keyboard.type('tomsmith');
    await page.keyboard.press('Tab'); // focus moves to password
    await page.keyboard.type('SuperSecretPassword!');
    await page.keyboard.press('Enter'); // submits the form

    await expect(page).toHaveURL(/secure/);
  });

  test('test checkbox states', async ({ page }) => {
    await page.goto('https://the-internet.herokuapp.com/checkboxes');

    const checkbox1 = page.getByRole('checkbox').nth(0);
    const checkbox2 = page.getByRole('checkbox').nth(1);

    // Check checkbox 1 (initially unchecked)
    await checkbox1.check();
    await expect(checkbox1).toBeChecked();

    // Uncheck checkbox 2 (initially checked)
    await checkbox2.uncheck();
    await expect(checkbox2).not.toBeChecked();

    // Use setChecked to toggle
    await checkbox1.setChecked(false);
    await expect(checkbox1).not.toBeChecked();
  });

  test('dropdown selection', async ({ page }) => {
    await page.goto('https://the-internet.herokuapp.com/dropdown');

    const dropdown = page.getByLabel('Select an option');

    // Select by value
    await dropdown.selectOption('1');
    await expect(dropdown).toHaveValue('1');

    // Select by label
    await dropdown.selectOption({ label: 'Option 2' });
    await expect(dropdown).toHaveValue('2');
  });
});
```
