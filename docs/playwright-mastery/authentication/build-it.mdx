# Authentication — Build It

Follow these steps to implement authentication reuse in your Playwright test suite.

---

## Step 1: Create auth.setup.ts

This file logs in once via the API and saves the browser state. It runs as a setup project before any test projects.

```typescript
// auth.setup.ts
import { test as setup, expect } from '@playwright/test';

const authFile = '.auth/user.json';

setup('authenticate via API', async ({ page }) => {
  // Perform authentication via API request
  const response = await page.request.post('https://example.com/api/login', {
    data: {
      username: 'testuser@example.com',
      password: 'password123',
    },
  });

  expect(response.ok()).toBeTruthy();

  // Save the authenticated state (cookies, localStorage)
  await page.context().storageState({ path: authFile });
});
```

Create the `.auth/` directory and add it to your `.gitignore`:

```bash
mkdir -p .auth
echo ".auth/" >> .gitignore
```

---

## Step 2: Configure the Setup Project

In your `playwright.config.ts`, add a setup project that runs `auth.setup.ts` first, then make your test projects depend on it.

```typescript
// playwright.config.ts
import { defineConfig, devices } from '@playwright/test';

export default defineConfig({
  testDir: './tests',
  fullyParallel: true,
  workers: process.env.CI ? 4 : 2,
  projects: [
    {
      name: 'setup',
      testMatch: 'auth.setup.ts',
    },
    {
      name: 'chromium',
      dependencies: ['setup'],
      use: {
        ...devices['Desktop Chrome'],
        storageState: '.auth/user.json',
      },
    },
    {
      name: 'firefox',
      dependencies: ['setup'],
      use: {
        ...devices['Desktop Firefox'],
        storageState: '.auth/user.json',
      },
    },
  ],
});
```

The `dependencies: ['setup']` line ensures Playwright runs the auth setup before any chromium or firefox tests. The `storageState: '.auth/user.json'` line tells each test to load the saved state before the first page load.

---

## Step 3: Write Tests That Use Authentication

Once your config is set up, tests automatically have the authenticated state. No login code needed.

```typescript
// tests/dashboard.spec.ts
import { test, expect } from '@playwright/test';

test('dashboard shows user info', async ({ page }) => {
  // Navigate to dashboard — already authenticated
  await page.goto('/dashboard');

  // Verify user-specific content is visible
  await expect(page.locator('[data-testid="user-email"]')).toHaveText(
    'testuser@example.com'
  );
  await expect(page.locator('[data-testid="welcome-message"]')).toContainText(
    'Welcome back'
  );
});

test('user can view their orders', async ({ page }) => {
  await page.goto('/orders');
  await expect(page.locator('[data-testid="order-list"]')).toBeVisible();
});
```

## Step 4: Multi-Role Authentication

Real applications have multiple roles (admin, user, guest). Create separate setup files for each role.

```typescript
// auth.admin.setup.ts
import { test as setup, expect } from '@playwright/test';

const adminFile = '.auth/admin.json';

setup('authenticate as admin', async ({ page }) => {
  const response = await page.request.post('https://example.com/api/login', {
    data: {
      username: 'admin@example.com',
      password: 'admin-pass-123',
    },
  });
  expect(response.ok()).toBeTruthy();
  await page.context().storageState({ path: adminFile });
});
```

```typescript
// auth.user.setup.ts
import { test as setup, expect } from '@playwright/test';

const userFile = '.auth/user.json';

      password: 'user-pass-456',
    },
  });

```

```typescript

});
```

## Step 5: Build Something Real

Build a complete multi-role authentication system with session expiry handling.

```typescript
// auth.setup.ts
import { test as setup, expect } from '@playwright/test';

// Read credentials from environment variables — never hardcode
const ADMIN_USER = process.env.ADMIN_USER || 'admin@example.com';
const ADMIN_PASS = process.env.ADMIN_PASS || 'admin-default';
const REGULAR_USER = process.env.REGULAR_USER || 'user@example.com';
const REGULAR_PASS = process.env.REGULAR_PASS || 'user-default';

setup('authenticate as admin', async ({ page }) => {
  const response = await page.request.post('https://example.com/api/login', {
    data: {
      username: ADMIN_USER,
      password: ADMIN_PASS,
    },
  });
  expect(response.ok()).toBeTruthy();
  await page.context().storageState({ path: '.auth/admin.json' });
});

setup('authenticate as regular user', async ({ page }) => {
  const response = await page.request.post('https://example.com/api/login', {
    data: {
      username: REGULAR_USER,
      password: REGULAR_PASS,
    },
  });
  expect(response.ok()).toBeTruthy();
  await page.context().storageState({ path: '.auth/user.json' });
});
```

```typescript
// playwright.config.ts
import { defineConfig, devices } from '@playwright/test';

export default defineConfig({
  testDir: './tests',
  fullyParallel: true,
  retries: 2,
  workers: process.env.CI ? 4 : 2,
  projects: [
    {
      name: 'setup',
      testMatch: 'auth.setup.ts',
    },
    {
      name: 'chromium-admin',
      dependencies: ['setup'],
      use: {
        ...devices['Desktop Chrome'],
        storageState: '.auth/admin.json',
      },
    },
    {
      name: 'chromium-user',
      dependencies: ['setup'],
      use: {
        ...devices['Desktop Chrome'],
        storageState: '.auth/user.json',
      },
    },
  ],
});
```

```typescript
// tests/session-expiry.spec.ts
import { test, expect } from '@playwright/test';

test.use({ storageState: '.auth/user.json' });

test('handles session expiry gracefully', async ({ page }) => {
  await page.goto('/dashboard');
  await expect(page.locator('[data-testid="dashboard"]')).toBeVisible();

  // Simulate token expiry by clearing cookies
  await page.context().clearCookies();
  await page.reload();

  // Should redirect to login page
  await expect(page).toHaveURL(/\/login/);
});

test('re-login after session expiry', async ({ page }) => {
  await page.goto('/dashboard');

  // Clear cookies to simulate expired session
  await page.context().clearCookies();
  await page.reload();
  await expect(page).toHaveURL(/\/login/);

  // Perform login via UI (fallback when storageState expired)
  await page.fill('[name="email"]', 'user@example.com');
  await page.fill('[name="password"]', 'user-default');
  await page.click('[data-testid="login-button"]');

  // Verify redirected back to dashboard
  await expect(page).toHaveURL(/\/dashboard/);
  await expect(page.locator('[data-testid="dashboard"]')).toBeVisible();
});
```

This complete example demonstrates:
- Credentials read from environment variables (never hardcoded)
- Two roles with separate storage state files
- Session expiry handling with graceful degradation
- Re-login flow when the saved session expires
