# Page Object Model — Build It

Build a complete Page Object Model structure step by step. You will create page classes, wire them with Playwright fixtures, and write tests that use them.

## Step 1: Create a Login Page Object

Create `pages/LoginPage.ts`:

```typescript
import { type Page, type Locator } from '@playwright/test';

export class LoginPage {
  readonly page: Page;
  readonly usernameInput: Locator;
  readonly passwordInput: Locator;
  readonly submitButton: Locator;
  readonly errorMessage: Locator;

  constructor(page: Page) {
    this.page = page;
    this.usernameInput = page.locator('#username');
    this.passwordInput = page.locator('#password');
    this.submitButton = page.locator('button[type="submit"]');
    this.errorMessage = page.locator('.error-message');
  }

  async goto(): Promise<void> {
    await this.page.goto('/login');
  }

  async login(username: string, password: string): Promise<void> {
    await this.usernameInput.fill(username);
    await this.passwordInput.fill(password);
    await this.submitButton.click();
  }

  async getErrorMessage(): Promise<string> {
    return (await this.errorMessage.textContent()) ?? '';
  }
}
```

Locators are defined once in the constructor. Methods use business-language names like `login()` and `getErrorMessage()`. The `goto()` method centralizes the URL, so changing the route updates one place.

## Step 2: Create a Dashboard Page Object

Create `pages/DashboardPage.ts`:

```typescript
import { type Page, type Locator } from '@playwright/test';

export class DashboardPage {
  readonly page: Page;
  readonly greeting: Locator;
  readonly statsTable: Locator;
  readonly logoutButton: Locator;
  readonly statRows: Locator;

  constructor(page: Page) {
    this.page = page;
    this.greeting = page.locator('.greeting');
    this.statsTable = page.locator('.stats-table');
    this.logoutButton = page.locator('button:has-text("Logout")');
    this.statRows = page.locator('.stats-table tr');
  }

  async getGreetingText(): Promise<string> {
    return (await this.greeting.textContent()) ?? '';
  }

  async getStatRowCount(): Promise<number> {
    return this.statRows.count();
  }

  async getStatValue(statName: string): Promise<string> {
    const row = this.statRows.filter({ hasText: statName });
    const valueCell = row.locator('td:last-child');
    return (await valueCell.textContent()) ?? '';
  }

  async logout(): Promise<void> {
    await this.logoutButton.click();
  }
}
```

Each method returns a meaningful value or `void`. The `getStatValue()` method demonstrates locator chaining — finding a row by text and extracting a cell value.

## Step 3: Create Fixtures for Dependency Injection

Create `fixtures.ts`:

```typescript
import { test as base } from '@playwright/test';
import { LoginPage } from './pages/LoginPage';
import { DashboardPage } from './pages/DashboardPage';

// Extend the base test with page object fixtures
export const test = base.extend<{
  loginPage: LoginPage;
  dashboardPage: DashboardPage;
}>({
  loginPage: async ({ page }, use) => {
    const loginPage = new LoginPage(page);
    await use(loginPage);
  },

  dashboardPage: async ({ page }, use) => {
    const dashboardPage = new DashboardPage(page);
    await use(dashboardPage);
  },
});

export { expect } from '@playwright/test';
```

Fixtures handle creation and cleanup. The `use` function provides the page object to the test and tears it down after. This is dependency injection — your tests never call `new LoginPage()` directly.

## Step 4: Write Tests Using Fixtures

Create `tests/login-flow.spec.ts`:

```typescript
import { test, expect } from '../fixtures';

test.describe('Login flow', () => {
  test('successful login redirects to dashboard', async ({ loginPage, dashboardPage }) => {
    await loginPage.goto();
    await loginPage.login('admin', 'password123');

    // Verify we reached the dashboard
    const greeting = await dashboardPage.getGreetingText();
    expect(greeting).toContain('Welcome');
  });

  test('shows error for invalid credentials', async ({ loginPage }) => {
    await loginPage.goto();
    await loginPage.login('invalid', 'wrong');

    const error = await loginPage.getErrorMessage();
    expect(error).toContain('Invalid username or password');
  });

  test('shows error for empty fields', async ({ loginPage }) => {
    await loginPage.goto();
    await loginPage.login('', '');

    const error = await loginPage.getErrorMessage();
    expect(error).toContain('required');
  });
});
```

The tests are clean and declarative. They call business methods like `login()` and `getGreetingText()` instead of Playwright primitives. If the login form changes, you edit `LoginPage.ts` — not a single test.

## Step 5: Build Something Real

Now build a complete login-to-dashboard flow using POM with a task management page. This example includes error state testing for bad passwords and empty fields.

Create `pages/TodoPage.ts`:

```typescript
import { type Page, type Locator } from '@playwright/test';

export interface TodoItem {
  id: number;
  text: string;
  completed: boolean;
}

export class TodoPage {
  readonly page: Page;
  readonly newTodoInput: Locator;
  readonly addButton: Locator;
  readonly todoList: Locator;
  readonly todoItems: Locator;
  readonly clearCompletedButton: Locator;
  readonly itemCount: Locator;

  constructor(page: Page) {
    this.page = page;
    this.newTodoInput = page.locator('#new-todo');
    this.addButton = page.locator('button:has-text("Add")');
    this.todoList = page.locator('.todo-list');
    this.todoItems = page.locator('.todo-item');
    this.clearCompletedButton = page.locator('button:has-text("Clear completed")');
    this.itemCount = page.locator('.todo-count');
  }

  async goto(): Promise<void> {
    await this.page.goto('/todos');
  }

  async addTodo(text: string): Promise<void> {
    await this.newTodoInput.fill(text);
    await this.addButton.click();
  }

  async getTodoCount(): Promise<number> {
    return this.todoItems.count();
  }

  async getItemCountText(): Promise<string> {
    return (await this.itemCount.textContent()) ?? '';
  }

  async toggleTodo(index: number): Promise<void> {
    const checkbox = this.todoItems.nth(index).locator('input[type="checkbox"]');
    await checkbox.check();
  }

  async getTodoText(index: number): Promise<string> {
    return (await this.todoItems.nth(index).textContent()) ?? '';
  }

  async clearCompleted(): Promise<void> {
    await this.clearCompletedButton.click();
  }

  async getTodos(): Promise<TodoItem[]> {
    const items: TodoItem[] = [];
    const count = await this.todoItems.count();
    for (let i = 0; i < count; i++) {
      const item = this.todoItems.nth(i);
      const text = (await item.textContent()) ?? '';
      const isChecked = await item.locator('input[type="checkbox"]').isChecked();
      items.push({ id: i, text: text.trim(), completed: isChecked });
    }
    return items;
  }
}
```

Now add the fixture to `fixtures.ts`:

```typescript
import { test as base } from '@playwright/test';
import { LoginPage } from './pages/LoginPage';
import { DashboardPage } from './pages/DashboardPage';
import { TodoPage } from './pages/TodoPage';

export const test = base.extend<{
  loginPage: LoginPage;
  dashboardPage: DashboardPage;
  todoPage: TodoPage;
}>({
  loginPage: async ({ page }, use) => {
    const loginPage = new LoginPage(page);
    await use(loginPage);
  },

  dashboardPage: async ({ page }, use) => {
    const dashboardPage = new DashboardPage(page);
    await use(dashboardPage);
  },

  todoPage: async ({ page }, use) => {
    const todoPage = new TodoPage(page);
    await use(todoPage);
  },
});

export { expect } from '@playwright/test';
```

Write the complete test suite in `tests/todo-flow.spec.ts`:

```typescript
import { test, expect } from '../fixtures';

test.describe('Todo app flow', () => {
  test('add and verify a new todo item', async ({ todoPage }) => {
    await todoPage.goto();

    await todoPage.addTodo('Buy groceries');
    await todoPage.addTodo('Write tests');
    await todoPage.addTodo('Review PR');

    expect(await todoPage.getTodoCount()).toBe(3);
    expect(await todoPage.getItemCountText()).toContain('3 items left');
  });

  test('mark a todo as completed and clear it', async ({ todoPage }) => {
    await todoPage.goto();
    await todoPage.addTodo('Task to complete');

    await todoPage.toggleTodo(0);

    const todos = await todoPage.getTodos();
    expect(todos[0].completed).toBe(true);

    await todoPage.clearCompleted();
    expect(await todoPage.getTodoCount()).toBe(0);
  });

  test('getTodoText returns the text of a specific item', async ({ todoPage }) => {
    await todoPage.goto();
    await todoPage.addTodo('Unique task');

    const text = await todoPage.getTodoText(0);
    expect(text).toContain('Unique task');
  });
});
```

Every piece of code above is complete and ready to run. To execute the suite:

```bash
npx playwright test tests/todo-flow.spec.ts
```

If you point the tests at a real application, the page objects handle all the locator and interaction logic. Your tests stay focused on the business flow.
