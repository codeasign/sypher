# Build Visual Testing

## Step 1: Enable Screenshots in Config

Set the screenshot behavior in `playwright.config.ts` so screenshots are captured on failure for debugging.

```typescript
import { defineConfig } from '@playwright/test';

export default defineConfig({
  use: {
    // Capture screenshot on failure for CI debugging
    screenshot: 'only-on-failure',
    // Capture trace on failure for deeper analysis
    trace: 'retain-on-failure',
  },
  // Store snapshots alongside tests
  snapshotDir: './__snapshots__',
});
```

## Step 2: Capture Your First Screenshot

Playwright creates a baseline image on the first run. Subsequent runs compare against it.

```typescript
import { test, expect } from '@playwright/test';

test('home page matches baseline', async ({ page }) => {
  await page.goto('https://example.com');

  // First run: creates baseline at __snapshots__/home-page-chromium-darwin.png
  // Subsequent runs: compare against the baseline
  await expect(page).toHaveScreenshot('home-page.png');
});
```

The baseline is stored as:
```
__snapshots__/home-page-chromium-win32.png
__snapshots__/home-page-chromium-darwin.png
```

Each platform and browser gets its own baseline automatically.

## Step 3: Configure Thresholds

Not all pixel differences are bugs. Antialiasing, GPU rendering, and OS-level font rendering introduce tiny differences. Set thresholds to ignore acceptable noise.

```typescript
test('dashboard with threshold', async ({ page }) => {
  await page.goto('https://example.com/dashboard');

  await expect(page).toHaveScreenshot('dashboard.png', {
    // Ignore up to 100 different pixels
    maxDiffPixels: 100,
    // Per-channel color tolerance (0.2 = 20% difference allowed)
    threshold: 0.2,
  });
});
```

## Step 4: Mask Dynamic Content

Timestamps, loading spinners, ads, and random data cause flaky visual tests. Mask them with a colored overlay.

```typescript
test('mask dynamic content', async ({ page }) => {
  await page.goto('https://example.com/dashboard');

  await expect(page).toHaveScreenshot('dashboard-masked.png', {
    mask: [
      // Mask the current-time widget
      page.getByTestId('current-time'),
      // Mask any ad containers
      page.locator('.ad-banner'),
      // Mask user avatar that changes
      page.locator('[data-testid="user-avatar"]'),
    ],
    // Color of the mask overlay (default: pink/lime)
    maskColor: '#FF00FF',
  });
});
```

## Step 5: Style Override for Screenshots

Sometimes you need to hide or modify elements specifically for the screenshot without affecting the actual page behavior.

```typescript
test('style override for screenshot', async ({ page }) => {
  await page.goto('https://example.com/page');

  await expect(page).toHaveScreenshot('styled.png', {
    style: `
      /* Hide third-party embeds */
      [data-testid="ad"], iframe, .cookie-banner {
        visibility: hidden !important;
      }
      /* Fix animation to known state */
      .spinner {
        animation: none !important;
        opacity: 1 !important;
      }
    `,
  });
});
```

## Step 6: Build Something Real

Build a visual snapshot suite for a dashboard page. Include a masked timestamp widget, a threshold for cross-browser antialiasing, a full-page table capture, and an intentional regression detection test.

```typescript
import { test, expect } from '@playwright/test';

test.describe('Dashboard Visual Regression', () => {

  test('dashboard loads correctly', async ({ page }) => {
    await page.goto('https://example.com/dashboard');

    // Wait for data to finish loading
    await page.waitForSelector('[data-testid="chart"]');
    await page.waitForTimeout(500); // let animations settle

    await expect(page).toHaveScreenshot('dashboard-main.png', {
      // Mask dynamic elements
      mask: [
        page.getByTestId('current-time'),
        page.getByTestId('notification-badge'),
      ],
      // Antialiasing tolerance
      maxDiffPixels: 200,
      threshold: 0.3,
    });
  });

  test('data table renders fully', async ({ page }) => {
    await page.goto('https://example.com/dashboard');

    // Capture the entire scrollable table
    const table = page.locator('[data-testid="data-table"]');
    await expect(table).toHaveScreenshot('data-table-full.png', {
      fullPage: true,
      // Mask pagination (random page numbers)
      mask: [page.locator('.pagination')],
    });
  });

  test('detect layout regression', async ({ page }) => {
    await page.goto('https://example.com/dashboard');

    // Simulate a layout regression: hide a section
    await page.addStyleTag({
      content: '[data-testid="sidebar"] { display: none !important; }',
    });

    // This should fail because sidebar is missing
    const result = await expect(page).toHaveScreenshot('dashboard-main.png', {
      maxDiffPixels: 50,
    });

    // The test will produce a diff image showing the missing sidebar
    console.log('Expected regression detected — diff image generated');
  });

  test('responsive mobile view', async ({ page }) => {
    // Set mobile viewport
    await page.setViewportSize({ width: 375, height: 812 });
    await page.goto('https://example.com/dashboard');

    await expect(page).toHaveScreenshot('dashboard-mobile.png', {
      fullPage: true,
      mask: [page.getByTestId('current-time')],
    });
  });
});
```
