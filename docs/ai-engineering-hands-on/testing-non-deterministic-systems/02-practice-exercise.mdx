## The Build — Step by Step

You will build a statistical test suite that measures the distribution of LLM outputs across multiple runs and determines whether a system is stable enough for production.

### Step 1: Create a multi-sample generator

Create a file called `sampler.py`:

```python
"""
Generate multiple samples from an LLM for statistical analysis.
"""

import os
import json
from openai import OpenAI


def sample_responses(
    prompt: str,
    system_prompt: str = "You are a helpful assistant.",
    n: int = 10,
    temperature: float = 0.7,
    model: str = "gpt-4o-mini",
) -> list[str]:
    """
    Generate n responses to the same prompt.

    Args:
        prompt: The user prompt to send.
        system_prompt: System instructions.
        n: Number of responses to generate.
        temperature: Sampling temperature (0.0 = deterministic).
        model: The model to use.

    Returns:
        List of n response strings.
    """
    client = OpenAI(api_key=os.environ["OPENAI_API_KEY"])
    responses = []

    for i in range(n):
        response = client.chat.completions.create(
            model=model,
            messages=[
                {"role": "system", "content": system_prompt},
                {"role": "user", "content": prompt},
            ],
            temperature=temperature,
        )
        responses.append(
            response.choices[0].message.content
        )

    return responses


def sample_with_stats(
    prompt: str,
    n: int = 10,
    temperature: float = 0.7,
    model: str = "gpt-4o-mini",
) -> dict:
    """
    Generate samples and return basic statistics.
    """
    responses = sample_responses(
        prompt, n=n, temperature=temperature, model=model
    )

    # Basic stats
    lengths = [len(r) for r in responses]
    word_counts = [len(r.split()) for r in responses]

    stats = {
        "n": n,
        "temperature": temperature,
        "model": model,
        "response_lengths": {
            "mean": sum(lengths) / len(lengths),
            "min": min(lengths),
            "max": max(lengths),
            "std": (
                (sum((l - sum(lengths)/len(lengths)) ** 2
                     for l in lengths) / len(lengths)) ** 0.5
            ),
        },
        "word_counts": {
            "mean": sum(word_counts) / len(word_counts),
            "min": min(word_counts),
            "max": max(word_counts),
        },
        "unique_responses": len(set(responses)),
        "responses": responses,
    }

    return stats


if __name__ == "__main__":
    results = sample_with_stats(
        "What is the capital of France?",
        n=5,
        temperature=0.7,
    )
    print(f"Responses generated: {results['n']}")
    print(f"Unique responses: {results['unique_responses']}")
    print(f"Mean length: {results['response_lengths']['mean']:.0f} chars")
    print(f"Std length: {results['response_lengths']['std']:.0f} chars")
```

### Step 2: Create an LLM-as-judge scorer

Create a file called `scorer.py`:

```python
"""
Score LLM responses using an LLM judge.
"""

import os
import json
from openai import OpenAI


def score_response(
    prompt: str,
    response: str,
    dimension: str = "overall_quality",
    rubric: str = "Is this response correct, relevant, and well-written?",
    model: str = "gpt-4o-mini",
) -> float:
    """
    Score a single response on a 0-1 scale.

    Args:
        prompt: The original prompt.
        response: The response to score.
        dimension: The quality dimension being measured.
        rubric: Scoring criteria.
        model: Judge model to use.

    Returns:
        Float score between 0.0 and 1.0.
    """
    client = OpenAI(api_key=os.environ["OPENAI_API_KEY"])

    scoring_prompt = f"""You are an expert evaluator.

Dimension: {dimension}
Rubric: {rubric}

Original prompt: {prompt}

Response: {response}

Score this response from 0.0 to 1.0. Return JSON with "score" and "reason"."""

    result = client.chat.completions.create(
        model=model,
        messages=[{"role": "user", "content": scoring_prompt}],
        temperature=0.0,
        response_format={"type": "json_object"},
    )

    data = json.loads(
        result.choices[0].message.content
    )
    return float(data["score"])


def score_responses(
    prompt: str,
    responses: list[str],
    dimension: str = "overall_quality",
    rubric: str = "Is this response correct, relevant, and well-written?",
    model: str = "gpt-4o-mini",
) -> list[float]:
    """
    Score all responses for a prompt.
    """
    scores = []
    for i, response in enumerate(responses):
        score = score_response(
            prompt, response, dimension, rubric, model
        )
        scores.append(score)
        print(f"  Response {i+1}: {score:.3f}")
    return scores
```

### Step 3: Build the statistical test harness

Create a file called `stats_harness.py`:

```python
"""
Statistical analysis for non-deterministic LLM outputs.
"""

import numpy as np
from scipy import stats as scipy_stats


def analyze_scores(scores: list[float]) -> dict:
    """
    Compute statistical properties of a score distribution.
    """
    arr = np.array(scores)

    return {
        "n": len(scores),
        "mean": float(np.mean(arr)),
        "median": float(np.median(arr)),
        "std": float(np.std(arr, ddof=1)),
        "variance": float(np.var(arr, ddof=1)),
        "min": float(np.min(arr)),
        "max": float(np.max(arr)),
        "p5": float(np.percentile(arr, 5)),
        "p25": float(np.percentile(arr, 25)),
        "p75": float(np.percentile(arr, 75)),
        "p95": float(np.percentile(arr, 95)),
        "range": float(np.max(arr) - np.min(arr)),
    }


def is_stable(
    scores: list[float],
    min_mean: float = 0.8,
    max_std: float = 0.15,
) -> tuple[bool, str]:
    """
    Check if the score distribution is stable enough for production.

    Criteria:
    - Mean score >= min_mean
    - Standard deviation <= max_std
    - P5 percentile >= min_mean - 0.2
    """
    stats = analyze_scores(scores)
    issues = []

    if stats["mean"] < min_mean:
        issues.append(
            f"mean {stats['mean']:.3f} < {min_mean}"
        )
    if stats["std"] > max_std:
        issues.append(
            f"std {stats['std']:.3f} > {max_std}"
        )
    if stats["p5"] < min_mean - 0.2:
        issues.append(
            f"p5 {stats['p5']:.3f} < {min_mean - 0.2}"
        )

    if issues:
        return False, "Unstable: " + "; ".join(issues)
    return True, "Stable"


def compare_versions(
    baseline_scores: list[float],
    candidate_scores: list[float],
    alpha: float = 0.05,
) -> dict:
    """
    Compare two versions using statistical tests.

    Returns dict with test results.
    """
    baseline_stats = analyze_scores(baseline_scores)
    candidate_stats = analyze_scores(candidate_scores)

    # Welch's t-test (does not assume equal variance)
    t_stat, p_value = scipy_stats.ttest_ind(
        baseline_scores, candidate_scores, equal_var=False
    )

    # Levene test for variance equality
    levene_stat, levene_p = scipy_stats.levene(
        baseline_scores, candidate_scores
    )

    mean_diff = candidate_stats["mean"] - baseline_stats["mean"]

    result = {
        "baseline": baseline_stats,
        "candidate": candidate_stats,
        "mean_difference": mean_diff,
        "t_test": {
            "t_statistic": float(t_stat),
            "p_value": float(p_value),
            "significant": p_value < alpha,
            "interpretation": (
                "Significant difference"
                if p_value < alpha
                else "No significant difference"
            ),
        },
        "variance_test": {
            "levene_statistic": float(levene_stat),
            "p_value": float(levene_p),
            "variances_different": levene_p < alpha,
        },
        "effect_size": float(
            mean_diff / baseline_stats["std"]
            if baseline_stats["std"] > 0
            else 0
        ),
    }

    return result
```

### Step 4: Run the full non-deterministic test suite

Create a file called `run_nd_test.py`:

```python
"""
Full non-deterministic test suite.
Generates samples, scores them, analyzes the distribution.
"""

from sampler import sample_responses
from scorer import score_responses
from stats_harness import analyze_scores, is_stable

PROMPTS = [
    "What is the tallest mountain in the world?",
    "Explain why the sky is blue in one sentence.",
    "Write a haiku about machine learning.",
]

RUBRIC = (
    "Is the response factually correct, clearly written, "
    "and appropriate for a general audience?"
)

for prompt in PROMPTS:
    print(f"\n{'='*60}")
    print(f"Prompt: {prompt}")
    print(f"{'='*60}")

    # Generate samples at different temperatures
    for temp in [0.0, 0.7, 1.0]:
        print(f"\nTemperature: {temp}")

        responses = sample_responses(
            prompt, n=10, temperature=temp
        )
        scores = score_responses(
            prompt, responses, rubric=RUBRIC
        )

        stats = analyze_scores(scores)
        stable, reason = is_stable(scores)

        print(f"  Mean: {stats['mean']:.3f}")
        print(f"  Std:  {stats['std']:.3f}")
        print(f"  P5:   {stats['p5']:.3f}")
        print(f"  P95:  {stats['p95']:.3f}")
        print(f"  {reason}")
```

### Step 5: Compare two versions with statistical significance

Create a file called `compare_versions.py`:

```python
"""
Compare two model versions using statistical tests.
"""

from sampler import sample_responses
from scorer import score_responses
from stats_harness import compare_versions

prompt = "Explain quantum computing in simple terms."
rubric = "Is the explanation accurate, clear, and accessible?"

# Version A: gpt-4o-mini at temperature 0.7
print("Sampling Version A (gpt-4o-mini, temp=0.7)...")
responses_a = sample_responses(
    prompt, n=15, temperature=0.7, model="gpt-4o-mini"
)
scores_a = score_responses(
    prompt, responses_a, rubric=rubric
)

# Version B: gpt-4o-mini at temperature 0.3
print("\nSampling Version B (gpt-4o-mini, temp=0.3)...")
responses_b = sample_responses(
    prompt, n=15, temperature=0.3, model="gpt-4o-mini"
)
scores_b = score_responses(
    prompt, responses_b, rubric=rubric
)

# Compare
print("\n" + "=" * 60)
print("COMPARISON RESULTS")
print("=" * 60)

result = compare_versions(scores_a, scores_b)

print(f"Version A (temp 0.7): mean={result['baseline']['mean']:.3f}, "
      f"std={result['baseline']['std']:.3f}")
print(f"Version B (temp 0.3): mean={result['candidate']['mean']:.3f}, "
      f"std={result['candidate']['std']:.3f}")
print(f"Mean difference: {result['mean_difference']:+.3f}")
print(f"T-test p-value: {result['t_test']['p_value']:.4f}")
print(f"Significant: {result['t_test']['significant']}")
print(f"Effect size: {result['effect_size']:.3f}")
print(f"Variances different: "
```

### Step 6: Create a pytest-based non-deterministic test suite

Create a file called `test_nd_suite.py`:

```python
"""
Pytest-based non-deterministic test suite.
Tests pass based on statistical properties, not individual outputs.
"""

import pytest
from sampler import sample_responses
from scorer import score_responses
from stats_harness import analyze_scores, is_stable, compare_versions

# Mark the whole module
pytestmark = pytest.mark.nondeterministic


class TestResponseStability:
    """Tests for non-deterministic system stability."""

    @pytest.mark.parametrize("temperature", [0.0, 0.3, 0.7])
    def test_stable_at_temperature(
        self, temperature
    ):
        """System must be stable at each temperature."""
        prompt = "What is the capital of France?"
        rubric = "Must be factually correct."

        responses = sample_responses(
            prompt, n=10, temperature=temperature
        )
        scores = score_responses(
            prompt, responses, rubric=rubric
        )

        stable, reason = is_stable(
            scores, min_mean=0.8, max_std=0.15
        )

        assert stable, (
            f"Temperature {temperature}: {reason}"
        )

    def test_zero_temp_deterministic(self):
        """At temperature 0.0, outputs should be nearly identical."""
        prompt = "What is 2+2?"
        rubric = "Must be correct."

        responses = sample_responses(
            prompt, n=5, temperature=0.0
        )
        scores = score_responses(
            prompt, responses, rubric=rubric
        )

        stats = analyze_scores(scores)
        assert stats["std"] < 0.05, (
            f"Zero-temp std too high: {stats['std']:.3f}"
        )

    def test_high_temp_rejected(self):
        """Very high temperature should be rejected as unstable."""
        prompt = "Describe a sunset."
        rubric = "Should be descriptive and coherent."

        responses = sample_responses(
            prompt, n=10, temperature=1.5
        )
        scores = score_responses(
            prompt, responses, rubric=rubric
        )

        stable, _ = is_stable(
            scores, min_mean=0.6, max_std=0.2
        )
        # High temperature may or may not pass — log the result
        print(f"High temp stability: {stable}")
        print(f"Mean: {sum(scores)/len(scores):.3f}, "
              f"Std: {__import__('numpy').std(scores, ddof=1):.3f}")
```

## Common Errors at This Stage

### Error 1: Too few samples for statistical significance
```
T-test p-value: 0.45 — No significant difference found
```
**Cause:** With only 3-5 samples per version, the test lacks statistical power to detect real differences.
**Fix:** Increase sample size to at least 10-15 per version. The required sample size depends on the effect size you want to detect.

### Error 2: Temperature 0.0 still produces different outputs
```
Zero-temp std too high: 0.12
```
**Cause:** The model or API may not guarantee deterministic outputs even at temperature 0.0. Some APIs have a separate `seed` parameter.
**Fix:** Check if the API supports a `seed` parameter (OpenAI does) and pass a fixed seed. Also check that you are not using `top_p` in combination with temperature.

### Error 3: Judge model adds noise
```
The scores themselves have high variance because the judge is non-deterministic
```
**Cause:** The judge model (gpt-4o-mini) is also non-deterministic, adding noise to the measurement.
**Fix:** Set `temperature=0.0` on the judge call. If the API supports `seed`, set it. Average over 2-3 judge calls per sample.

### Error 4: All scores are the same at temperature 0.0
```
All 5 responses are identical — unique_responses = 1
```
**Cause:** This is expected behavior at temperature 0.0 for simple prompts. The model always picks the argmax token.
**Fix:** This is not an error. For deterministic prompts, you can write traditional assertion-based tests alongside the statistical tests.
