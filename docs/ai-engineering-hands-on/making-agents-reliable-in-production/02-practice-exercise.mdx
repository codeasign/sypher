You will build a reliable production agent with: bounded execution, error-handled tool calls, retry with backoff, state checkpointing, cost tracking, and structured logging.

### Step 1: Build the reliability core

Create a file called `reliability.py`:

```python
"""
Core reliability infrastructure for production agents.
"""

import time
import random
import json
import sqlite3
from datetime import datetime
from dataclasses import dataclass, asdict
from typing import Optional


@dataclass
class AgentState:
    """Persistent state for an agent session."""
    session_id: str
    step: int = 0
    total_tokens: int = 0
    total_cost: float = 0.0
    conversation_history: list = None
    tool_results: list = None
    status: str = "running"
    error: Optional[str] = None
    created_at: str = None
    updated_at: str = None

    def __post_init__(self):
        if self.conversation_history is None:
            self.conversation_history = []
        if self.tool_results is None:
            self.tool_results = []
        if self.created_at is None:
            self.created_at = datetime.now().isoformat()
        self.updated_at = datetime.now().isoformat()


class TokenBudget:
    """Tracks token usage and enforces budgets."""

    MODEL_COSTS = {
        "gpt-4o-mini": {"input": 0.15e-6, "output": 0.60e-6},
        "gpt-4o": {"input": 2.50e-6, "output": 10.00e-6},
    }

    def __init__(
        self,
        max_tokens: int = 4000,
        max_cost: float = 0.10,
        model: str = "gpt-4o-mini",
    ):
        self.max_tokens = max_tokens
        self.max_cost = max_cost
        self.model = model
        self.total_input_tokens = 0
        self.total_output_tokens = 0

    def add_usage(
        self, input_tokens: int, output_tokens: int
    ) -> dict:
        """Record token usage and return updated stats."""
        self.total_input_tokens += input_tokens
        self.total_output_tokens += output_tokens
        cost = self._calculate_cost(input_tokens, output_tokens)

        stats = {
            "total_input_tokens": self.total_input_tokens,
            "total_output_tokens": self.total_output_tokens,
            "total_tokens": (self.total_input_tokens
                             + self.total_output_tokens),
            "total_cost": self._calculate_cost(
                self.total_input_tokens, self.total_output_tokens
            ),
        }

        # Check budgets
        if stats["total_tokens"] > self.max_tokens:
            raise BudgetExceededError(
                f"Token budget exceeded: {stats['total_tokens']} > "
                f"{self.max_tokens}"
            )
        if stats["total_cost"] > self.max_cost:
            raise BudgetExceededError(
                f"Cost budget exceeded: ${stats['total_cost']:.4f} > "
                f"${self.max_cost:.4f}"
            )

        return stats

    def _calculate_cost(
        self, input_tokens: int, output_tokens: int
    ) -> float:
        costs = self.MODEL_COSTS.get(self.model, self.MODEL_COSTS["gpt-4o-mini"])
        return input_tokens * costs["input"] + output_tokens * costs["output"]

    @property
    def total_cost(self) -> float:
        return self._calculate_cost(
            self.total_input_tokens, self.total_output_tokens
        )


class BudgetExceededError(Exception):
    """Raised when an agent exceeds its budget."""
    pass


def retry_with_backoff(
    func,
    max_retries: int = 3,
    base_delay: float = 1.0,
    max_delay: float = 30.0,
) -> tuple:
    """
    Call a function with exponential backoff retry.

    Args:
        func: A callable to execute.
        max_retries: Maximum number of retry attempts.
        base_delay: Initial delay in seconds.
        max_delay: Maximum delay in seconds.

    Returns:
        (result, attempts, errors) tuple.
    """
    last_error = None
    for attempt in range(max_retries + 1):
        try:
            result = func()
            return result, attempt + 1, None
        except Exception as e:
            last_error = e
            if attempt < max_retries:
                delay = min(
                    max_delay,
                    base_delay * (2 ** attempt) + random.uniform(0, 0.5)
                )
                print(f"  Retry {attempt + 1}/{max_retries} "
                      f"after {delay:.1f}s: {e}")
                time.sleep(delay)

    return None, max_retries + 1, last_error


class StructuredLogger:
    """Logs agent steps in structured JSON format."""

    def __init__(self, log_path: str = "agent_log.jsonl"):
        self.log_path = log_path

    def log(
        self,
        session_id: str,
        step: int,
        action: str,
        details: dict,
    ):
        """Write a structured log entry."""
        entry = {
            "timestamp": datetime.now().isoformat(),
            "session_id": session_id,
            "step": step,
            "action": action,
        }
        entry.update(details)

        with open(self.log_path, "a") as f:
            f.write(json.dumps(entry) + "\n")

        print(f"[LOG] Step {step}: {action} — "
              f"{details.get('tool', details.get('status', ''))}")
```

### Step 2: Build the state persistence

Create a file called `state_store.py`:

```python
"""
State persistence for agent sessions.
"""

import json
import sqlite3
from datetime import datetime
from reliability import AgentState


class StateStore:
    """Persists agent state to SQLite."""

    def __init__(self, db_path: str = "agent_state.db"):
        self.conn = sqlite3.connect(db_path)
        self.conn.execute("""
            CREATE TABLE IF NOT EXISTS agent_sessions (
                session_id TEXT PRIMARY KEY,
                state_json TEXT,
                created_at TEXT,
                updated_at TEXT
            )
        """)
        self.conn.commit()

    def save(self, state: AgentState):
        """Save or update agent state."""
        state.updated_at = datetime.now().isoformat()
        self.conn.execute(
            """INSERT OR REPLACE INTO agent_sessions
               (session_id, state_json, created_at, updated_at)
               VALUES (?, ?, ?, ?)""",
            (
                state.session_id,
                json.dumps(asdict(state)),
                state.created_at,
                state.updated_at,
            ),
        )
        self.conn.commit()

    def load(self, session_id: str) -> AgentState:
        """Load agent state by session ID."""
        cursor = self.conn.execute(
            "SELECT state_json FROM agent_sessions WHERE session_id = ?",
            (session_id,),
        )
        row = cursor.fetchone()
        if row is None:
            raise KeyError(f"Session {session_id} not found")
        data = json.loads(row[0])
        return AgentState(**data)

    def list_sessions(
        self, limit: int = 10
    ) -> list[dict]:
        """List recent sessions."""
        cursor = self.conn.execute(
            """SELECT session_id, status, step, updated_at
               FROM agent_sessions
               ORDER BY updated_at DESC LIMIT ?""",
            (limit,),
        )
        return [
            {
                "session_id": row[0],
                "status": row[1],
                "steps": row[2],
                "updated": row[3],
            }
            for row in cursor.fetchall()
        ]
```

### Step 3: Build the reliable agent

Create a file called `reliable_agent.py`:

```python
"""
A production-reliable agent with all safeguards.
"""

import os
import uuid
from openai import OpenAI
from reliability import (
    AgentState, TokenBudget, BudgetExceededError,
    retry_with_backoff, StructuredLogger,
)
from state_store import StateStore


class ReliableAgent:
    """Agent with production reliability infrastructure."""

    def __init__(
        self,
        system_prompt: str = None,
        model: str = "gpt-4o-mini",
        max_steps: int = 10,
        max_tokens: int = 4000,
        max_cost: float = 0.10,
    ):
        self.client = OpenAI(
            api_key=os.environ["OPENAI_API_KEY"]
        )
        self.model = model
        self.system_prompt = system_prompt or (
            "You are a helpful assistant with access to tools. "
            "You search for information and provide answers."
        )
        self.max_steps = max_steps
        self.budget = TokenBudget(
            max_tokens=max_tokens,
            max_cost=max_cost,
            model=model,
        )
        self.logger = StructuredLogger()
        self.state_store = StateStore()
        self.session_id = str(uuid.uuid4())[:8]

    def run(self, user_input: str) -> dict:
        """
        Run the agent with full reliability infrastructure.
        """
        state = AgentState(
            session_id=self.session_id,
            conversation_history=[
                {"role": "system", "content": self.system_prompt},
                {"role": "user", "content": user_input},
            ],
        )

        # Save initial state
        self.state_store.save(state)

        try:
            for step in range(1, self.max_steps + 1):
                state.step = step
                print(f"\n--- Step {step}/{self.max_steps} ---")

                # Call LLM
                result, attempts, error = retry_with_backoff(
                    lambda: self._call_llm(
                        state.conversation_history
                    ),
                )

                if error:
                    state.status = "error"
                    state.error = str(error)
                    self.state_store.save(state)
                    self.logger.log(
                        self.session_id, step, "llm_error",
                        {"error": str(error), "retries": attempts},
                    )
                    return self._build_response(state, error_msg=str(error))

                response = result["response"]
                tokens = result["tokens"]

                # Update budget
                budget_stats = self.budget.add_usage(
                    tokens["input"], tokens["output"]
                )

                # Log step
                self.logger.log(
                    self.session_id, step, "llm_call",
                    {
                        "input_tokens": tokens["input"],
                        "output_tokens": tokens["output"],
                        "cumulative_cost": budget_stats["total_cost"],
                        "response_preview": response[:100],
                    },
                )

                state.total_tokens = budget_stats["total_tokens"]
                state.total_cost = budget_stats["total_cost"]
                state.conversation_history.append(
                    {"role": "assistant", "content": response}
                )

                # Check if agent wants to stop
                if self._is_final_response(response):
                    state.status = "completed"
                    self.state_store.save(state)
                    return self._build_response(state)

                # Save checkpoint
                self.state_store.save(state)

            # Agent ran out of steps
            state.status = "max_steps_exceeded"
            self.state_store.save(state)
            return self._build_response(
                state,
                warning=(
                    f"Agent reached maximum {self.max_steps} steps. "
                    f"Response may be incomplete."
                ),
            )

        except BudgetExceededError as e:
            state.status = "budget_exceeded"
            state.error = str(e)
            self.state_store.save(state)
            return self._build_response(
                state, error_msg=str(e)
            )

    def _call_llm(self, messages: list[dict]) -> dict:
        """Make the LLM call with token tracking."""
        response = self.client.chat.completions.create(
            model=self.model,
            messages=messages,
            temperature=0.5,
        )
        return {
            "response": response.choices[0].message.content,
            "tokens": {
                "input": response.usage.prompt_tokens,
                "output": response.usage.completion_tokens,
            },
        }

    def _is_final_response(self, response: str) -> bool:
        """Check if the agent indicates it is done."""
        done_indicators = [
            "final answer", "in summary", "to summarize",
            "i have completed", "the answer is",
        ]
        return any(
            indicator in response.lower()
            for indicator in done_indicators
        )

    def _build_response(
        self,
        state: AgentState,
        error_msg: str = None,
        warning: str = None,
    ) -> dict:
        """Build the final response to return to the caller."""
        last_msg = (
            state.conversation_history[-1]["content"]
            if state.conversation_history
            else ""
        )

        return {
            "session_id": self.session_id,
            "status": state.status,
            "response": error_msg or last_msg,
            "warning": warning,
            "stats": {
                "steps": state.step,
                "total_tokens": state.total_tokens,
                "total_cost": state.total_cost,
            },
        }
```

"""
Test the retry with backoff mechanism.
"""

from reliability import retry_with_backoff
import time


def test_successful_call():
    """Test a function that succeeds on first try."""
    def working_func():
        return "Success!"

    result, attempts, error = retry_with_backoff(working_func)
    print(f"Working function: {attempts} attempt(s), "
          f"result={result}")


def test_retry_then_succeed():
    """Test a function that fails twice then succeeds."""
    call_count = [0]

    def flaky_func():
        call_count[0] += 1
        if call_count[0] < 3:
            raise ConnectionError(f"Attempt {call_count[0]} failed")
        return "Success after retry!"

    start = time.time()
    result, attempts, error = retry_with_backoff(
        flaky_func, max_retries=3, base_delay=0.1
    )
    elapsed = time.time() - start

    print(f"Flaky function: {attempts} attempt(s), "
          f"result={result}, elapsed={elapsed:.2f}s")


def test_all_retries_fail():
    """Test a function that always fails."""
    def always_fails():
        raise ValueError("Always fails")

    start = time.time()
    result, attempts, error = retry_with_backoff(
        always_fails, max_retries=2, base_delay=0.1
    )
    elapsed = time.time() - start

    print(f"Failing function: {attempts} attempt(s), "
          f"error={error}, elapsed={elapsed:.2f}s")


if __name__ == "__main__":
    test_successful_call()
    test_retry_then_succeed()
    test_all_retries_fail()

    print("\nNote: Backoff delays grow exponentially:")
    print("  Retry 1: wait 0.1s + jitter")
    print("  Retry 2: wait 0.2s + jitter")
    print("  Retry 3: wait 0.4s + jitter")

### Step 6: Create the CI test suite

Create a file called `test_reliability_suite.py`:

```python
"""
CI test suite for production reliability infrastructure.
"""

import pytest
from reliability import TokenBudget, BudgetExceededError, retry_with_backoff
from state_store import StateStore, AgentState


class TestTokenBudget:
    """Tests for token budget enforcement."""

    def test_budget_within_limits(self):
        budget = TokenBudget(max_tokens=1000, max_cost=0.05)
        stats = budget.add_usage(100, 50)
        assert stats["total_tokens"] == 150
        assert stats["total_cost"] > 0

    def test_budget_exceeded_tokens(self):
        budget = TokenBudget(max_tokens=200, max_cost=1.0)
        with pytest.raises(BudgetExceededError):
            budget.add_usage(300, 0)

    def test_budget_exceeded_cost(self):
        budget = TokenBudget(max_tokens=100000, max_cost=0.001)
        with pytest.raises(BudgetExceededError):
            budget.add_usage(5000, 1000)


class TestRetryWithBackoff:
    """Tests for retry mechanism."""

    def test_success_first_try(self):
        result, attempts, error = retry_with_backoff(
            lambda: "ok", max_retries=3
        )
        assert result == "ok"
        assert attempts == 1
        assert error is None

    def test_retry_eventually_succeeds(self):
        counter = [0]

        def flaky():
            counter[0] += 1
            if counter[0] < 3:
                raise ValueError("Not yet")
            return "finally"

        result, attempts, error = retry_with_backoff(
            flaky, max_retries=3, base_delay=0.01
        )
        assert result == "finally"
        assert attempts == 3

    def test_all_retries_fail(self):
        def always_fails():
            raise ValueError("Always")

        result, attempts, error = retry_with_backoff(
            always_fails, max_retries=2, base_delay=0.01
        )
        assert result is None
        assert attempts == 3
        assert error is not None


class TestStateStore:
    """Tests for state persistence."""

    def test_save_and_load(self):
        store = StateStore(":memory:")
        state = AgentState(session_id="test_session")
        store.save(state)
        loaded = store.load("test_session")
        assert loaded.session_id == "test_session"
        assert loaded.status == "running"

    def test_list_sessions(self):
        store = StateStore(":memory:")
        store.save(AgentState(session_id="s1"))
        store.save(AgentState(session_id="s2"))
        sessions = store.list_sessions()
        assert len(sessions) == 2
```

## Common Errors at This Stage

### Error 1: Budget exceeded unexpectedly
```
BudgetExceededError: Cost budget exceeded: $0.0015 > $0.0010
```
**Cause:** The max_cost is set too low for the number of steps the agent needs.
**Fix:** Calculate expected cost: steps * (input_tokens + output_tokens) * cost_per_token. Set max_cost to 2-3x the expected cost to allow for retries.

### Error 2: State persistence failure
```
sqlite3.OperationalError: database is locked
```
**Cause:** Multiple agent instances trying to write to the same SQLite database concurrently.
**Fix:** Use a database that supports concurrent writes (PostgreSQL, DynamoDB) or use a connection pool for SQLite.

### Error 3: Retry loop on permanent error
```
Retrying... Retrying... Retrying... (never succeeds)
```
**Cause:** The retry mechanism is retrying an error that will never succeed (e.g., invalid API key, wrong model name).
**Fix:** Distinguish between retriable errors (429, 503, 500) and permanent errors (401, 403, 404). Only retry the former.

### Error 4: Agent claims completion prematurely
```
"Final answer: The capital is Paris" but agent has only done 1 of 5 expected tasks
```
**Cause:** The agent thinks it is done when it is not. The `_is_final_response` method matched on "the answer is" too early.
**Fix:** Make the completion detection more specific. Only consider an agent done if it explicitly ends with a final-summary pattern, not if it simply provides one answer.
