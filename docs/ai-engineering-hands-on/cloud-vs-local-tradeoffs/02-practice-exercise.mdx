# Practice Exercise: Cloud vs Local Decision Analyzer

In this exercise, you will build a decision analysis tool that evaluates whether a given scenario is better suited for cloud inference, local inference, or a hybrid approach. The tool takes scenario parameters and produces a cost projection, privacy assessment, and recommendation.

## Learning Objectives

- Calculate the break-even point between cloud API costs and local hardware costs.
- Evaluate privacy requirements against regulatory compliance needs.
- Assess latency requirements against local hardware capabilities.
- Make a data-driven recommendation using a rule-based scoring system.
- Present a cost projection with a clear recommendation.

## Setup

Create a new Python file called `decision_analyzer.py` in your working directory. Python 3.10 or later is required. No cloud API keys are needed — all calculations are local.

## Step 1: Define the Scenario Data Model

Start by defining a data class that captures all the parameters of a deployment scenario.

```python
from dataclasses import dataclass
from typing import Optional
from enum import Enum


class PrivacyLevel(Enum):
    PUBLIC = "public"
    INTERNAL = "internal"
    CONFIDENTIAL = "confidential"
    REGULATED = "regulated"
    HIGHLY_RESTRICTED = "highly_restricted"


class ModelTier(Enum):
    SMALL = "small"            # 1-3B parameters, runs on any GPU
    MEDIUM = "medium"          # 7-13B parameters, consumer GPU
    LARGE = "large"            # 34-70B parameters, high-end GPU
    FRONTIER = "frontier"      # Only available via cloud API


@dataclass
class Scenario:
    name: str
    monthly_tokens: int
    model_tier: ModelTier
    privacy_level: PrivacyLevel
    max_latency_ms: int
    has_gpu_hardware: bool = False
    gpu_budget: float = 0.0
    cloud_api_budget: float = 0.0
    requires_offline: bool = False
    expected_growth_rate: float = 0.0
```

## Step 2: Build the Pricing Database

Create lookups that map model tiers to cloud API costs and local hardware costs.

```python
@dataclass
class CloudPricing:
    provider: str
    model_name: str
    input_price_per_1k: float
    output_price_per_1k: float


@dataclass
class HardwarePricing:
    gpu_name: str
    upfront_cost: float
    power_watts: int
    tokens_per_second: int
    ram_gb: int


CLOUD_PRICING = {
    ModelTier.SMALL: [
        CloudPricing("OpenAI", "gpt-4o-mini", 0.00015, 0.0006),
        CloudPricing("Anthropic", "claude-3-haiku", 0.00025, 0.00125),
        CloudPricing("Google", "gemini-1.5-flash", 0.000075, 0.0003),
    ],
    ModelTier.MEDIUM: [
        CloudPricing("OpenAI", "gpt-4o-mini", 0.00015, 0.0006),
        CloudPricing("Anthropic", "claude-3-sonnet", 0.003, 0.015),
        CloudPricing("Google", "gemini-1.5-pro", 0.0025, 0.01),
    ],
    ModelTier.LARGE: [
        CloudPricing("OpenAI", "gpt-4o", 0.0025, 0.01),
        CloudPricing("Anthropic", "claude-3-opus", 0.015, 0.075),
        CloudPricing("Google", "gemini-1.5-pro", 0.0025, 0.01),
    ],
    ModelTier.FRONTIER: [
        CloudPricing("OpenAI", "o3", 0.01, 0.04),
    ],
}

HARDWARE_PRICING = [
    HardwarePricing("RTX 3060 12GB", 300, 170, 45, 12),
    HardwarePricing("RTX 3090 24GB", 1200, 350, 60, 24),
    HardwarePricing("RTX 4090 24GB", 1600, 450, 80, 24),
    HardwarePricing("A6000 48GB", 5000, 300, 70, 48),
    HardwarePricing("A100 80GB", 15000, 400, 90, 80),
    HardwarePricing("Dual RTX 3090", 2400, 700, 100, 48),
]


def find_cloud_options(scenario: Scenario) -> list[CloudPricing]:
    return CLOUD_PRICING.get(scenario.model_tier, [])


def find_hardware_options(scenario: Scenario) -> list[HardwarePricing]:
    model_vram = {
        ModelTier.SMALL: 4, ModelTier.MEDIUM: 8,
        ModelTier.LARGE: 24, ModelTier.FRONTIER: 999,
    }
    required = model_vram.get(scenario.model_tier, 16)
    return [h for h in HARDWARE_PRICING if h.ram_gb >= required]
```

## Step 3: Implement the Cost Calculator

Compute monthly cloud costs, local hardware amortization, and the break-even point.

```python
@dataclass
class CostProjection:
    monthly_cloud_cost: float
    total_local_monthly: float
    hardware_upfront_cost: float
    break_even_months: float
    cloud_option: str
    hardware_option: str


def calculate_costs(scenario: Scenario) -> CostProjection:
    # Find cheapest cloud option
    cloud_options = find_cloud_options(scenario)
    cheapest_cloud = min(
        cloud_options,
        key=lambda c: c.input_price_per_1k + c.output_price_per_1k,
    ) if cloud_options else CloudPricing("N/A", "N/A", 0.0, 0.0)

    # Assume 60% input, 40% output tokens
    input_tokens = int(scenario.monthly_tokens * 0.6)
    output_tokens = int(scenario.monthly_tokens * 0.4)
    monthly_cloud = (
        (input_tokens / 1000) * cheapest_cloud.input_price_per_1k
        + (output_tokens / 1000) * cheapest_cloud.output_price_per_1k
    )

    # Find cheapest suitable hardware
    hardware_options = find_hardware_options(scenario)
    if not hardware_options:
        return CostProjection(
            monthly_cloud_cost=round(monthly_cloud, 2),
            total_local_monthly=0.0, hardware_upfront_cost=0.0,
            break_even_months=float("inf"),
            cloud_option=f"{cheapest_cloud.provider} {cheapest_cloud.model_name}",
            hardware_option="No suitable hardware",
        )

    cheapest = min(hardware_options, key=lambda h: h.upfront_cost)
    monthly_amortization = cheapest.upfront_cost / 24  # 24-month amortization
    kwh_per_month = cheapest.power_watts * 24 * 30 / 1000
    monthly_electricity = kwh_per_month * 0.12
    total_local = monthly_amortization + monthly_electricity

    # Break-even: months until local hardware pays for itself
    monthly_savings = monthly_cloud - monthly_electricity
    if monthly_savings <= 0:
        break_even = float("inf")
    else:
        break_even = cheapest.upfront_cost / monthly_savings

    return CostProjection(
        monthly_cloud_cost=round(monthly_cloud, 2),
        total_local_monthly=round(total_local, 2),
        hardware_upfront_cost=cheapest.upfront_cost,
        break_even_months=round(break_even, 1),
        cloud_option=f"{cheapest_cloud.provider} {cheapest_cloud.model_name}",
        hardware_option=cheapest.gpu_name,
    )
```

## Step 4: Implement the Privacy Assessment

Check whether local inference is required, optional, or unnecessary based on the data privacy level.

```python
class PrivacyVerdict(Enum):
    LOCAL_REQUIRED = "local_required"
    LOCAL_RECOMMENDED = "local_recommended"
    CLOUD_ALLOWED = "cloud_allowed"
    HYBRID_POSSIBLE = "hybrid_possible"


def assess_privacy(scenario: Scenario) -> tuple[PrivacyVerdict, str]:
    level = scenario.privacy_level
    if level == PrivacyLevel.HIGHLY_RESTRICTED:
        return (PrivacyVerdict.LOCAL_REQUIRED,
                "Air-gapped local inference required.")
    if level == PrivacyLevel.REGULATED:
        return (PrivacyVerdict.LOCAL_REQUIRED,
                "Local inference required for regulatory compliance.")
    if level == PrivacyLevel.CONFIDENTIAL:
        return (PrivacyVerdict.LOCAL_RECOMMENDED,
                "Local inference strongly recommended.")
    if level == PrivacyLevel.INTERNAL:
        return (PrivacyVerdict.HYBRID_POSSIBLE,
                "Cloud acceptable with data processing agreement.")
    return (PrivacyVerdict.CLOUD_ALLOWED,
            "Cloud inference is acceptable.")


def check_compliance_requirements(scenario: Scenario) -> list[str]:
    requirements = []
    level = scenario.privacy_level
    if level == PrivacyLevel.REGULATED:
        requirements.extend([
            "Sign Business Associate Agreement (BAA)",
            "Enable data encryption in transit and at rest",
            "Conduct a Data Protection Impact Assessment",
            "Implement access controls and audit logging",
        ])
    if level in (PrivacyLevel.REGULATED, PrivacyLevel.HIGHLY_RESTRICTED):
        requirements.extend([
            "Maintain data residency within approved regions",
            "Implement data retention and deletion policy",
        ])
    if level == PrivacyLevel.CONFIDENTIAL:
        requirements.extend([
            "Review cloud provider data usage policy",
            "Opt out of training data usage",
            "Implement PII redaction middleware",
        ])
    return requirements
```

## Step 5: Implement the Latency and Capability Assessment

Evaluate whether local hardware can meet latency requirements and whether the model tier is available locally.

```python
@dataclass
class LatencyAssessment:
    can_meet_latency: bool
    estimated_local_ms: int
    estimated_cloud_ms: int
    verdict: str


def assess_latency(scenario: Scenario) -> LatencyAssessment:
    avg_tokens = 200
    model_speeds = {
        ModelTier.SMALL: 50, ModelTier.MEDIUM: 40,
        ModelTier.LARGE: 25, ModelTier.FRONTIER: 0,
    }
    speed = model_speeds.get(scenario.model_tier, 25)

    if speed == 0:
        return LatencyAssessment(
            False, 0, 300 + int((avg_tokens / 60) * 1000),
            "Frontier model not available locally.",
        )

    local_ms = int((avg_tokens / speed) * 1000)
    cloud_ms = 100 + 100 + int((avg_tokens / 60) * 1000)
    can_meet = local_ms <= scenario.max_latency_ms

    if can_meet:
        v = f"Local inference meets latency ({local_ms}ms < {scenario.max_latency_ms}ms)."
    else:
        v = f"Local inference may not meet latency ({local_ms}ms > {scenario.max_latency_ms}ms)."

    return LatencyAssessment(can_meet, local_ms, cloud_ms, v)


def assess_capability(scenario: Scenario) -> str:
    if scenario.model_tier == ModelTier.FRONTIER:
        return "Frontier models are only available via cloud API."
    if scenario.model_tier == ModelTier.LARGE:
        return "Large open-weight models exist but may underperform frontier by 5-15%."
    return f"Models at {scenario.model_tier.value} tier run well on consumer hardware."
```

## Step 6: Build the Recommendation Engine

Combine all assessments into a final recommendation.

```python
class Recommendation(Enum):
    CLOUD = "cloud"
    LOCAL = "local"
    HYBRID = "hybrid"
    NOT_FEASIBLE = "not_feasible"


@dataclass
class AnalysisResult:
    scenario_name: str
    recommendation: Recommendation
    reasoning: str
    cost: CostProjection
    privacy_verdict: tuple[PrivacyVerdict, str]
    latency: LatencyAssessment
    capability: str
    compliance: list[str]
    risks: list[str]


def analyze(scenario: Scenario) -> AnalysisResult:
    cost = calculate_costs(scenario)
    privacy_verdict = assess_privacy(scenario)
    latency = assess_latency(scenario)
    capability = assess_capability(scenario)
    compliance = check_compliance_requirements(scenario)
    risks = []
    priv_v, _ = privacy_verdict

    # Decision tree
    if priv_v == PrivacyVerdict.LOCAL_REQUIRED:
        if scenario.model_tier == ModelTier.FRONTIER:
            rec, reason = Recommendation.NOT_FEASIBLE, (
                "Privacy requires local, but frontier model is cloud-only. "
                "Consider a smaller model with RAG or negotiate a BAA."
            )
        elif cost.hardware_upfront_cost > scenario.gpu_budget > 0:
            rec, reason = Recommendation.NOT_FEASIBLE, (
                f"Privacy requires local, but hardware (${cost.hardware_upfront_cost}) "
                f"exceeds budget (${scenario.gpu_budget})."
            )
        else:
            rec, reason = Recommendation.LOCAL, "Privacy regulations require local inference."
            if not latency.can_meet_latency:
                risks.append("Latency may exceed requirements. Consider GPU upgrade.")

    elif priv_v == PrivacyVerdict.LOCAL_RECOMMENDED:
        if scenario.model_tier == ModelTier.FRONTIER:
            rec, reason = Recommendation.CLOUD, (
                "Privacy recommends local, but frontier is cloud-only. "
                "Use a provider with zero-retention agreement."
            )
        elif cost.break_even_months < 12:
            rec, reason = Recommendation.LOCAL, (
                f"Local breaks even in {cost.break_even_months} months "
                f"and provides better privacy."
            )
        else:
            rec, reason = Recommendation.HYBRID, (
                "Privacy recommends local, but cloud is cheaper at this volume. "
                "Use local for sensitive data, cloud for the rest."
            )

    else:  # Cloud or hybrid allowed
        if scenario.requires_offline:
            rec, reason = Recommendation.LOCAL, "Offline operation required."
        elif cost.break_even_months < 6:
            rec, reason = Recommendation.LOCAL, (
                f"Local breaks even in {cost.break_even_months} months "
                f"with lower latency."
            )
        elif cost.break_even_months < 24:
            rec, reason = Recommendation.HYBRID, (
                f"Local breaks even in {cost.break_even_months} months. "
                "Use hybrid: local for steady traffic, cloud for bursts."
            )
        else:
            rec, reason = Recommendation.CLOUD, "Cloud API is most economical."

    return AnalysisResult(
        scenario.name, rec, reason, cost, privacy_verdict,
        latency, capability, compliance, risks,
    )


def print_report(result: AnalysisResult) -> None:
    print("=" * 60)
    print(f"  SCENARIO: {result.scenario_name}")
    print(f"  RECOMMENDATION: {result.recommendation.value.upper()}")
    print(f"  {result.reasoning}")
    print()
    print(f"  COST: Cloud=${result.cost.monthly_cloud_cost}/mo "
          f"Local=${result.cost.total_local_monthly}/mo "
          f"(HW=${result.cost.hardware_upfront_cost})")
    print(f"  BREAK-EVEN: {result.cost.break_even_months} months")
    print(f"  PRIVACY: {result.privacy_verdict[0].value}")
    if result.compliance:
        print(f"  COMPLIANCE: {', '.join(result.compliance)}")
    print(f"  LATENCY: Local={result.latency.estimated_local_ms}ms "
          f"Cloud={result.latency.estimated_cloud_ms}ms")
    if result.risks:
        print(f"  RISKS: {'; '.join(result.risks)}")
    print()
```

## Step 8: Extend the Tool

The decision analyzer can be extended with additional capabilities:

```python
def project_annual_cost(scenario: Scenario, cloud_cost_per_month: float) -> list[dict]:
    """Project 12 months of costs with compounding growth."""
    projections = []
    current_tokens = scenario.monthly_tokens
    for month in range(1, 13):
        cost = (current_tokens / scenario.monthly_tokens) * cloud_cost_per_month
        projections.append({
            "month": month,
            "tokens": current_tokens,
            "cloud_cost": round(cost, 2),
        })
        current_tokens = int(current_tokens * (1 + scenario.expected_growth_rate))
    return projections
```
