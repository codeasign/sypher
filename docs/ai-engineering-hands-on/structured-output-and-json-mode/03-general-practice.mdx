## Tier 1 — Beginner

### Exercise 1: Simple Classification Schema

**Task:** Create a script that classifies a product review as "positive", "negative", or "mixed" using structured output with a schema that has `sentiment` (string, enum) and `summary` (string) fields.

**Hint:** Define a schema with `type: "object"`, two properties (`sentiment` as enum, `summary` as string), and `strict: true`.

### Solution

```python
import os
import json
from openai import OpenAI

client = OpenAI(api_key=os.environ["OPENAI_API_KEY"])

schema = {
    "name": "classify_review",
    "schema": {
        "type": "object",
        "properties": {
            "sentiment": {
                "type": "string",
                "enum": ["positive", "negative", "mixed"]
            },
            "summary": {
                "type": "string",
                "description": "One-sentence summary of the review"
            }
        },
        "required": ["sentiment", "summary"],
        "additionalProperties": False
    },
    "strict": True
}

response = client.chat.completions.create(
    model="gpt-4o-mini",
    messages=[
        {"role": "developer", "content": "Classify this product review."},
        {"role": "user", "content": "The product works well but the battery life is terrible."}
    ],
    response_format={"type": "json_schema", "json_schema": schema}
)

result = json.loads(response.choices[0].message.content)
print(f"Sentiment: {result['sentiment']}")
print(f"Summary: {result['summary']}")
```

Expected output:
```
Sentiment: mixed
Summary: The product works well but has poor battery life.
```

### Exercise 2: Array of Simple Objects

**Task:** Extract a list of shopping items from the text "I need to buy milk, eggs, and bread." using a schema that returns an array of `{item: string, quantity: integer}` objects.

**Hint:** Use `type: "array"` with `items` containing the object schema.

### Solution

```python
import os
import json
from openai import OpenAI

client = OpenAI(api_key=os.environ["OPENAI_API_KEY"])

schema = {
    "name": "extract_items",
    "schema": {
        "type": "object",
        "properties": {
            "items": {
                "type": "array",
                "items": {
                    "type": "object",
                    "properties": {
                        "item": {"type": "string"},
                        "quantity": {"type": "integer"}
                    },
                    "required": ["item", "quantity"],
                    "additionalProperties": False
                }
            }
        },
        "required": ["items"],
        "additionalProperties": False
    },
    "strict": True
}

response = client.chat.completions.create(
    model="gpt-4o-mini",
    messages=[
        {"role": "developer", "content": "Extract shopping items from the text."},
        {"role": "user", "content": "I need to buy milk, eggs, and bread."}
    ],
    response_format={"type": "json_schema", "json_schema": schema}
)

result = json.loads(response.choices[0].message.content)
for item in result["items"]:
    print(f"{item['item']} × {item['quantity']}")
```

### Exercise 4: Error Handling for Refusals

**Task:** Write a safe extraction function that takes text and a schema, calls the API, and returns the parsed data OR a default value if the model refuses.

**Hint:** Check `response.choices[0].message.refusal` before parsing.

### Solution

```python
import os
import json
from openai import OpenAI

client = OpenAI(api_key=os.environ["OPENAI_API_KEY"])

def safe_extract(schema, text, default=None):
    response = client.chat.completions.create(
        model="gpt-4o-mini",
        messages=[
            {"role": "developer", "content": "Extract the requested information."},
            {"role": "user", "content": text}
        ],
        response_format={"type": "json_schema", "json_schema": schema}
    )
    if response.choices[0].message.refusal:
        print(f"Refusal: {response.choices[0].message.refusal}")
        return default
    return json.loads(response.choices[0].message.content)

schema = {
    "name": "extract_book",
    "schema": {
        "type": "object",
        "properties": {
            "title": {"type": "string"},
            "author": {"type": "string"}
        },
        "required": ["title", "author"],
        "additionalProperties": False
    },
    "strict": True
}

result = safe_extract(schema, "1984 by George Orwell")
if result:
    print(f"Book: {result['title']} by {result['author']}")
```

## Tier 2 — Intermediate

### Exercise 1: Multi-Output Extraction Pipeline

**Task:** Build a pipeline that extracts entities from a document using structured output, then classifies the document topic using a second structured output call. Print both results.

**Hint:** Chain two API calls — first extracts entities, second classifies topic.

### Solution

```python
import os
import json
from openai import OpenAI

client = OpenAI(api_key=os.environ["OPENAI_API_KEY"])

entity_schema = {
    "name": "extract_entities",
    "schema": {
        "type": "object",
        "properties": {
            "entities": {
                "type": "array",
                "items": {
                    "type": "object",
                    "properties": {
                        "name": {"type": "string"},
                        "type": {"type": "string", "enum": ["person", "org", "place", "concept"]}
                    },
                    "required": ["name", "type"],
                    "additionalProperties": False
                }
            }
        },
        "required": ["entities"],
        "additionalProperties": False
    },
    "strict": True
}

topic_schema = {
    "name": "classify_topic",
    "schema": {
        "type": "object",
        "properties": {
            "topic": {"type": "string", "enum": ["tech", "science", "history", "business", "other"]},
            "confidence": {"type": "number"}
        },
        "required": ["topic", "confidence"],
        "additionalProperties": False
    },
    "strict": True
}

doc = "Apple's latest chip uses a 3-nanometer process, making it the most powerful mobile processor ever built."

# Step 1: Extract entities
r1 = client.chat.completions.create(
    model="gpt-4o-mini",
    messages=[{"role": "user", "content": f"Extract entities: {doc}"}],
    response_format={"type": "json_schema", "json_schema": entity_schema}
)
entities = json.loads(r1.choices[0].message.content)

# Step 2: Classify topic
r2 = client.chat.completions.create(
    model="gpt-4o-mini",
    messages=[{"role": "user", "content": f"Classify topic: {doc}"}],
    response_format={"type": "json_schema", "json_schema": topic_schema}
)
topic = json.loads(r2.choices[0].message.content)

print(f"Entities: {[e['name'] for e in entities['entities']]}")
print(f"Topic: {topic['topic']} (confidence: {topic['confidence']})")
```

### Exercise 2: Schema Validation Before the Call

**Task:** Write a function that validates a schema against OpenAI's supported features before sending it. Check that there are no `anyOf`, `oneOf`, `$ref`, or `null` types. Return a list of issues or `True` if valid.

**Hint:** Recursively walk the schema properties. Check each value's type.

### Solution

```python
def validate_schema(schema_obj, path=""):
    issues = []
    obj = schema_obj.get("schema", schema_obj)

    def walk(node, path):
        if not isinstance(node, dict):
            return
        for key in node:
            if key in ("anyOf", "oneOf", "allOf"):
                issues.append(f"{path}: '{key}' is not supported")
            if key == "$ref":
                issues.append(f"{path}: '$ref' is not supported")
            if key == "type" and node[key] == "null":
                issues.append(f"{path}: 'null' type is not supported")
            if key == "items" and isinstance(node[key], dict):
                walk(node[key], f"{path}.items")
            if key == "properties" and isinstance(node[key], dict):
                for prop_name, prop_schema in node[key].items():
                    walk(prop_schema, f"{path}.properties.{prop_name}")

    walk(obj, path or "root")
    return issues if issues else True

# Test
good_schema = {
    "name": "test",
    "schema": {
        "type": "object",
        "properties": {
            "name": {"type": "string"}
        },
        "required": ["name"]
    }
}

bad_schema = {
    "name": "test2",
    "schema": {
        "type": "object",
        "properties": {
            "value": {"anyOf": [{"type": "string"}, {"type": "integer"}]}
        }
    }
}

print("Good schema:", validate_schema(good_schema))
print("Bad schema:", validate_schema(bad_schema))
```

### Exercise 3: Retry on Schema Violation

**Task:** Create a function that calls structured output and retries up to 3 times if the response contains a `refusal`. Each retry should use a slightly reworded prompt.

**Hint:** Maintain a list of prompt variations. Try each in sequence until one succeeds.

### Solution

```python
import os
import json
from openai import OpenAI

client = OpenAI(api_key=os.environ["OPENAI_API_KEY"])

def retry_structured(schema, text, max_retries=3):
    prompts = [
        "Extract the requested information from this text.",
        "Parse this text and return the structured data requested.",
        "Analyze this text and output the data in the specified format.",
    ]

    schema_config = {"type": "json_schema", "json_schema": schema}

    for attempt in range(max_retries):
        prompt = prompts[attempt % len(prompts)]
        response = client.chat.completions.create(
            model="gpt-4o-mini",
            messages=[
                {"role": "developer", "content": prompt},
                {"role": "user", "content": text}
            ],
            response_format=schema_config
        )

        if not response.choices[0].message.refusal:
            return json.loads(response.choices[0].message.content)

        print(f"Attempt {attempt + 1} refused, retrying...")

    return None

# Test with a schema
schema = {
    "name": "extract_product",
    "schema": {
        "type": "object",
        "properties": {
            "product_name": {"type": "string"},
            "price": {"type": "number"},
            "in_stock": {"type": "boolean"}
        },
        "required": ["product_name", "price", "in_stock"],
        "additionalProperties": False
    },
    "strict": True
}

result = retry_structured(schema, "The新款 MacBook Pro costs $2499 and is available now.")
if result:
    print(f"Product: {result['product_name']}, ${result['price']}, In stock: {result['in_stock']}")
else:
    print("All attempts failed.")
```

## Tier 3 — Advanced

### Exercise 1: Dynamic Schema Generator

**Task:** Write a function that takes a natural language description like "I need a schema to extract name, email, and phone number from a contact" and returns a working JSON Schema that can be used with OpenAI's Structured Outputs. The function must only use supported features.

**Hint:** Parse the description for field types. Default `string` unless "number" or "integer" is mentioned. Always mark all fields as `required`.

### Solution

```python
def generate_schema(name, description):
    """Generate a JSON Schema from a natural language description."""
    type_map = {
        "boolean": "boolean", "bool": "boolean",
        "integer": "integer", "int": "integer", "number": "number",
        "float": "number", "decimal": "number"
    }

    # Parse: "name (string), age (integer), active (boolean)"
    fields = []
    parts = [p.strip() for p in description.replace(", ", ",").split(",")]
    for part in parts:
        # Match "field (type)" or just "field"
        if "(" in part and part.endswith(")"):
            field_name, field_type = part.split("(", 1)
            field_type = field_type.rstrip(")").strip().lower()
            field_name = field_name.strip()
        else:
            field_name = part.strip()
            field_type = "string"

        json_type = type_map.get(field_type, "string")
        field_def = {"type": json_type}

        if json_type == "string" and field_name in ("email", "phone", "url", "color"):
            field_def["description"] = f"The {field_name} value"

        fields.append((field_name, field_def))

    properties = {name: defn for name, defn in fields}
    required = [name for name, _ in fields]

    return {
        "name": name.replace(" ", "_"),
        "schema": {
            "type": "object",
            "properties": properties,
            "required": required,
            "additionalProperties": False
        },
        "strict": True
    }

# Test
schema = generate_schema("contact_info", "name, email (string), phone, age (integer)")
import json
print(json.dumps(schema, indent=2))
```

### Exercise 2: Multi-Schema Router

**Task:** Create a router that inspects user input and selects one of several schemas to use for extraction. For example, if the input mentions "invoice," use the invoice schema; if it mentions "recipe," use the recipe schema.

**Hint:** Use a simple keyword check (`if "invoice" in text.lower()`) to route, or use a lightweight classification call first.

### Solution

```python
import os
import json
from openai import OpenAI

client = OpenAI(api_key=os.environ["OPENAI_API_KEY"])

schemas = {
    "invoice": {
        "name": "extract_invoice",
        "schema": {
            "type": "object",
            "properties": {
                "vendor": {"type": "string"},
                "total": {"type": "number"},
                "date": {"type": "string"}
            },
            "required": ["vendor", "total", "date"],
            "additionalProperties": False
        },
        "strict": True
    },
    "recipe": {
        "name": "extract_recipe",
        "schema": {
            "type": "object",
            "properties": {
                "title": {"type": "string"},
                "servings": {"type": "integer"},
                "cook_time_minutes": {"type": "integer"}
            },
            "required": ["title", "servings", "cook_time_minutes"],
            "additionalProperties": False
        },
        "strict": True
    }
}

def route_and_extract(text):
    if "invoice" in text.lower():
        schema = schemas["invoice"]
    elif any(w in text.lower() for w in ["recipe", "cook", "ingredient"]):
        schema = schemas["recipe"]
    else:
        print("No matching schema found.")
        return None

    response = client.chat.completions.create(
        model="gpt-4o-mini",
        messages=[
            {"role": "developer", "content": f"Extract data using the {schema['name']} schema."},
            {"role": "user", "content": text}
        ],
        response_format={"type": "json_schema", "json_schema": schema}
    )

    if response.choices[0].message.refusal:
        return None
    return json.loads(response.choices[0].message.content)

# Test
texts = [
    "INVOICE #123: Acme Corp, $500, Jan 15 2024",
    "Recipe: Pancakes, serves 4, takes 20 minutes"
]

for text in texts:
    result = route_and_extract(text)
    if result:
        print(f"Input: {text[:40]}...")
        print(f"  Result: {json.dumps(result)}")
        print()
```

### Exercise 3: Streaming Structured Output

**Task:** For this advanced exercise, research how to handle structured output in a streaming context. The challenge: structured output tokens are guaranteed to form valid JSON, but intermediate chunks may not be valid JSON individually. Create a script that streams a structured output and uses a streaming JSON parser (like `ijson`) to progressively extract fields as they arrive.

**Hint:** Use `stream=True` with `response_format` (supported by OpenAI). Each chunk still contains deltas, but they progressively build toward valid JSON. A streaming JSON parser can extract completed fields before the stream finishes.

### Solution

```python
import os
import json
from openai import OpenAI

client = OpenAI(api_key=os.environ["OPENAI_API_KEY"])

schema = {
    "name": "generate_story",
    "schema": {
        "type": "object",
        "properties": {
            "title": {"type": "string"},
            "characters": {
                "type": "array",
                "items": {"type": "string"}
            },
            "summary": {"type": "string"}
        },
        "required": ["title", "characters", "summary"],
        "additionalProperties": False
    },
    "strict": True
}

stream = client.chat.completions.create(
    model="gpt-4o-mini",
    messages=[
        {"role": "user", "content": "Write a short story about a robot chef."}
    ],
    response_format={"type": "json_schema", "json_schema": schema},
    stream=True
)

collected = []
for chunk in stream:
    if chunk.choices[0].delta.content is not None:
        content = chunk.choices[0].delta.content
        collected.append(content)
        # Try to parse progressively (may fail on partial JSON)
        try:
            partial = json.loads("".join(collected))
            if "title" in partial and "characters" in partial:
                # Clear the display and show current state
                os.system("cls" if os.name == "nt" else "clear")
                print(f"Title: {partial.get('title', '...')}")
                print(f"Characters: {', '.join(partial.get('characters', ['...']))}")
                print(f"Summary: {partial.get('summary', '...')[:50]}...")
        except json.JSONDecodeError:
            pass  # Partial JSON, continue accumulating

print("\n--- Complete ---")
result = json.loads("".join(collected))
print(json.dumps(result, indent=2))
```

This progressive display works because structured output guarantees the tokens will eventually form valid JSON. The intermediate parsing attempts may fail (partial JSON is often invalid), but once a field is complete, it becomes parseable.
