### Exercise 1: Add Logging to a Bare Agent

You are given a minimal agent loop that has no logging at all. Your task is to add structured logging that records every step.

**Task:** Take the following agent code and add logging that writes a JSON Lines file for each run. Each log entry must include: run ID, timestamp, step number, event type, and token count.

```python
from anthropic import Anthropic

client = Anthropic()
MODEL = "claude-sonnet-4-20250514"

TOOLS = [
    {
        "name": "get_time",
        "description": "Get the current time for a timezone",
        "input_schema": {
            "type": "object",
            "properties": {
                "timezone": {
                    "type": "string",
                    "description": "Timezone name like UTC, America/New_York"
                }
            },
            "required": ["timezone"]
        }
    }
]

SYSTEM = "You are a time assistant. Use the get_time tool to answer time questions."


def get_time(timezone: str) -> str:
    from datetime import datetime as dt
    import pytz
    try:
        tz = pytz.timezone(timezone)
        return dt.now(tz).strftime("%H:%M:%S")
    except Exception as e:
        return f"Error: {e}"


def run(input_text: str):
    messages = [{"role": "user", "content": input_text}]
    for _ in range(5):
        response = client.messages.create(
            model=MODEL, system=SYSTEM, max_tokens=512,
            messages=messages, tools=TOOLS,
        )
        if response.stop_reason == "tool_use":
            tb = response.content[-1]
            result = get_time(**tb.input)
            messages.append({"role": "assistant", "content": response.content})
            messages.append({
                "role": "user",
                "content": [{"type": "tool_result",
                             "tool_use_id": tb.id,
                             "content": result}]
            })
        else:
            return response.content[0].text


print(run("What time is it in New York?"))
```

**Hint:** Create a logger that writes JSON Lines files. Import `uuid.uuid4()` for run IDs, `datetime.now(timezone.utc)` for timestamps. After each LLM response, record the tokens, finish reason, and any tool call details to a JSON Lines file in a `logs/` directory.

**Solution:**

```python
import json, uuid, os
from datetime import datetime, timezone
from anthropic import Anthropic

client = Anthropic()
MODEL = "claude-sonnet-4-20250514"
os.makedirs("logs", exist_ok=True)

TOOLS = [
    {
        "name": "get_time",
        "description": "Get the current time for a timezone",
        "input_schema": {
            "type": "object",
            "properties": {
                "timezone": {
                    "type": "string",
                    "description": "Timezone name like UTC, America/New_York"
                }
            },
            "required": ["timezone"]
        }
    }
]

SYSTEM = "You are a time assistant. Use the get_time tool to answer time questions."


def get_time(timezone: str) -> str:
    from datetime import datetime as dt
    import pytz
    try:
        tz = pytz.timezone(timezone)
        return dt.now(tz).strftime("%H:%M:%S")
    except Exception as e:
        return f"Error: {e}"


def run(input_text: str):
    run_id = str(uuid.uuid4())[:8]
    log_file = f"logs/run_{run_id}.jsonl"
    messages = [{"role": "user", "content": input_text}]

    for step_num in range(5):
        ts = datetime.now(timezone.utc).isoformat()
        response = client.messages.create(
            model=MODEL, system=SYSTEM, max_tokens=512,
            messages=messages, tools=TOOLS,
        )
        entry = {
            "run_id": run_id, "timestamp": ts, "step_number": step_num + 1,
            "input_tokens": response.usage.input_tokens,
            "output_tokens": response.usage.output_tokens,
            "finish_reason": response.stop_reason,
        }
        if response.stop_reason == "tool_use":
            tb = response.content[-1]
            entry["event_type"] = "tool_call"
            entry["tool_name"] = tb.name
            entry["tool_input"] = tb.input
            result = get_time(**tb.input)
            entry["tool_result"] = result
            messages.append({"role": "assistant", "content": response.content})
            messages.append({"role": "user", "content": [
                {"type": "tool_result", "tool_use_id": tb.id, "content": result}
            ]})
        else:
            entry["event_type"] = "final_response"
            entry["output"] = response.content[0].text if response.content else ""
            with open(log_file, "a") as f:
                f.write(json.dumps(entry) + "\n")
            return response.content[0].text

        with open(log_file, "a") as f:
            f.write(json.dumps(entry) + "\n")

    return "No final response"


print(run("What time is it in New York?"))
```

Expected output: The agent prints the time in New York. A JSON Lines file appears in `logs/` with each entry containing a run ID, timestamp, step number, and token counts.

### Exercise 2: Read the Logs and Identify Tool Calls

Write a script that reads a JSON Lines log file and prints a human-readable summary of each step.

**Task:** Create `read_logs.py` that accepts a log file path as a command-line argument and prints each step with its event type, tool name (if applicable), and token counts.

**Solution:**

```python
import json, sys

with open(sys.argv[1]) as f:
    lines = f.readlines()

total_in = total_out = 0
for line in lines:
    e = json.loads(line)
    step = e.get("step_number", "?")
    event = e.get("event_type", "?")
    inp = e.get("input_tokens", 0)
    out = e.get("output_tokens", 0)
    total_in += inp
    total_out += out

    if event == "tool_call":
        tool = e.get("tool_name", "?")
        params = json.dumps(e.get("tool_input", {}))
        print(f"Step {step}: [{event}] tool={tool} params={params} "
              f"tokens=({inp}+{out})")
    elif event == "final_response":
        output = e.get("output", "")[:60]
        print(f"Step {step}: [{event}] output=\"{output}...\" "
              f"tokens=({inp}+{out})")
    else:
        print(f"Step {step}: [{event}] tokens=({inp}+{out})")

print(f"Total: {total_in} input + {total_out} output = {total_in + total_out} tokens")
```

Run: `python read_logs.py logs/run_*.jsonl`

Expected output: A step-by-step summary showing the tool call with its parameters, the final response, and the total token usage.

### Exercise 4: Detect a Runaway Loop

Write a loop detector that flags repeated identical tool calls.

**Task:** Write a function that scans log entries and returns `True` if three or more consecutive tool calls have the same name and same input parameters.

**Solution:**

```python
import json, sys

def detect_loop(entries: list) -> bool:
    calls = [e for e in entries if e.get("event_type") == "tool_call"]
    for i in range(2, len(calls)):
        if (calls[i-2]["tool_name"] == calls[i-1]["tool_name"] == calls[i]["tool_name"]
                and calls[i-2].get("tool_input") == calls[i-1].get("tool_input")
                == calls[i].get("tool_input")):
            return True
    return False

with open(sys.argv[1]) as f:
    entries = [json.loads(line) for line in f]
print(f"Loop detected: {detect_loop(entries)}")
```

Test with a simulated loop log:

```bash

{"step":2,"event_type":"tool_call","tool_name":"get_time","tool_input":{"timezone":"UTC"}}
{"step":3,"event_type":"tool_call","tool_name":"get_time","tool_input":{"timezone":"UTC"}}' > logs/loop.jsonl
python detect_loop.py logs/loop.jsonl
```

Expected output: `Loop detected: True` for the loop log, `False` for a normal log.

### Exercise 3: Score Results and Identify Failure Patterns

Analyze evaluation results to find patterns.

**Task:** Write `analyze_results.py` that reads `eval_results.json` and identifies: (1) the lowest scoring test case, (2) the most expensive test case, (3) any trajectory-outcome gap, and (4) any unexpected tool calls.

**Solution:**

```python
import json

with open("eval_results.json") as f:
    results = json.load(f)

lowest = min(results, key=lambda r: r["total_score"])
print(f"Lowest score: {lowest['test_id']} ({lowest['total_score']:.2f})")

most_tokens = max(results, key=lambda r: r["tokens"]["total"])
print(f"Most tokens: {most_tokens['test_id']} ({most_tokens['tokens']['total']})")

biggest_gap = max(results, key=lambda r: abs(r["trajectory_score"] - r["outcome_score"]))
gap = abs(biggest_gap["trajectory_score"] - biggest_gap["outcome_score"])
print(f"Biggest traj-outcome gap: {biggest_gap['test_id']} ({gap:.2f})")
```

Expected output: A report showing which test case needs the most attention. The "invalid_date" case often has the largest gap because the agent may call the tool correctly but fail to handle the "yesterday" date gracefully.

### Exercise 4: Compare Two Prompt Versions

Extend the comparison script to handle three prompt variants.

**Task:** Compare three prompts -- basic, explicit, and verbose -- against the same dataset. Report which prompt wins on each test case.

**Solution:**

```python
import json
import agent
from agent import run_agent, logger
from evaluator import evaluate_run
from eval_dataset import EVAL_DATASET

PROMPTS = {
    "A (basic)": "You are a helpful weather assistant.",
    "B (explicit)": "You are a helpful weather assistant. Always call get_weather before answering. If data is missing, say so.",
    "C (verbose)": "You are a helpful weather assistant. Call get_weather for each location. Summarize results. If data is missing, apologize. If the date is unusual, explain you only have today and tomorrow data.",
}

all_results = {}
for name, prompt in PROMPTS.items():
    agent.SYSTEM_PROMPT = prompt
    results = []
    for tc in EVAL_DATASET:
        steps = run_agent(tc["input"])
        score = evaluate_run(logger, tc["expected_tools"], tc["expected_keywords"])
        score["test_id"] = tc["id"]
        results.append(score)
    all_results[name] = results

agent.SYSTEM_PROMPT = "You are a helpful weather assistant."

print(f"{'Test':<20}", end="")
for name in PROMPTS:
    print(f" {name:>15}", end="")
print(f" {'Best':>10}")
print("-" * 60)
for i, tc in enumerate(EVAL_DATASET):
    row = f"{tc['id']:<20}"
    scores = {}
    for name in PROMPTS:
        s = all_results[name][i]["total_score"]
        scores[name] = s
        row += f" {s:>15.2f}"

```

### Exercise 1: Implement a Regression Test Pipeline

Build a pipeline that runs against a stored baseline and reports regressions.

**Task:** Create `regression_pipeline.py` that loads a stored baseline, runs the agent against the dataset, compares scores, and flags any test case where the total score dropped by more than 0.1. Update the baseline if all tests pass.

**Solution:**

```python
import json, sys, os
from agent import run_agent, logger
from evaluator import evaluate_run
from eval_dataset import EVAL_DATASET

BASELINE_FILE = "baseline_results.json"
THRESHOLD = 0.1

baseline = {}
if os.path.exists(BASELINE_FILE):
    with open(BASELINE_FILE) as f:
        baseline = json.load(f)

all_pass = True
results = []
for tc in EVAL_DATASET:
    steps = run_agent(tc["input"])
    score = evaluate_run(logger, tc["expected_tools"], tc["expected_keywords"])
    score["test_id"] = tc["id"]
    results.append(score)

    curr = score["total_score"]
    prev = baseline.get(tc["id"])
    if prev is not None:
        diff = curr - prev
        if diff < -THRESHOLD:
            print(f"REGRESSION: {tc['id']} ({curr:.2f} vs baseline {prev:.2f})")
            all_pass = False
        else:
            print(f"OK: {tc['id']} ({curr:.2f})")
    else:
        print(f"OK (new): {tc['id']} ({curr:.2f})")

if all_pass:
    with open(BASELINE_FILE, "w") as f:
        json.dump({r["test_id"]: r["total_score"] for r in results}, f, indent=2)
    print("Pipeline PASSED -- baseline updated")
    sys.exit(0)
else:
    print("Pipeline FAILED -- regressions detected")
    sys.exit(1)
```

Run twice: first run creates the baseline, second run compares against it. Expected output: the first run creates `baseline_results.json`, the second run reports no regressions (assuming the agent is deterministic).

### Exercise 2: Add PII Redaction to Logs

Implement a PII redaction filter that protects sensitive data before writing logs.

**Task:** Modify the `AgentLogger.log_step` to redact emails, phone numbers, credit card numbers, and API keys. Use regex patterns compiled with `re.compile()`.

**Solution:**

```python
import re, json, os, uuid
from datetime import datetime, timezone

PATTERNS = {
    re.compile(r'\b[A-Za-z0-9._%+-]+@[A-Za-z0-9.-]+\.[A-Z|a-z]{2,}\b'): "[EMAIL]",
    re.compile(r'\b\+?\d[\d\-.\s()]{7,}\d\b'): "[PHONE]",
    re.compile(r'\b\d{4}[-.\s]?\d{4}[-.\s]?\d{4}[-.\s]?\d{4}\b'): "[CC]",
    re.compile(r'\bsk-[A-Za-z0-9]{20,}\b'): "[KEY]",
}

def redact(obj):
    if isinstance(obj, str):
        for pattern, replacement in PATTERNS.items():
            obj = pattern.sub(replacement, obj)
        return obj
    if isinstance(obj, dict):
        return {k: redact(v) for k, v in obj.items()}
    return obj

class SafeLogger:
    def __init__(self, log_dir="logs"):
        self.log_dir = log_dir
        os.makedirs(log_dir, exist_ok=True)

    def start_run(self):
        self.run_id = str(uuid.uuid4())[:8]
        self.steps = []
        return self.run_id

    def log_step(self, data: dict):
        data["run_id"] = self.run_id
        data["timestamp"] = datetime.now(timezone.utc).isoformat()
        self.steps.append(data)
        redacted = redact(data)
        with open(os.path.join(self.log_dir, f"run_{self.run_id}.jsonl"), "a") as f:
            f.write(json.dumps(redacted) + "\n")
```

Test: pass a log entry with `{"tool_input": {"email": "user@example.com", "cc": "4111-1111-1111-1111"}}` and verify the output contains `[EMAIL]` and `[CC]` instead of the actual values.

### Exercise 3: Build a Dashboard Showing Evaluation Scores Over Time

Create a script that reads multiple evaluation result files and prints a trend table.

**Task:** Create `dashboard.py` that finds all `eval_results_*.json` files, aggregates by date, and prints a table with columns: date, test count, average trajectory score, average outcome score, average total score, and average tokens per run.

**Solution:**

```python
import json, glob, re

def compute_avgs(results):
    n = len(results)
    if n == 0:
        return {"count": 0, "avg_traj": 0, "avg_outcome": 0, "avg_total": 0, "avg_tokens": 0}
    return {
        "count": n,
        "avg_traj": round(sum(r["trajectory_score"] for r in results) / n, 3),
        "avg_outcome": round(sum(r["outcome_score"] for r in results) / n, 3),
        "avg_total": round(sum(r["total_score"] for r in results) / n, 3),
        "avg_tokens": round(sum(r["tokens"]["total"] for r in results) / n, 1),
    }

files = sorted(glob.glob("eval_results_*.json"))
if not files:
    print("No eval_results_*.json files found.")
    exit()

rows = []
for f in files:
    match = re.search(r"eval_results_(\d{4}-\d{2}-\d{2})\.json", f)
    date = match.group(1) if match else "unknown"
    with open(f) as fh:
        results = json.load(fh)
    rows.append((date, compute_avgs(results)))

print(f"{'Date':<14} {'Cases':<7} {'Traj':<8} {'Outcome':<9} {'Total':<8} {'Tokens':<8}")
print("-" * 55)
for date, s in rows:
    print(f"{date:<14} {s['count']:<7} {s['avg_traj']:<8} {s['avg_outcome']:<9} "
          f"{s['avg_total']:<8} {s['avg_tokens']:<8}")

if len(rows) >= 2:
    delta = rows[-1][1]["avg_total"] - rows[0][1]["avg_total"]
    direction = "improving" if delta > 0 else "declining"
    print(f"\nTrend: scores are {direction} (delta={delta:+.3f} over {len(rows)} runs)")
```
