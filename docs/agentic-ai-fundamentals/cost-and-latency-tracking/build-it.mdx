# Build It: Implementing Cost and Latency Tracking

⏱️ 25 minutes

## Breaking It Down

### 1. Per-Session Cost Calculation

Track token usage per LLM call and compute real-time cost:

```python
# cost_tracker.py
from dataclasses import dataclass, field
from datetime import datetime

MODEL_PRICING = {
    "claude-sonnet-4-6": {"input_per_m": 3.00, "output_per_m": 15.00},
    "claude-opus-4-8": {"input_per_m": 15.00, "output_per_m": 75.00},
    "claude-haiku-4-5": {"input_per_m": 0.25, "output_per_m": 1.25},
    "gpt-4o": {"input_per_m": 2.50, "output_per_m": 10.00},
}

@dataclass
class LLMCallCost:
    model: str
    input_tokens: int
    output_tokens: int
    input_cost: float = 0.0
    output_cost: float = 0.0
    total_cost: float = 0.0

    def __post_init__(self):
        pricing = MODEL_PRICING.get(self.model, {"input_per_m": 3.0, "output_per_m": 15.0})
        self.input_cost = self.input_tokens / 1_000_000 * pricing["input_per_m"]
        self.output_cost = self.output_tokens / 1_000_000 * pricing["output_per_m"]
        self.total_cost = round(self.input_cost + self.output_cost, 6)

class SessionCostTracker:
    def __init__(self):
        self.llm_calls: list[LLMCallCost] = []
        self.tool_costs: list[dict] = []
        self.session_start = datetime.now()

    def record_llm_call(self, model: str, input_tokens: int, output_tokens: int):
        self.llm_calls.append(LLMCallCost(model, input_tokens, output_tokens))

    def record_tool_cost(self, tool_name: str, cost_usd: float, duration_ms: int):
        self.tool_costs.append({
            "tool": tool_name,
            "cost": cost_usd,
            "duration_ms": duration_ms,
        })

    def summary(self) -> dict:
        total_llm_cost = sum(c.total_cost for c in self.llm_calls)
        total_tool_cost = sum(t["cost"] for t in self.tool_costs)
        total_input_tokens = sum(c.input_tokens for c in self.llm_calls)
        total_output_tokens = sum(c.output_tokens for c in self.llm_calls)
        total_tool_duration = sum(t["duration_ms"] for t in self.tool_costs)

        return {
            "llm_cost_usd": round(total_llm_cost, 4),
            "tool_cost_usd": round(total_tool_cost, 4),
            "total_cost_usd": round(total_llm_cost + total_tool_cost, 4),
            "total_input_tokens": total_input_tokens,
            "total_output_tokens": total_output_tokens,
            "total_tokens": total_input_tokens + total_output_tokens,
            "total_tool_duration_ms": total_tool_duration,
            "llm_calls_made": len(self.llm_calls),
            "tool_calls_made": len(self.tool_costs),
        }
```

### 2. Latency Breakdown Instrumentation

Measure latency at each stage of the pipeline:

```python
# latency_profiler.py
import time
from dataclasses import dataclass, field
from typing import List

@dataclass
class LatencySpan:
    stage: str
    start_ms: float
    end_ms: float
    duration_ms: float = 0.0

    def __post_init__(self):
        self.duration_ms = round(self.end_ms - self.start_ms, 1)

class LatencyProfiler:
    def __init__(self):
        self.spans: List[LatencySpan] = []
        self._start = time.time() * 1000  # ms since epoch

    def begin_stage(self, stage: str) -> float:
        """Mark the start of a pipeline stage."""
        now = time.time() * 1000
        self.spans.append(LatencySpan(stage=stage, start_ms=now, end_ms=0))
        return now

    def end_stage(self, stage: str):
        """Mark the end of the most recent instance of a stage."""
        now = time.time() * 1000
        for span in reversed(self.spans):
            if span.stage == stage and span.end_ms == 0:
                span.end_ms = now
                span.duration_ms = round(now - span.start_ms, 1)
                return

    def breakdown(self) -> dict:
        """Return a latency breakdown by stage."""
        stages = {}
        for span in self.spans:
            if span.duration_ms > 0:
                stages[span.stage] = stages.get(span.stage, 0) + span.duration_ms
        return stages

    def total_ms(self) -> float:
        return round(time.time() * 1000 - self._start, 1)
```

Usage:

```python
profiler = LatencyProfiler()

profiler.begin_stage("prompt_construction")
system_prompt = build_prompt(user_input)
profiler.end_stage("prompt_construction")

profiler.begin_stage("llm_inference")
response = await llm.generate(system_prompt + user_input)
profiler.end_stage("llm_inference")

for tool_call in response.tool_calls:
    profiler.begin_stage(f"tool:{tool_call.name}")
    result = await execute_tool(tool_call)
    profiler.end_stage(f"tool:{tool_call.name}")

print(profiler.breakdown())
# {'prompt_construction': 12.3, 'llm_inference': 5432.1, 'tool:search_web': 843.2}
```

### 3. Realistic Cost Calculation Example

Here is a realistic cost estimate for a typical agent session:

```python
def estimate_session_cost(tool_calls: int, model: str = "claude-sonnet-4-6") -> dict:
    """
    Estimate cost for a typical agent session based on parameter counts.
    
    Assumptions:
    - System prompt: ~2000 tokens
    - Each LLM round: 2000 input (system + prior context) + 500 output tokens
    - N tool calls means N+1 LLM rounds (initial + N follow-ups)
    """
    rounds = tool_calls + 1  # initial response + one per tool call
    input_per_round = 2000 + (1000 * tool_calls)  # context grows with each round
    output_per_round = 500

    total_input = sum(input_per_round for _ in range(rounds))
    total_output = output_per_round * rounds

    pricing = MODEL_PRICING[model]
    input_cost = total_input / 1_000_000 * pricing["input_per_m"]
    output_cost = total_output / 1_000_000 * pricing["output_per_m"]

    return {
        "model": model,
        "llm_rounds": rounds,
        "tool_calls": tool_calls,
        "total_input_tokens": total_input,
        "total_output_tokens": total_output,
        "input_cost_usd": round(input_cost, 4),
        "output_cost_usd": round(output_cost, 4),
        "total_cost_usd": round(input_cost + output_cost, 4),
    }

# Example: a session with 6 tool calls using Sonnet
print(estimate_session_cost(tool_calls=6))
# {'total_input_tokens': 42000, 'total_output_tokens': 3500,
#  'input_cost_usd': 0.126, 'output_cost_usd': 0.0525, 'total_cost_usd': 0.1785}

# Same session using Opus:
print(estimate_session_cost(tool_calls=6, model="claude-opus-4-8"))
# {'total_input_tokens': 42000, 'total_output_tokens': 3500,
#  'input_cost_usd': 0.63, 'output_cost_usd': 0.2625, 'total_cost_usd': 0.8925}
```

A session with 6 tool calls costs about $0.18 on Sonnet and $0.89 on Opus. At 10,000 sessions per month, that is $1,785/month on Sonnet vs $8,925/month on Opus — a 5x difference from model choice alone.

### 4. Setting Cost and Latency Budgets

Define budget thresholds and alert when exceeded:

```python
# budgets.py
class AgentBudget:
    def __init__(self):
        self.max_cost_per_session: float = 0.50      # $0.50 max per session
        self.max_latency_per_session_ms: float = 60000  # 60 seconds max
        self.max_tool_calls_per_session: int = 15
        self.max_cost_per_day_usd: float = 50.00       # $50 daily budget
        self.daily_spend: float = 0.0

    def check_session(self, cost_summary: dict, latency_ms: float, tool_calls: int) -> list:
        """Check a session against budget limits. Returns list of violation messages."""
        violations = []

        if cost_summary["total_cost_usd"] > self.max_cost_per_session:
            violations.append(
                f"Cost over budget: ${cost_summary['total_cost_usd']:.4f} "
                f"(limit: ${self.max_cost_per_session:.2f})"
            )

        if latency_ms > self.max_latency_per_session_ms:
            violations.append(
                f"Latency over budget: {latency_ms:.0f}ms "
                f"(limit: {self.max_latency_per_session_ms}ms)"
            )

        if tool_calls > self.max_tool_calls_per_session:
            violations.append(
                f"Tool calls over budget: {tool_calls} "
                f"(limit: {self.max_tool_calls_per_session})"
            )

        self.daily_spend += cost_summary["total_cost_usd"]
        if self.daily_spend > self.max_cost_per_day_usd:
            violations.append(
                f"Daily spend over budget: ${self.daily_spend:.2f} "
                f"(limit: ${self.max_cost_per_day_usd:.2f})"
            )

        return violations
```

### 5. Building an Optimization Dashboard

Track the highest-impact optimization opportunities:

```python
# optimization_dashboard.py
def analyze_optimization_opportunities(session_logs: list) -> dict:
    """Analyze logs to find the biggest cost and latency optimization opportunities."""
    total_sessions = len(session_logs)
    if total_sessions == 0:
        return {}

    total_cost = sum(s["cost_usd"] for s in session_logs)
    total_latency = sum(s["latency_ms"] for s in session_logs)
    total_tool_calls = sum(s["tool_calls"] for s in session_logs)

    # Find sessions in the top 10% by cost
    sorted_by_cost = sorted(session_logs, key=lambda s: s["cost_usd"], reverse=True)
    top_10_pct = max(1, total_sessions // 10)
    expensive_cost = sum(s["cost_usd"] for s in sorted_by_cost[:top_10_pct])

    return {
        "total_sessions": total_sessions,
        "total_cost_usd": round(total_cost, 2),
        "avg_cost_per_session": round(total_cost / total_sessions, 4),
        "avg_latency_ms": round(total_latency / total_sessions, 0),
        "avg_tool_calls": round(total_tool_calls / total_sessions, 1),
        "p95_latency_ms": sorted(session_logs, key=lambda s: s["latency_ms"])[
            int(total_sessions * 0.95)
        ]["latency_ms"] if total_sessions > 1 else 0,
        "top_10_pct_cost_usd": round(expensive_cost, 2),
        "top_10_pct_share": round(expensive_cost / total_cost * 100, 1),
        "estimated_monthly_cost": round(total_cost * 30, 2),
    }
```

## How It Works Internally

Cost tracking works by intercepting every LLM API call and recording the token counts from the API response. Most LLM providers return `input_tokens` and `output_tokens` in the response metadata. Multiply by the model's per-token pricing to get the cost.

Latency tracking uses wall-clock timestamps at the start and end of each pipeline stage. The key metric is not average latency (which is pulled up by outliers) but p95 latency — the value below which 95% of sessions fall. A p95 of 30 seconds means 1 in 20 sessions takes longer than 30 seconds, which is a better target for optimization.

Tool call count is the primary cost and latency driver. Each tool call adds one full LLM round-trip (2-10 seconds) plus the tool execution time. Reducing tool calls from 8 to 4 typically cuts both cost and latency by 40-50%.
