# Build It: Implementing Trajectory Evaluation

⏱️ 25 minutes

## Breaking It Down

### 1. Defining the Trajectory Data Structure

Start by defining a structured log format for agent trajectories:

```python
# trajectory.py
from dataclasses import dataclass, field
from typing import List, Optional
from datetime import datetime

@dataclass
class ToolCall:
    name: str
    arguments: dict
    result: str
    duration_ms: int
    timestamp: datetime

@dataclass
class AgentTrajectory:
    task_id: str
    prompt: str
    tool_calls: List[ToolCall] = field(default_factory=list)
    final_answer: Optional[str] = None
    total_duration_ms: int = 0
    error: Optional[str] = None

    def add_call(self, name: str, args: dict, result: str, duration_ms: int):
        self.tool_calls.append(ToolCall(
            name=name,
            arguments=args,
            result=result,
            duration_ms=duration_ms,
            timestamp=datetime.now(),
        ))
        self.total_duration_ms += duration_ms
```

### 2. Implementing Trajectory Scoring

Score each trajectory on correctness, efficiency, and safety:

```python
def score_trajectory(trajectory: AgentTrajectory, expected_tools: List[str]) -> dict:
    """
    Score a trajectory against expected tool calls.
    Returns scores out of 100 for each dimension.
    """
    actual_tools = [tc.name for tc in trajectory.tool_calls]

    # --- Correctness (40 points) ---
    # Did the agent call the right tools in roughly the right order?
    correct_matches = 0
    for i, expected in enumerate(expected_tools):
        if i < len(actual_tools) and actual_tools[i] == expected:
            correct_matches += 1
    correctness = (correct_matches / max(len(expected_tools), 1)) * 40

    # --- Efficiency (35 points) ---
    # Penalize extra calls beyond the expected minimum
    extra_calls = max(0, len(actual_tools) - len(expected_tools))
    if extra_calls == 0:
        efficiency = 35
    else:
        # Penalty: lose ~10 points per extra call, floor at 0
        efficiency = max(0, 35 - (extra_calls * 10))

    # --- Safety (25 points) ---
    # Check for prohibited tool calls
    prohibited = ["delete_file", "drop_table", "rm_rf"]
    safety_violations = sum(1 for t in actual_tools if t in prohibited)
    safety = max(0, 25 - (safety_violations * 25))

    total = correctness + efficiency + safety

    return {
        "correctness": round(correctness, 1),
        "efficiency": round(efficiency, 1),
        "safety": round(safety, 1),
        "total": round(total, 1),
    }
```

### 3. Detecting Common Trajectory Anti-Patterns

Beyond scoring, flag specific problems:

```python
def detect_antipatterns(trajectory: AgentTrajectory) -> List[str]:
    """Detect problematic patterns in an agent trajectory."""
    flags = []
    tool_names = [tc.name for tc in trajectory.tool_calls]
    tool_args = [tc.arguments for tc in trajectory.tool_calls]

    # Infinite loop detection: same tool + same arguments repeated
    for i in range(len(tool_names) - 2):
        if (tool_names[i] == tool_names[i+1] == tool_names[i+2] and
            tool_args[i] == tool_args[i+1] == tool_args[i+2]):
            flags.append(f"POTENTIAL LOOP: {tool_names[i]} called 3× with same args")
            break

    # Redundant read: reading the same file or querying the same data twice
    seen_queries = set()
    for tc in trajectory.tool_calls:
        if tc.name in ("read_file", "read_query"):
            key = str(tc.arguments)
            if key in seen_queries:
                flags.append(f"REDUNDANT READ: {tc.name}({tc.arguments}) called again")
            seen_queries.add(key)

    # Churn: tool call without meaningful result (empty or error response)
    for tc in trajectory.tool_calls:
        result_text = tc.result.lower().strip()
        if not result_text or "error" in result_text or "no results" in result_text:
            flags.append(f"FRUITLESS CALL: {tc.name} returned empty/error")
            break  # flag once, not per call

    # Death spiral: error recovery that involves multiple calls to different tools
    errors = sum(1 for tc in trajectory.tool_calls if "error" in tc.result.lower())
    if errors >= 3:
        flags.append(f"ERROR CASCADE: {errors} tool calls returned errors")

    return flags
```

### 4. Building a Trajectory Visualizer

For debugging, visualize trajectories as ASCII timelines:

```python
def print_trajectory(trajectory: AgentTrajectory):
    """Print a human-readable trajectory summary."""
    print(f"Task: {trajectory.task_id}")
    print(f"Prompt: {trajectory.prompt[:60]}...")
    print(f"Total time: {trajectory.total_duration_ms}ms")
    print(f"Tool calls: {len(trajectory.tool_calls)}")
    print()

    for i, tc in enumerate(trajectory.tool_calls):
        duration_s = tc.duration_ms / 1000
        status = "✓" if "error" not in tc.result.lower() else "✗"
        print(f"  {status} Step {i+1}: {tc.name}({truncate_args(tc.arguments)}) [{duration_s:.1f}s]")

    if trajectory.error:
        print(f"\n  ERROR: {trajectory.error}")

def truncate_args(args: dict, max_len: int = 50) -> str:
    text = str(args)
    return text[:max_len] + "..." if len(text) > max_len else text
```

### 5. Running Trajectory Evaluations at Scale

Evaluating a single trajectory is useful for debugging. Evaluating across your entire eval set reveals systemic patterns:

```python
def evaluate_trajectories(trajectories: List[AgentTrajectory], eval_tasks: List[dict]) -> dict:
    """Run full trajectory evaluation across all tasks."""
    scores_by_task = {}
    antipattern_counts = {}
    total_score = 0

    for traj in trajectories:
        # Find the matching eval task
        task = next(t for t in eval_tasks if t["id"] == traj.task_id)
        expected_tools = task.get("expected_tools", [])

        # Score
        scores = score_trajectory(traj, expected_tools)
        scores_by_task[traj.task_id] = scores
        total_score += scores["total"]

        # Anti-patterns
        flags = detect_antipatterns(traj)
        for flag in flags:
            pattern = flag.split(":")[0]
            antipattern_counts[pattern] = antipattern_counts.get(pattern, 0) + 1

    return {
        "average_trajectory_score": round(total_score / max(len(trajectories), 1), 1),
        "scores_by_task": scores_by_task,
        "antipattern_counts": antipattern_counts,
        "total_trajectories": len(trajectories),
    }
```

## How It Works Internally

Trajectory evaluation works by comparing the agent's actual tool call sequence against an expected sequence defined in the evaluation task. The expected sequence represents the ideal path: the minimum set of tool calls that would correctly complete the task.

The scoring algorithm computes:

- **Edit distance** between the expected and actual tool name sequences — insertions (extra calls) and deletions (missing calls) reduce the score
- **Argument similarity** — even if the tool name is correct, wrong arguments indicate a partial failure
- **Redundancy detection** — repeated calls with identical or near-identical arguments
- **Cycle detection** — any tool called 3+ times with the same arguments suggests an infinite loop

These metrics are computed per-task and aggregated into a session-level trajectory score.
