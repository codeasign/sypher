# When to Use Multi-Agent — Build It

*Reading time: 14 minutes*

## Breaking It Down

### 1. The Decision Framework

Before you write any agent code, evaluate whether a multi-agent architecture is justified. The framework below scores a task on three dimensions.

```python
from dataclasses import dataclass


@dataclass
class ArchitectureScore:
    """Score a task's suitability for multi-agent architecture."""
    tool_count: int
    distinct_domains: list[str]
    quality_critical: bool  # Is incorrect output costly?
    latency_budget_ms: int
    tool_conflicts: bool    # Do tools have similar descriptions?
    context_overflow: bool  # Does the system prompt exceed 4000 tokens?

    def recommendation(self) -> str:
        """Return an architecture recommendation based on the score."""
        reasons = []

        if self.tool_count > 30:
            reasons.append("Tool count exceeds 30 — split by domain")

        if len(self.distinct_domains) > 1:
            reasons.append(f"Found {len(self.distinct_domains)} distinct domains")

        if self.tool_conflicts:
            reasons.append("Tool descriptions conflict — specialization needed")

        if self.context_overflow:
            reasons.append("System prompt too large — split reduces per-agent prompt size")

        if self.quality_critical and not reasons:
            reasons.append("Quality critical — consider adding a supervisor")

        if not reasons:
            return "Single agent recommended"

        if len(reasons) <= 2:
            return f"Multi-agent advisable: {'; '.join(reasons)}"

        return f"Multi-agent strongly recommended: {'; '.join(reasons)}"


# Example evaluations
scenarios = [
    ArchitectureScore(
        tool_count=12,
        distinct_domains=["customer_support"],
        quality_critical=False,
        latency_budget_ms=5000,
        tool_conflicts=False,
        context_overflow=False,
    ),
    ArchitectureScore(
        tool_count=45,
        distinct_domains=["billing", "knowledge_base", "account"],
        quality_critical=True,
        latency_budget_ms=10000,
        tool_conflicts=True,
        context_overflow=True,
    ),
]

for i, score in enumerate(scenarios, 1):
    print(f"Scenario {i}: {score.recommendation()}")
```

### 2. Complexity Cost Calculator

Every agent you add has a real cost. This calculator estimates the price of multi-agent vs. single-agent approaches.

```python
@dataclass
class CostCalculator:
    """Estimate the token and latency cost of agent architectures."""

    # Average token counts
    system_prompt_tokens: int = 1500
    tools_per_agent: int = 8
    agent_round_trips: int = 3

    def single_agent_cost(self, tool_count: int) -> dict:
        """Estimate cost for a single agent with all tools."""
        prompt_overhead = self.system_prompt_tokens + (tool_count * 80)
        per_turn = prompt_overhead + 500  # user input + output
        total_tokens = per_turn * self.agent_round_trips
        return {
            "architecture": "single",
            "total_tokens": total_tokens,
            "estimated_cost_usd": round(total_tokens * 3e-6, 4),
            "estimated_latency_ms": self.agent_round_trips * 2000,
        }

    def multi_agent_cost(self, tools_per_agent: list[int]) -> dict:
        """Estimate cost for N agents with distributed tools."""
        total_tokens = 0
        for count in tools_per_agent:
            prompt_overhead = self.system_prompt_tokens + (count * 80)
            per_turn = prompt_overhead + 500
            total_tokens += per_turn * self.agent_round_trips

        return {
            "architecture": f"multi ({len(tools_per_agent)} agents)",
            "total_tokens": total_tokens,
            "estimated_cost_usd": round(total_tokens * 3e-6, 4),
            "estimated_latency_ms": max(
                count * 2000 for count in tools_per_agent
            ),
        }


calc = CostCalculator()
single = calc.single_agent_cost(tool_count=45)
multi = calc.multi_agent_cost([15, 15, 15])

print(f"Single agent: {single['total_tokens']} tokens, ${single['estimated_cost_usd']}")
print(f"Multi-agent: {multi['total_tokens']} tokens, ${multi['estimated_cost_usd']}")
```

### 3. Breaking Up an Existing Agent

When you decide to split a monolithic agent, use this process:

```python
@dataclass
class AgentSplitter:
    """Analyze a monolithic agent and propose a split plan."""

    def analyze_tools(self, tool_definitions: list[dict]) -> dict[str, list]:
        """Group tools by domain to identify natural split points."""
        domains = {}

        for tool in tool_definitions:
            desc = tool.get("description", "").lower()
            name = tool.get("name", "")

            if any(kw in desc for kw in ["billing", "payment", "invoice", "refund"]):
                domains.setdefault("billing", []).append(name)
            elif any(kw in desc for kw in ["search", "query", "knowledge", "faq"]):
                domains.setdefault("knowledge", []).append(name)
            elif any(kw in desc for kw in ["profile", "account", "user", "team"]):
                domains.setdefault("account", []).append(name)
            else:
                domains.setdefault("general", []).append(name)

        return domains

    def split_report(self, tool_definitions: list[dict]) -> str:
        """Generate a report recommending how to split the agent."""
        domains = self.analyze_tools(tool_definitions)
        lines = ["## Agent Split Analysis"]
        lines.append(f"Total tools: {len(tool_definitions)}")

        for domain, tools in sorted(domains.items()):
            lines.append(f"\n### {domain.capitalize()} Domain ({len(tools)} tools)")
            for t in tools:
                lines.append(f"  - {t}")

        return "\n".join(lines)


# Example: analyze a monolithic customer support agent
tools = [
    {"name": "lookup_invoice", "description": "Look up a customer invoice"},
    {"name": "process_refund", "description": "Issue a refund to a customer"},
    {"name": "search_kb", "description": "Search the knowledge base for articles"},
    {"name": "check_order_status", "description": "Check the status of an order"},
    {"name": "update_profile", "description": "Update customer profile"},
    {"name": "escalate_ticket", "description": "Escalate a support ticket"},
]

splitter = AgentSplitter()
report = splitter.split_report(tools)
print(report)
```

## How It Works Internally

The decision framework works because LLMs have a finite "decision bandwidth" per call. As the number of tools and instructions grows, the model's attention is distributed across more options, reducing the probability of picking the correct one on any given turn. By splitting tools across agents, you increase the signal-to-noise ratio for each agent, improving per-call accuracy at the cost of additional calls. The trade-off is favorable when the accuracy improvement outweighs the latency and cost increase.

## Build Something Real

Build a system that evaluates itself. Given a task description and tool list, it recommends an architecture and optionally auto-generates the multi-agent split.

```python
@dataclass
class ArchitectureAdvisor:
    """Self-evaluating system that recommends its own architecture."""

    def recommend(self, task: str, tools: list[dict],
                  quality_sensitive: bool = False) -> dict:
        """Analyze a task and recommend an architecture."""
        tool_count = len(tools)

        # Extract domains from tool descriptions
        domains_found = set()
        for t in tools:
            desc = t.get("description", "").lower()
            for domain in ["billing", "knowledge", "account", "data",
                           "search", "admin", "communication"]:
                if domain in desc:
                    domains_found.add(domain)

        # Decision logic
        if tool_count <= 15 and len(domains_found) <= 1:
            return {
                "recommendation": "single_agent",
                "rationale": f"Only {tool_count} tools in one domain — single agent is optimal.",
                "agents": 1,
            }

        if tool_count > 30:
            return {
                "recommendation": "multi_agent_specialist",
                "rationale": f"{tool_count} tools across {len(domains_found)} domains — split into specialist agents.",
                "agents": min(len(domains_found), 5),
                "domains": list(domains_found),
            }

        if quality_sensitive:
            return {
                "recommendation": "single_agent_with_supervisor",
                "rationale": "Quality-critical task — add a supervisor to a single agent.",
                "agents": 2,
            }

        return {
            "recommendation": "single_agent",
            "rationale": "Moderate tool count with simple domain separation.",
            "agents": 1,
        }


advisor = ArchitectureAdvisor()

cases = [
    ("Answer customer questions about billing and technical support",
     [{"name": "lookup_invoice", "description": "Look up billing invoice"},
      {"name": "search_kb", "description": "Search knowledge base for technical articles"},
      {"name": "check_service_status", "description": "Check if a service is operational"}],
     False),
    ("Full customer support pipeline with billing, knowledge, account, and admin",
     [{"name": t, "description": d} for t, d in [
         ("lookup_invoice", "Look up billing invoice"),
         ("process_refund", "Process billing refund"),
         ("search_kb", "Search knowledge base for articles"),
         ("update_profile", "Update customer account profile"),
         ("manage_team", "Manage team members in account"),
         ("escalate_ticket", "Escalate admin ticket"),
         ("run_report", "Run analytics report on data"),
     ]], True),
]

for task, tool_list, quality in cases:
    result = advisor.recommend(task, tool_list, quality)
    print(f"\nTask: {task[:50]}...")
    print(f"Recommendation: {result['recommendation']}")
    print(f"Rationale: {result['rationale']}")
```

This advisory system is itself a practical application of the decision framework: it uses a deterministic rule set rather than an LLM call because the decision criteria are well-defined and do not require reasoning.

## Performance Considerations

The most expensive thing you can do is add agents unnecessarily. Each additional agent multiplies your token consumption by the agent's round-trip count. Before adding a supervisor, check whether deterministic quality checks (word count, format validation, keyword presence) are sufficient. Before splitting into specialist agents, check whether better tool descriptions would fix the accuracy problem.
