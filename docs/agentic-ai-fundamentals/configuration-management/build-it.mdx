# Build It: Implementing Agent Configuration Management

⏱️ 25 minutes

## Breaking It Down

### 1. Layered Config with Pydantic Models

Pydantic provides validation, type coercion, and clear error messages for configuration values. Define a base configuration model that represents all configurable parameters for your agent:

```python
# config_schema.py
from pydantic import BaseModel, Field, field_validator
from typing import List, Optional
from enum import Enum


class LogLevel(str, Enum):
    DEBUG = "debug"
    INFO = "info"
    WARNING = "warning"
    ERROR = "error"


class ModelConfig(BaseModel):
    name: str = "claude-sonnet-4-6"
    temperature: float = Field(default=0.7, ge=0.0, le=1.0)
    max_tokens: int = Field(default=4096, ge=1, le=65536)
    system_prompt: str = "You are a helpful assistant."

    @field_validator("system_prompt")
    @classmethod
    def prompt_not_empty(cls, v: str) -> str:
        if not v.strip():
            raise ValueError("system_prompt must not be empty")
        return v


class ToolConfig(BaseModel):
    enabled_tools: List[str] = ["search_web", "read_file"]
    max_tool_calls_per_session: int = Field(default=25, ge=1)
    tool_timeout_seconds: int = Field(default=30, ge=1)


class MemoryConfig(BaseModel):
    type: str = "short_term"  # short_term, long_term, episodic
    max_context_tokens: int = Field(default=32000, ge=1000)
    persist_path: Optional[str] = None


class ObservabilityConfig(BaseModel):
    log_level: LogLevel = LogLevel.INFO
    tracing_enabled: bool = False
    cost_tracking_enabled: bool = True
    metrics_port: int = Field(default=9090, ge=1024, le=65535)


class AgentConfig(BaseModel):
    """Top-level configuration for an agent system."""

    environment: str = "development"
    model: ModelConfig = ModelConfig()
    tools: ToolConfig = ToolConfig()
    memory: MemoryConfig = MemoryConfig()
    observability: ObservabilityConfig = ObservabilityConfig()
```

Using Pydantic means invalid configuration is caught at startup with a clear error message rather than surfacing as mysterious runtime behavior. If someone sets temperature to 1.5 or max_tokens to zero, the validation error tells them exactly which field is wrong and what the allowed range is.

### 3. Layered Config Loader

The loader merges defaults, environment-specific settings, and environment variable overrides into a single validated configuration object:

```python
# config_loader.py
import os
import yaml
from pathlib import Path
from typing import Optional
from config_schema import AgentConfig


def load_yaml(path: Path) -> dict:
    """Load a YAML file and return its contents as a dictionary."""
    with open(path, "r") as f:
        return yaml.safe_load(f) or {}


def deep_merge(base: dict, override: dict) -> dict:
    """Recursively merge override into base. Lists are replaced, not merged."""
    result = base.copy()
    for key, value in override.items():
        if key in result and isinstance(result[key], dict) and isinstance(value, dict):
            result[key] = deep_merge(result[key], value)
        else:
            result[key] = value
    return result


class ConfigLoader:
    """Loads and resolves layered configuration."""

    def __init__(self, config_dir: str = "config"):
        self.config_dir = Path(config_dir)

    def load(self, environment: Optional[str] = None) -> AgentConfig:
        """Load configuration with layered resolution.

        Resolution order (later wins):
        1. Defaults (config/default.yaml)
        2. Environment file (config/{environment}.yaml)
        3. Environment variables (AGENT__* prefixed)
        """
        env = environment or os.environ.get("AGENT_ENV", "development")

        # Layer 1: defaults
        default_path = self.config_dir / "default.yaml"
        config_data = load_yaml(default_path) if default_path.exists() else {}

        # Layer 2: environment-specific
        env_path = self.config_dir / f"{env}.yaml"
        if env_path.exists():
            env_config = load_yaml(env_path)
            config_data = deep_merge(config_data, env_config)

        # Layer 3: environment variable overrides
        env_var_overrides = self._load_env_vars()
        if env_var_overrides:
            config_data = deep_merge(config_data, env_var_overrides)

        # Validate and return
        config_data["environment"] = env
        return AgentConfig(**config_data)

    def _load_env_vars(self) -> dict:
        """Load AGENT__* environment variables as nested config overrides.

        AGENT__MODEL__TEMPERATURE=0.5 becomes {"model": {"temperature": 0.5}}
        """
        result = {}
        prefix = "AGENT__"
        for key, value in os.environ.items():
            if key.startswith(prefix):
                parts = key[len(prefix) :].lower().split("__")
                current = result
                for part in parts[:-1]:
                    current = current.setdefault(part, {})
                # Cast string value to appropriate Python type
                raw: str = value
                if raw.lower() in ("true", "false"):
                    current[parts[-1]] = raw.lower() == "true"
                else:
                    try:
                        if "." in raw:
                            current[parts[-1]] = float(raw)
                        else:
                            current[parts[-1]] = int(raw)
                    except ValueError:
                        current[parts[-1]] = raw
        return result
```

Usage:

```python
loader = ConfigLoader(config_dir="config")

# Load development config
dev_config = loader.load("development")
print(dev_config.model.name)  # claude-sonnet-4-6

# Load production config
prod_config = loader.load("production")
print(prod_config.model.name)  # claude-opus-4-8
print(prod_config.observability.tracing_enabled)  # True
```

The `AGENT_ENV` environment variable controls which environment file is loaded. This lets the deployment platform control the environment without modifying code or config files.

### 4. Secrets Management with Environment Variables

Secrets must never be stored in configuration files. Use environment variables for simple deployments and a secrets vault for production:

```python
# secrets_manager.py
import os
from typing import Optional


class SecretsManager:
    """Resolves secrets from environment variables."""

    def get(self, key: str, required: bool = True) -> Optional[str]:
        """Get a secret value. Raises an error if required and missing."""
        value = os.environ.get(key)
        if value:
            return value

        if required:
            raise RuntimeError(
                f"Required secret '{key}' not found in environment variables"
            )
        return None

    def inject_into_config(self, config: dict, secret_map: dict) -> dict:
        """Replace placeholder values in config with real secrets.

        secret_map maps config paths to environment variable names:
        {"model.api_key": "ANTHROPIC_API_KEY"}
        """
        result = config.copy()
        for config_path, env_var in secret_map.items():
            parts = config_path.split(".")
            current = result
            for part in parts[:-1]:
                if part not in current:
                    current[part] = {}
                current = current[part]
            # Only overwrite if the env var is set
            value = os.environ.get(env_var)
            if value:
                current[parts[-1]] = value
        return result
```

Define a secret mapping alongside each environment config:

```yaml
# config/production.yaml
# ... (the normal config values)
secret_map:
  model.api_key: ANTHROPIC_API_KEY
  memory.connection_string: MEMORY_DB_URL
  observability.api_key: OBSERVABILITY_API_KEY
```

Production secrets live in the environment or a vault, never in configuration files. The environment file only references the environment variable names.

### 5. Runtime Config Reload

Agent systems that run for hours or days need the ability to reload configuration without restarting. Implement a reload mechanism using file modification time polling:

```python
# config_reloader.py
import time
import threading
from pathlib import Path
from typing import Callable, Optional
from config_loader import ConfigLoader
from config_schema import AgentConfig


class ConfigReloader:
    """Watches config files for changes and triggers reload callbacks."""

    def __init__(self, config_dir: str, environment: str):
        self.loader = ConfigLoader(config_dir)
        self.environment = environment
        self.current_config: AgentConfig = self.loader.load(environment)
        self._watched_files: set[Path] = set()
        self._mtimes: dict[Path, float] = {}
        self._callbacks: list[Callable[[AgentConfig], None]] = []
        self._running = False
        self._thread: Optional[threading.Thread] = None

    def on_reload(self, callback: Callable[[AgentConfig], None]):
        """Register a callback to run when config is reloaded."""
        self._callbacks.append(callback)

    def _check_for_changes(self):
        """Check if any watched files have changed since last check."""
        changes_detected = False
        for path in self._watched_files:
            if path.exists():
                mtime = path.stat().st_mtime
                if path in self._mtimes and mtime > self._mtimes[path]:
                    changes_detected = True
                self._mtimes[path] = mtime

        if changes_detected:
            new_config = self.loader.load(self.environment)
            self.current_config = new_config
            for callback in self._callbacks:
                try:
                    callback(new_config)
                except Exception as e:
                    print(f"Config reload callback failed: {e}")

    def start(self, interval_seconds: int = 30):
        """Start polling config files for changes."""
        config_dir = self.loader.config_dir
        self._watched_files = set(config_dir.glob("*.yaml"))
        for path in self._watched_files:
            if path.exists():
                self._mtimes[path] = path.stat().st_mtime

        self._running = True

        def poll_loop():
            while self._running:
                self._check_for_changes()
                time.sleep(interval_seconds)

        self._thread = threading.Thread(target=poll_loop, daemon=True)
        self._thread.start()

    def stop(self):
        """Stop the polling loop."""
        self._running = False
        if self._thread:
            self._thread.join(timeout=5)
```

Usage in an agent application:

```python
reloader = ConfigReloader(config_dir="config", environment="production")
reloader.start(interval_seconds=30)


def on_config_update(new_config: AgentConfig):
    print(f"Config reloaded. Model: {new_config.model.name}")
    # Re-initialize agent components with new config
    agent.update_model(new_config.model)
    agent.update_tools(new_config.tools)


reloader.on_reload(on_config_update)

# The agent now picks up configuration changes within 30 seconds
# without needing a restart.
```

### 6. Feature Flags for Agent Behaviors

Feature flags let you toggle agent capabilities without redeployment. They integrate naturally with the config system:

```python
# feature_flags.py
from pydantic import BaseModel, Field


class FeatureFlags(BaseModel):
    """Feature flags control agent behavior at runtime."""

    enable_experimental_search: bool = False
    use_reasoning_agent: bool = False
    enable_parallel_tool_calls: bool = True
    enable_human_approval_gate: bool = False
    max_retry_on_tool_error: int = Field(default=3, ge=0)
    log_full_trajectory: bool = False
```

Add feature flags to the top-level `AgentConfig`:

```python
class AgentConfig(BaseModel):
    environment: str = "development"
    model: ModelConfig = ModelConfig()
    tools: ToolConfig = ToolConfig()
    memory: MemoryConfig = MemoryConfig()
    observability: ObservabilityConfig = ObservabilityConfig()
    feature_flags: FeatureFlags = FeatureFlags()
```

Usage in agent logic:

```python
def execute_tool_call(tool_name: str, agent_config: AgentConfig):
    if tool_name == "experimental_search":
        if not agent_config.feature_flags.enable_experimental_search:
            return {"error": "experimental_search is disabled"}
        return execute_experimental_search()

    if agent_config.feature_flags.enable_parallel_tool_calls:
        # Execute independent tool calls in parallel
        return execute_parallel(tool_name)

    return execute_sequential(tool_name)
```

Feature flag values can be set per environment in YAML:

```yaml
# config/production.yaml
feature_flags:
  enable_experimental_search: false
  use_reasoning_agent: true
  enable_parallel_tool_calls: true
  enable_human_approval_gate: true
```

```yaml
# config/development.yaml
feature_flags:
  enable_experimental_search: true
  use_reasoning_agent: false
  enable_parallel_tool_calls: false
  enable_human_approval_gate: false
```

This pattern lets you gradually roll out features. Enable `experimental_search` in development first, test it, then enable it in staging, and finally turn it on in production — all without changing code.

## Performance Considerations

Config reload polling introduces minimal overhead. Checking file modification times for a handful of YAML files takes under a millisecond. At a 30-second polling interval, the CPU impact is negligible.

The more important consideration is the cost of regenerating components when configuration changes. If your agent has expensive initialization steps (loading embedding models into memory, establishing database connections, compiling tool definitions), update those components lazily rather than eagerly on every config reload. Use a version counter:

```python
class ConfigAwareComponent:
    def __init__(self):
        self._config_version = 0
        self._cache: dict = {}

    def update_config(self, new_config: AgentConfig):
        self._config_version += 1

    def get_or_recompute(self, key: str, config: AgentConfig, compute_fn):
        cached = self._cache.get(key)
        if cached and cached["version"] == self._config_version:
            return cached["value"]

        value = compute_fn(config)
        self._cache[key] = {"value": value, "version": self._config_version}
        return value
```

This pattern ensures that components only recompute when configuration actually changes, not every time the reloader polls.

## Security Considerations

Secrets must be handled with care at every stage of the configuration lifecycle:

- **Never log configuration values** — a debug log that prints the full config object can leak API keys and database passwords into your log aggregation system. Implement a sanitized representation:

```python
def safe_config_dump(config: AgentConfig) -> dict:
    """Return a configuration dict with secrets redacted."""
    data = config.model_dump()
    # Redact known secret fields
    secrets_patterns = ["api_key", "password", "token", "secret", "connection_string"]
    for section, values in data.items():
        if isinstance(values, dict):
            for key in list(values.keys()):
                if any(pattern in key.lower() for pattern in secrets_patterns):
                    values[key] = "***REDACTED***"
    return data
```

- **Use environment-specific access controls** — production secrets should be accessible only to the production deployment service account. Developers should not have access to production secrets. If using a vault, enforce path-based policies:

```hcl
path "secret/agent/production/*" {
  capabilities = ["read"]
  allowed_entities = ["prod-agent-service-account"]
}

path "secret/agent/development/*" {
  capabilities = ["create", "read", "update", "delete", "list"]
  allowed_entities = ["developer-role"]
}
```

- **Use short-lived credentials** — prefer temporary tokens with automatic rotation over static API keys. Cloud secret managers support automatic credential rotation. Vault's dynamic secrets engine can generate credentials that expire after a configurable TTL.

- **Validate secrets at startup** — do not discover a missing API key when your agent makes its first API call. Validate all required secrets during configuration loading:

```python
def validate_secrets(config: AgentConfig, required_secrets: list[str]):
    missing = [s for s in required_secrets if s not in os.environ]
    if missing:
        raise RuntimeError(f"Missing required secrets: {', '.join(missing)}")
```

## Build Something Real

Combine all the pieces into a single production-ready configuration system:

```python
# agent_config.py  -- Complete config system
import os
import yaml
import time
import threading
from pathlib import Path
from typing import Optional, Callable
from pydantic import BaseModel, Field
from enum import Enum


# ── Configuration Models ──────────────────────────────────────────

class LogLevel(str, Enum):
    DEBUG = "debug"
    INFO = "info"
    WARNING = "warning"
    ERROR = "error"


class ModelConfig(BaseModel):
    name: str = Field(default="claude-sonnet-4-6")
    temperature: float = Field(default=0.7, ge=0.0, le=1.0)
    max_tokens: int = Field(default=4096, ge=1, le=65536)
    system_prompt: str = Field(default="You are a helpful assistant.")


class ToolConfig(BaseModel):
    enabled_tools: list = Field(default=["search_web", "read_file"])
    max_tool_calls_per_session: int = Field(default=25, ge=1)
    tool_timeout_seconds: int = Field(default=30, ge=1)


class MemoryConfig(BaseModel):
    type: str = Field(default="short_term")
    max_context_tokens: int = Field(default=32000, ge=1000)


class ObservabilityConfig(BaseModel):
    log_level: LogLevel = Field(default=LogLevel.INFO)
    tracing_enabled: bool = Field(default=False)


class FeatureFlags(BaseModel):
    enable_experimental_search: bool = False
    use_reasoning_agent: bool = False
    enable_parallel_tool_calls: bool = True


class AgentConfig(BaseModel):
    environment: str = Field(default="development")
    model: ModelConfig = Field(default=ModelConfig())
    tools: ToolConfig = Field(default=ToolConfig())
    memory: MemoryConfig = Field(default=MemoryConfig())
    observability: ObservabilityConfig = Field(default=ObservabilityConfig())
    feature_flags: FeatureFlags = Field(default=FeatureFlags())
    secret_map: dict = Field(default={})


# ── YAML Loader ───────────────────────────────────────────────────

def load_yaml_file(path: Path) -> dict:
    with open(path, "r") as f:
        return yaml.safe_load(f) or {}


def deep_merge(base: dict, override: dict) -> dict:
    result = base.copy()
    for key, value in override.items():
        if key in result and isinstance(result[key], dict) and isinstance(value, dict):
            result[key] = deep_merge(result[key], value)
        else:
            result[key] = value
    return result


# ── Secrets Manager ───────────────────────────────────────────────

class SecretsManager:
    def get(self, key: str, required: bool = True) -> Optional[str]:
        value = os.environ.get(key)
        if value:
            return value
        if required:
            raise RuntimeError(f"Required secret '{key}' not found")
        return None

    def inject(self, config: dict, secret_map: dict) -> dict:
        result = config.copy()
        for config_key, env_var in secret_map.items():
            parts = config_key.split(".")
            current = result
            for part in parts[:-1]:
                current = current.setdefault(part, {})
            current[parts[-1]] = self.get(env_var)
        return result


# ── Config Manager ────────────────────────────────────────────────

class ConfigManager:
    """Complete configuration manager with layered loading, secrets, and hot-reload."""

    def __init__(self, config_dir: str = "config"):
        self.config_dir = Path(config_dir)
        self.secrets = SecretsManager()
        self.current_config: Optional[AgentConfig] = None
        self._reloader: Optional["ConfigReloader"] = None
        self._load_callbacks: list[Callable] = []

    def on_load(self, callback: Callable):
        self._load_callbacks.append(callback)

    def load(self, environment: Optional[str] = None) -> AgentConfig:
        env = environment or os.environ.get("AGENT_ENV", "development")

        # Layer 1: defaults
        config_data = {}
        default_path = self.config_dir / "default.yaml"
        if default_path.exists():
            config_data = load_yaml_file(default_path)

        # Layer 2: environment override
        env_path = self.config_dir / f"{env}.yaml"
        if env_path.exists():
            env_data = load_yaml_file(env_path)
            config_data = deep_merge(config_data, env_data)

        # Layer 3: secrets injection
        secret_map = config_data.pop("secret_map", {})
        config_data = self.secrets.inject(config_data, secret_map)

        # Layer 4: environment variable overrides
        config_data = deep_merge(config_data, self._load_env_overrides())

        config_data["environment"] = env
        self.current_config = AgentConfig(**config_data)

        for cb in self._load_callbacks:
            cb(self.current_config)
        return self.current_config

    def _load_env_overrides(self) -> dict:
        result = {}
        prefix = "AGENT__"
        for key, value in os.environ.items():
            if key.startswith(prefix):
                parts = key[len(prefix) :].lower().split("__")
                current = result
                for part in parts[:-1]:
                    current = current.setdefault(part, {})
                raw: str = value
                if raw.lower() in ("true", "false"):
                    current[parts[-1]] = raw.lower() == "true"
                else:
                    try:
                        current[parts[-1]] = int(raw) if "." not in raw else float(raw)
                    except ValueError:
                        current[parts[-1]] = raw
        return result

    def start_hot_reload(
        self,
        interval: int = 30,
        callback: Optional[Callable] = None,
    ):
        self._reloader = ConfigReloader(self, interval, callback)
        self._reloader.start()

    def stop_hot_reload(self):
        if self._reloader:
            self._reloader.stop()


class ConfigReloader:
    def __init__(
        self,
        manager: ConfigManager,
        interval: int,
        callback: Optional[Callable] = None,
    ):
        self.manager = manager
        self.interval = interval
        self.callback = callback
        self._running = False
        self._thread: Optional[threading.Thread] = None
        self._mtime_cache: dict[str, float] = {}

    def _scan_mtimes(self):
        return {
            str(p): p.stat().st_mtime
            for p in self.manager.config_dir.glob("*.yaml")
            if p.exists()
        }

    def start(self):
        self._mtime_cache = self._scan_mtimes()
        self._running = True

        def poll():
            while self._running:
                time.sleep(self.interval)
                current = self._scan_mtimes()
                changed = any(
                    current.get(p, 0) > self._mtime_cache.get(p, 0)
                    for p in set(current) | set(self._mtime_cache)
                )
                if changed:
                    env = (
                        self.manager.current_config.environment
                        if self.manager.current_config
                        else "development"
                    )
                    new_config = self.manager.load(env)
                    self._mtime_cache = current
                    if self.callback:
                        self.callback(new_config)

        self._thread = threading.Thread(target=poll, daemon=True)
        self._thread.start()

    def stop(self):
        self._running = False
        if self._thread:
            self._thread.join(timeout=5)


# ── Usage ─────────────────────────────────────────────────────────

if __name__ == "__main__":
    config_manager = ConfigManager(config_dir="config")
    config = config_manager.load("production")

    print(f"Environment: {config.environment}")
    print(f"Model: {config.model.name}")
    print(f"Experimental search: {config.feature_flags.enable_experimental_search}")

    def on_config_changed(new_config: AgentConfig):
        print(f"Config updated: model={new_config.model.name}")

    config_manager.start_hot_reload(interval=30, callback=on_config_changed)
```

This single module gives you:

- Validated configuration via Pydantic models
- Layered YAML override per environment
- Secrets injected from environment variables, never stored in config files
- Hot-reload that picks up file changes without restarting the process
- Feature flags that can be toggled per environment

To use it, create your `config/default.yaml` and `config/production.yaml` files, set `AGENT_ENV=production` in your deployment, set the required environment variables for secrets, and call `ConfigManager().load()` once at startup.
