# ReAct Pattern — Build It

*Reading time: 14 minutes*

## Breaking It Down

### 1. Thought-Action-Observation in One Turn

In the ReAct pattern, each turn produces a thought followed by an action. The model is prompted to externalize its reasoning before requesting a tool.

```python
import anthropic

client = anthropic.Anthropic()


def react_turn(context: list[dict]) -> tuple[str | None, str | None]:
    """
    Execute one ReAct turn.
    Returns (thought, action) where action may be a tool call or a final answer.
    """
    response = client.messages.create(
        model="claude-3-5-sonnet-20241022",
        max_tokens=500,
        temperature=0,
        system=(
            "You are a ReAct agent. For each turn, you must:\n"
            "1. State your thought (what you are thinking and why)\n"
            "2. Take an action (call a tool or give the final answer)\n\n"
            "Format:\n"
            "Thought: <your reasoning>\n"
            "Action: <tool name> with <parameters>\n"
            "-or-\n"
            "Thought: <your reasoning>\n"
            "Final Answer: <your answer>"
        ),
        messages=context,
        tools=[{
            "name": "search_knowledge",
            "description": "Search the knowledge base for information",
            "input_schema": {
                "type": "object",
                "properties": {
                    "query": {"type": "string", "description": "Search query"}
                },
                "required": ["query"]
            }
        }],
    )

    content = response.content[0]

    if content.type == "tool_use":
        # The thought is in the text block (if present) or inferred
        return "Tool requested", content
    else:
        return None, content.text


# Simple ReAct loop
messages = [{"role": "user", "content": "What is the capital of France and what is its population?"}]

for turn in range(5):
    thought, action = react_turn(messages)

    if action and not isinstance(action, str):
        # It is a tool call
        result = f"Paris is the capital of France. Population: 2.1 million (city proper)."
        messages.append({"role": "assistant", "content": action})
        messages.append({
            "role": "user",
            "content": [
                {"type": "tool_result", "tool_use_id": action.id, "content": result}
            ]
        })
        print(f"Turn {turn + 1}: Thought → Called tool, got result")
    else:
        print(f"Turn {turn + 1}: {action}")
        break
```

### 2. Structured ReAct with Tags

Use explicit XML tags to separate thought, action, and observation in the prompt.

```python
def structured_react(user_input: str) -> str:
    """ReAct with structured XML tags for the reasoning trace."""
    messages = [{"role": "user", "content": user_input}]
    tools = [{
        "name": "search_knowledge",
        "description": "Search the knowledge base",
        "input_schema": {
            "type": "object",
            "properties": {"query": {"type": "string"}},
            "required": ["query"]
        }
    }]

    for turn in range(8):
        response = client.messages.create(
            model="claude-3-5-sonnet-20241022",
            max_tokens=500,
            temperature=0,
            system=(
                "You are a reasoning agent. For each step:\n"
                "<thought>Explain what you know and what you need to find out.</thought>\n"
                "<action>Either call a tool or provide the final answer.</action>\n"
                "Do NOT repeat information already in the observation."
            ),
            messages=messages,
            tools=tools,
        )

        content = response.content[0]

        if content.type == "tool_use":
            # Simulate tool execution
            result = f"Information about '{content.input['query']}': [relevant data]"
            messages.append({"role": "assistant", "content": content})
            messages.append({
                "role": "user",
                "content": [
                    {"type": "tool_result", "tool_use_id": content.id, "content": result}
                ]
            })
        else:
            return content.text

    return "Max turns reached"


output = structured_react("Compare the population of France and Germany.")
print(output)
```

### 3. Extracting the Reasoning Trace

The reasoning trace (the thought steps) is valuable for debugging and auditing. Extract and log it.

```python
import re


class ReActTracer:
    """Extracts and logs the reasoning trace from a ReAct agent."""

    def __init__(self):
        self.trace: list[dict] = []

    def add_turn(self, thought: str | None, action_type: str, action_detail: str,
                 observation: str | None = None):
        self.trace.append({
            "thought": thought or "No explicit thought",
            "action_type": action_type,
            "action_detail": action_detail,
            "observation": observation,
        })

    def extract_thought(self, text: str) -> str | None:
        """Extract thought from XML tags or natural language."""
        # Try XML tags first
        match = re.search(r"<thought>(.*?)</thought>", text, re.DOTALL)
        if match:
            return match.group(1).strip()

        # Try "Thought:" prefix
        for line in text.split("\n"):
            if line.startswith("Thought:"):
                return line.replace("Thought:", "").strip()

        return None

    def print_trace(self):
        for i, t in enumerate(self.trace, 1):
            print(f"\n--- Turn {i} ---")
            print(f"Thought: {t['thought']}")
            print(f"Action: {t['action_type']} → {t['action_detail']}")
            if t['observation']:
                print(f"Observation: {t['observation'][:80]}...")


tracer = ReActTracer()
tracer.add_turn("I need to find the capital of France.",
                "tool_call", "search_knowledge('capital of France')",
                "Paris")
tracer.add_turn("Now I need the population.",
                "tool_call", "search_knowledge('population of France')",
                "67 million")
tracer.add_turn("I have both pieces of information.",
                "final_answer", "The capital is Paris and the population is 67 million.")
tracer.print_trace()
```

## How It Works Internally

The ReAct pattern works because the model's reasoning tokens serve as a scratch pad. When the model writes "Thought: I need to find X before I can answer Y," it implicitly commits to a plan. The reasoning tokens guide the next token predictions toward the correct tool call.

The explicit thought-action structure also helps with planning. By externalizing the reasoning, the model can reference its own earlier thoughts and avoid contradicting itself.

## Build Something Real

Build a full ReAct agent with structured traces, error handling, and cost tracking.

```python
import anthropic
import json
import re
import time

client = anthropic.Anthropic()


class ReActAgent:
    """Complete ReAct agent with structured reasoning traces."""

    def __init__(self):
        self.trace: list[dict] = []
        self.knowledge_base = {
            "python": "Python is a high-level programming language created by Guido van Rossum in 1991.",
            "javascript": "JavaScript is a scripting language created by Brendan Eich in 1995.",
            "typescript": "TypeScript is a typed superset of JavaScript created by Microsoft in 2012.",
        }
        self.tools = [{
            "name": "search_knowledge",
            "description": "Search the knowledge base for information about a programming language",
            "input_schema": {
                "type": "object",
                "properties": {
                    "language": {"type": "string", "description": "Programming language name"}
                },
                "required": ["language"]
            }
        }]

    def _execute(self, name: str, args: dict) -> str:
        if name == "search_knowledge":
            lang = args["language"].lower()
            for key, value in self.knowledge_base.items():
                if key in lang or lang in key:
                    return value
            return f"No information found about '{args['language']}'."
        return f"Unknown tool: {name}"

    def run(self, question: str) -> dict:
        """Run the ReAct agent and return the answer with full trace."""
        messages = [{"role": "user", "content": question}]
        start = time.time()
        total_input_tokens = 0
        total_output_tokens = 0

        for turn in range(10):
            turn_start = time.time()

            response = client.messages.create(
                model="claude-3-5-sonnet-20241022",
                max_tokens=500,
                temperature=0,
                system=(
                    "You are a ReAct agent. For each turn:\n"
                    "<thought>Your reasoning about what to do next.</thought>\n"
                    "<action>Either call a tool or state the final answer.</action>"
                ),
                messages=messages,
                tools=self.tools,
            )

            total_input_tokens += response.usage.input_tokens
            total_output_tokens += response.usage.output_tokens
            content = response.content[0]

            if content.type == "tool_use":
                thought = f"Calling {content.name} to find information"
                result = self._execute(content.name, content.input)

                self.trace.append({
                    "turn": turn + 1,
                    "thought": thought,
                    "action": f"{content.name}({content.input})",
                    "observation": result[:100],
                    "time_ms": round((time.time() - turn_start) * 1000),
                })

                messages.append({"role": "assistant", "content": content})
                messages.append({
                    "role": "user",
                    "content": [
                        {"type": "tool_result", "tool_use_id": content.id, "content": result}
                    ]
                })
            else:
                # Extract thought from the text
                thought = None
                match = re.search(r"<thought>(.*?)</thought>", content.text, re.DOTALL)
                if match:
                    thought = match.group(1).strip()

                self.trace.append({
                    "turn": turn + 1,
                    "thought": thought or "Final answer",
                    "action": "final_answer",
                    "observation": content.text[:200],
                    "time_ms": round((time.time() - turn_start) * 1000),
                })

                total_time = round((time.time() - start) * 1000)
                return {
                    "answer": content.text,
                    "trace": self.trace,
                    "total_turns": len(self.trace),
                    "total_time_ms": total_time,
                    "total_input_tokens": total_input_tokens,
                    "total_output_tokens": total_output_tokens,
                }


agent = ReActAgent()
result = agent.run("Compare Python and TypeScript. What are their origins and creators?")
print(f"Answer:\n{result['answer']}\n")
print(f"Trace ({result['total_turns']} turns, {result['total_time_ms']}ms):")
for t in result['trace']:
    print(f"  Turn {t['turn']}: Thought: {t['thought'][:60]}...")
    print(f"           Action: {t['action']}")
```

The ReAct trace gives you complete visibility into the agent's decision process. Each thought explains why the agent chose the next action, and each observation shows what the agent learned.

## Performance Considerations

ReAct adds token overhead because the model generates reasoning tokens on every turn. Measure the ratio of reasoning tokens to action tokens. If reasoning is consistently over 70% of the output, consider whether a smaller model could handle the task.

## Security Considerations

The reasoning trace is visible to the user in some implementations. Ensure the trace does not expose sensitive information (API keys, internal instructions, business logic) that should remain hidden.
