# Transports — Build It

*Reading time: 14 minutes*

## Breaking It Down

### 1. Stdio Transport — Server Side

The stdio transport is the simplest MCP transport. The server reads JSON-RPC messages from stdin and writes responses to stdout. Stderr is reserved for logging.

```python
import json
import sys
from typing import Callable


class StdioServerTransport:
    """MCP server transport over stdin/stdout."""

    def __init__(self, handler: Callable[[str], str | None]):
        self._handler = handler

    def run(self):
        """Read messages from stdin, dispatch, and write responses."""
        for line in sys.stdin:
            line = line.strip()
            if not line:
                continue

            response = self._handler(line)
            if response:
                print(response, flush=True)

    @staticmethod
    def send_response(response: str):
        """Write a response line to stdout."""
        print(response, flush=True)

    @staticmethod
    def log(message: str):
        """Write a log message to stderr (not stdout)."""
        print(message, file=sys.stderr, flush=True)


# Example usage:
def sample_handler(raw: str) -> str | None:
    """Simple handler that echoes back (for illustration)."""
    msg = json.loads(raw)
    if msg.get("method") == "ping":
        return json.dumps({
            "jsonrpc": "2.0",
            "id": msg.get("id"),
            "result": {},
        })
    return None
```

### 2. Stdio Transport — Client Side

The client spawns the server process and communicates through its stdin/stdout pipes.

```python
import subprocess


class StdioClientTransport:
    """MCP client transport over a child process."""

    def __init__(self, server_command: list[str]):
        self._proc = subprocess.Popen(
            server_command,
            stdin=subprocess.PIPE,
            stdout=subprocess.PIPE,
            stderr=subprocess.PIPE,
            text=True,
        )

    def send_message(self, message: str) -> str:
        """Send a JSON-RPC message and read one response."""
        self._proc.stdin.write(message + "\n")
        self._proc.stdin.flush()
        return self._proc.stdout.readline().strip()

    def read_stderr(self) -> str:
        """Read any pending stderr output (server logs)."""
        return self._proc.stderr.read()

    def close(self):
        """Terminate the server process."""
        self._proc.terminate()
        self._proc.wait()

    @property
    def is_alive(self) -> bool:
        """Check if the server process is still running."""
        return self._proc.poll() is None
```

### 3. SSE Transport — Server Side

SSE transport uses HTTP. The server exposes an SSE endpoint for pushing events to the client and a POST endpoint for receiving messages.

```python
from http.server import HTTPServer, BaseHTTPRequestHandler
from urllib.parse import urlparse
import json


class SseServerTransport(BaseHTTPRequestHandler):
    """MCP server transport over SSE/HTTP."""

    # Shared state across handlers (simplified for illustration)
    sse_connections = []
    message_handler = None

    def do_GET(self):
        """Handle SSE stream connection."""
        if self.path == "/sse":
            self.send_response(200)
            self.send_header("Content-Type", "text/event-stream")
            self.send_header("Cache-Control", "no-cache")
            self.send_header("Connection", "keep-alive")
            self.end_headers()

            # Register this connection
            SseServerTransport.sse_connections.append(self)

            # Send endpoint info
            self.wfile.write(
                f"event: endpoint\ndata: /message\n\n".encode()
            )
            self.wfile.flush()

            # Keep connection open
            while True:
                # In production, use threading and a message queue
                import time
                time.sleep(30)
                self.wfile.write(b": keepalive\n\n")
                self.wfile.flush()

        else:
            self.send_response(404)
            self.end_headers()

    def do_POST(self):
        """Handle incoming JSON-RPC messages."""
        if self.path == "/message":
            content_length = int(self.headers.get("Content-Length", 0))
            body = self.rfile.read(content_length).decode()

            if SseServerTransport.message_handler:
                response = SseServerTransport.message_handler(body)
                if response:
                    self.send_response(200)
                    self.send_header("Content-Type", "application/json")
                    self.end_headers()
                    self.wfile.write(response.encode())

                    # Also push to SSE connections
                    for conn in SseServerTransport.sse_connections:
                        conn.wfile.write(
                            f"data: {response}\n\n".encode()
                        )
                        conn.wfile.flush()
                else:
                    self.send_response(202)
                    self.end_headers()
            else:
                self.send_response(500)
                self.end_headers()
        else:
            self.send_response(404)
            self.end_headers()

    def log_message(self, format, *args):
        """Suppress default HTTP server logging."""
        pass
```

### 4. SSE Transport — Client Side

The SSE client connects to the SSE endpoint and sends messages via HTTP POST.

```python
import json
import requests
import sseclient  # pip install sseclient-py
from threading import Thread


class SseClientTransport:
    """MCP client transport over SSE."""

    def __init__(self, base_url: str = "http://localhost:8000"):
        self._base_url = base_url
        self._sse_url = f"{base_url}/sse"
        self._message_url = None
        self._response_queue = []
        self._running = False

    def connect(self) -> str:
        """Connect to the SSE stream and get the message endpoint."""
        response = requests.get(self._sse_url, stream=True)
        client = sseclient.SSEClient(response)

        for event in client.events():
            if event.event == "endpoint":
                self._message_url = f"{self._base_url}{event.data}"
                break

        return self._message_url

    def send_message(self, message: str) -> str:
        """Send a JSON-RPC message via HTTP POST."""
        response = requests.post(
            self._message_url,
            data=message,
            headers={"Content-Type": "application/json"},
        )
        return response.text

    def close(self):
        """Close the SSE connection."""
        self._running = False
```

### 5. Transport-Agnostic Server

The key MCP design insight: the same server logic works with any transport. Build your handler once, then plug it into stdio or SSE.

```python
from dataclasses import dataclass
from typing import Callable


@dataclass
class TransportAgnosticServer:
    """MCP server that works with any transport."""
    _handler: Callable | None = None

    def set_handler(self, handler: Callable[[str], str | None]):
        """Set the message handler that processes JSON-RPC."""
        self._handler = handler

    def run_stdio(self):
        """Run over stdio transport."""
        if not self._handler:
            raise ValueError("Handler not set")
        transport = StdioServerTransport(self._handler)
        transport.run()

    def run_sse(self, host: str = "0.0.0.0", port: int = 8000):
        """Run over SSE transport."""
        if not self._handler:
            raise ValueError("Handler not set")
        SseServerTransport.message_handler = self._handler
        server = HTTPServer((host, port), SseServerTransport)
        print(f"MCP SSE server running on http://{host}:{port}")
        server.serve_forever()


# Build the shared handler
def create_handler(tools: dict) -> Callable:
    """Create a message handler with tool routing."""
    def handler(raw: str) -> str | None:
        try:
            msg = json.loads(raw)
        except json.JSONDecodeError:
            return None

        method = msg.get("method")
        msg_id = msg.get("id")

        if method == "tools/list":
            return json.dumps({
                "jsonrpc": "2.0", "id": msg_id,
                "result": {
                    "tools": [
                        {"name": n, "description": d.get("desc", ""),
                         "inputSchema": d.get("schema", {"type": "object",
                                                         "properties": {}})}
                        for n, d in tools.items()
                    ],
                },
            })

        if method == "tools/call":
            tool = tools.get(msg.get("params", {}).get("name"))
            if tool:
                result = tool["handler"](
                    **msg.get("params", {}).get("arguments", {})
                )
                return json.dumps({
                    "jsonrpc": "2.0", "id": msg_id,
                    "result": {
                        "content": [{"type": "text", "text": str(result)}]
                    },
                })

        return None

    return handler

# server.run_stdio()  # or server.run_sse()
print("Server logic written once — runs on any transport.")
```

## How It Works Internally

Stdio transport is synchronous and ordered: one request in, one response out. SSE transport is asynchronous: the server can push multiple events to the SSE stream, and the client sends requests independently over POST. The JSON-RPC `id` field is what correlates requests to responses in both cases, but SSE requires explicit ID tracking whereas stdio can rely on ordering.

## Performance Considerations

Stdio transport adds approximately 0.1-0.5ms per message (process pipe overhead). SSE transport adds 5-50ms per message (HTTP round trip, network latency). For latency-sensitive tools (file reads, simple computations), stdio is the clear choice. For tools that involve network calls themselves (querying a remote database), the transport overhead is negligible compared to the tool's own latency, so SSE is fine.

## Security Considerations

With stdio transport, the server runs as the same user as the client — it inherits the client's permissions. With SSE transport, the server runs as the HTTP server's user, which may have different permissions. Always ensure the server user has only the permissions it needs, regardless of transport.
