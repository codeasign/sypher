# Few-Shot Prompting — Build It

*Reading time: 12 minutes*

## Breaking It Down

### 1. Zero-Shot vs. Few-Shot

Start by seeing the difference between no examples and a few examples:

```python
import anthropic

client = anthropic.Anthropic()


def zero_shot_classify(text: str, categories: list[str]) -> str:
    """Classify without examples."""
    cats = ", ".join(categories)
    response = client.messages.create(
        model="claude-3-5-sonnet-20241022",
        max_tokens=10,
        temperature=0,
        system=f"Classify the text into one category: {cats}. Reply with only the category.",
        messages=[{"role": "user", "content": text}]
    )
    return response.content[0].text.strip()


def few_shot_classify(text: str, examples: list[tuple[str, str]]) -> str:
    """Classify using examples."""
    messages = []
    for input_text, label in examples:
        messages.append({"role": "user", "content": input_text})
        messages.append({"role": "assistant", "content": label})

    messages.append({"role": "user", "content": text})

    response = client.messages.create(
        model="claude-3-5-sonnet-20241022",
        max_tokens=10,
        temperature=0,
        system="Classify customer feedback into: BUG, FEATURE_REQUEST, or GENERAL.",
        messages=messages
    )
    return response.content[0].text.strip()


# Test with ambiguous text
test_text = "The search feature does not return results for partial matches."

print("Zero-shot classification:")
print(zero_shot_classify(test_text, ["BUG", "FEATURE_REQUEST", "GENERAL"]))

examples = [
    ("The app crashes when I click save.", "BUG"),
    ("It would be nice to have dark mode.", "FEATURE_REQUEST"),
    ("What are your business hours?", "GENERAL"),
]
print("\nFew-shot classification:")
print(few_shot_classify(test_text, examples))
```

The zero-shot classifier might label "search feature does not return results" as FEATURE_REQUEST or GENERAL. The few-shot classifier, seeing the pattern that feature requests are phrased as wishes ("It would be nice to..."), correctly classifies it as BUG.

### 2. How Example Count Affects Accuracy

Test how many examples you need:

```python
def classify_with_example_count(
    text: str,
    categories: list[str],
    examples: list[tuple[str, str]],
    n: int
) -> str:
    """Use the first N examples for classification."""
    selected = examples[:n]
    messages = []
    for input_text, label in selected:
        messages.append({"role": "user", "content": input_text})
        messages.append({"role": "assistant", "content": label})

    messages.append({"role": "user", "content": text})

    response = client.messages.create(
        model="claude-3-5-sonnet-20241022",
        max_tokens=10,
        temperature=0,
        system=f"Classify into: {', '.join(categories)}.",
        messages=messages
    )
    return response.content[0].text.strip()


examples_pool = [
    ("The login button is not working.", "BUG"),
    ("I would like an export to CSV feature.", "FEATURE_REQUEST"),
    ("Thanks for the quick response!", "GENERAL"),
    ("The page loads very slowly on mobile.", "BUG"),
    ("Can you add a dark mode option?", "FEATURE_REQUEST"),
    ("How do I reset my password?", "GENERAL"),
    ("The dropdown menu overlaps the text.", "BUG"),
]

test_cases = [
    "The payment form does not validate my zip code.",
    "Can you support saving filters?",
    "What is your return policy?",
]

for n in [0, 1, 3, 5]:
    print(f"\n=== {n} example(s) ===")
    for case in test_cases:
        result = classify_with_example_count(
            case,
            ["BUG", "FEATURE_REQUEST", "GENERAL"],
            examples_pool,
            n
        )
        print(f"  [{result:16s}] {case[:50]}")
```

With 0 examples, the model may use categories you did not define. With 1 example, it tends to mimic that single pattern. With 3-5 examples covering diverse cases, classification stabilizes.

### 3. Example Ordering Effects

The order of examples matters. Compare forward and backward ordering:

```python
def classify_with_examples(examples: list[tuple[str, str]], text: str) -> str:
    """Classify using the given examples in the given order."""
    messages = []
    for input_text, label in examples:
        messages.append({"role": "user", "content": input_text})
        messages.append({"role": "assistant", "content": label})
    messages.append({"role": "user", "content": text})

    response = client.messages.create(
        model="claude-3-5-sonnet-20241022",
        max_tokens=10,
        temperature=0,
        messages=messages
    )
    return response.content[0].text.strip()


base_examples = [
    ("The API returns an empty response.", "BUG"),
    ("I want to be able to sort by date.", "FEATURE_REQUEST"),
    ("Do you offer student discounts?", "GENERAL"),
]

# The last example has the most influence (recency bias)
reordered = list(reversed(base_examples))
# Now the last example is GENERAL — may bias toward that category

test = "The export feature generates corrupted files."
print(f"Normal order:     {classify_with_examples(base_examples, test)}")
print(f"Reversed order:   {classify_with_examples(reordered, test)}")
```

If the last example in the normal order is FEATURE_REQUEST, the model may bias toward classifying ambiguous inputs as FEATURE_REQUEST. Reversing the order shifts the bias.

## How It Works Internally

In-context learning works through the attention mechanism. When the model sees a sequence of examples, the attention layers learn which patterns in the input predict which patterns in the output. The first example establishes the pattern; subsequent examples reinforce or refine it.

The model does not memorize the examples — it extracts the latent rule connecting inputs and outputs. This is why example quality matters more than quantity: five well-chosen examples that clearly demonstrate the pattern are more effective than twenty noisy examples.

Recency bias in example ordering occurs because the attention layers give more weight to tokens that are closer to the current position. The last example before the actual input has the strongest influence on the output distribution.

## Build Something Real

Build a few-shot intent classifier for a customer support system:

```python
import anthropic

client = anthropic.Anthropic()


SUPPORT_EXAMPLES = [
    # (user_message, intent)
    ("Where is my order?", "ORDER_STATUS"),
    ("I have not received my package yet.", "ORDER_STATUS"),
    ("Can I return this item?", "RETURN_REQUEST"),
    ("I want to send this back for a refund.", "RETURN_REQUEST"),
    ("My account was charged twice.", "BILLING_ISSUE"),
    ("I was overcharged for shipping.", "BILLING_ISSUE"),
    ("The product arrived damaged.", "PRODUCT_ISSUE"),
    ("The screen is cracked out of the box.", "PRODUCT_ISSUE"),
    ("How do I change my shipping address?", "ACCOUNT_HELP"),
    ("I forgot my password.", "ACCOUNT_HELP"),
]


def classify_intent_fewshot(
    message: str,
    num_examples: int = 5
) -> dict:
    """Classify customer message intent using few-shot examples."""
    # Select diverse examples: one of each intent type
    selected = SUPPORT_EXAMPLES[:num_examples]

    messages = []
    for user_msg, intent in selected:
        messages.append({"role": "user", "content": user_msg})
        messages.append({"role": "assistant", "content": intent})

    messages.append({"role": "user", "content": message})

    response = client.messages.create(
        model="claude-3-5-sonnet-20241022",
        max_tokens=20,
        temperature=0,
        system=(
            "Classify customer support messages into one intent: "
            "ORDER_STATUS, RETURN_REQUEST, BILLING_ISSUE, PRODUCT_ISSUE, or ACCOUNT_HELP."
        ),
        messages=messages
    )

    return {
        "intent": response.content[0].text.strip(),
        "input_tokens": response.usage.input_tokens,
        "output_tokens": response.usage.output_tokens,
    }


# Test against new messages
test_messages = [
    "Can you track my delivery?",
    "The item I received is defective.",
    "I need a refund for order #9876.",
]

for msg in test_messages:
    result = classify_intent_fewshot(msg, 5)
    print(f"Message: {msg}")
    print(f"Intent:  {result['intent']}")
    print(f"Tokens:  {result['input_tokens']}in / {result['output_tokens']}out\n")
```

The few-shot examples teach the model to distinguish between RETURN_REQUEST ("I want to send this back") and PRODUCT_ISSUE ("arrived damaged"), which are often confused in zero-shot prompting.
