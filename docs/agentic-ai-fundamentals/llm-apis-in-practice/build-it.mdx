# LLM APIs in Practice — Build It

*Reading time: 12 minutes*

## Breaking It Down

### 1. Streaming Responses

Streaming returns tokens incrementally, which drastically reduces perceived latency:

```python
import anthropic

client = anthropic.Anthropic()

# Non-streaming: wait for the complete response
response = client.messages.create(
    model="claude-3-5-sonnet-20241022",
    max_tokens=200,
    messages=[{"role": "user", "content": "Write a short poem about AI."}]
)
print("Non-streaming:", response.content[0].text[:80])

# Streaming: receive tokens as they are generated
print("\nStreaming:")
with client.messages.stream(
    model="claude-3-5-sonnet-20241022",
    max_tokens=200,
    messages=[{"role": "user", "content": "Write a short poem about AI."}]
) as stream:
    for text in stream.text_stream:
        print(text, end="", flush=True)
```

The stream yields text chunks as the model generates them. The user sees output appearing word by word, which feels faster even if total generation time is the same.

### 2. A Production API Client

Build a reusable client that handles errors, retries, and cost tracking:

```python
import anthropic
import time
import logging
from dataclasses import dataclass, field

logging.basicConfig(level=logging.INFO)
logger = logging.getLogger("llm_client")


@dataclass
class LLMResponse:
    """Standardized response from our API client."""
    text: str
    model: str
    input_tokens: int
    output_tokens: int
    latency_ms: float
    error: str | None = None


@dataclass
class LLMClientConfig:
    """Configuration for the LLM API client."""
    model: str = "claude-3-5-sonnet-20241022"
    max_tokens: int = 1000
    temperature: float = 0.0
    max_retries: int = 3
    base_delay: float = 1.0
    request_timeout: float = 120.0


class LLMClient:
    """Production-grade LLM API client with retry, logging, and cost tracking."""

    def __init__(self, config: LLMClientConfig | None = None):
        self.config = config or LLMClientConfig()
        self.client = anthropic.Anthropic(timeout=self.config.request_timeout)
        self.total_cost: float = 0.0
        self.request_count: int = 0

    def generate(
        self,
        system_prompt: str,
        messages: list[dict],
        stream: bool = False
    ) -> LLMResponse:
        """Send a request with retry logic and cost tracking."""
        start = time.time()
        last_error: Exception | None = None

        for attempt in range(self.config.max_retries):
            try:
                if stream:
                    return self._stream_request(system_prompt, messages, start)
                return self._sync_request(system_prompt, messages, start)

            except anthropic.RateLimitError as e:
                wait = self.config.base_delay * (2 ** attempt)
                logger.warning(
                    f"Rate limited (attempt {attempt + 1}). "
                    f"Waiting {wait:.1f}s"
                )
                time.sleep(wait)
                last_error = e

            except anthropic.APIStatusError as e:
                if e.status_code >= 500 and attempt < self.config.max_retries - 1:
                    wait = self.config.base_delay * (2 ** attempt)
                    logger.warning(
                        f"Server error {e.status_code} "
                        f"(attempt {attempt + 1}). Retrying in {wait:.1f}s"
                    )
                    time.sleep(wait)
                    last_error = e
                else:
                    logger.error(f"Non-retriable API error: {e}")
                    raise

            except Exception as e:
                logger.error(f"Unexpected error: {e}")
                raise

        # All retries exhausted
        elapsed = (time.time() - start) * 1000
        return LLMResponse(
            text="",
            model=self.config.model,
            input_tokens=0,
            output_tokens=0,
            latency_ms=elapsed,
            error=str(last_error)
        )

    def _sync_request(
        self,
        system_prompt: str,
        messages: list[dict],
        start: float
    ) -> LLMResponse:
        """Make a non-streaming request."""
        response = self.client.messages.create(
            model=self.config.model,
            max_tokens=self.config.max_tokens,
            temperature=self.config.temperature,
            system=system_prompt,
            messages=messages
        )
        elapsed = (time.time() - start) * 1000
        self.request_count += 1

        # Track cost (example: Claude 3.5 Sonnet pricing)
        cost = self._calculate_cost(
            response.usage.input_tokens,
            response.usage.output_tokens
        )
        self.total_cost += cost

        logger.info(
            f"Request #{self.request_count}: "
            f"in={response.usage.input_tokens} "
            f"out={response.usage.output_tokens} "
            f"cost=${cost:.6f} "
            f"latency={elapsed:.0f}ms"
        )

        return LLMResponse(
            text=response.content[0].text,
            model=response.model,
            input_tokens=response.usage.input_tokens,
            output_tokens=response.usage.output_tokens,
            latency_ms=elapsed
        )

    def _stream_request(
        self,
        system_prompt: str,
        messages: list[dict],
        start: float
    ) -> LLMResponse:
        """Make a streaming request and collect the full response."""
        collected = []

        with self.client.messages.stream(
            model=self.config.model,
            max_tokens=self.config.max_tokens,
            temperature=self.config.temperature,
            system=system_prompt,
            messages=messages
        ) as stream:
            for text in stream.text_stream:
                collected.append(text)

        response = stream.get_final_message()
        elapsed = (time.time() - start) * 1000
        self.request_count += 1

        cost = self._calculate_cost(
            response.usage.input_tokens,
            response.usage.output_tokens
        )
        self.total_cost += cost

        return LLMResponse(
            text="".join(collected),
            model=response.model,
            input_tokens=response.usage.input_tokens,
            output_tokens=response.usage.output_tokens,
            latency_ms=elapsed
        )

    def _calculate_cost(self, input_tokens: int, output_tokens: int) -> float:
        """Calculate cost in USD for Claude 3.5 Sonnet."""
        INPUT_RATE = 3.0 / 1_000_000    # $3 per million input tokens
        OUTPUT_RATE = 15.0 / 1_000_000  # $15 per million output tokens
        return (input_tokens * INPUT_RATE) + (output_tokens * OUTPUT_RATE)


# Usage example
config = LLMClientConfig(max_retries=3)
client = LLMClient(config)

result = client.generate(
    system_prompt="You are a helpful assistant.",
    messages=[{"role": "user", "content": "What are the three main cloud providers?"}],
    stream=False
)

print(f"Response: {result.text}")
print(f"Cost: ${result.input_tokens * 3 / 1_000_000 + result.output_tokens * 15 / 1_000_000:.6f}")
print(f"Total session cost so far: ${client.total_cost:.6f}")
```

## How It Works Internally

When you send a streaming request, the API uses server-sent events (SSE) over HTTP. The connection stays open, and the server pushes each token as it is generated. The SDK deserializes these events and yields them as text chunks.

Under the hood, the model's forward pass generates one token at a time. Each token requires the full attention computation over the entire context. The streaming API sends each token as soon as it is decoded rather than waiting for the full sequence. From the model's perspective, generation is identical — only the delivery mechanism changes.

Exponential backoff works by doubling the wait time after each failed attempt: 1s, 2s, 4s, 8s. This prevents your retries from overwhelming the API while keeping total wait time bounded. Adding jitter (randomizing the wait by ±50%) prevents synchronized retries of thundering herd problem.

## Build Something Real

Build a batch processor that sends prompts to an LLM with production-quality reliability:

```python
import anthropic
import time
import json
from pathlib import Path

client = anthropic.Anthropic()


def process_batch(
    prompts: list[dict[str, str]],
    model: str = "claude-3-5-sonnet-20241022",
    max_retries: int = 3
) -> list[dict]:
    """Process a batch of prompts with per-item retries and logging."""
    results = []

    for i, item in enumerate(prompts):
        for attempt in range(max_retries):
            try:
                start = time.time()
                response = client.messages.create(
                    model=model,
                    max_tokens=500,
                    temperature=0,
                    system=item.get("system", ""),
                    messages=[{"role": "user", "content": item["prompt"]}]
                )
                elapsed = (time.time() - start) * 1000

                results.append({
                    "id": item.get("id", i),
                    "success": True,
                    "output": response.content[0].text,
                    "input_tokens": response.usage.input_tokens,
                    "output_tokens": response.usage.output_tokens,
                    "latency_ms": round(elapsed, 1)
                })
                break  # Success — exit retry loop

            except (anthropic.RateLimitError, anthropic.APIStatusError) as e:
                if attempt < max_retries - 1:
                    wait = 2 ** attempt
                    time.sleep(wait)
                else:
                    results.append({
                        "id": item.get("id", i),
                        "success": False,
                        "error": str(e)
                    })

    return results


# Example: classify multiple customer messages
batch_prompts = [
    {"id": 1, "system": "Classify the sentiment.", "prompt": "I love this product!"},
    {"id": 2, "system": "Classify the sentiment.", "prompt": "This is terrible, I want a refund."},
    {"id": 3, "system": "Classify the sentiment.", "prompt": "The package arrived on Tuesday."},
]

batch_results = process_batch(batch_prompts)
for r in batch_results:
    status = "OK" if r["success"] else "FAIL"
    tokens = f"{r.get('input_tokens', 0)}in/{r.get('output_tokens', 0)}out"
    latency = f"{r.get('latency_ms', 0)}ms"
    print(f"[{status}] ID={r['id']} | {tokens} | {latency}")
    if r["success"]:
        print(f"  Output: {r['output']}")
```
