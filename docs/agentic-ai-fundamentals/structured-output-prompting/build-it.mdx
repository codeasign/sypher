# Structured Output Prompting — Build It

*Reading time: 12 minutes*

## Breaking It Down

### 1. Prompt-Only JSON Output

The simplest approach: describe the JSON structure in the prompt:

```python
import anthropic
import json

client = anthropic.Anthropic()


def extract_json_by_prompt(text: str) -> dict:
    """Extract structured data by describing the JSON format in the prompt."""
    response = client.messages.create(
        model="claude-3-5-sonnet-20241022",
        max_tokens=300,
        temperature=0,
        messages=[{
            "role": "user",
            "content": (
                "Extract the following information from the text and return it as JSON.\n\n"
                "Schema:\n"
                "{\n"
                '  "name": "full name of the person",\n'
                '  "email": "email address",\n'
                '  "urgency": "HIGH, MEDIUM, or LOW",\n'
                '  "request_type": "the type of request",\n'
                '  "description": "one-sentence summary"\n'
                "}\n\n"
                "Return ONLY valid JSON. No other text.\n\n"
                "Text: " + text
            )
        }]
    )
    return response.content[0].text


email = """
Dear Support Team,

My name is Maria Garcia and my email is maria.garcia@example.com.
I am writing because my account was compromised and someone made
unauthorized purchases. This is extremely urgent.

Best,
Maria
"""

result = extract_json_by_prompt(email)
print("Prompt-only JSON output:")
print(result)

# Attempt to parse it
try:
    parsed = json.loads(result)
    print(f"\nParsed successfully: {parsed['name']} - {parsed['urgency']}")
except json.JSONDecodeError as e:
    print(f"\nParse failed: {e}")
```

The prompt-only approach usually works, but can occasionally include markdown code fences, extra text, or malformed JSON.

### 2. JSON Mode with Response Validation

Wrap the prompt in a validator that catches and retries on malformed output:

```python
def extract_with_validation(
    text: str,
    schema: dict,
    max_retries: int = 2
) -> dict | None:
    """Extract structured data with retry on malformed output."""
    schema_description = json.dumps(schema, indent=2)

    for attempt in range(max_retries + 1):
        response = client.messages.create(
            model="claude-3-5-sonnet-20241022",
            max_tokens=300,
            temperature=0,
            messages=[{
                "role": "user",
                "content": (
                    "Extract information from the text and return valid JSON "
                    "matching this schema:\n\n"
                    f"{schema_description}\n\n"
                    "Return ONLY valid JSON. No explanation. "
                    "No markdown formatting. No code fences.\n\n"
                    f"Text: {text}"
                )
            }]
        )

        raw = response.content[0].text

        # Clean the output — strip markdown code fences if present
        cleaned = raw.strip()
        if cleaned.startswith("```"):
            # Extract content between code fences
            lines = cleaned.split("\n")
            cleaned = "\n".join(lines[1:-1]) if len(lines) > 2 else ""

        try:
            parsed = json.loads(cleaned)
            # Validate that expected keys exist
            for key in schema.keys():
                if key not in parsed:
                    raise ValueError(f"Missing key: {key}")
            return parsed
        except (json.JSONDecodeError, ValueError) as e:
            if attempt == max_retries:
                print(f"Failed after {max_retries} retries: {e}")
                return None
            continue

    return None


schema = {
    "name": "string",
    "email": "string",
    "urgency": "string",
    "request_type": "string",
    "description": "string"
}

result = extract_with_validation(email, schema)
if result:
    print(f"Validated extraction: {result}")
```

The validator handles the most common failure cases: markdown code fences, extra whitespace, and missing keys.

### 3. Few-Shot JSON Generation

Show the model exactly what format you expect:

```python
def extract_with_fewshot_json(text: str) -> dict:
    """Extract structured data using few-shot JSON examples."""
    response = client.messages.create(
        model="claude-3-5-sonnet-20241022",
        max_tokens=300,
        temperature=0,
        messages=[
            {
                "role": "user",
                "content": "My router keeps disconnecting every 10 minutes."
            },
            {
                "role": "assistant",
                "content": json.dumps({
                    "issue": "Intermittent connectivity",
                    "category": "NETWORK",
                    "priority": "HIGH",
                    "affected_users": 1,
                    "steps_taken": []
                })
            },
            {
                "role": "user",
                "content": "Can you tell me the price of the premium plan?"
            },
            {
                "role": "assistant",
                "content": json.dumps({
                    "issue": "Pricing inquiry",
                    "category": "BILLING",
                    "priority": "LOW",
                    "affected_users": 0,
                    "steps_taken": []
                })
            },
            {
                "role": "user",
                "content": text
            }
        ]
    )
    return json.loads(response.content[0].text)


new_email = "Our entire team cannot access the shared drive. Please help!"
result = extract_with_fewshot_json(new_email)
print(f"Few-shot JSON: {result}")
```

The few-shot examples show the model the exact JSON keys, value types, and nesting structure. It learns the pattern from the examples, not from schema descriptions.

## How It Works Internally

When you prompt for JSON output, the model selects tokens that are statistically likely to form valid JSON structures. The training data contains vast amounts of JSON, so the model has a strong prior for what valid JSON looks like.

JSON mode (available in some APIs) modifies the sampling process to mask out tokens that would produce invalid JSON. The model can only generate tokens that continue a valid JSON structure. This eliminates JSON syntax errors but does not guarantee the content matches your schema — the keys and value types are still controlled by the prompt.

Tool use / function calling is the most robust approach because the schema is defined at the API level, not in the prompt. The API validates the structure server-side and returns a structured tool call object that is guaranteed to match the schema. This is why production systems should prefer tool use over prompt-only structured output.

## Build Something Real

Build a structured data extraction pipeline that handles real-world messiness:

```python
import anthropic
import json
import re

client = anthropic.Anthropic()


def clean_json_string(raw: str) -> str:
    """Clean a raw model response to extract valid JSON."""
    # Remove markdown code fences
    raw = re.sub(r"```(?:json)?\s*", "", raw).strip()

    # Find the first { and last }
    start = raw.find("{")
    end = raw.rfind("}")

    if start == -1 or end == -1:
        # Try finding array brackets
        start = raw.find("[")
        end = raw.rfind("]")

    if start != -1 and end != -1 and end > start:
        return raw[start:end + 1]

    return raw


def extract_structured(
    text: str,
    schema: dict,
    temperature: float = 0.0
) -> dict | None:
    """Extract structured data with comprehensive error handling."""
    schema_str = json.dumps(schema, indent=2)

    response = client.messages.create(
        model="claude-3-5-sonnet-20241022",
        max_tokens=500,
        temperature=temperature,
        system=(
            "You extract structured data from text. "
            "Return only valid JSON matching the provided schema. "
            "Do not include any text outside the JSON object."
        ),
        messages=[{
            "role": "user",
            "content": (
                f"Extract data matching this schema:\n{schema_str}\n\n"
                f"Text:\n{text}"
            )
        }]
    )

    raw = response.content[0].text
    cleaned = clean_json_string(raw)

    try:
        data = json.loads(cleaned)
        return data
    except json.JSONDecodeError:
        # Last resort: try to fix common JSON issues
        # Remove trailing commas
        cleaned = re.sub(r",\s*}", "}", cleaned)
        cleaned = re.sub(r",\s*]", "]", cleaned)
        try:
            return json.loads(cleaned)
        except json.JSONDecodeError:
            return None


# Test with a messy input
messy_email = """
From: Bob Johnson <bob.j@test.com>
To: support@company.com
Subject: URGENT: Can't login since yesterday

HELP! I've been trying to login since yesterday and it keeps saying
"invalid password". I tried resetting but the reset email never arrives.
I have a deadline today and I'm locked out of my account. Please fix ASAP.
- Bob (account: bob.j, plan: premium)
"""

schema = {
    "contact_name": "string",
    "contact_email": "string",
    "issue_category": "LOGIN | BILLING | TECHNICAL | OTHER",
    "priority": "CRITICAL | HIGH | MEDIUM | LOW",
    "account_id": "string or null",
    "description": "string (max 2 sentences)"
}

result = extract_structured(messy_email, schema)
if result:
    print(f"Extracted: {json.dumps(result, indent=2)}")
else:
    print("Extraction failed — could not parse structured output.")
```

This pipeline handles real-world inputs with a cleaning step, retry logic, and fallback parsing for common JSON formatting issues.
