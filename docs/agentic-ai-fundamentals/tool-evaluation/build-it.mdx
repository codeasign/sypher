# Build It: Implementing Tool Evaluation

⏱️ 25 minutes

## Breaking It Down

### 1. Instrumenting Tool Call Logging

Start by adding structured logging to every tool call in your MCP server:

```python
# tool_logger.py
import json
import time
from dataclasses import dataclass, field, asdict
from datetime import datetime
from typing import Any

@dataclass
class ToolInvocation:
    tool_name: str
    arguments: dict
    result_summary: str  # first 200 chars of result
    duration_ms: int
    error: str = ""
    task_id: str = ""
    timestamp: str = field(default_factory=lambda: datetime.now().isoformat())

class ToolLogger:
    def __init__(self):
        self.invocations: list[ToolInvocation] = []

    def log_call(self, tool_name: str, args: dict, result: Any, duration_ms: int, task_id: str = ""):
        result_str = str(result)[:200]
        error = ""
        if isinstance(result, Exception):
            error = str(result)
            result_str = ""
        self.invocations.append(ToolInvocation(
            tool_name=tool_name,
            arguments=args,
            result_summary=result_str,
            duration_ms=duration_ms,
            error=error,
            task_id=task_id,
        ))

    def export(self) -> str:
        return json.dumps([asdict(inv) for inv in self.invocations], indent=2)

logger = ToolLogger()

# Usage inside a tool function:
@server.tool("deploy_service")
async def deploy_service(service: str, version: str) -> TextContent:
    start = time.time()
    try:
        result = await actual_deploy(service, version)
        logger.log_call("deploy_service", {"service": service, "version": version}, result, int((time.time() - start) * 1000))
        return TextContent(type="text", text=result)
    except Exception as e:
        logger.log_call("deploy_service", {"service": service, "version": version}, e, int((time.time() - start) * 1000))
        return TextContent(type="text", text=f"Error: {e}")
```

### 2. Computing Tool Metrics

From the logged invocations, compute per-tool metrics:

```python
# tool_metrics.py
from collections import Counter, defaultdict
from typing import List

def compute_tool_metrics(invocations: List[ToolInvocation]) -> dict:
    """Compute per-tool metrics from invocation logs."""
    total = len(invocations)
    if total == 0:
        return {}

    by_tool = defaultdict(list)
    for inv in invocations:
        by_tool[inv.tool_name].append(inv)

    metrics = {}
    for tool_name, calls in by_tool.items():
        error_calls = [c for c in calls if c.error]
        successful = len(calls) - len(error_calls)

        # Invocation metrics
        invocation_share = len(calls) / total * 100

        # Duration metrics
        avg_duration = sum(c.duration_ms for c in calls) / len(calls)
        min_duration = min(c.duration_ms for c in calls)
        max_duration = max(c.duration_ms for c in calls)

        # Error metrics
        error_rate = len(error_calls) / len(calls) * 100 if calls else 0

        metrics[tool_name] = {
            "total_calls": len(calls),
            "invocation_share_pct": round(invocation_share, 1),
            "successful_calls": successful,
            "error_rate_pct": round(error_rate, 1),
            "avg_duration_ms": round(avg_duration, 0),
            "min_duration_ms": min_duration,
            "max_duration_ms": max_duration,
            "frequency_rank": 0,  # set below
        }

    # Rank tools by frequency
    sorted_tools = sorted(metrics.items(), key=lambda x: x[1]["total_calls"], reverse=True)
    for rank, (name, _) in enumerate(sorted_tools, 1):
        metrics[name]["frequency_rank"] = rank

    return metrics
```

### 3. Detecting Tool Design Problems

Use heuristics to flag potential tool design issues:

```python
def analyze_tool_diagnostics(metrics: dict, tool_definitions: dict) -> List[dict]:
    """Detect potential tool design problems from usage patterns."""
    issues = []

    for tool_name, tool_metric in metrics.items():
        # Problem 1: High error rate
        if tool_metric["error_rate_pct"] > 20:
            issues.append({
                "severity": "high",
                "tool": tool_name,
                "issue": "High error rate",
                "detail": f"{tool_metric['error_rate_pct']}% of calls return errors",
                "suggestion": "Check parameter schema and tool implementation for bugs",
            })

        # Problem 2: Very fast calls with errors (validation failures)
        if tool_metric["error_rate_pct"] > 10 and tool_metric["avg_duration_ms"] < 100:
            issues.append({
                "severity": "medium",
                "tool": tool_name,
                "issue": "Rapid validation failures",
                "detail": f"Average duration {tool_metric['avg_duration_ms']}ms with {tool_metric['error_rate_pct']}% errors",
                "suggestion": "Input validation may be rejecting valid parameters — check parameter descriptions",
            })

        # Problem 3: Never-used tool
        if tool_metric["total_calls"] == 0:
            issues.append({
                "severity": "medium",
                "tool": tool_name,
                "issue": "Zero invocations",
                "detail": "Tool is never called by the agent",
                "suggestion": "Improve the tool description or check if it overlaps with another tool",
            })

    # Problem 4: Tool confusion — two tools called with similar frequency for same task
    call_patterns = defaultdict(set)
    for inv in logger.invocations:
        call_patterns[inv.task_id].add(inv.tool_name)

    for task_id, tools_used in call_patterns.items():
        if len(tools_used) >= 3:
            # Three or more different tools used for one task suggests confusion
            issues.append({
                "severity": "medium",
                "tool": ", ".join(sorted(tools_used)),
                "issue": "Tool confusion",
                "detail": f"Task '{task_id}' used {len(tools_used)} different tools",
                "suggestion": "Tool descriptions may overlap — clarify boundaries or merge tools",
            })

    return issues
```

### 5. Running Tool Evaluation as a Report

Combine everything into a single report:

```python
def generate_tool_report(invocations: List[ToolInvocation], tool_defs: dict) -> dict:
    """Generate a complete tool evaluation report."""
    metrics = compute_tool_metrics(invocations)
    issues = analyze_tool_diagnostics(metrics, tool_defs)

    # Overall tool health score (0-100)
    avg_error_rate = sum(m["error_rate_pct"] for m in metrics.values()) / max(len(metrics), 1)
    unused_tools = sum(1 for m in metrics.values() if m["total_calls"] == 0)
    health_score = max(0, 100 - avg_error_rate - (unused_tools * 10))

    return {
        "health_score": round(health_score, 1),
        "total_tools": len(tool_defs),
        "tools_used": len(metrics),
        "tools_unused": unused_tools,
        "tool_metrics": metrics,
        "design_issues": issues,
        "total_invocations": len(invocations),
    }
```

## Performance Considerations

Tool evaluation logging adds minimal overhead — each invocation record is a few hundred bytes, and recording takes under 1ms. The more significant cost is storage: an agent making 50 tool calls per session over 1000 sessions produces 50,000 records. Store them in a time-series database or log aggregator rather than in memory.

For large-scale evaluation, batch-process logs in daily or hourly increments rather than computing metrics on every invocation. A summary cache updated every 10 minutes is sufficient for dashboards.
