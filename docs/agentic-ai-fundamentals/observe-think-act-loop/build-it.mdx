# Observe-Think-Act Loop — Build It

*Reading time: 12 minutes*

## Breaking It Down

### 1. A Bare Loop

Start with the simplest possible observe-think-act loop.

```python
import anthropic

client = anthropic.Anthropic()


def bare_loop(user_input: str) -> str:
    """The simplest observe-think-act loop."""
    messages = [{"role": "user", "content": user_input}]

    for turn in range(10):  # Hard limit
        # OBSERVE: Receive the latest message (the user input or tool result)
        # THINK + ACT: Send to the model, get a response
        response = client.messages.create(
            model="claude-3-5-sonnet-20241022",
            max_tokens=500,
            temperature=0,
            messages=messages,
        )

        output = response.content[0].text

        # Check if the response is a question or request for more info
        if "?" in output:
            # Need more input — loop back to observe
            messages.append({"role": "assistant", "content": output})
            continue

        # Done: produce the final output
        return output

    return "Loop limit reached without convergence."


result = bare_loop("What is the capital of Japan?")
print(result)
```

This loop works for simple conversations, but it has no tools, no termination check, and no way to distinguish thinking from acting.

### 2. Adding Observation from Tools

Now add tools. The observation comes from tool results, not just user messages.

```python
def search_kb(query: str) -> str:
    """Simple knowledge base lookup."""
    data = {
        "capital of Japan": "Tokyo",
        "population of Japan": "125 million",
        "currency of Japan": "Japanese Yen (JPY)",
    }
    for key, value in data.items():
        if query.lower() in key:
            return value
    return f"No information found for: {query}"


def observe_think_act_loop(user_input: str) -> str:
    """Full observe-think-act loop with tools."""
    messages = [{"role": "user", "content": user_input}]
    tools = [{
        "name": "search_knowledge_base",
        "description": "Search the knowledge base for information",
        "input_schema": {
            "type": "object",
            "properties": {
                "query": {"type": "string"}
            },
            "required": ["query"]
        }
    }]

    for turn in range(10):
        response = client.messages.create(
            model="claude-3-5-sonnet-20241022",
            max_tokens=500,
            temperature=0,
            messages=messages,
            tools=tools,
        )

        content = response.content[0]

        if content.type == "tool_use":
            # ACT phase: execute the tool
            result = search_kb(content.input["query"])

            # OBSERVE phase: the result is the new observation
            messages.append({"role": "assistant", "content": content})
            messages.append({
                "role": "user",
                "content": [
                    {"type": "tool_result", "tool_use_id": content.id, "content": result}
                ]
            })
            # Loop: back to THINK with the new observation
        else:
            # THINK phase produced a final answer
            return content.text

    return "Loop limit reached."


result = observe_think_act_loop("Tell me about Japan's capital and population.")
print(result)
```

### 3. Adding Termination Conditions

A good loop knows when to stop. Add checks for task completion, stalling, and error states.

```python
class LoopController:
    """Controls the observe-think-act loop with termination conditions."""

    def __init__(self, max_turns: int = 15, max_time_s: int = 60):
        self.max_turns = max_turns
        self.max_time_s = max_time_s
        self.turn_count = 0
        self.start_time = __import__('time').time()
        self.last_states: list[str] = []

    def should_continue(self, current_state: str) -> tuple[bool, str]:
        """
        Check if the loop should continue.
        Returns (continue_flag, reason).
        """
        self.turn_count += 1

        # Check turn limit
        if self.turn_count > self.max_turns:
            return False, f"Exceeded max turns ({self.max_turns})"

        # Check time limit
        elapsed = __import__('time').time() - self.start_time
        if elapsed > self.max_time_s:
            return False, f"Exceeded max time ({self.max_time_s}s)"

        # Check for stalling (same state repeated 3 times)
        self.last_states.append(current_state)
        if len(self.last_states) > 3:
            self.last_states.pop(0)
        if len(self.last_states) == 3 and len(set(self.last_states)) == 1:
            return False, "Agent is stuck in a loop (same state 3 times)"

        return True, "ok"


def controlled_loop(user_input: str) -> str:
    """Observe-think-act loop with termination checks."""
    messages = [{"role": "user", "content": user_input}]
    controller = LoopController(max_turns=10)
    tools = [{
        "name": "search_knowledge_base",
        "description": "Search for information",
        "input_schema": {
            "type": "object",
            "properties": {
                "query": {"type": "string"}
            },
            "required": ["query"]
        }
    }]

    for turn in range(controller.max_turns + 1):
        # THINK
        response = client.messages.create(
            model="claude-3-5-sonnet-20241022",
            max_tokens=500,
            temperature=0,
            messages=messages,
            tools=tools,
        )

        content = response.content[0]

        if content.type == "tool_use":
            # ACT
            result = search_kb(content.input["query"])

            # Check if we should continue
            should_continue, reason = controller.should_continue(
                f"tool_call:{content.name}"
            )
            if not should_continue:
                return f"Loop stopped: {reason}"

            # OBSERVE
            messages.append({"role": "assistant", "content": content})
            messages.append({
                "role": "user",
                "content": [
                    {"type": "tool_result", "tool_use_id": content.id, "content": result}
                ]
            })
        else:
            return content.text

    return "Loop limit reached."
```

## How It Works Internally

The observe-think-act loop is a state machine with three states:

- **OBSERVE**: New information arrives (user message or tool result). Append to context.
- **THINK**: Send context to model. Model produces a response (text or tool request).
- **ACT**: If tool request, execute the tool and go to OBSERVE. If text, check if done.
- **DONE**: Return the final text.

The termination condition is checked after each ACT phase. This prevents the agent from looping on failed tool calls.

## Build Something Real

Build a full loop with a running task tracker and structured output.

```python
import anthropic
import json
import time

client = anthropic.Anthropic()


class ObserveThinkActAgent:
    """Complete agent with the observe-think-act pattern."""

    def __init__(self):
        self.tools = [
            {
                "name": "search_knowledge_base",
                "description": "Search the knowledge base",
                "input_schema": {
                    "type": "object",
                    "properties": {
                        "query": {"type": "string"}
                    },
                    "required": ["query"]
                }
            },
            {
                "name": "verify_fact",
                "description": "Verify a fact against known data",
                "input_schema": {
                    "type": "object",
                    "properties": {
                        "fact": {"type": "string"},
                        "source": {"type": "string"}
                    },
                    "required": ["fact"]
                }
            },
        ]
        self._kb = {
            "tokyo capital": "Tokyo is the capital of Japan.",
            "japan population": "Japan's population is approximately 125 million.",
        }

    def _execute(self, name: str, args: dict) -> str:
        if name == "search_knowledge_base":
            return self._kb.get(args["query"].lower(), "Not found.")
        if name == "verify_fact":
            for key, value in self._kb.items():
                if args["fact"].lower() in value.lower():
                    return f"Verified: {value}"
            return "Could not verify this fact."
        return f"Unknown tool: {name}"

    def run(self, task: str) -> dict:
        """Run the loop and return a structured result."""
        messages = [{"role": "user", "content": task}]
        turns = []
        start = time.time()

        for turn_num in range(15):
            turn_start = time.time()

            response = client.messages.create(
                model="claude-3-5-sonnet-20241022",
                max_tokens=500,
                temperature=0,
                messages=messages,
                tools=self.tools,
            )

            content = response.content[0]
            turn_time = time.time() - turn_start

            turns.append({
                "turn": turn_num + 1,
                "type": content.type,
                "time_ms": round(turn_time * 1000),
                "input_tokens": response.usage.input_tokens,
                "output_tokens": response.usage.output_tokens,
            })

            if content.type == "tool_use":
                result = self._execute(content.name, content.input)
                messages.append({"role": "assistant", "content": content})
                messages.append({
                    "role": "user",
                    "content": [
                        {"type": "tool_result", "tool_use_id": content.id, "content": result}
                    ]
                })
            else:
                total_time = time.time() - start
                return {
                    "answer": content.text,
                    "turns": turns,
                    "total_turns": len(turns),
                    "total_time_ms": round(total_time * 1000),
                }

        return {"answer": "Max turns reached", "turns": turns,
                "total_turns": len(turns), "total_time_ms": round((time.time() - start) * 1000)}


agent = ObserveThinkActAgent()
result = agent.run("What is the capital of Japan and what is its population?")
print(f"Answer: {result['answer']}")
print(f"Turns: {result['total_turns']}, Time: {result['total_time_ms']}ms")
```

The loop now tracks turn timing, token usage, and termination reasons. This data is essential for tuning loop parameters in production.

## Performance Considerations

Each turn adds an API round trip. Keep turns fast by using small models for simple tool calls and reserving large models for reasoning-heavy turns. Set the max turns based on the task complexity — a lookup task rarely needs more than 3 turns.
