# Building an MCP Server — Build It

*Reading time: 16 minutes*

## Breaking It Down

### 1. Project Setup

Start by installing the MCP Python SDK and scaffolding your project.

```bash
pip install mcp httpx
```

Create this project structure:

```
mcp-server/
  server.py           # Main server entry point
  tools/
    __init__.py
    search.py         # Search tool implementation
    file_ops.py       # File operation tools
  resources/
    __init__.py
    config.py         # Configuration resources
    docs.py           # Documentation resources
```

### 2. Server Scaffold with the MCP SDK

The MCP Python SDK provides the `Server` class that handles lifecycle management, JSON-RPC parsing, and transport abstraction. You provide the tool and resource implementations.

```python
import asyncio
import json
from typing import Any

from mcp.server import Server, NotificationOptions
from mcp.server.models import InitializationOptions
from mcp.types import (
    GetPromptResult,
    Prompt,
    PromptArgument,
    PromptMessage,
    ReadResourceResult,
    Resource,
    ResourceTemplate,
    TextContent,
    Tool,
)


# Create the server instance
server = Server("example-server")


# Register the tools list handler
@server.list_tools()
async def handle_list_tools() -> list[Tool]:
    """Return the list of tools this server provides."""
    return [
        Tool(
            name="greet",
            description="Greet someone by name with a customizable message",
            inputSchema={
                "type": "object",
                "properties": {
                    "name": {
                        "type": "string",
                        "description": "The name of the person to greet",
                    },
                    "greeting": {
                        "type": "string",
                        "description": "Optional greeting word",
                    },
                },
                "required": ["name"],
            },
        ),
    ]


# Register the tool call handler
@server.call_tool()
async def handle_call_tool(name: str, arguments: dict) -> list[TextContent]:
    """Execute a tool by name with the given arguments."""
    if name == "greet":
        greeting = arguments.get("greeting", "Hello")
        name_val = arguments["name"]
        return [TextContent(type="text", text=f"{greeting}, {name_val}!")]

    raise ValueError(f"Unknown tool: {name}")


# Register resource handlers
@server.list_resources()
async def handle_list_resources() -> list[Resource]:
    """Return the list of available resources."""
    return [
        Resource(
            uri="info://server/status",
            name="Server Status",
            description="Current server status information",
            mimeType="application/json",
        ),
    ]


@server.read_resource()
async def handle_read_resource(uri: str) -> str:
    """Read a resource by its URI."""
    if uri == "info://server/status":
        return json.dumps({
            "status": "running",
            "version": "1.0.0",
            "uptime": "72h",
        })

    raise ValueError(f"Unknown resource: {uri}")


async def main():
    """Run the server over stdio transport."""
    async with mcp.server.stdio() as (read_stream, write_stream):
        await server.run(
            read_stream,
            write_stream,
            InitializationOptions(
                server_name="example-server",
                server_version="1.0.0",
                capabilities=server.get_capabilities(
                    notification_options=NotificationOptions(),
                    experimental_capabilities={},
                ),
            ),
        )


if __name__ == "__main__":
    asyncio.run(main())
```

### 3. Building a Practical File Search Server

Now build something real: a server that searches files and returns results. This combines file system access with structured tool definitions.

```python
import asyncio
import json
import os
from pathlib import Path

from mcp.server import Server, NotificationOptions
from mcp.server.models import InitializationOptions
from mcp.types import TextContent, Tool


server = Server("file-search-server")
WORKSPACE = Path(os.environ.get("MCP_WORKSPACE", "."))


@server.list_tools()
async def list_tools() -> list[Tool]:
    return [
        Tool(
            name="search_files",
            description=(
                "Search for files in the workspace matching a pattern. "
                "Returns file paths, sizes, and modification times."
            ),
            inputSchema={
                "type": "object",
                "properties": {
                    "pattern": {
                        "type": "string",
                        "description": "File pattern to match (e.g., '*.py', '*.md')",
                    },
                    "max_results": {
                        "type": "number",
                        "description": "Maximum number of results to return",
                        "default": 20,
                    },
                },
                "required": ["pattern"],
            },
        ),
        Tool(
            name="search_text",
            description=(
                "Search for text within files in the workspace. "
                "Returns matching file paths and line numbers."
            ),
            inputSchema={
                "type": "object",
                "properties": {
                    "query": {
                        "type": "string",
                        "description": "Text to search for within files",
                    },
                    "file_pattern": {
                        "type": "string",
                        "description": "Only search files matching this pattern (e.g., '*.py')",
                    },
                    "max_results": {
                        "type": "number",
                        "description": "Maximum results per file",
                        "default": 10,
                    },
                },
                "required": ["query"],
            },
        ),
    ]


@server.call_tool()
async def call_tool(name: str, arguments: dict) -> list[TextContent]:
    if name == "search_files":
        return await handle_search_files(arguments)
    elif name == "search_text":
        return await handle_search_text(arguments)
    raise ValueError(f"Unknown tool: {name}")


async def handle_search_files(args: dict) -> list[TextContent]:
    """Search for files matching a pattern."""
    pattern = args["pattern"]
    max_results = min(args.get("max_results", 20), 100)

    results = []
    for path in WORKSPACE.rglob(pattern):
        if len(results) >= max_results:
            break
        stat = path.stat()
        results.append(
            f"{path.relative_to(WORKSPACE)}  {stat.st_size} bytes"
        )

    if not results:
        return [TextContent(type="text", text="No files found.")]

    return [
        TextContent(
            type="text",
            text=f"Found {len(results)} files:\n" + "\n".join(results),
        )
    ]


async def handle_search_text(args: dict) -> list[TextContent]:
    """Search for text within files."""
    query = args["query"].lower()
    file_pattern = args.get("file_pattern", "*")
    max_results = min(args.get("max_results", 10), 50)

    results = []
    for path in WORKSPACE.rglob(file_pattern):
        if not path.is_file():
            continue
        try:
            text = path.read_text(errors="ignore")
            matches = []
            for i, line in enumerate(text.splitlines(), 1):
                if query in line.lower():
                    matches.append(f"  Line {i}: {line.strip()[:80]}")
                    if len(matches) >= max_results:
                        break
            if matches:
                rel_path = path.relative_to(WORKSPACE)
                results.append(f"\n{rel_path}:")
                results.extend(matches)
        except Exception:
            continue

    if not results:
        return [TextContent(type="text", text="No matches found.")]

    return [
        TextContent(
            type="text",
            text="Search results:" + "\n".join(results),
        )
    ]


async def main():
    async with mcp.server.stdio() as (read_stream, write_stream):
        await server.run(
            read_stream,
            write_stream,
            InitializationOptions(
                server_name="file-search-server",
                server_version="1.0.0",
                capabilities=server.get_capabilities(
                    notification_options=NotificationOptions(),
                    experimental_capabilities={},
                ),
            ),
        )


if __name__ == "__main__":
    asyncio.run(main())
```

### 4. Adding Resource Templates

Resource templates let clients access dynamic URIs. For example, a file server can expose any file as a resource identified by its path.

```python
@server.list_resource_templates()
async def list_resource_templates() -> list[ResourceTemplate]:
    """Declare available resource templates."""
    return [
        ResourceTemplate(
            uriTemplate="file://{path}",
            name="File Content",
            description="Read the content of any file in the workspace",
            mimeType="text/plain",
        ),
    ]


# Update the read_resource handler to handle templates
@server.read_resource()
async def read_resource(uri: str) -> str:
    """Read a resource, supporting both static and templated URIs."""

    # Handle static resources
    if uri == "info://server/status":
        return json.dumps({"status": "running"})

    # Handle templated resources: file://{path}
    if uri.startswith("file://"):
        path_part = uri[len("file://"):]
        full_path = (WORKSPACE / path_part).resolve()

        # Security: prevent path traversal
        if not str(full_path).startswith(str(WORKSPACE.resolve())):
            raise ValueError("Path traversal detected")

        if not full_path.exists():
            raise ValueError(f"File not found: {path_part}")

        return full_path.read_text(errors="replace")

    raise ValueError(f"Unknown resource: {uri}")


# Register this handler on the existing server object
# server.read_resource = read_resource
```

### 5. Testing with the MCP Inspector

The MCP Inspector is a web-based tool for testing MCP servers. Run it alongside your server to verify initialization, tool calls, and resource reads.

```bash
# Install and run the MCP Inspector
npx @modelcontextprotocol/inspector python server.py
```

The Inspector lets you:
- See the initialization handshake in real time
- List tools and call them with custom arguments
- List and read resources
- View the raw JSON-RPC messages

## Build Something Real

Build a complete "System Info Server" that exposes system information as resources and diagnostic tools. This server is immediately useful for any agent that needs to understand its environment.

```python
import asyncio
import json
import os
import platform
import shutil
from datetime import datetime, timezone

from mcp.server import Server, NotificationOptions
from mcp.server.models import InitializationOptions
from mcp.types import Resource, TextContent, Tool


server = Server("system-info-server")


# ── Resources ─────────────────────────────────────────────────

@server.list_resources()
async def list_resources() -> list[Resource]:
    return [
        Resource(
            uri="system://info",
            name="System Information",
            description="Operating system, CPU, and memory info",
            mimeType="application/json",
        ),
        Resource(
            uri="system://environment",
            name="Environment Variables",
            description="Current environment variables",
            mimeType="application/json",
        ),
        Resource(
            uri="system://disk",
            name="Disk Usage",
            description="Disk usage for the current volume",
            mimeType="application/json",
        ),
    ]


@server.read_resource()
async def read_resource(uri: str) -> str:
    if uri == "system://info":
        return json.dumps({
            "os": platform.system(),
            "os_version": platform.version(),
            "python_version": platform.python_version(),
            "hostname": platform.node(),
            "cpus": os.cpu_count(),
        }, indent=2)

    if uri == "system://environment":
        return json.dumps(dict(os.environ), indent=2)

    if uri == "system://disk":
        usage = shutil.disk_usage(".")
        return json.dumps({
            "total_gb": round(usage.total / (2**30), 1),
            "used_gb": round(usage.used / (2**30), 1),
            "free_gb": round(usage.free / (2**30), 1),
            "percent_used": round(usage.used / usage.total * 100, 1),
        }, indent=2)

    raise ValueError(f"Unknown resource: {uri}")


# ── Tools ────────────────────────────────────────────────────

@server.list_tools()
async def list_tools() -> list[Tool]:
    return [
        Tool(
            name="check_port",
            description=(
                "Check if a network port is available on localhost. "
                "Returns whether the port is open and which process is using it."
            ),
            inputSchema={
                "type": "object",
                "properties": {
                    "port": {
                        "type": "number",
                        "description": "Port number to check (1-65535)",
                    },
                },
                "required": ["port"],
            },
        ),
        Tool(
            name="disk_cleanup_report",
            description=(
                "Analyze the current directory and report which "
                "file types consume the most disk space."
            ),
            inputSchema={
                "type": "object",
                "properties": {
                    "path": {
                        "type": "string",
                        "description": "Directory to analyze",
                    },
                    "min_size_mb": {
                        "type": "number",
                        "description": "Only report files larger than this (MB)",
                        "default": 1,
                    },
                },
            },
        ),
    ]


@server.call_tool()
async def call_tool(name: str, arguments: dict) -> list[TextContent]:
    if name == "check_port":
        port = int(arguments["port"])
        import socket
        sock = socket.socket(socket.AF_INET, socket.SOCK_STREAM)
        result = sock.connect_ex(("127.0.0.1", port))
        sock.close()
        status = "open" if result == 0 else "closed"
        return [TextContent(
            type="text",
            text=f"Port {port} is {status}.",
        )]

    if name == "disk_cleanup_report":
        target = Path(arguments.get("path", "."))
        min_bytes = arguments.get("min_size_mb", 1) * 1024 * 1024
        sizes = {}
        for f in target.rglob("*"):
            if f.is_file() and f.stat().st_size >= min_bytes:
                ext = f.suffix or "(no extension)"
                sizes[ext] = sizes.get(ext, 0) + f.stat().st_size

        sorted_types = sorted(sizes.items(), key=lambda x: -x[1])
        lines = ["Disk usage by file type:"]
        for ext, total_bytes in sorted_types[:10]:
            mb = round(total_bytes / (1024 * 1024), 1)
            lines.append(f"  {ext}: {mb} MB")
        return [TextContent(type="text", text="\n".join(lines))]

    raise ValueError(f"Unknown tool: {name}")


async def main():
    async with mcp.server.stdio() as (read_stream, write_stream):
        await server.run(
            read_stream,
            write_stream,
            InitializationOptions(
                server_name="system-info-server",
                server_version="1.0.0",
                capabilities=server.get_capabilities(
                    notification_options=NotificationOptions(),
                    experimental_capabilities={},
                ),
            ),
        )


if __name__ == "__main__":
    asyncio.run(main())
```
