# Build It: Implementing Agent Observability

⏱️ 25 minutes

## Breaking It Down

### 1. Structured Logging with Trace IDs

Start by adding a trace ID to every agent session and logging every event with context:

```python
# agent_logger.py
import json
import uuid
from datetime import datetime
from typing import Any

class AgentLogger:
    def __init__(self, session_id: str = None):
        self.session_id = session_id or str(uuid.uuid4())
        self.trace_id = str(uuid.uuid4())
        self.events: list[dict] = []

    def _log(self, event_type: str, payload: dict, severity: str = "INFO"):
        self.events.append({
            "timestamp": datetime.now().isoformat(),
            "session_id": self.session_id,
            "trace_id": self.trace_id,
            "event_type": event_type,
            "severity": severity,
            "payload": payload,
        })

    def log_user_input(self, text: str):
        self._log("user_input", {"text": text[:500]})

    def log_llm_call(self, model: str, prompt_tokens: int, max_tokens: int):
        self._log("llm_call", {
            "model": model,
            "prompt_tokens": prompt_tokens,
            "max_tokens": max_tokens,
        })

    def log_llm_response(self, response_text: str, completion_tokens: int):
        self._log("llm_response", {
            "text": response_text[:500],
            "completion_tokens": completion_tokens,
        })

    def log_tool_call(self, tool_name: str, arguments: dict, duration_ms: int):
        self._log("tool_call", {
            "tool": tool_name,
            "arguments": arguments,
            "duration_ms": duration_ms,
        })

    def log_tool_result(self, tool_name: str, result_summary: str, success: bool):
        self._log("tool_result", {
            "tool": tool_name,
            "result": result_summary[:300],
            "success": success,
        })

    def log_error(self, source: str, error: str):
        self._log("error", {"source": source, "error": str(error)}, severity="ERROR")

    def export(self) -> str:
        return json.dumps(self.events, indent=2)
```

Usage in an agent session:

```python
logger = AgentLogger()
logger.log_user_input("Find the top 5 customers by revenue this month")
logger.log_llm_call("claude-sonnet-4-6", prompt_tokens=2048, max_tokens=4096)
# ... after model responds with a tool call:
logger.log_tool_call("read_query", {"sql": "SELECT ..."}, duration_ms=320)
logger.log_tool_result("read_query", "Returned 5 rows", success=True)
```

### 2. Adding OpenTelemetry Tracing

OpenTelemetry provides a standard way to create spans that connect across servers:

```python
# tracing.py
from opentelemetry import trace
from opentelemetry.exporter.otlp.proto.http.trace_exporter import OTLPSpanExporter
from opentelemetry.sdk.trace import TracerProvider
from opentelemetry.sdk.trace.export import BatchSpanProcessor

# Initialize tracer
provider = TracerProvider()
processor = BatchSpanProcessor(OTLPSpanExporter(endpoint="http://localhost:4318/v1/traces"))
provider.add_span_processor(processor)
trace.set_tracer_provider(provider)
tracer = trace.get_tracer("agent-tracer")

# Usage inside MCP server:
@server.tool("read_query")
async def read_query(sql: str) -> TextContent:
    with tracer.start_as_current_span("db_query") as span:
        span.set_attribute("db.system", "sqlite")
        span.set_attribute("db.statement", sql[:200])
        start = time.time()
        try:
            result = execute_sql(sql)
            duration = (time.time() - start) * 1000
            span.set_attribute("db.duration_ms", duration)
            span.set_attribute("db.rows_returned", len(result))
            return TextContent(type="text", text=str(result))
        except Exception as e:
            span.record_exception(e)
            raise
```

### 3. Building Metrics Collection

Track key agent metrics over time:

```python
# metrics.py
from collections import defaultdict
from datetime import datetime, timedelta
import statistics

class AgentMetrics:
    def __init__(self):
        self.tool_counts = defaultdict(int)
        self.tool_durations = defaultdict(list)
        self.tool_errors = defaultdict(int)
        self.token_usage = []
        self.session_durations = []
        self.session_count = 0

    def record_tool_call(self, tool_name: str, duration_ms: int, error: bool = False):
        self.tool_counts[tool_name] += 1
        self.tool_durations[tool_name].append(duration_ms)
        if error:
            self.tool_errors[tool_name] += 1

    def record_token_usage(self, prompt: int, completion: int):
        self.token_usage.append({"prompt": prompt, "completion": completion, "total": prompt + completion})

    def record_session_end(self, duration_ms: int):
        self.session_durations.append(duration_ms)
        self.session_count += 1

    def summary(self) -> dict:
        """Compute aggregate metrics for dashboards."""
        total_tokens = sum(t["total"] for t in self.token_usage)
        avg_cost_per_session = (sum(t["prompt"] for t in self.token_usage) * 3
            + sum(t["completion"] for t in self.token_usage) * 15) / max(len(self.token_usage), 1) / 1_000_000

        return {
            "total_sessions": self.session_count,
            "avg_session_duration_ms": round(statistics.mean(self.session_durations), 0) if self.session_durations else 0,
            "total_tokens_used": total_tokens,
            "avg_tokens_per_session": round(total_tokens / max(self.session_count, 1), 0),
            "avg_cost_per_session_usd": round(avg_cost_per_session, 4),
            "total_tool_calls": sum(self.tool_counts.values()),
            "tool_breakdown": {
                name: {
                    "count": count,
                    "avg_duration_ms": round(statistics.mean(self.tool_durations[name]), 0),
                    "error_rate_pct": round(self.tool_errors[name] / count * 100, 1),
                }
                for name, count in sorted(self.tool_counts.items(), key=lambda x: x[1], reverse=True)
            },
        }
```

### 4. Integrating with an Observability Backend

Push logs and metrics to an aggregator like Grafana Loki or Grafana Mimir:

```python
# export_observability.py
import httpx
import json

async def push_logs(logger: AgentLogger, endpoint: str):
    """Push structured logs to a log aggregator."""
    async with httpx.AsyncClient() as client:
        payload = {
            "streams": [
                {
                    "stream": {
                        "session_id": logger.session_id,
                        "trace_id": logger.trace_id,
                    },
                    "values": [
                        [str(event["timestamp"]), json.dumps(event)]
                        for event in logger.events
                    ],
                }
            ]
        }
        await client.post(f"{endpoint}/loki/api/v1/push", json=payload)

async def push_metrics(metrics: AgentMetrics, endpoint: str):
    """Push aggregated metrics to a time-series database."""
    summary = metrics.summary()
    # In production, format as Prometheus metrics or push to Grafana Mimir
    print(f"Metrics pushed: {json.dumps(summary, indent=2)}")
```

### 5. Building a Real-Time Dashboard Query

Query aggregated logs for debugging:

```python
# dashboard_queries.py
def agent_dashboard_queries():
    """Return the key queries for an agent observability dashboard."""
    return {
        "sessions_last_24h": """
            SELECT count(distinct session_id)
            FROM agent_logs
            WHERE timestamp > now() - interval '24 hours'
        """,
        "avg_latency_by_tool": """
            SELECT tool, avg(duration_ms) as avg_ms, count(*) as calls
            FROM tool_calls
            WHERE timestamp > now() - interval '24 hours'
            GROUP BY tool
            ORDER BY calls DESC
        """,
        "error_rate_trend": """
            SELECT date_trunc('hour', timestamp) as hour,
                   count(*) as total,
                   sum(case when severity='ERROR' then 1 else 0 end) as errors
            FROM agent_logs
            WHERE timestamp > now() - interval '7 days'
            GROUP BY hour
            ORDER BY hour
        """,
        "top_slow_sessions": """
            SELECT session_id, max(duration_ms) as total_time
            FROM tool_calls
            WHERE timestamp > now() - interval '24 hours'
            GROUP BY session_id
            ORDER BY total_time DESC
            LIMIT 10
        """,
    }
```

## How It Works Internally

Observability in agent systems follows the same distributed tracing principles as microservices, but with a key difference: the "service boundaries" are conceptual rather than network-based. A single agent session spans:

1. **User input** — enters the system
2. **System prompt construction** — the host prepends instructions and tool definitions
3. **LLM inference** — the model generates a response (this is the longest span, often 2-10 seconds)
4. **Tool call** — the model's tool call is dispatched to the MCP server
5. **Tool execution** — the server processes the request (filesystem I/O, API call, database query)
6. **Result processing** — the result is returned to the model for the next reasoning step

Each of these spans gets a `trace_id` that links it to the parent session. The OpenTelemetry exporter sends these spans to a collector (Grafana Tempo, Jaeger, or Honeycomb), which reconstructs the full waterfall diagram.
