# Planning and Decision Making — Build It

*Reading time: 14 minutes*

## Breaking It Down

### 1. Flat Planning

The agent generates a complete plan upfront, then executes each step in order.

```python
import anthropic
import json

client = anthropic.Anthropic()


def flat_plan_agent(task: str) -> str:
    """Generate a plan upfront, then execute step by step."""
    # Phase 1: Plan
    plan_prompt = (
        f"Task: {task}\n\n"
        f"Break this task into a numbered list of steps. "
        f"Each step should be a single tool action or decision. "
        f"Respond with only the numbered steps, one per line."
    )

    plan_response = client.messages.create(
        model="claude-3-5-sonnet-20241022",
        max_tokens=500,
        temperature=0,
        messages=[{"role": "user", "content": plan_prompt}]
    )

    plan = plan_response.content[0].text
    print(f"=== PLAN ===\n{plan}\n")

    # Phase 2: Execute each step
    steps = [s.strip() for s in plan.split("\n") if s.strip() and s[0].isdigit()]
    results = []

    for step in steps:
        execute_prompt = (
            f"Original task: {task}\n"
            f"Current plan step: {step}\n"
            f"Previous results: {json.dumps(results)}\n\n"
            f"Execute this step and report the result. "
            f"If this step requires information from a previous step, "
            f"use the previous results."
        )

        step_response = client.messages.create(
            model="claude-3-5-sonnet-20241022",
            max_tokens=500,
            temperature=0,
            messages=[{"role": "user", "content": execute_prompt}],
        )

        step_result = step_response.content[0].text
        results.append({"step": step, "result": step_result})
        print(f"Step result: {step_result}\n")

    # Phase 3: Synthesize
    synthesis_prompt = (
        f"Original task: {task}\n\n"
        f"Step results:\n{json.dumps(results, indent=2)}\n\n"
        f"Synthesize these results into a final answer for the user."
    )

    final_response = client.messages.create(
        model="claude-3-5-sonnet-20241022",
        max_tokens=500,
        temperature=0,
        messages=[{"role": "user", "content": synthesis_prompt}]
    )

    return final_response.content[0].text


result = flat_plan_agent(
    "Research and compare three cloud providers: AWS, Azure, and GCP. "
    "Compare their AI/ML services, pricing models, and free tier offerings."
)
print(f"=== FINAL ===\n{result}")
```

Flat planning works when all steps are known in advance. The agent cannot adapt if an intermediate step fails — it blindly executes the original plan.

### 2. Dynamic Replanning

The agent maintains a plan but revises it based on execution results.

```python
class DynamicPlanner:
    """Agent that replans when a step fails or returns unexpected results."""

    def __init__(self):
        self.plan: list[str] = []
        self.completed: list[dict] = []
        self.max_replans = 3

    def create_plan(self, task: str):
        """Generate an initial plan."""
        response = client.messages.create(
            model="claude-3-5-sonnet-20241022",
            max_tokens=500,
            temperature=0,
            messages=[{
                "role": "user",
                "content": (
                    f"Task: {task}\n"
                    f"Create a step-by-step plan. Each step should be a concrete action. "
                    f"Number each step. Respond with only the numbered steps."
                )
            }]
        )
        self.plan = [
            s.strip() for s in response.content[0].text.split("\n")
            if s.strip() and s[0].isdigit()
        ]

    def execute_step(self, step: str) -> dict:
        """Execute a single plan step."""
        context = f"Executing step: {step}\n\n"
        for c in self.completed:
            context += f"Step '{c['step']}' completed with: {c['result']}\n"

        response = client.messages.create(
            model="claude-3-5-sonnet-20241022",
            max_tokens=300,
            temperature=0,
            messages=[{
                "role": "user",
                "content": (
                    f"{context}\n"
                    f"Execute this step and report what happened. "
                    f"If the step succeeded, start with 'SUCCESS:'. "
                    f"If it failed, start with 'FAILED:' and explain why."
                )
            }]
        )
        result = response.content[0].text
        return {"step": step, "result": result, "succeeded": result.startswith("SUCCESS")}

    def replan(self, task: str, failed_step: str) -> list[str]:
        """Generate a new plan from the current state."""
        context = f"Original task: {task}\n"
        for c in self.completed:
            context += f"✓ {c['step']}\n"
        context += f"✗ {failed_step} (FAILED)\n"

        response = client.messages.create(
            model="claude-3-5-sonnet-20241022",
            max_tokens=500,
            temperature=0,
            messages=[{
                "role": "user",
                "content": (
                    f"{context}\n"
                    f"A step in the plan failed. Create a revised plan that "
                    f"works around the failure. List only the remaining steps."
                )
            }]
        )
        return [
            s.strip() for s in response.content[0].text.split("\n")
            if s.strip() and s[0].isdigit()
        ]

    def run(self, task: str) -> str:
        """Execute the task with dynamic replanning."""
        self.create_plan(task)
        replan_count = 0

        while self.plan and replan_count <= self.max_replans:
            step = self.plan.pop(0)
            print(f"Executing: {step[:60]}...")
            result = self.execute_step(step)
            self.completed.append(result)

            if not result["succeeded"]:
                print(f"  FAILED. Replanning...")
                replan_count += 1
                remaining = self.replan(task, step)
                if remaining:
                    self.plan = remaining + self.plan
                else:
                    return "Could not recover from failure. No alternative plan available."

        # Synthesize final result
        synthesis = client.messages.create(
            model="claude-3-5-sonnet-20241022",
            max_tokens=300,
            temperature=0,
            messages=[{
                "role": "user",
                "content": (
                    f"Task: {task}\n\n"
                    f"Completed steps:\n{json.dumps(self.completed, indent=2)}\n\n"
                    f"Provide a final summary."
                )
            }]
        )
        return synthesis.content[0].text


planner = DynamicPlanner()
result = planner.run("Plan a weekend trip: find a destination within 200 miles, "
                     "check if there are hotels under $150/night, and create a packing list.")
print(f"\n=== FINAL RESULT ===\n{result}")
```

### 3. Decision Strategies

Choose between actions using different strategies.

```python
def greedy_choice(options: list[dict], context: str) -> dict:
    """Greedy: pick the option with the highest immediate score."""
    scores = {o["name"]: o.get("score", 0) for o in options}
    best = max(scores, key=scores.get)
    return next(o for o in options if o["name"] == best)


def llm_choice(options: list[dict], context: str) -> dict:
    """LLM-assisted: ask the model to evaluate and choose."""
    options_text = "\n".join(
        f"- {o['name']}: {o.get('description', 'No description')}"
        for o in options
    )

    response = client.messages.create(
        model="claude-3-5-sonnet-20241022",
        max_tokens=200,
        temperature=0,
        messages=[{
            "role": "user",
            "content": (
                f"Context: {context}\n\n"
                f"Options:\n{options_text}\n\n"
                f"Which option is best? Respond with the option name only."
            )
        }]
    )
    chosen_name = response.content[0].text.strip()
    return next(o for o in options if o["name"] in chosen_name)


# Example: choose a restaurant
options = [
    {"name": "Italian Bistro", "score": 7, "description": "Pasta and pizza, $$"},
    {"name": "Sushi Bar", "score": 9, "description": "Fresh sushi, $$$"},
    {"name": "Taco Stand", "score": 6, "description": "Street tacos, $"},
]

context = "Group of 4 friends, one is vegetarian. Budget is moderate."
print(f"Greedy choice: {greedy_choice(options, context)['name']}")
print(f"LLM choice: {llm_choice(options, context)['name']}")
```

## How It Works Internally

Planning in agents mirrors classical AI planning but uses the LLM as the planner instead of a symbolic planner. The LLM's advantage is that it can interpret natural language goals and generate plans for open-ended tasks. The disadvantage is that LLM-generated plans may miss steps or include impossible actions.

The replanning loop is a simple state machine:

1. **Plan** — decompose task into steps
2. **Execute** — run one step
3. **Evaluate** — did the step succeed?
4. If yes: continue to next step
5. If no: replan from current state, go to step 2

## Build Something Real

Build a trip planning agent that uses hierarchical planning.

```python
import anthropic
import json

client = anthropic.Anthropic()


class TripPlannerAgent:
    """A hierarchical planning agent for trip planning."""

    def __init__(self):
        self.tools = [
            {
                "name": "search_flights",
                "description": "Search for flights between two cities",
                "input_schema": {
                    "type": "object",
                    "properties": {
                        "origin": {"type": "string"},
                        "destination": {"type": "string"},
                        "date": {"type": "string"}
                    },
                    "required": ["origin", "destination", "date"]
                }
            },
            {
                "name": "search_hotels",
                "description": "Search for hotels in a city",
                "input_schema": {
                    "type": "object",
                    "properties": {
                        "city": {"type": "string"},
                        "max_price": {"type": "number"}
                    },
                    "required": ["city"]
                }
            },
        ]

    def _simulate_search_flights(self, origin: str, destination: str, date: str) -> str:
        """Simulated flight search."""
        return json.dumps([
            {"airline": "SkyWays", "price": 350, "departure": "08:00", "arrival": "10:30"},
            {"airline": "AirConnect", "price": 280, "departure": "14:00", "arrival": "16:30"},
        ])

    def _simulate_search_hotels(self, city: str, max_price: float | None = None) -> str:
        """Simulated hotel search."""
        hotels = [
            {"name": "City Inn", "price": 120, "rating": 4.0},
            {"name": "Grand Hotel", "price": 200, "rating": 4.5},
            {"name": "Budget Stay", "price": 75, "rating": 3.5},
        ]
        if max_price:
            hotels = [h for h in hotels if h["price"] <= max_price]
        return json.dumps(hotels)

    def decompose(self, task: str) -> list[str]:
        """Decompose the task into subtasks (hierarchical planning)."""
        response = client.messages.create(
            model="claude-3-5-sonnet-20241022",
            max_tokens=600,
            temperature=0,
            messages=[{
                "role": "user",
                "content": (
                    f"Task: {task}\n\n"
                    f"Decompose this into 2-4 high-level subtasks. "
                    f"Each subtask should be a self-contained piece of work. "
                    f"Format as a JSON list of strings. Example:\n"
                    f'["Find flights to destination", "Book hotel", "Plan activities"]'
                )
            }]
        )
        text = response.content[0].text
        # Extract JSON list from response
        import re
        match = re.search(r'\[.*?\]', text, re.DOTALL)
        if match:
            return json.loads(match.group())
        return [task]  # Fallback: treat as single task

    def execute_subtask(self, subtask: str, context: str) -> str:
        """Execute a subtask using the agent loop."""
        messages = [
            {"role": "user", "content": f"Context: {context}\n\nSubtask: {subtask}"}
        ]

        for turn in range(8):
            response = client.messages.create(
                model="claude-3-5-sonnet-20241022",
                max_tokens=500,
                temperature=0,
                system="You are a trip planning assistant. Complete the subtask.",
                messages=messages,
                tools=self.tools,
            )

            content = response.content[0]
            if content.type == "tool_use":
                if content.name == "search_flights":
                    result = self._simulate_search_flights(**content.input)
                elif content.name == "search_hotels":
                    result = self._simulate_search_hotels(**content.input)
                else:
                    result = f"Unknown tool: {content.name}"

                messages.append({"role": "assistant", "content": content})
                messages.append({
                    "role": "user",
                    "content": [
                        {"type": "tool_result", "tool_use_id": content.id, "content": result}
                    ]
                })
            else:
                return content.text

        return "Subtask did not complete."

    def run(self, task: str) -> str:
        """Run the hierarchical planner."""
        print("=== DECOMPOSING TASK ===")
        subtasks = self.decompose(task)
        for i, st in enumerate(subtasks, 1):
            print(f"  Subtask {i}: {st}")

        subtask_results = []
        for i, subtask in enumerate(subtasks):
            print(f"\n=== EXECUTING SUBTASK {i + 1} ===")
            context = f"Original task: {task}\n" + "\n".join(
                f"Subtask {j + 1} result: {r}"
                for j, r in enumerate(subtask_results)
            )
            result = self.execute_subtask(subtask, context)
            print(f"  Result: {result[:100]}...")
            subtask_results.append(result)

        # Synthesize final answer
        print(f"\n=== SYNTHESIZING FINAL PLAN ===")
        synthesis = client.messages.create(
            model="claude-3-5-sonnet-20241022",
            max_tokens=500,
            temperature=0,
            messages=[{
                "role": "user",
                "content": (
                    f"Task: {task}\n\n"
                    f"Subtask results:\n{json.dumps(subtask_results, indent=2)}\n\n"
                    f"Combine these into a complete trip plan for the user."
                )
            }]
        )
        return synthesis.content[0].text


planner = TripPlannerAgent()
result = planner.run(
    "Plan a 3-day trip from New York to Chicago for next month. "
    "Find flights under $400 and hotels under $150 per night."
)
print(f"\n=== FINAL TRIP PLAN ===\n{result}")
```
