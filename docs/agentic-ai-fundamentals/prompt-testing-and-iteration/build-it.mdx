# Prompt Testing and Iteration — Build It

*Reading time: 12 minutes*

## Breaking It Down

### 1. Building a Test Dataset

A test dataset is a list of (input, expected_output) pairs. For a classification task:

```python
import anthropic

client = anthropic.Anthropic()


# Format: (input_text, expected_category)
TEST_DATASET = [
    # BUG cases
    ("The app crashes when I click the submit button.", "BUG"),
    ("Login returns a 500 error every time.", "BUG"),
    ("The search results do not match the query.", "BUG"),
    ("Images fail to load on the product page.", "BUG"),

    # FEATURE_REQUEST cases
    ("It would be great to have dark mode support.", "FEATURE_REQUEST"),
    ("Can you add export to CSV functionality?", "FEATURE_REQUEST"),
    ("I wish the dashboard had real-time updates.", "FEATURE_REQUEST"),

    # GENERAL cases
    ("What are your business hours?", "GENERAL"),
    ("How do I reset my password?", "GENERAL"),
    ("Thanks for the quick response!", "GENERAL"),

    # Edge cases
    ("", "GENERAL"),  # empty input
    ("""Very long message about multiple topics...  testing...  okay.""", "GENERAL"),  # ambiguous
]


def run_test_suite(
    system_prompt: str,
    dataset: list[tuple[str, str]]
) -> dict:
    """Run a prompt against every test case and compute metrics."""
    correct = 0
    total = len(dataset)
    results = []

    for text, expected in dataset:
        response = client.messages.create(
            model="claude-3-5-sonnet-20241022",
            max_tokens=10,
            temperature=0,
            system=system_prompt,
            messages=[{"role": "user", "content": text or "empty"}]
        )

        predicted = response.content[0].text.strip()
        is_correct = predicted == expected
        if is_correct:
            correct += 1

        results.append({
            "input": text[:50],
            "expected": expected,
            "predicted": predicted,
            "correct": is_correct
        })

    accuracy = correct / total if total > 0 else 0
    errors = [r for r in results if not r["correct"]]

    return {
        "accuracy": round(accuracy, 3),
        "correct": correct,
        "total": total,
        "results": results,
        "errors": errors,
        "error_rate": round(1 - accuracy, 3)
    }


# Test a simple prompt
system_prompt = (
    "Classify the text into BUG, FEATURE_REQUEST, or GENERAL. "
    "Respond with only the category."
)

evaluation = run_test_suite(system_prompt, TEST_DATASET)
print(f"Accuracy: {evaluation['accuracy']} ({evaluation['correct']}/{evaluation['total']})")

if evaluation["errors"]:
    print(f"\nErrors ({len(evaluation['errors'])}):")
    for err in evaluation["errors"]:
        print(f"  Expected={err['expected']} Got={err['predicted']} | {err['input']}")
```

### 2. A/B Testing Two Prompt Versions

Compare two prompt designs on the same dataset to pick the better one:

```python
def compare_prompts(
    prompt_a: str,
    prompt_b: str,
    dataset: list[tuple[str, str]],
    label_a: str = "Prompt A",
    label_b: str = "Prompt B"
) -> dict:
    """Compare two prompt versions side by side."""
    results_a = run_test_suite(prompt_a, dataset)
    results_b = run_test_suite(prompt_b, dataset)

    comparison = {
        label_a: {
            "accuracy": results_a["accuracy"],
            "errors": results_a["errors"],
        },
        label_b: {
            "accuracy": results_b["accuracy"],
            "errors": results_b["errors"],
        },
        "improvement": round(
            results_b["accuracy"] - results_a["accuracy"], 3
        ),
    }

    return comparison


prompt_v1 = (
    "Classify the text into BUG, FEATURE_REQUEST, or GENERAL. "
    "Respond with only the category."
)

prompt_v2 = (
    "You are a support ticket classifier with high accuracy.\n\n"
    "Rules:\n"
    "- BUG: The user reports something that is broken or not working\n"
    "- FEATURE_REQUEST: The user asks for something new or improved\n"
    "- GENERAL: Questions, feedback, or anything else\n\n"
    "Classify the text. Respond with only the category."
)

comparison = compare_prompts(prompt_v1, prompt_v2, TEST_DATASET)

for name, data in [("v1", prompt_v1), ("v2", prompt_v2)]:
    print(f"Prompt {name}: accuracy = {comparison[f'Prompt {name.upper()}']['accuracy']}")

print(f"\nImprovement: {comparison['improvement']}")

# Show which errors v2 fixed and which it introduced
errors_v1 = set((e["input"], e["expected"]) for e in comparison["Prompt A"]["errors"])
errors_v2 = set((e["input"], e["expected"]) for e in comparison["Prompt B"]["errors"])

fixed = errors_v1 - errors_v2
new = errors_v2 - errors_v1

if fixed:
    print(f"\nFixed by v2: {len(fixed)} cases")
if new:
    print(f"New errors in v2: {len(new)} cases")
```

### 3. Iterative Prompt Refinement

Track improvements across multiple iterations:

```python
def refine_prompt(
    base_prompt: str,
    dataset: list[tuple[str, str]],
    iterations: list[str]
) -> list[dict]:
    """Run multiple prompt iterations and track improvements."""
    history = []

    for i, prompt in enumerate(iterations):
        eval_result = run_test_suite(prompt, dataset)
        history.append({
            "iteration": i + 1,
            "accuracy": eval_result["accuracy"],
            "errors": eval_result["errors"],
            "prompt": prompt[:80],
        })

    return history


prompt_v1 = (
    "Classify the text into BUG, FEATURE_REQUEST, or GENERAL. "
    "Respond with only the category."
)

prompt_v2 = (
    "Classify the text into one of: BUG, FEATURE_REQUEST, GENERAL.\n\n"
    "BUG = something is broken or not working\n"
    "FEATURE_REQUEST = asking for something new\n"
    "GENERAL = everything else\n\n"
    "Category:"
)

prompt_v3 = prompt_v1 + (
    "\n\nExamples:\n"
    "App crashes on startup → BUG\n"
    "Please add dark mode → FEATURE_REQUEST\n"
    "What are your hours → GENERAL"
)

history = refine_prompt(prompt_v1, TEST_DATASET, [prompt_v1, prompt_v2, prompt_v3])

print("Refinement History:")
for entry in history:
    print(f"  Iteration {entry['iteration']}: accuracy={entry['accuracy']}, "
          f"errors={len(entry['errors'])}, prompt='{entry['prompt']}'")
```

## How It Works Internally

Prompt evaluation is an inference pipeline: for each test case, you send the prompt to the LLM, get the output, and compare it to the expected output. The comparison can be exact match (for classification), structural match (for JSON validation), or semantic similarity (for generation tasks).

The key insight is that prompt evaluation is not unit testing — it is model testing. The prompt is not deterministic code; it is a probabilistic instruction. A single test failure does not necessarily mean the prompt is wrong. You need enough test cases to distinguish signal from noise.

A minimum of 20-30 test cases gives you a statistically meaningful accuracy estimate. For production systems, aim for 100+ test cases covering all known categories and edge cases.

## Build Something Real

Build a complete prompt testing framework with reporting:

```python
import anthropic
import json
from dataclasses import dataclass
from datetime import datetime

client = anthropic.Anthropic()


@dataclass
class TestCase:
    """A single test case with input, expected output, and metadata."""
    input_text: str
    expected: str
    category: str = "general"
    notes: str = ""


@dataclass
class EvaluationReport:
    """Report from running a prompt against a test suite."""
    prompt_name: str
    prompt_text: str
    accuracy: float
    total: int
    passed: int
    failures: list[dict]
    timestamp: str

    def summary(self) -> str:
        return (
            f"[{self.timestamp}] {self.prompt_name}: "
            f"{self.passed}/{self.total} passed "
            f"({round(self.accuracy * 100, 1)}%)"
        )


class PromptTestSuite:
    """A test suite for evaluating and iterating on prompts."""

    def __init__(self, test_cases: list[TestCase]):
        self.test_cases = test_cases

    def evaluate(self, prompt_name: str, system_prompt: str) -> EvaluationReport:
        """Evaluate a prompt against all test cases."""
        passed = 0
        failures = []

        for tc in self.test_cases:
            response = client.messages.create(
                model="claude-3-5-sonnet-20241022",
                max_tokens=50,
                temperature=0,
                system=system_prompt,
                messages=[{"role": "user", "content": tc.input_text or "empty"}]
            )

            predicted = response.content[0].text.strip()
            if predicted == tc.expected:
                passed += 1
            else:
                failures.append({
                    "input": tc.input_text[:60],
                    "expected": tc.expected,
                    "predicted": predicted,
                    "category": tc.category,
                })

        accuracy = passed / len(self.test_cases) if self.test_cases else 0.0

        return EvaluationReport(
            prompt_name=prompt_name,
            prompt_text=system_prompt[:100],
            accuracy=accuracy,
            total=len(self.test_cases),
            passed=passed,
            failures=failures,
            timestamp=datetime.now().isoformat()
        )

    def compare(self, prompt_sets: dict[str, str]) -> list[EvaluationReport]:
        """Compare multiple prompt versions."""
        reports = []
        for name, prompt in prompt_sets.items():
            report = self.evaluate(name, prompt)
            reports.append(report)
            print(report.summary())
        return reports


# Build the test suite
test_cases = [
    TestCase("The app crashes when I click submit", "BUG", "crash"),
    TestCase("I would like dark mode please", "FEATURE_REQUEST", "feature"),
    TestCase("What are your hours of operation?", "GENERAL", "general"),
    TestCase("Login returns 500 error", "BUG", "broken"),
    TestCase("Can you add CSV export?", "FEATURE_REQUEST", "enhancement"),
]

suite = PromptTestSuite(test_cases)

# Compare versions
prompts = {
    "v1-basic": "Classify as BUG, FEATURE_REQUEST, or GENERAL. Respond with only the label.",
    "v2-with-definitions": (
        "You are a classifier. Categories:\n"
        "BUG = broken or crashing\n"
        "FEATURE_REQUEST = asking for new capability\n"
        "GENERAL = everything else\n"
        "Respond with only the label."
    ),
}

reports = suite.compare(prompts)

# Print detailed failure analysis
for report in reports:
    if report.failures:
        print(f"\n{report.prompt_name} failures:")
        for f in report.failures:
            print(f"  {f['input']} → predicted '{f['predicted']}', expected '{f['expected']}'")
```

This test suite framework lets you run evaluations automatically, compare versions, and track prompt quality over time.
