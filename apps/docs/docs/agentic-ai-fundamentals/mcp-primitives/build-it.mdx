# MCP Primitives — Build It

*Reading time: 14 minutes*

## Breaking It Down

### 1. Implementing Tools

Tools are the most commonly used MCP primitive. They follow the same structure as LLM tool definitions but are served dynamically.

```python
import json


class ToolPrimitive:
    """Build MCP tool definitions and handle tool calls."""

    @staticmethod
    def define(name: str, description: str,
               properties: dict, required: list[str] = None) -> dict:
        """Create a tool definition in MCP format."""
        return {
            "name": name,
            "description": description,
            "inputSchema": {
                "type": "object",
                "properties": properties,
                "required": required or [],
            },
        }

    @staticmethod
    def success(text: str) -> dict:
        """Create a successful tool response."""
        return {
            "content": [{"type": "text", "text": text}],
        }

    @staticmethod
    def error(message: str) -> dict:
        """Create a tool error response."""
        return {
            "content": [{"type": "text", "text": f"Error: {message}"}],
            "isError": True,
        }


# Example: define a search tool
search_tool = ToolPrimitive.define(
    name="search_docs",
    description="Search the documentation for matching content",
    properties={
        "query": {
            "type": "string",
            "description": "Search query string",
        },
        "max_results": {
            "type": "number",
            "description": "Maximum number of results",
            "default": 5,
        },
    },
    required=["query"],
)

print(json.dumps(search_tool, indent=2))
```

### 2. Implementing Resources

Resources expose data identified by a URI scheme. The server advertises resource templates that tell clients what URIs are available.

```python
class ResourcePrimitive:
    """Build MCP resource definitions and handle resource reads."""

    @staticmethod
    def define(uri: str, name: str, description: str = "",
               mime_type: str = "text/plain") -> dict:
        """Create a resource definition."""
        return {
            "uri": uri,
            "name": name,
            "description": description,
            "mimeType": mime_type,
        }

    @staticmethod
    def content(uri: str, text: str, mime_type: str = "text/plain") -> dict:
        """Create a resource content response."""
        return {
            "contents": [
                {
                    "uri": uri,
                    "mimeType": mime_type,
                    "text": text,
                }
            ],
        }

    @staticmethod
    def template(uri_template: str, name: str,
                 description: str = "") -> dict:
        """Create a resource template for dynamic URIs."""
        return {
            "uriTemplate": uri_template,
            "name": name,
            "description": description,
        }


# Example: define static resources and a resource template
static_resources = [
    ResourcePrimitive.define(
        uri="config://app/settings",
        name="Application Settings",
        description="Global application configuration",
        mime_type="application/json",
    ),
]

resource_templates = [
    ResourcePrimitive.template(
        uri_template="file://{path}",
        name="File Contents",
        description="Read any file by its path",
    ),
]

print("Static resources:", json.dumps(static_resources, indent=2))
print("\nResource templates:", json.dumps(resource_templates, indent=2))
```

### 3. Implementing Prompts

Prompts are server-side templates that clients can retrieve and use. They are useful for capturing prompt engineering expertise in reusable form.

```python
class PromptPrimitive:
    """Build MCP prompt definitions and handle prompt retrieval."""

    @staticmethod
    def define(name: str, description: str = "",
               arguments: list[dict] = None) -> dict:
        """Create a prompt definition."""
        return {
            "name": name,
            "description": description,
            "arguments": arguments or [],
        }

    @staticmethod
    def render(name: str, arguments: dict) -> dict:
        """Render a prompt template with the given arguments."""
        if name == "analyze_log":
            log_content = arguments.get("log", "")
            level = arguments.get("level", "error")
            return {
                "messages": [
                    {
                        "role": "user",
                        "content": {
                            "type": "text",
                            "text": (
                                f"Analyze the following log entries at "
                                f"severity level '{level}' or higher.\n\n"
                                f"{log_content}\n\n"
                                "For each entry, provide:\n"
                                "1. The timestamp and severity\n"
                                "2. A one-line summary of the issue\n"
                                "3. A suggested action"
                            ),
                        },
                    }
                ],
            }

        if name == "summarize_document":
            document = arguments.get("document", "")
            return {
                "messages": [
                    {
                        "role": "user",
                        "content": {
                            "type": "text",
                            "text": (
                                "Summarize the following document in "
                                "3-5 bullet points. Focus on actionable "
                                "information.\n\n"
                                f"{document}"
                            ),
                        },
                    }
                ],
            }

        return {"messages": []}


# Example: define available prompts
available_prompts = [
    PromptPrimitive.define(
        name="analyze_log",
        description="Analyze log entries at a given severity level",
        arguments=[
            {
                "name": "log",
                "description": "Log content to analyze",
                "required": True,
            },
            {
                "name": "level",
                "description": "Minimum severity level",
                "required": False,
            },
        ],
    ),
    PromptPrimitive.define(
        name="summarize_document",
        description="Summarize a document into bullet points",
        arguments=[
            {
                "name": "document",
                "description": "Document text to summarize",
                "required": True,
            },
        ],
    ),
]

print("Available prompts:", json.dumps(available_prompts, indent=2))

# Test rendering
rendered = PromptPrimitive.render("analyze_log", {
    "log": "ERROR 2025-06-01: Connection timeout\nWARN 2025-06-01: Retry attempt 1",
    "level": "warn",
})
print("\nRendered prompt:", json.dumps(rendered, indent=2))
```

### 4. Combining Primitives in One Server

A real MCP server typically exposes all three primitives. Here is how they work together in the server's message dispatch.

```python
from dataclasses import dataclass


@dataclass
class McpPrimitiveServer:
    """MCP server that handles all three primitives."""

    def __init__(self):
        self._tools: dict[str, dict] = {}
        self._resources: list[dict] = []
        self._prompts: list[dict] = {}
        self._initialized = False

    def add_tool(self, name: str, description: str,
                 input_schema: dict, handler: callable):
        self._tools[name] = {
            "definition": {
                "name": name, "description": description,
                "inputSchema": input_schema,
            },
            "handler": handler,
        }

    def add_resource(self, uri: str, name: str, mime_type: str,
                     reader: callable):
        self._resources.append({
            "uri": uri, "name": name, "mimeType": mime_type,
            "reader": reader,
        })

    def add_prompt(self, name: str, renderer: callable):
        self._prompts[name] = renderer

    def handle_request(self, method: str, params: dict) -> dict:
        """Route a method call to the appropriate handler."""
        if method == "tools/list":
            return {
                "tools": [t["definition"] for t in self._tools.values()]
            }

        if method == "tools/call":
            tool = self._tools.get(params.get("name"))
            if not tool:
                return ToolPrimitive.error(f"Unknown tool: {params.get('name')}")
            try:
                result = tool["handler"](**params.get("arguments", {}))
                return ToolPrimitive.success(str(result))
            except Exception as e:
                return ToolPrimitive.error(str(e))

        if method == "resources/list":
            return {
                "resources": [
                    {"uri": r["uri"], "name": r["name"],
                     "mimeType": r["mimeType"]}
                    for r in self._resources
                ]
            }

        if method == "resources/read":
            for r in self._resources:
                if r["uri"] == params.get("uri"):
                    text = r["reader"]()
                    return ResourcePrimitive.content(
                        r["uri"], text, r["mimeType"]
                    )
            return {"error": f"Resource not found: {params.get('uri')}"}

        if method == "prompts/list":
            return {
                "prompts": [
                    {"name": name, "description": p.__doc__ or ""}
                    for name, p in self._prompts.items()
                ]
            }

        if method == "prompts/get":
            renderer = self._prompts.get(params.get("name"))
            if not renderer:
                return {"error": f"Prompt not found: {params.get('name')}"}
            return renderer(**params.get("arguments", {}))

        return {"error": f"Method not found: {method}"}
```

## How It Works Internally

When a client sends `tools/list`, the server returns all tool definitions. The client's model then decides which tool to call. This is different from resources, where the client fetches data directly without model mediation. Tools go through the model's reasoning process; resources are direct data reads. This distinction is why choosing the right primitive matters — it determines whether the model decides to use something (Tool) or simply receives it (Resource).

## Build Something Real

Build a "Documentation Server" that exposes documentation pages as Resources, search as a Tool, and analysis templates as Prompts.

```python
from dataclasses import dataclass


@dataclass
class DocServer:
    """MCP server exposing documentation via all three primitives."""

    def build(self) -> McpPrimitiveServer:
        server = McpPrimitiveServer()

        # Tool: search documents
        def search_docs(query: str, max_results: int = 5):
            docs = {
                "agent": "An agent is an LLM driving a loop...",
                "tool": "Tools are callable functions...",
                "memory": "Memory stores agent context...",
            }
            results = [
                f"{name}: {text[:80]}..."
                for name, text in docs.items()
                if query.lower() in text.lower()
            ][:max_results]
            return "\n\n".join(results) if results else "No results found."

        server.add_tool(
            name="search_docs",
            description="Search documentation content",
            input_schema={
                "type": "object",
                "properties": {
                    "query": {"type": "string"},
                    "max_results": {"type": "number", "default": 5},
                },
                "required": ["query"],
            },
            handler=search_docs,
        )

        # Resource: homepage
        def read_homepage():
            return "# Documentation\n\nWelcome to the documentation."

        server.add_resource(
            uri="docs://home",
            name="Documentation Homepage",
            mime_type="text/markdown",
            reader=read_homepage,
        )

        # Prompt: summarize
        def summarize_prompt(**args):
            text = args.get("text", "")
            return {
                "messages": [{
                    "role": "user",
                    "content": {
                        "type": "text",
                        "text": f"Summarize this documentation:\n\n{text}",
                    },
                }]
            }

        server.add_prompt("summarize_doc", summarize_prompt)

        return server


# Build and verify
doc_server = DocServer().build()
print("Tools:", len(doc_server._tools))
print("Resources:", len(doc_server._resources))
print("Prompts:", len(doc_server._prompts))
```
