# Human in the Loop — Build It

*Reading time: 12 minutes*

## Breaking It Down

### 1. A Simple Approval Gate

The approval gate pauses the agent before executing a high-risk tool.

```python
import anthropic

client = anthropic.Anthropic()


class ApprovalGate:
    """
    An approval gate that pauses the agent and asks a human for permission.
    For this lesson, the approval function simulates human input via console.
    """

    def __init__(self, risky_tools: list[str]):
        self.risky_tools = risky_tools

    def needs_approval(self, tool_name: str, tool_args: dict) -> bool:
        """Check if a tool call needs human approval."""
        return tool_name in self.risky_tools

    def ask_human(self, tool_name: str, tool_args: dict) -> tuple[bool, str | None]:
        """
        Present the tool call to a human for approval.
        Returns (approved, modified_args).
        """
        print(f"\n⚠ APPROVAL REQUIRED")
        print(f"Tool: {tool_name}")
        print(f"Arguments: {tool_args}")
        print(f"Options: [a]pprove, [r]eject, [m]odify")

        choice = input("Choice (a/r/m): ").strip().lower()

        if choice == "a":
            return True, None
        elif choice == "r":
            return False, None
        elif choice == "m":
            modified = input("Enter modified arguments (JSON): ").strip()
            try:
                import json
                return True, json.loads(modified)
            except json.JSONDecodeError:
                print("Invalid JSON. Rejecting.")
                return False, None
        else:
            print("Invalid choice. Rejecting for safety.")
            return False, None


# Simulated risky tools
gate = ApprovalGate(risky_tools=["send_email", "delete_record", "process_refund"])
approved, modified = gate.ask_human("send_email", {"to": "user@example.com", "body": "Hello"})
print(f"Approved: {approved}, Modified args: {modified}")
```

In a real system, the `ask_human` function would send a notification to a dashboard, Slack channel, or email queue rather than blocking on console input.

### 2. Agent with Approval Gates

Integrate the approval gate into the agent loop.

```python
def agent_with_gates(user_input: str) -> str:
    """Agent loop with human approval gates on risky tools."""
    messages = [{"role": "user", "content": user_input}]
    gate = ApprovalGate(risky_tools=["send_email", "delete_record"])

    tools = [
        {
            "name": "lookup_record",
            "description": "Look up a record by ID",
            "input_schema": {
                "type": "object",
                "properties": {
                    "record_id": {"type": "string"}
                },
                "required": ["record_id"]
            }
        },
        {
            "name": "send_email",
            "description": "Send an email to a user",
            "input_schema": {
                "type": "object",
                "properties": {
                    "to": {"type": "string"},
                    "subject": {"type": "string"},
                    "body": {"type": "string"}
                },
                "required": ["to", "subject", "body"]
            }
        },
    ]

    for turn in range(10):
        response = client.messages.create(
            model="claude-3-5-sonnet-20241022",
            max_tokens=500,
            temperature=0,
            messages=messages,
            tools=tools,
        )

        content = response.content[0]

        if content.type == "tool_use":
            if gate.needs_approval(content.name, content.input):
                approved, modified = gate.ask_human(content.name, content.input)
                if not approved:
                    result = f"Human rejected: {content.name} was not approved."
                else:
                    args = modified if modified else content.input
                    result = f"Executed {content.name} with {args} (approved by human)"
            else:
                # Safe tool — execute directly
                result = f"Executed {content.name} — result: [data]"

            messages.append({"role": "assistant", "content": content})
            messages.append({
                "role": "user",
                "content": [
                    {"type": "tool_result", "tool_use_id": content.id, "content": result}
                ]
            })
        else:
            return content.text

    return "Max turns reached"


# Run with a request that triggers an approval gate
result = agent_with_gates("Look up record ID 42 and then send an email to alice@example.com "
                          "thanking her for her order.")
print(result)
```

### 3. Escalation Paths

When the agent cannot handle a task confidently, it should escalate to a human.

```python
def escalation_agent(user_input: str, confidence_threshold: float = 0.7) -> str:
    """Agent that escalates to a human when confidence is below threshold."""
    messages = [{"role": "user", "content": user_input}]

    # First, assess confidence
    confidence_response = client.messages.create(
        model="claude-3-5-sonnet-20241022",
        max_tokens=100,
        temperature=0,
        messages=[{
            "role": "user",
            "content": (
                f"On a scale of 0.0 to 1.0, how confident are you that you can "
                f"correctly handle this request without human help? "
                f"Respond with only a number.\n\nRequest: {user_input}"
            )
        }]
    )

    try:
        confidence = float(confidence_response.content[0].text.strip())
    except ValueError:
        confidence = 0.5

    print(f"Confidence: {confidence:.2f}")

    if confidence < confidence_threshold:
        return (
            f"ESCALATED TO HUMAN (confidence: {confidence:.2f})\n"
            f"Reason: The agent is not confident it can handle this request.\n"
            f"Original request: {user_input}\n"
            f"Context: This request requires human judgment."
        )

    # Proceed with normal agent handling
    tools = [{
        "name": "lookup_information",
        "description": "Look up information",
        "input_schema": {
            "type": "object",
            "properties": {"query": {"type": "string"}},
            "required": ["query"]
        }
    }]

    for turn in range(5):
        response = client.messages.create(
            model="claude-3-5-sonnet-20241022",
            max_tokens=500,
            temperature=0,
            messages=messages,
            tools=tools,
        )

        content = response.content[0]
        if content.type == "tool_use":
            result = f"Information about '{content.input['query']}': [data]"
            messages.append({"role": "assistant", "content": content})
            messages.append({
                "role": "user",
                "content": [
                    {"type": "tool_result", "tool_use_id": content.id, "content": result}
                ]
            })
        else:
            return content.text

    return "Normal handling complete"


# Test with a request the agent might not be confident about
result = escalation_agent("Should I approve a $5,000 refund for a customer "
                          "who claims they never received their order?")
print(result)
```

## How It Works Internally

The human-in-the-loop pattern modifies the agent loop with a gate check between the "model decides" step and the "execute action" step:

1. Model requests a tool call
2. Orchestrator checks: does this tool need approval?
3. If yes: pause the loop, present context to human, wait for response
4. If approved: execute the tool (possibly with human-modified arguments)
5. If rejected: return a rejection message to the model

The loop remains paused during step 3. In a real system, this pause can last minutes or hours. The agent should save its state and resume when the human responds.

## Build Something Real

Build a support agent with configurable approval gates and escalation paths.

```python
import anthropic
import json

client = anthropic.Anthropic()


class SupportAgent:
    """Customer support agent with configurable human oversight."""

    def __init__(self):
        self.gates = {
            "process_refund": {"threshold": 100.0},  # Auto-approve under $100
            "send_email": {},
            "delete_account": {},
        }
        self.tools = [
            {
                "name": "lookup_order",
                "description": "Look up order details",
                "input_schema": {
                    "type": "object",
                    "properties": {
                        "order_id": {"type": "string"}
                    },
                    "required": ["order_id"]
                }
            },
            {
                "name": "process_refund",
                "description": "Process a refund for an order",
                "input_schema": {
                    "type": "object",
                    "properties": {
                        "order_id": {"type": "string"},
                        "amount": {"type": "number"},
                        "reason": {"type": "string"}
                    },
                    "required": ["order_id", "amount", "reason"]
                }
            },
            {
                "name": "send_email",
                "description": "Send an email to a customer",
                "input_schema": {
                    "type": "object",
                    "properties": {
                        "to": {"type": "string"},
                        "subject": {"type": "string"},
                        "body": {"type": "string"}
                    },
                    "required": ["to", "subject", "body"]
                }
            },
            {
                "name": "delete_account",
                "description": "Delete a customer account",
                "input_schema": {
                    "type": "object",
                    "properties": {
                        "account_id": {"type": "string"},
                        "reason": {"type": "string"}
                    },
                    "required": ["account_id", "reason"]
                }
            },
        ]
        self.order_data = {
            "ORD-1001": {"total": 75.00, "customer": "alice@example.com", "status": "delivered"},
            "ORD-1002": {"total": 250.00, "customer": "bob@example.com", "status": "shipped"},
        }

    def check_gate(self, tool_name: str, args: dict) -> tuple[bool, str | None]:
        """
        Check if a tool call passes the approval gate.
        Returns (approved, rejection_reason).
        """
        gate_config = self.gates.get(tool_name)
        if not gate_config:
            return True, None  # No gate configured — auto-approve

        if tool_name == "process_refund":
            threshold = gate_config.get("threshold", 0)
            if args.get("amount", 0) <= threshold:
                return True, None
            return False, (
                f"Refund ${args['amount']:.2f} exceeds auto-approval threshold "
                f"of ${threshold:.2f}. Needs manager approval."
            )

        # All other gated tools always need human approval
        return False, f"{tool_name} requires human approval."

    def escalate(self, tool_name: str, args: dict, reason: str) -> str:
        """Escalate to a human for decision."""
        return (
            f"ESCALATED: {tool_name} requires human review.\n"
            f"Context: {json.dumps(args, indent=2)}\n"
            f"Reason: {reason}\n"
            f"Status: Pending human decision."
        )

    def run(self, user_input: str) -> str:
        """Run the support agent with approval gates."""
        messages = [{"role": "user", "content": user_input}]

        for turn in range(15):
            response = client.messages.create(
                model="claude-3-5-sonnet-20241022",
                max_tokens=500,
                temperature=0,
                system="You are a customer support agent. Be helpful but follow the rules.",
                messages=messages,
                tools=self.tools,
            )

            content = response.content[0]
            if content.type == "tool_use":
                approved, reason = self.check_gate(content.name, content.input)

                if not approved:
                    result = self.escalate(content.name, content.input, reason)
                elif content.name == "lookup_order":
                    order = self.order_data.get(content.input["order_id"])
                    result = json.dumps(order) if order else "Order not found."
                elif content.name == "process_refund":
                    result = (
                        f"Refund of ${content.input['amount']:.2f} for "
                        f"{content.input['order_id']} processed."
                    )
                elif content.name == "send_email":
                    result = f"Email sent to {content.input['to']}."
                else:
                    result = f"Action '{content.name}' completed."

                messages.append({"role": "assistant", "content": content})
                messages.append({
                    "role": "user",
                    "content": [
                        {"type": "tool_result", "tool_use_id": content.id, "content": result}
                    ]
                })
            else:
                return content.text

        return "Support agent reached max turns."


agent = SupportAgent()
result = agent.run(
    "Look up order ORD-1002. The customer wants a refund of $250. "
    "Also process a refund of $75 for order ORD-1001."
)
print(result)
```

The support agent handles small refunds automatically and escalates large ones to a human. The gates are configurable per tool, with thresholds for auto-approval.

## Performance Considerations

Human approval adds latency — minutes or hours instead of seconds. Design your system so the agent can continue working on other tasks while waiting for approval. Use a queue-based architecture where pending approvals are stored and resumed when the human responds.
