## Breaking It Down

### 1. Basic MCP Client with SDK

The MCP Python SDK provides a client that handles connection management, initialization, and message passing. You provide the server command and tool integration.

```bash
pip install mcp httpx anthropic
```

```python
import asyncio
import json

from mcp import ClientSession, StdioServerParameters
from mcp.client.stdio import stdio_client


async def connect_to_server(server_command: list[str]):
    """Connect to an MCP server over stdio and return a session."""
    server_params = StdioServerParameters(
        command=server_command[0],
        args=server_command[1:],
    )

    # The context manager handles spawning the process
    # and setting up the transport
    async with stdio_client(server_params) as (read, write):
        async with ClientSession(read, write) as session:
            # Step 1: Initialize
            await session.initialize()

            # Step 2: List tools
            tools_result = await session.list_tools()
            print(f"Server offers {len(tools_result.tools)} tools:")
            for tool in tools_result.tools:
                print(f"  - {tool.name}: {tool.description}")

            # Step 3: Call a tool
            result = await session.call_tool(
                "greet",
                {"name": "MCP Developer"},
            )
            print(f"\nTool result: {result.content[0].text}")

            # Session closes automatically on exit
```

### 2. Translating MCP Tools to Agent Format

The key integration step: convert MCP tool definitions to the format expected by your LLM API.

```python
from mcp.types import Tool as McpTool


def mcp_tool_to_api_format(mcp_tool: McpTool) -> dict:
    """Convert an MCP tool definition to Anthropic/OpenAI API format."""
    return {
        "name": mcp_tool.name,
        "description": mcp_tool.description,
        "input_schema": mcp_tool.inputSchema,
    }


def translate_all_tools(mcp_tools: list[McpTool]) -> list[dict]:
    """Translate all MCP tools to API format."""
    return [mcp_tool_to_api_format(t) for t in mcp_tools]


# Example MCP tool from a server
example_mcp_tool = McpTool(
    name="search_files",
    description="Search for files matching a pattern",
    inputSchema={
        "type": "object",
        "properties": {
            "pattern": {"type": "string"},
            "max_results": {"type": "number"},
        },
        "required": ["pattern"],
    },
)

api_tool = mcp_tool_to_api_format(example_mcp_tool)
print(json.dumps(api_tool, indent=2))
```

### 3. Routing Agent Tool Calls to MCP

When the agent decides to call a tool, the client intercepts the call and routes it to the correct MCP server.

```python
class McpToolRouter:
    """Routes tool calls from the agent to the correct MCP server."""

    def __init__(self):
        self._sessions: dict[str, ClientSession] = {}
        self._tool_map: dict[str, str] = {}  # tool_name -> server_name

    def add_server(self, name: str, session: ClientSession,
                   tools: list[McpTool]):
        """Register a server and its tools."""
        self._sessions[name] = session
        for tool in tools:
            self._tool_map[tool.name] = name

    async def call_tool(self, tool_name: str,
                        arguments: dict) -> str:
        """Call a tool by name, routing to the right server."""
        server_name = self._tool_map.get(tool_name)
        if not server_name:
            return f"Error: unknown tool '{tool_name}'"

        session = self._sessions[server_name]
        try:
            result = await session.call_tool(tool_name, arguments)
            # Extract text content from the result
            texts = [
                item.text for item in result.content
                if hasattr(item, 'text')
            ]
            return "\n".join(texts) if texts else str(result)
        except Exception as e:
            return f"Error calling {tool_name}: {e}"
```

### 5. Multi-Server Client

Production agents often need tools from multiple MCP servers. Here is a client that connects to several servers and merges their tools.

```python
from dataclasses import dataclass, field


@dataclass
class MultiServerMcpClient:
    """MCP client that connects to multiple servers and merges tools."""

    servers: dict[str, list[str]] = field(default_factory=dict)
    _sessions: dict[str, ClientSession] = field(default_factory=dict)
    _contexts: list = field(default_factory=list)

    def add_server_config(self, name: str, command: list[str]):
        """Configure a server to connect to."""
        self.servers[name] = command

    async def connect_all(self):
        """Connect to all configured servers."""
        all_tools = []

        for name, cmd in self.servers.items():
            server_params = StdioServerParameters(
                command=cmd[0],
                args=cmd[1:],
            )

            # Store the context for proper cleanup
            ctx = stdio_client(server_params)
            self._contexts.append(ctx)
            read, write = await ctx.__aenter__()

            session = ClientSession(read, write)
            self._contexts.append(session)
            await session.__aenter__()
            await session.initialize()

            self._sessions[name] = session

            tools_result = await session.list_tools()
            print(f"Server '{name}': {len(tools_result.tools)} tools")
            all_tools.extend(tools_result.tools)

        return all_tools

    async def call_tool(self, tool_name: str,
                        arguments: dict) -> str:
        """Call a tool, routing to the correct server."""
        for name, session in self._sessions.items():
            # Check if this server has the tool
            tools_result = await session.list_tools()
            if any(t.name == tool_name for t in tools_result.tools):
                result = await session.call_tool(tool_name, arguments)
                texts = [
                    item.text for item in result.content
                    if hasattr(item, 'text')
                ]
                return "\n".join(texts)

        return f"Error: tool '{tool_name}' not found on any server"

    async def disconnect_all(self):
        """Clean up all connections."""
        for ctx in reversed(self._contexts):
            try:
                await ctx.__aexit__(None, None, None)
            except Exception:
                pass
        self._sessions.clear()
        self._contexts.clear()
```

## How It Works Internally

When the agent requests tools, the MCP client calls `tools/list` on each connected server, then translates the MCP-format tool definitions to the LLM API's format (primarily renaming `inputSchema` to `input_schema` to match Anthropic/OpenAI conventions). The translated tools are merged into a single list and passed to the LLM as the `tools` parameter.

When the LLM returns a `tool_use` block, the client matches the tool name to the server that provides it, calls `tools/call` via the MCP session, and returns the result as a `tool_result` message in the agent's conversation history.

## Build Something Real

Build a complete MCP client that connects to a file-search MCP server and a system-info server, discovers all tools, and presents them to an agent for use.

```python
import asyncio
from mcp import ClientSession, StdioServerParameters
from mcp.client.stdio import stdio_client


async def build_and_run_client():
    """Build a complete multi-server MCP client."""
    client = MultiServerMcpClient()

    # Configure servers
    client.add_server_config(
        "file-search",
        ["python", "file_search_server.py"],
    )
    client.add_server_config(
        "system-info",
        ["python", "system_info_server.py"],
    )

    # Connect and discover tools
    print("Connecting to MCP servers...")
    all_tools = await client.connect_all()
    print(f"\nTotal tools available: {len(all_tools)}")
    for t in all_tools:
        print(f"  • {t.name}: {t.description}")

    # Example: call a tool from the file-search server
    print("\n--- Calling search_files ---")
    result = await client.call_tool("search_files", {
        "pattern": "*.md",
        "max_results": 5,
    })
    print(result)

    # Example: call a tool from the system-info server
    print("\n--- Calling check_port ---")
    result = await client.call_tool("check_port", {"port": 8080})
    print(result)

    # Clean up
    await client.disconnect_all()

```
