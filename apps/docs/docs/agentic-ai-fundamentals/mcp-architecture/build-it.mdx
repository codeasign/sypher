# MCP Architecture — Build It

*Reading time: 14 minutes*

## Breaking It Down

### 1. The Server Side

An MCP server listens for JSON-RPC messages, dispatches them to the appropriate handler, and returns responses. The server maintains lifecycle state — it tracks whether initialization has completed.

```python
import json
import sys
from typing import Any


class McpServer:
    """A minimal MCP server that implements the full lifecycle."""

    def __init__(self, name: str = "example-server", version: str = "0.1.0"):
        self._name = name
        self._version = version
        self._initialized = False
        self._tools: dict[str, dict] = {}
        self._tool_handlers: dict[str, callable] = {}

    def register_tool(self, name: str, description: str,
                      input_schema: dict, handler: callable):
        """Register a tool with the server."""
        self._tools[name] = {
            "name": name,
            "description": description,
            "inputSchema": input_schema,
        }
        self._tool_handlers[name] = handler

    def handle_message(self, raw: str) -> str | None:
        """Parse and handle a single JSON-RPC message."""
        try:
            msg = json.loads(raw)
        except json.JSONDecodeError as e:
            return self._error(None, -32700, f"Parse error: {e}")

        method = msg.get("method")
        msg_id = msg.get("id")
        params = msg.get("params", {})

        # Handle initialize even before initialized
        if method == "initialize":
            return self._handle_initialize(msg_id, params)

        # All other methods require initialization
        if not self._initialized:
            return self._error(msg_id, -32000, "Server not initialized")

        if method == "notifications/initialized":
            return None  # Notification, no response expected

        if method == "tools/list":
            return self._handle_tools_list(msg_id)

        if method == "tools/call":
            return self._handle_tool_call(msg_id, params)

        if method == "resources/list":
            return self._success(msg_id, {"resources": []})

        if method == "ping":
            return self._success(msg_id, {})

        return self._error(msg_id, -32601, f"Method not found: {method}")

    def _handle_initialize(self, msg_id: int, params: dict) -> str:
        """Handle the initialize handshake."""
        client_info = params.get("clientInfo", {})
        client_version = params.get("protocolVersion", "unknown")
        print(f"Client connected: {client_info.get('name')} "
              f"(protocol: {client_version})", file=sys.stderr)

        return self._success(msg_id, {
            "protocolVersion": "2024-11-05",
            "capabilities": {
                "tools": {},
                "resources": {},
            },
            "serverInfo": {
                "name": self._name,
                "version": self._version,
            },
        })

    def handle_initialized(self):
        """Mark the server as initialized."""
        self._initialized = True

    def _handle_tools_list(self, msg_id: int) -> str:
        """Return the list of registered tools."""
        return self._success(msg_id, {
            "tools": list(self._tools.values()),
        })

    def _handle_tool_call(self, msg_id: int, params: dict) -> str:
        """Execute a tool and return the result."""
        name = params.get("name")
        arguments = params.get("arguments", {})

        handler = self._tool_handlers.get(name)
        if not handler:
            return self._error(msg_id, -32602, f"Unknown tool: {name}")

        try:
            result = handler(**arguments)
            return self._success(msg_id, {
                "content": [{"type": "text", "text": str(result)}]
            })
        except Exception as e:
            return self._error(msg_id, -32603, f"Tool error: {e}")

    def _success(self, msg_id: int | None, result: dict) -> str:
        return json.dumps({"jsonrpc": "2.0", "id": msg_id, "result": result})

    def _error(self, msg_id: int | None, code: int, message: str) -> str:
        return json.dumps({
            "jsonrpc": "2.0",
            "id": msg_id,
            "error": {"code": code, "message": message},
        })

    def run_stdio(self):
        """Run the server over stdin/stdout."""
        buffer = ""
        for line in sys.stdin:
            response = self.handle_message(line.strip())
            if response:
                print(response, flush=True)
            # After receiving initialized notification
            if '"method":"notifications/initialized"' in line:
                self.handle_initialized()
```

### 2. The Client Side

The client manages the lifecycle and sends properly ordered messages.

```python
import json
import subprocess


class McpClient:
    """MCP client that manages the full lifecycle."""

    def __init__(self, server_command: list[str]):
        self._proc = subprocess.Popen(
            server_command,
            stdin=subprocess.PIPE,
            stdout=subprocess.PIPE,
            stderr=subprocess.PIPE,
            text=True,
        )
        self._next_id = 0
        self._initialized = False

    def _request(self, method: str, params: dict = None) -> dict:
        """Send a request and wait for one response line."""
        self._next_id += 1
        payload = {
            "jsonrpc": "2.0",
            "id": self._next_id,
            "method": method,
            "params": params or {},
        }
        self._proc.stdin.write(json.dumps(payload) + "\n")
        self._proc.stdin.flush()
        response = self._proc.stdout.readline().strip()
        return json.loads(response) if response else {}

    def _notify(self, method: str, params: dict = None):
        """Send a notification (no response expected)."""
        payload = {
            "jsonrpc": "2.0",
            "method": method,
            "params": params or {},
        }
        self._proc.stdin.write(json.dumps(payload) + "\n")
        self._proc.stdin.flush()

    def connect(self) -> dict:
        """Run the full initialization handshake."""
        # Step 1: Initialize
        init_result = self._request("initialize", {
            "protocolVersion": "2024-11-05",
            "capabilities": {"tools": {}, "resources": {}},
            "clientInfo": {"name": "mcp-client", "version": "1.0.0"},
        })
        # Step 2: Send initialized notification
        self._notify("notifications/initialized")
        self._initialized = True
        return init_result

    def list_tools(self) -> list[dict]:
        """List tools from the server."""
        result = self._request("tools/list")
        return result.get("result", {}).get("tools", [])

    def call_tool(self, name: str, arguments: dict = None) -> dict:
        """Call a tool on the server."""
        result = self._request("tools/call", {
            "name": name,
            "arguments": arguments or {},
        })
        return result.get("result", {})

    def close(self):
        """Shut down the connection."""
        self._proc.terminate()
        self._proc.wait()
```

### 4. Transport Switching

The same server logic works with either transport. The only difference is how messages are read and written.

## How It Works Internally

The architecture enforces a strict ordering constraint: initialization before operation. The server must buffer or reject any request received before initialization completes. This prevents race conditions where a tool call arrives before the server has registered all its tools.

The `id` field in JSON-RPC messages allows the client to correlate responses to requests even if the transport reorders messages. On SSE transport, the server may send multiple responses on the same stream — the `id` ensures the client knows which response matches which request.

## Build Something Real

Build a complete MCP file server and client that handles the full lifecycle with error handling.

```python
import json
import os
from datetime import datetime, timezone


@dataclass
class FileServer(McpServer):
    """An MCP server for file operations with full lifecycle management."""

    def __init__(self, workspace: str = "."):
        super().__init__(name="file-server", version="1.0.0")
        self._workspace = os.path.abspath(workspace)

        self.register_tool(
            name="read_file",
            description="Read a file from the workspace",
            input_schema={
                "type": "object",
                "properties": {
                    "path": {
                        "type": "string",
                        "description": "Path relative to workspace"
                    }
                },
                "required": ["path"],
            },
            handler=self._handle_read_file,
        )

        self.register_tool(
            name="list_files",
            description="List files in the workspace directory",
            input_schema={
                "type": "object",
                "properties": {},
            },
            handler=self._handle_list_files,
        )

    def _safe_path(self, path: str) -> str:
        """Resolve and validate a file path within the workspace."""
        full = os.path.normpath(os.path.join(self._workspace, path))
        if not full.startswith(self._workspace):
            raise PermissionError("Path traversal denied")
        return full

    def _handle_read_file(self, path: str) -> str:
        full_path = self._safe_path(path)
        if not os.path.isfile(full_path):
            return f"Error: file not found: {path}"
        with open(full_path, "r") as f:
            return f.read()

    def _handle_list_files(self) -> str:
        entries = []
        for name in os.listdir(self._workspace):
            full = os.path.join(self._workspace, name)
            stat = os.stat(full)
            entries.append(
                f"{name}  {stat.st_size} bytes  "
                f"{datetime.fromtimestamp(stat.st_mtime, timezone.utc)}"
            )
        return "\n".join(entries) if entries else "(empty directory)"


# Usage:
# server = FileServer("/tmp/workspace")
# server.run_stdio()
#
# The client connects, initializes, lists tools, and calls them
# through the standard MCP lifecycle.
```
