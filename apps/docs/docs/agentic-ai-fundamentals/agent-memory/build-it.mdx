# Agent Memory — Build It

*Reading time: 14 minutes*

## Breaking It Down

### 1. Sliding Window Memory

The simplest memory strategy: keep the most recent conversation turns.

```python
import anthropic

client = anthropic.Anthropic()


class SlidingWindowMemory:
    """Keeps only the most recent N messages."""

    def __init__(self, window_size: int = 10):
        self.window_size = window_size
        self.messages: list[dict] = []

    def add(self, role: str, content: any):
        """Add a message and trim if over the window size."""
        self.messages.append({"role": role, "content": content})
        # Trim from the start, but always keep the system context
        if len(self.messages) > self.window_size:
            excess = len(self.messages) - self.window_size
            self.messages = self.messages[excess:]

    def get_context(self) -> list[dict]:
        """Return the current message window."""
        return list(self.messages)


memory = SlidingWindowMemory(window_size=6)
memory.add("user", "What is the capital of France?")
memory.add("assistant", "Paris.")
memory.add("user", "What about Germany?")
memory.add("assistant", "Berlin.")
memory.add("user", "What about Italy?")
memory.add("assistant", "Rome.")
memory.add("user", "What was my first question?")  # This pushes out the first turn

print(f"Messages in window: {len(memory.get_context())}")
for msg in memory.get_context():
    print(f"  {msg['role']}: {str(msg['content'][:50])}")

# The first question "What is the capital of France?" is gone
```

Sliding windows are simple but lossy. Important information drops out when the window is full.

### 2. Summarizing Memory

Replace old messages with summaries instead of dropping them.

```python
class SummarizingMemory:
    """Summarizes older messages to preserve information compactly."""

    def __init__(self, max_messages: int = 8, summary_model: str = "claude-3-haiku-20240307"):
        self.max_messages = max_messages
        self.summary_model = summary_model
        self.messages: list[dict] = []
        self.summary: str | None = None

    def add(self, role: str, content: any):
        self.messages.append({"role": role, "content": content})
        if len(self.messages) > self.max_messages:
            self._summarize_old_messages()

    def _summarize_old_messages(self):
        """Summarize old messages and replace them with the summary."""
        old_messages = self.messages[:len(self.messages) - self.max_messages // 2]

        # Only summarize if we have actual content
        if not old_messages:
            return

        conversation_text = "\n".join(
            f"{m['role']}: {m['content']}" for m in old_messages
        )

        response = client.messages.create(
            model=self.summary_model,
            max_tokens=200,
            temperature=0,
            messages=[{
                "role": "user",
                "content": (
                    f"Summarize this conversation history in 2-3 sentences, "
                    f"keeping all key facts and decisions:\n\n{conversation_text}"
                )
            }]
        )

        self.summary = response.content[0].text
        # Remove old messages, keep recent ones
        self.messages = self.messages[len(old_messages):]

    def get_context(self) -> list[dict]:
        """Return the full context including any summary."""
        context = []
        if self.summary:
            context.append({
                "role": "user",
                "content": f"[Previous conversation summary: {self.summary}]"
            })
        context.extend(self.messages)
        return context


memory = SummarizingMemory(max_messages=4)

# Simulate a longer conversation
for i in range(8):
    memory.add("user", f"This is turn number {i + 1}.")
    memory.add("assistant", f"Acknowledging turn {i + 1}.")

print("Context after 8 turns:")
for msg in memory.get_context():
    role = msg["role"]
    content = str(msg["content"][:80])
    print(f"  {role}: {content}")
```

Summarizing memory preserves information at the cost of reduced fidelity. Important numbers or specific instructions may be lost in the summary.

### 3. Retrieval-Augmented Memory

Store messages in a vector database and retrieve the most relevant ones for each step.

```python
import json
from typing import Any


class SimpleVectorStore:
    """A minimal vector store using keyword overlap as a similarity proxy."""

    def __init__(self):
        self.documents: list[dict] = []

    def add(self, text: str, metadata: dict | None = None):
        """Store a document with its text."""
        self.documents.append({
            "text": text,
            "metadata": metadata or {},
        })

    def search(self, query: str, top_k: int = 3) -> list[dict]:
        """Simple keyword-based retrieval. In production, use embeddings."""
        query_terms = set(query.lower().split())
        scored = []
        for doc in self.documents:
            doc_terms = set(doc["text"].lower().split())
            overlap = len(query_terms & doc_terms)
            if overlap > 0:
                scored.append((overlap, doc))
        scored.sort(key=lambda x: x[0], reverse=True)
        return [doc for _, doc in scored[:top_k]]


class RetrievalMemory:
    """Memory backed by a vector store for long-term recall."""

    def __init__(self, max_short_term: int = 10):
        self.short_term: list[dict] = []
        self.max_short_term = max_short_term
        self.long_term = SimpleVectorStore()

    def add(self, role: str, content: Any):
        """Add a message. Store in short-term and long-term memory."""
        entry = {"role": role, "content": content}
        self.short_term.append(entry)

        # Also store in long-term memory (text only, not tool blocks)
        if isinstance(content, str):
            self.long_term.add(content, {"role": role})

        # Trim short-term if needed
        if len(self.short_term) > self.max_short_term:
            # Move oldest message summary to long-term
            oldest = self.short_term.pop(0)
            if isinstance(oldest["content"], str):
                self.long_term.add(
                    f"[archived] {oldest['role']}: {oldest['content']}",
                    {"role": oldest["role"], "archived": True}
                )

    def get_context(self, query: str | None = None) -> list[dict]:
        """Get context with relevant long-term memories injected."""
        context = list(self.short_term)

        if query:
            # Retrieve relevant memories
            memories = self.long_term.search(query, top_k=2)
            for mem in memories:
                # Avoid duplicating what is already in short-term
                if not any(
                    isinstance(s["content"], str) and s["content"] == mem["text"]
                    for s in self.short_term
                ):
                    context.insert(0, {
                        "role": "system",
                        "content": f"[Relevant memory: {mem['text']}]"
                    })

        return context


# Simulate a conversation where earlier facts are retrieved later
memory = RetrievalMemory(max_short_term=4)

memory.add("user", "My name is Alice and my account number is ACCT-789.")
memory.add("assistant", "Noted, Alice. I have your account on file.")
memory.add("user", "What is my account number?")
memory.add("assistant", "Your account number is ACCT-789.")

# The question about account number would match via keyword retrieval
# Even if the first exchange fell out of short-term memory
context = memory.get_context("account number Alice")
print(f"Context entries: {len(context)}")
for msg in context:
    content = str(msg["content"][:80])
    print(f"  [{msg['role']}] {content}")
```

## How It Works Internally

All three memory strategies manage the same constraint: the model's context window has a fixed maximum. The difference is what happens when new information arrives and the window is full.

- **Sliding window**: Drops the oldest content. O(1) time, content is lost.
- **Summarization**: Compresses old content into a shorter form. O(n) time for summarization call, information is compressed but preserved.
- **Retrieval**: Stores everything externally and selectively retrieves. O(log n) search, preserves full fidelity at the cost of external storage and retrieval latency.

## Build Something Real

Build an agent with tiered memory that uses all three strategies.

```python
import anthropic
import json
from typing import Any

client = anthropic.Anthropic()


class TieredMemory:
    """Three-tier memory: short-term (sliding), medium (summaries), long-term (retrieval)."""

    def __init__(self, short_term_limit: int = 8):
        self.short_term: list[dict] = []
        self.short_term_limit = short_term_limit
        self.summaries: list[str] = []
        self.long_term = SimpleVectorStore()
        self.turn_count = 0

    def add(self, role: str, content: Any):
        self.turn_count += 1
        self.short_term.append({"role": role, "content": content})

        # Archive to long-term if it is a user or assistant message
        if isinstance(content, str) and content.strip():
            self.long_term.add(
                f"[Turn {self.turn_count}] {role}: {content[:500]}",
                {"role": role, "turn": self.turn_count}
            )

        # Check if short-term needs compaction
        if len(self.short_term) >= self.short_term_limit:
            self._compact()

    def _compact(self):
        """Summarize older short-term messages and archive them."""
        to_summarize = self.short_term[:len(self.short_term) // 2]
        if not to_summarize:
            return

        text = "\n".join(
            f"{m['role']}: {m['content']}"
            for m in to_summarize if isinstance(m['content'], str)
        )
        if not text.strip():
            self.short_term = self.short_term[len(to_summarize):]
            return

        try:
            response = client.messages.create(
                model="claude-3-haiku-20240307",
                max_tokens=150,
                temperature=0,
                messages=[{
                    "role": "user",
                    "content": f"Summarize:\n\n{text}"
                }]
            )
            summary = response.content[0].text
            self.summaries.append(summary)
            self.long_term.add(
                f"[Summary {len(self.summaries)}]: {summary}",
                {"role": "summary", "turn": self.turn_count}
            )
        except Exception:
            # If summarization fails, just drop the old messages
            pass

        self.short_term = self.short_term[len(to_summarize):]

    def build_context(self, current_query: str) -> list[dict]:
        """Build the full context with all memory tiers."""
        context = []

        # 1. Most recent summaries (medium-term)
        recent_summaries = self.summaries[-2:]
        for s in recent_summaries:
            context.append({
                "role": "user",
                "content": f"[Context from earlier: {s}]"
            })

        # 2. Relevant long-term memories
        memories = self.long_term.search(current_query, top_k=2)
        seen = set()
        for mem in memories:
            text = mem["text"]
            if text not in seen and not any(
                str(m.get("content", "")) == f"[Context from earlier: {text}]"
                for m in context
            ):
                seen.add(text)
                context.append({
                    "role": "user",
                    "content": f"[Related memory: {text}]"
                })

        # 3. Current short-term context
        context.extend(self.short_term)

        return context


def agent_with_memory(user_input: str, memory: TieredMemory) -> str:
    """Run the agent with tiered memory."""
    context = memory.build_context(user_input)

    context.append({"role": "user", "content": user_input})

    response = client.messages.create(
        model="claude-3-5-sonnet-20241022",
        max_tokens=500,
        temperature=0,
        system="You are a helpful assistant with good memory.",
        messages=context,
    )

    reply = response.content[0].text
    memory.add("user", user_input)
    memory.add("assistant", reply)
    return reply


memory = TieredMemory(short_term_limit=4)

queries = [
    "My name is Bob and my favorite color is green.",
    "I have a dog named Max, a golden retriever.",
    "I work at a robotics company called BotWorks.",
    "What is my name and what is my dog's name?",
    "Also, where do I work?",
]

for q in queries:
    print(f"\nUser: {q}")
    reply = agent_with_memory(q, memory)
    print(f"Agent: {reply}")
```

The tiered memory system keeps the most recent conversation in full fidelity in short-term memory. Summaries preserve the gist of earlier exchanges. Long-term retrieval can recall specific facts when needed, even if they were in neither short-term nor the most recent summaries.

## Performance Considerations

Each summarization call adds latency and cost. Tune the short-term window to your task: a task with long context windows (128K tokens) may never need summarization for a single session. Use retrieval-only memory for sessions under 50 turns.
