# What Is MCP? — Build It

*Reading time: 12 minutes*

## Breaking It Down

### 1. JSON-RPC 2.0 Basics

MCP uses JSON-RPC 2.0 as its message format. Every message is a JSON object with `jsonrpc`, a `method`, an `id` (for requests), and optional `params`.

```json
{
  "jsonrpc": "2.0",
  "id": 1,
  "method": "tools/list",
  "params": {}
}
```

A successful response carries the matching `id` and a `result` field:

```json
{
  "jsonrpc": "2.0",
  "id": 1,
  "result": {
    "tools": [
      {
        "name": "read_file",
        "description": "Read the contents of a file",
        "inputSchema": {
          "type": "object",
          "properties": {
            "path": {
              "type": "string",
              "description": "Path to the file"
            }
          },
          "required": ["path"]
        }
      }
    ]
  }
}
```

An error response includes an `error` object with a `code` and `message`:

```json
{
  "jsonrpc": "2.0",
  "id": 1,
  "error": {
    "code": -32602,
    "message": "Invalid params: path is required"
  }
}
```

### 2. The MCP Initialization Handshake

Before any tool calls or resource reads, the client and server negotiate capabilities through an `initialize` exchange.

```python
import json


def build_initialize_request(client_name: str, client_version: str) -> str:
    """Build the MCP initialization request."""
    request = {
        "jsonrpc": "2.0",
        "id": 1,
        "method": "initialize",
        "params": {
            "protocolVersion": "2024-11-05",
            "capabilities": {
                "tools": {},
                "resources": {},
                "prompts": {}
            },
            "clientInfo": {
                "name": client_name,
                "version": client_version
            }
        }
    }
    return json.dumps(request, indent=2)


# Example: build an init request from a custom agent
request = build_initialize_request("my-agent", "1.0.0")
print("=== MCP Initialize Request ===")
print(request)
```

A successful initialization response includes the server's capabilities:

```json
{
  "jsonrpc": "2.0",
  "id": 1,
  "result": {
    "protocolVersion": "2024-11-05",
    "capabilities": {
      "tools": {},
      "resources": {},
      "prompts": {}
    },
    "serverInfo": {
      "name": "filesystem-server",
      "version": "0.1.0"
    }
  }
}
```

After initialization, the client sends an `initialized` notification (no response expected):

```json
{
  "jsonrpc": "2.0",
  "method": "notifications/initialized",
  "params": {}
}
```

### 3. Discovering and Calling Tools

Once initialized, the client can list tools and call them.

```python
def build_tools_list_request() -> str:
    """Build a request to list available tools."""
    request = {
        "jsonrpc": "2.0",
        "id": 2,
        "method": "tools/list",
        "params": {}
    }
    return json.dumps(request, indent=2)


def build_tool_call_request(tool_name: str, arguments: dict,
                            request_id: int = 3) -> str:
    """Build a request to call a specific tool."""
    request = {
        "jsonrpc": "2.0",
        "id": request_id,
        "method": "tools/call",
        "params": {
            "name": tool_name,
            "arguments": arguments
        }
    }
    return json.dumps(request, indent=2)


# Example: list tools then call one
print("=== List Tools ===")
print(build_tools_list_request())

print("\n=== Call Tool ===")
print(build_tool_call_request("read_file", {"path": "/tmp/notes.txt"}))
```

### 4. Reading Resources

Resources are structured data endpoints that return content with a URI and MIME type.

```python
def build_resources_list_request() -> str:
    """Build a request to list available resources."""
    request = {
        "jsonrpc": "2.0",
        "id": 4,
        "method": "resources/list",
        "params": {}
    }
    return json.dumps(request, indent=2)


def build_resource_read_request(uri: str) -> str:
    """Build a request to read a specific resource."""
    request = {
        "jsonrpc": "2.0",
        "id": 5,
        "method": "resources/read",
        "params": {
            "uri": uri
        }
    }
    return json.dumps(request, indent=2)


# Example: list resources then read one
print("=== List Resources ===")
print(build_resources_list_request())

print("\n=== Read Resource ===")
print(build_resource_read_request("file:///config/app.json"))
```

## How It Works Internally

The MCP protocol is transport-agnostic. The same JSON-RPC messages work over stdin/stdout pipes (for local, child-process servers) and SSE/HTTP (for remote servers). The client does not need to know which transport the server uses — it just sends and receives JSON-RPC messages. The transport layer handles the serialization and delivery.

When a tool call returns, the result includes a `content` array where each element has a `type` field — typically `"text"` for plain text, but MCP also supports `"image"`, `"resource"`, and other types. The client renders these appropriately based on type.

## Build Something Real

Build a minimal MCP client in Python that connects to a server via stdio, initializes, lists tools, and calls one. This is the foundation of any MCP integration.

```python
import json
import subprocess
import sys


class McpClient:
    """Minimal MCP client that communicates over stdio."""

    def __init__(self, server_command: list[str]):
        self._request_id = 0
        self._proc = subprocess.Popen(
            server_command,
            stdin=subprocess.PIPE,
            stdout=subprocess.PIPE,
            stderr=subprocess.PIPE,
            text=True,
        )

    def _send_request(self, method: str, params: dict = None) -> dict:
        """Send a JSON-RPC request and wait for the response."""
        self._request_id += 1
        request = {
            "jsonrpc": "2.0",
            "id": self._request_id,
            "method": method,
            "params": params or {},
        }
        request_str = json.dumps(request) + "\n"
        self._proc.stdin.write(request_str)
        self._proc.stdin.flush()

        response_line = self._proc.stdout.readline()
        return json.loads(response_line.strip())

    def initialize(self):
        """Perform the MCP initialization handshake."""
        result = self._send_request("initialize", {
            "protocolVersion": "2024-11-05",
            "capabilities": {"tools": {}, "resources": {}, "prompts": {}},
            "clientInfo": {"name": "mcp-client-tutorial", "version": "1.0.0"},
        })
        # Send initialized notification
        notification = {
            "jsonrpc": "2.0",
            "method": "notifications/initialized",
            "params": {},
        }
        self._proc.stdin.write(json.dumps(notification) + "\n")
        self._proc.stdin.flush()
        return result

    def list_tools(self) -> list[dict]:
        """List all tools available on the server."""
        result = self._send_request("tools/list")
        return result.get("result", {}).get("tools", [])

    def call_tool(self, name: str, arguments: dict) -> dict:
        """Call a tool by name with the given arguments."""
        result = self._send_request("tools/call", {
            "name": name,
            "arguments": arguments,
        })
        return result.get("result", {})

    def close(self):
        """Close the connection to the server."""
        self._proc.terminate()
        self._proc.wait()
```

This client can connect to any MCP server that communicates over stdio. The same client structure works for SSE-based servers with a different transport implementation.
