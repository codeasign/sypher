# Prompt Templates — Build It

*Reading time: 12 minutes*

## Breaking It Down

### 1. Building a Prompt Template System in Python

Start with a simple but capable template class:

```python
import anthropic
from string import Template

client = anthropic.Anthropic()


class PromptTemplate:
    """A simple prompt template with variable substitution."""

    def __init__(self, system_template: str, user_template: str):
        self.system_template = Template(system_template)
        self.user_template = Template(user_template)

    def render(self, **variables) -> dict:
        """Render the template with the provided variables."""
        return {
            "system": self.system_template.safe_substitute(**variables),
            "user": self.user_template.safe_substitute(**variables),
        }

    def execute(self, **variables) -> str:
        """Render and execute the prompt, returning the response."""
        rendered = self.render(**variables)
        response = client.messages.create(
            model="claude-3-5-sonnet-20241022",
            max_tokens=300,
            temperature=0,
            system=rendered["system"],
            messages=[{"role": "user", "content": rendered["user"]}]
        )
        return response.content[0].text


# Define a classification template
classifier = PromptTemplate(
    system_template=(
        "You are a $role classifier.\n"
        "Classify text into one of: $categories.\n"
        "Respond with only the category name."
    ),
    user_template="Text: $text"
)

# Use it with different variables
result = classifier.execute(
    role="sentiment",
    categories="POSITIVE, NEGATIVE, NEUTRAL",
    text="This product exceeded my expectations!"
)
print(f"Classification: {result}")

# Same template, different parameters
result2 = classifier.execute(
    role="urgency",
    categories="URGENT, NORMAL, LOW",
    text="The server has been down for 3 hours with no response from support."
)
print(f"Urgency: {result2}")
```

The same template structure handles different classification tasks just by changing the variables.

### 2. Templates with Dynamic Context Injection

In production, you often inject dynamic content like database records or retrieved documents:

```python
class RAGPromptTemplate:
    """Template for RAG-style prompts with document context."""

    def __init__(self, system_template: str, user_template: str):
        self.system_template = Template(system_template)
        self.user_template = Template(user_template)

    def render_context(self, documents: list[str], max_tokens: int = 2000) -> str:
        """Format and truncate retrieved documents to fit within a token budget."""
        context_parts = []
        token_count = 0

        for doc in documents:
            # Approximate tokens: 1.3 tokens per word
            doc_tokens = int(len(doc.split()) * 1.3)

            if token_count + doc_tokens > max_tokens:
                # Truncate the document to fit remaining budget
                remaining = max_tokens - token_count
                words = int(remaining / 1.3)
                context_parts.append(doc[:words * 5])  # rough char estimate
                break

            context_parts.append(doc)
            token_count += doc_tokens

        return "\n---\n".join(context_parts)

    def execute(
        self,
        question: str,
        documents: list[str],
        instruction: str = "Answer the question based on the provided context."
    ) -> str:
        """Render with document context and execute."""
        context = self.render_context(documents)
        rendered_system = Template(self.system_template).safe_substitute(
            instruction=instruction
        )
        rendered_user = Template(self.user_template).safe_substitute(
            context=context,
            question=question
        )

        response = client.messages.create(
            model="claude-3-5-sonnet-20241022",
            max_tokens=400,
            temperature=0,
            system=rendered_system,
            messages=[{"role": "user", "content": rendered_user}]
        )
        return response.content[0].text


rag_template = RAGPromptTemplate(
    system_template="$instruction",
    user_template=(
        "Context:\n$context\n\n"
        "Question: $question\n\n"
        "Answer based only on the context provided."
    )
)

# Simulate retrieved documents
docs = [
    "The return policy allows returns within 30 days of purchase. Items must be unused and in original packaging.",
    "Electronics have a 14-day return window. Opened items are subject to a 15% restocking fee.",
    "To start a return, go to your account, find the order, and click 'Return Item'. You will receive a shipping label via email.",
]

question = "Can I return a laptop I opened?"
response = rag_template.execute(question, docs)
print(f"Q: {question}\nA: {response}")
```

### 3. Conditional Template Sections

Some templates need sections that appear only conditionally:

```python
class ConditionalPromptTemplate:
    """Template with conditional sections using simple boolean variables."""

    def __init__(self, system_template_parts: dict):
        self.parts = system_template_parts

    def build_system_prompt(self, **conditions) -> str:
        """Build system prompt by conditionally including sections."""
        sections = []

        if conditions.get("role"):
            sections.append(f"You are {conditions['role']}.")

        if conditions.get("format_instructions"):
            sections.append(f"Format: {conditions['format_instructions']}")

        if conditions.get("guardrails"):
            sections.append(f"Rules:\n" + "\n".join(f"- {r}" for r in conditions["guardrails"]))

        if conditions.get("examples"):
            sections.append("Examples:\n" + "\n".join(f"- {e}" for e in conditions["examples"]))

        sections.append("Keep responses concise and accurate.")

        return "\n\n".join(sections)

    def execute(self, user_message: str, **conditions) -> str:
        system_prompt = self.build_system_prompt(**conditions)
        response = client.messages.create(
            model="claude-3-5-sonnet-20241022",
            max_tokens=300,
            temperature=0,
            system=system_prompt,
            messages=[{"role": "user", "content": user_message}]
        )
        return response.content[0].text


conditional = ConditionalPromptTemplate({})

# Call with different conditions
response_a = conditional.execute(
    "Explain machine learning.",
    role="a 5th-grade teacher",
    format_instructions="Use simple analogies. Max 3 sentences."
)
print(f"With role + format:\n{response_a}\n")

response_b = conditional.execute(
    "Explain machine learning.",
    role="a senior data scientist",
    guardrails=["Do not oversimplify", "Include specific technical details"],
    examples=["Machine learning is a subset of AI that..."]
)
print(f"With role + guardrails + examples:\n{response_b}\n")
```

## How It Works Internally

Template rendering is a purely mechanical process — the template engine performs string substitution before the prompt is sent to the API. The LLM never sees the template variables; it only sees the rendered text.

The key consideration is token budgeting. When you render a template, you should estimate the token count of the rendered output to ensure it fits within the model's context window. Variable values (especially injected documents) can vary wildly in size, and a template that works for short inputs may produce prompts that exceed the limit for long inputs.

## Build Something Real

Build a template library for a multi-feature application:

```python
import anthropic
from string import Template
from dataclasses import dataclass, field

client = anthropic.Anthropic()


@dataclass
class PromptLibrary:
    """A library of versioned prompt templates for different features."""

    templates: dict = field(default_factory=lambda: {
        "classify": {
            "system": "You are a $task classifier. Categories: $labels. Respond with only the label.",
            "user": "Text: $text",
            "version": "1.0"
        },
        "summarize": {
            "system": "You are a summarizer. Summarize the following text in $max_words words or fewer.",
            "user": "$text",
            "version": "1.2"
        },
        "extract": {
            "system": "Extract the following fields from the text: $fields. Return as JSON.",
            "user": "Text: $text",
            "version": "2.0"
        },
        "rewrite": {
            "system": "You rewrite text to be $style in tone. Maintain all factual information.",
            "user": "$text",
            "version": "1.0"
        },
    })

    def execute(self, feature: str, **variables) -> str:
        """Execute a template from the library."""
        if feature not in self.templates:
            raise ValueError(f"Unknown feature: {feature}")

        tpl = self.templates[feature]
        system = Template(tpl["system"]).safe_substitute(**variables)
        user_msg = Template(tpl["user"]).safe_substitute(**variables)

        response = client.messages.create(
            model="claude-3-5-sonnet-20241022",
            max_tokens=300,
            temperature=0,
            system=system,
            messages=[{"role": "user", "content": user_msg}]
        )

        return response.content[0].text

    def list_features(self):
        """List all available features with their versions."""
        for name, tpl in self.templates.items():
            print(f"  {name:15s}  v{tpl['version']}")


library = PromptLibrary()
library.list_features()

# Use different features
print(library.execute("classify", task="sentiment", labels="POS, NEG, NEUTRAL", text="Amazing service!"))
print(library.execute("summarize", max_words="50", text="Long document text here..."[:200]))
```

This template library pattern keeps all prompts in one place, versioned, and accessible by feature name. Application code calls `library.execute("classify", ...)` without knowing or caring how the prompt is structured internally.
