# Reasoning vs. Execution — Build It

*Reading time: 12 minutes*

## Breaking It Down

### 1. Fused Reasoning and Execution

First, see what it looks like when the same call both plans and executes.

```python
import anthropic

client = anthropic.Anthropic()


def fused_agent(user_input: str) -> str:
    """The model plans and executes in the same loop."""
    messages = [{"role": "user", "content": user_input}]
    tools = [{
        "name": "search_db",
        "description": "Search the database for records",
        "input_schema": {
            "type": "object",
            "properties": {
                "query": {"type": "string", "description": "Search query"}
            },
            "required": ["query"]
        }
    }]

    for turn in range(10):
        response = client.messages.create(
            model="claude-3-5-sonnet-20241022",  # Same model for reasoning and execution
            max_tokens=1000,
            temperature=0,
            system="You are an agent that searches a database.",
            messages=messages,
            tools=tools,
        )
        content = response.content[0]
        if content.type == "tool_use":
            result = f"Results for '{content.input['query']}': [found 3 records]"
            messages.append({"role": "assistant", "content": content})
            messages.append({
                "role": "user",
                "content": [
                    {"type": "tool_result", "tool_use_id": content.id, "content": result}
                ]
            })
        else:
            return content.text
    return "No response"
```

Every tool call uses the same expensive model. The planning tokens — "I should search the database for records matching this query" — cost the same as the execution tokens.

### 2. Separated Reasoning and Execution

Split the loop: a planner model decides what to do, and an executor model formats the tool call.

```python
def separated_agent(user_input: str) -> str:
    """Planner (large model) + executor (small model)."""
    messages = [{"role": "user", "content": user_input}]
    tools = [{
        "name": "search_db",
        "description": "Search the database for records",
        "input_schema": {
            "type": "object",
            "properties": {
                "query": {"type": "string", "description": "Search query"}
            },
            "required": ["query"]
        }
    }]

    for turn in range(10):
        # REASONING: use a powerful model for planning
        plan_response = client.messages.create(
            model="claude-3-5-sonnet-20241022",
            max_tokens=300,
            temperature=0,
            system=(
                "Decide the next action. Respond with exactly one tool call "
                "or the text 'FINAL:' followed by your answer."
            ),
            messages=messages,
            tools=tools,
        )

        plan_content = plan_response.content[0]

        if plan_content.type == "tool_use":
            # EXECUTION: use a cheaper model to call the tool
            # (In this case the executor just validates and passes through)
            exec_response = client.messages.create(
                model="claude-3-haiku-20240307",
                max_tokens=500,
                temperature=0,
                messages=[{
                    "role": "user",
                    "content": (
                        f"Execute this tool call:\n"
                        f"Tool: {plan_content.name}\n"
                        f"Arguments: {plan_content.input}\n"
                        f"Return the arguments exactly as provided."
                    )
                }]
            )
            # Execute the tool (our code always runs the actual function)
            result = f"Results for '{plan_content.input['query']}': [found 3 records]"

            messages.append({"role": "assistant", "content": plan_content})
            messages.append({
                "role": "user",
                "content": [
                    {"type": "tool_result", "tool_use_id": plan_content.id, "content": result}
                ]
            })
        else:
            return plan_content.text
    return "No response"
```

The expensive model (Sonnet) handles only the reasoning — deciding which tool to call. The actual tool execution is handled by a cheaper model or direct code.

### 3. Token Cost Comparison

Measure the cost difference between fused and separated approaches.

```python
def count_tokens(messages: list[dict], system: str) -> int:
    """Estimate token count (rough approximation)."""
    total = len(system.split())
    for msg in messages:
        content = msg.get("content", "")
        if isinstance(content, str):
            total += len(content.split())
        elif isinstance(content, list):
            for block in content:
                if isinstance(block, dict):
                    total += len(str(block.get("content", "")).split())
    return total


# Simulate a 5-turn conversation
planning_tokens_per_turn = 400  # Sonnet-style reasoning tokens
execution_tokens_per_turn = 80  # Haiku-style execution tokens

fused_cost = 5 * planning_tokens_per_turn * 3.0  # $3 per 1M tokens (example rate)
separated_cost = (
    5 * planning_tokens_per_turn * 3.0 +     # reasoning using Sonnet
    5 * execution_tokens_per_turn * 0.25      # execution using Haiku
)

print(f"Fused cost (arbitrary units): {fused_cost:.2f}")
print(f"Separated cost: {separated_cost:.2f}")
print(f"Savings: {(1 - separated_cost / fused_cost) * 100:.0f}%")
```

The exact savings depend on your model pricing and token counts. In this example, the separated approach costs about 20% less because execution tokens use a much cheaper model.

## How It Works Internally

The separation works because reasoning and execution have different requirements:

- **Reasoning** needs a model with strong planning capability, large context, and the ability to consider multiple alternatives. These are characteristics of larger, more expensive models.
- **Execution** needs a model that can follow instructions precisely, produce structured output, and respond quickly. Smaller models are often better at following structured formats without creative deviation.

When you separate the two, the reasoning model produces a structured decision (a tool call request). Your code validates it, executes it, and returns the result. The execution model never needs to plan — it only needs to format.

## Build Something Real

Build a research agent with separated planning and execution stages.

```python
import anthropic
import json
from typing import Any

client = anthropic.Anthropic()


class ResearchAgent:
    """A research agent with separated planning and execution."""

    def __init__(self):
        self.tools = [
            {
                "name": "web_search",
                "description": "Search the web for information on a topic",
                "input_schema": {
                    "type": "object",
                    "properties": {
                        "query": {"type": "string", "description": "Search query"},
                        "num_results": {
                            "type": "integer", "description": "Number of results",
                            "default": 5
                        }
                    },
                    "required": ["query"]
                }
            },
            {
                "name": "extract_text",
                "description": "Extract text content from a URL",
                "input_schema": {
                    "type": "object",
                    "properties": {
                        "url": {"type": "string", "description": "Page URL"}
                    },
                    "required": ["url"]
                }
            },
        ]
        self.tool_results: dict[str, Any] = {}

    def plan_next_action(self, context: list[dict]) -> dict | str:
        """Use a powerful model to plan the next action."""
        response = client.messages.create(
            model="claude-3-5-sonnet-20241022",
            max_tokens=500,
            temperature=0,
            system=(
                "You are a research planner. Decide the next action. "
                "Respond with a tool call or a final answer."
            ),
            messages=context,
            tools=self.tools,
        )
        return response.content[0]

    def execute_tool(self, tool_name: str, tool_input: dict) -> str:
        """Execute a tool call — this is where the actual work happens."""
        if tool_name == "web_search":
            return self._simulate_search(tool_input["query"], tool_input.get("num_results", 5))
        elif tool_name == "extract_text":
            return self._simulate_extract(tool_input["url"])
        return f"Unknown tool: {tool_name}"

    def _simulate_search(self, query: str, num_results: int) -> str:
        """Simulated search results."""
        results = [
            {"title": f"Result 1 for: {query}", "url": f"https://example.com/1"},
            {"title": f"Result 2 for: {query}", "url": f"https://example.com/2"},
        ]
        return json.dumps(results[:num_results])

    def _simulate_extract(self, url: str) -> str:
        """Simulated text extraction."""
        return f"Extracted content from {url}: [sample text for analysis]"

    def run(self, question: str, max_plans: int = 8) -> str:
        """Run the research agent with separated planning."""
        context = [{"role": "user", "content": question}]

        for plan_count in range(max_plans):
            decision = self.plan_next_action(context)

            if isinstance(decision, dict) and hasattr(decision, 'type') and decision.type == "tool_use":
                tool_result = self.execute_tool(decision.name, decision.input)
                print(f"[Plan {plan_count + 1}] {decision.name}({decision.input})")

                context.append({"role": "assistant", "content": decision})
                context.append({
                    "role": "user",
                    "content": [
                        {
                            "type": "tool_result",
                            "tool_use_id": decision.id,
                            "content": tool_result
                        }
                    ]
                })
            else:
                # Final answer
                return decision if isinstance(decision, str) else decision.text

        return "Research plan did not converge within the plan limit."


agent = ResearchAgent()
result = agent.run("Research the environmental impact of electric vehicles.")
print(f"\nFinal report:\n{result}")
```

The planner (Sonnet) spends its tokens on deciding search queries and extraction targets. The execution phase handles the actual search work. Each phase uses the right model for the job.
