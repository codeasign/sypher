# Orchestrator and Supervisor Patterns — Build It

*Reading time: 15 minutes*

## Breaking It Down

### 1. The Orchestrator Pattern

An orchestrator receives a high-level task, splits it into sub-tasks, assigns each to a worker agent, collects results, and produces a final answer. The workers are single-purpose agents that do not talk to each other.

```python
from dataclasses import dataclass, field
from typing import Callable


@dataclass
class SubTask:
    """A unit of work assigned to a worker agent."""
    id: str
    description: str
    assigned_to: str | None = None
    result: str | None = None
    status: str = "pending"  # pending | running | completed | failed


@dataclass
class Orchestrator:
    """Central coordinator that routes work and assembles results."""
    workers: dict[str, Callable] = field(default_factory=dict)

    def register_worker(self, name: str, handler: Callable):
        """Register a worker agent by name."""
        self.workers[name] = handler

    def decompose(self, task: str) -> list[SubTask]:
        """Break a task into sub-tasks (rule-based for clarity)."""
        subtasks = []
        if "research" in task.lower():
            subtasks.append(SubTask(id="1", description="Gather background data",
                                    assigned_to="researcher"))
        if "write" in task.lower():
            subtasks.append(SubTask(id="2", description="Draft the document",
                                    assigned_to="writer"))
        if "review" in task.lower():
            subtasks.append(SubTask(id="3", description="Review for quality",
                                    assigned_to="reviewer"))
        return subtasks

    def run(self, task: str) -> str:
        """Execute the full orchestration flow."""
        subtasks = self.decompose(task)
        results = {}

        for st in subtasks:
            handler = self.workers.get(st.assigned_to)
            if not handler:
                st.status = "failed"
                st.result = f"No worker registered for {st.assigned_to}"
                continue
            st.status = "running"
            try:
                st.result = handler(st.description)
                st.status = "completed"
            except Exception as e:
                st.result = f"Error: {e}"
                st.status = "failed"
            results[st.assigned_to] = st.result

        # Assemble final output from all results
        parts = [f"## {name.upper()}\n{res}" for name, res in results.items()]
        return "\n\n".join(parts)
```

### 2. Worker Agent Handlers

Each worker is a function that performs a specific task. In production these would be full agent loops with their own tools and models, but the interface is always the same: input string in, result string out.

```python
def research_worker(task: str) -> str:
    """Simulate a research agent gathering information."""
    return (
        f"Research findings for: {task}\n"
        "- Market size: $4.2B (2025), projected $8.7B (2030)\n"
        "- Key competitors: Company A, Company B, Company C\n"
        "- Growth rate: 15.6% CAGR"
    )


def writing_worker(task: str) -> str:
    """Simulate a writing agent producing a draft."""
    return (
        f"Draft for: {task}\n"
        "The market is experiencing rapid growth driven by AI adoption. "
        "Companies A through C dominate the space, but new entrants are "
        "gaining share through specialized vertical solutions."
    )


def review_worker(task: str) -> str:
    """Simulate a review agent checking quality."""
    return (
        f"Review for: {task}\n"
        "Status: PASSED\n"
        "Notes: All required sections present. Data sources need citations."
    )


# Wire up the orchestrator
orchestrator = Orchestrator()
orchestrator.register_worker("researcher", research_worker)
orchestrator.register_worker("writer", writing_worker)
orchestrator.register_worker("reviewer", review_worker)

result = orchestrator.run("Research, write, and review a market analysis report.")
print(result)
```

### 3. The Supervisor Pattern

A supervisor does not do work — it evaluates work and decides whether to accept it, request changes, or escalate. This pattern is essential for quality-critical pipelines where a single bad output can cascade downstream.

```python
from enum import Enum


class Verdict(Enum):
    ACCEPT = "accept"
    REVISE = "revise"
    ESCALATE = "escalate"


@dataclass
class Supervisor:
    """Evaluates worker output and decides next action."""
    max_revisions: int = 3
    escalation_threshold: int = 5  # severity score threshold

    def evaluate(self, task: str, output: str,
                 attempt: int) -> tuple[Verdict, str]:
        """Evaluate worker output and return a verdict with feedback."""
        issues = []
        severity = 0

        # Quality checks
        if len(output.split()) < 20:
            issues.append("Output is too short")
            severity += 2

        if "error" in output.lower() or "unable" in output.lower():
            issues.append("Output contains failure language")
            severity += 3

        if "todo" in output.lower() or "placeholder" in output.lower():
            issues.append("Output contains placeholders")
            severity += 2

        if attempt >= self.max_revisions:
            return Verdict.ESCALATE, (
                f"Escalating after {attempt} failed attempts. "
                f"Issues: {', '.join(issues)}"
            )

        if severity > self.escalation_threshold:
            return Verdict.ESCALATE, (
                f"Critical issues found: {', '.join(issues)}"
            )

        if issues:
            return Verdict.REVISE, (
                f"Revision needed: {', '.join(issues)}"
            )

        return Verdict.ACCEPT, "Output meets quality standards."


# Test the supervisor
supervisor = Supervisor()

outputs = [
    "Short output.",
    ("This is a longer but still problematic output with placeholder text "
     "like TODO: add data here and error handling is missing."),
    # Third attempt could pass
]

for i, out in enumerate(outputs):
    verdict, feedback = supervisor.evaluate(
        "Write report", out, attempt=i + 1
    )
    print(f"Attempt {i + 1}: {verdict.value}")
    print(f"  Feedback: {feedback}")
    print()
```

### 4. Combining Orchestrator and Supervisor

The most powerful pattern combines both: the orchestrator routes work, and a supervisor evaluates each worker's output before the orchestrator assembles the result. If a worker fails the supervisor's check, the orchestrator can either retry the worker or escalate.

```python
@dataclass
class SupervisedOrchestrator:
    """Orchestrator with per-worker quality gates."""
    workers: dict[str, Callable] = field(default_factory=dict)
    supervisor: Supervisor = field(default_factory=Supervisor)

    def register_worker(self, name: str, handler: Callable):
        self.workers[name] = handler

    def decompose(self, task: str) -> list[SubTask]:
        """Simple rule-based decomposition."""
        steps = [
            ("researcher", f"Research for: {task}"),
            ("writer", f"Write based on research: {task}"),
            ("reviewer", f"Review the draft: {task}"),
        ]
        return [
            SubTask(id=str(i), description=desc, assigned_to=name)
            for i, (name, desc) in enumerate(steps, 1)
        ]

    def run(self, task: str) -> str:
        subtasks = self.decompose(task)
        results = {}

        for st in subtasks:
            handler = self.workers.get(st.assigned_to)
            if not handler:
                st.status = "failed"
                st.result = "Unknown worker"
                continue

            for attempt in range(self.supervisor.max_revisions):
                st.status = "running"
                try:
                    output = handler(st.description)
                except Exception as e:
                    output = f"Error: {e}"

                verdict, feedback = self.supervisor.evaluate(
                    st.description, output, attempt + 1
                )

                if verdict == Verdict.ACCEPT:
                    st.result = output
                    st.status = "completed"
                    results[st.assigned_to] = output
                    break
                elif verdict == Verdict.ESCALATE:
                    st.result = f"ESCALATED: {feedback}"
                    st.status = "failed"
                    results[st.assigned_to] = st.result
                    break
                # else REVISE — loop continues for next attempt

            # If all attempts exhausted
            if st.status != "completed":
                st.status = "failed"
                st.result = f"Failed after {self.supervisor.max_revisions} attempts"
                results[st.assigned_to] = st.result

        parts = [f"## {name.upper()}\n{res}" for name, res in results.items()]
        return "\n\n".join(parts)
```

## How It Works Internally

When an orchestrator delegates to a worker, the worker runs its own complete agent loop — perceiving its task description, reasoning about it, calling tools, and producing output. The orchestrator's role is purely in the control plane: it manages which worker runs when and how results are combined.

A supervisor, by contrast, operates on the output plane. It does not need an LLM at all — it can be a set of rules, a smaller model, or a full agent loop. The key architectural property is that the supervisor sees the worker's output but not its internal state. This separation of concerns means you can upgrade or replace workers without changing the supervision logic.

## Build Something Real

Build a document generation pipeline that uses a supervised orchestrator to research a topic, write a summary, review it, and revise until it passes quality gates.

```python
import json
from dataclasses import dataclass, field
from typing import Callable
from enum import Enum


class QualityLevel(Enum):
    PASS = "pass"
    REVISE = "revise"
    FAIL = "fail"


@dataclass
class DocumentAgent:
    """Production document-generation pipeline with quality control."""
    temperature: float = 0.3
    max_tokens: int = 2000

    def research(self, topic: str) -> dict:
        """Research phase: gather structured data about a topic."""
        return {
            "topic": topic,
            "key_facts": [
                f"{topic} is a rapidly evolving field",
                "Market adoption has increased 40% year over year",
                "Three major vendors dominate the ecosystem",
            ],
            "sources": ["Industry Report 2025", "Market Analysis Q2"],
            "confidence": 0.85,
        }

    def write(self, research: dict) -> str:
        """Writing phase: produce a structured document from research data."""
        lines = [
            f"# {research['topic']} Analysis\n",
            "## Executive Summary",
            f"{research['topic']} represents a growing opportunity in the "
            "enterprise software market. Key drivers include digital "
            "transformation initiatives and increasing AI adoption.",
            "",
            "## Key Findings",
        ]
        for fact in research["key_facts"]:
            lines.append(f"- {fact}")
        lines.extend([
            "",
            "## Sources",
        ])
        for source in research["sources"]:
            lines.append(f"- {source}")
        return "\n".join(lines)

    def review(self, document: str, min_words: int = 30) -> QualityLevel:
        """Review phase: check document quality against criteria."""
        word_count = len(document.split())
        if word_count < min_words:
            return QualityLevel.REVISE
        if "# " not in document:
            return QualityLevel.REVISE
        if "TODO" in document or "placeholder" in document.lower():
            return QualityLevel.FAIL
        return QualityLevel.PASS

    def run_pipeline(self, topic: str) -> dict:
        """Run the full supervised document pipeline."""
        print(f"Starting research on: {topic}")
        data = self.research(topic)

        for attempt in range(3):
            print(f"Writing attempt {attempt + 1}...")
            doc = self.write(data)
            quality = self.review(doc)

            if quality == QualityLevel.PASS:
                print("Document passed review.")
                return {"status": "success", "document": doc, "attempts": attempt + 1}
            elif quality == QualityLevel.REVISE:
                print(f"Document needs revision. Retrying...")
                # Add more detail on retry
                data["key_facts"].append(f"Additional fact iteration {attempt + 2}")
            else:
                print("Document failed quality check. Escalating.")
                return {"status": "escalated", "document": doc, "attempts": attempt + 1}

        return {"status": "failed", "document": None, "attempts": 3}


agent = DocumentAgent()
result = agent.run_pipeline("Multi-Agent Systems")
print(json.dumps(result["status"]))
if result.get("document"):
    print(result["document"][:200] + "...")
```

## Security Considerations

In a multi-agent system, each agent should operate with the minimum privileges needed for its role. A research agent should not have write access to the database. A reviewer should not be able to modify worker code. Implement a permissions matrix that maps each agent role to its allowed tools and data scopes.
