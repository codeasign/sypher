# Loop Safety and Recovery — Build It

*Reading time: 14 minutes*

## Breaking It Down

### 1. Hard Limits

Hard limits are the last line of defense. They halt execution unconditionally.

```python
import time


class HardLimits:
    """Fixed boundaries the agent cannot exceed."""

    def __init__(self, max_turns: int = 25, max_time_s: int = 120,
                 max_tokens: int = 50000, max_consecutive_errors: int = 5):
        self.max_turns = max_turns
        self.max_time_s = max_time_s
        self.max_tokens = max_tokens
        self.max_consecutive_errors = max_consecutive_errors

        self.turn_count = 0
        self.start_time = time.time()
        self.total_tokens = 0
        self.consecutive_errors = 0

    def check(self) -> tuple[bool, str]:
        """
        Check all limits. Returns (ok, message).
        If any limit is exceeded, ok is False.
        """
        self.turn_count += 1

        if self.turn_count > self.max_turns:
            return False, f"Turn limit exceeded ({self.max_turns})"

        elapsed = time.time() - self.start_time
        if elapsed > self.max_time_s:
            return False, f"Time limit exceeded ({self.max_time_s}s)"

        if self.total_tokens > self.max_tokens:
            return False, f"Token limit exceeded ({self.max_tokens})"

        if self.consecutive_errors > self.max_consecutive_errors:
            return False, f"Too many consecutive errors ({self.max_consecutive_errors})"

        return True, "OK"

    def record_tokens(self, count: int):
        """Record token usage from a turn."""
        self.total_tokens += count

    def record_error(self, is_error: bool):
        """Track consecutive errors."""
        if is_error:
            self.consecutive_errors += 1
        else:
            self.consecutive_errors = 0


# Example usage in a loop
limits = HardLimits(max_turns=5)
for i in range(10):
    ok, msg = limits.check()
    if not ok:
        print(f"Loop stopped at turn {i + 1}: {msg}")
        break
    limits.record_tokens(500)
    print(f"Turn {i + 1}: OK")
```

### 2. Retry with Backoff

When a tool call fails, retry with increasing delays.

```python
import time
import random


def retry_with_backoff(func, args: dict, max_retries: int = 3,
                       base_delay: float = 1.0, backoff_factor: float = 2.0):
    """
    Execute a function with exponential backoff retry.
    """
    last_error = None

    for attempt in range(max_retries + 1):
        try:
            result = func(**args)
            if attempt > 0:
                print(f"  Retry {attempt} succeeded")
            return result
        except Exception as e:
            last_error = e
            if attempt < max_retries:
                delay = base_delay * (backoff_factor ** attempt)
                # Add jitter to prevent thundering herd
                delay *= 0.5 + random.random() * 0.5
                print(f"  Attempt {attempt + 1} failed: {e}. Retrying in {delay:.1f}s...")
                time.sleep(delay)
            else:
                print(f"  All {max_retries} retries exhausted.")
                raise last_error


# Simulated flaky tool
flaky_calls = 0


def flaky_tool(value: str) -> str:
    global flaky_calls
    flaky_calls += 1
    if flaky_calls < 3:  # Fail the first two times
        raise ConnectionError("Simulated network error")
    return f"Processed: {value}"


try:
    result = retry_with_backoff(flaky_tool, {"value": "test"}, max_retries=3)
    print(f"Final result: {result}")
except Exception as e:
    print(f"Failed permanently: {e}")
```

### 3. Circuit Breaker

A circuit breaker detects failure patterns and halts the agent before it wastes resources.

```python
class CircuitBreaker:
    """
    Tracks failures and opens the circuit when the error rate exceeds a threshold.
    States: CLOSED (normal) → OPEN (tripped) → HALF_OPEN (testing recovery)
    """

    def __init__(self, threshold: float = 0.5, window_size: int = 10,
                 recovery_timeout: float = 30.0):
        self.threshold = threshold  # Error rate that triggers the break
        self.window_size = window_size
        self.recovery_timeout = recovery_timeout

        self.results: list[bool] = []  # True = success, False = failure
        self.state = "CLOSED"
        self.tripped_at: float | None = None

    def record(self, success: bool):
        """Record the result of a tool call."""
        self.results.append(success)
        if len(self.results) > self.window_size:
            self.results.pop(0)

        if self.state == "CLOSED":
            # Check if we should trip
            if len(self.results) >= self.window_size:
                error_rate = 1.0 - (sum(self.results) / len(self.results))
                if error_rate >= self.threshold:
                    self.state = "OPEN"
                    self.tripped_at = time.time()
                    print(f"⚠ Circuit BREAKER TRIPPED (error rate: {error_rate:.0%})")

        elif self.state == "OPEN":
            # Check if enough time has passed to try again
            if time.time() - (self.tripped_at or 0) > self.recovery_timeout:
                self.state = "HALF_OPEN"
                print("→ Circuit HALF-OPEN, testing recovery...")

    def allow_request(self) -> bool:
        """Check if a new tool call is allowed."""
        if self.state == "CLOSED":
            return True
        if self.state == "HALF_OPEN":
            # Allow one request to test recovery
            return True
        if self.state == "OPEN":
            remaining = (self.tripped_at or 0) + self.recovery_timeout - time.time()
            print(f"  Circuit OPEN, try again in {remaining:.0f}s")
            return False
        return True


# Simulate a circuit breaker in action
breaker = CircuitBreaker(threshold=0.4, window_size=5, recovery_timeout=2.0)

for i in range(20):
    if not breaker.allow_request():
        print(f"  Call {i + 1}: BLOCKED by circuit breaker")
        time.sleep(0.5)
        continue

    # Simulate a call that starts failing
    success = i < 5 or i >= 12  # Fail calls 5-11
    breaker.record(success)
    print(f"Call {i + 1}: {'✓' if success else '✗'}  State: {breaker.state}")
    time.sleep(0.2)
```

## How It Works Internally

The circuit breaker implements a finite state machine:

- **CLOSED**: Normal operation. Errors are tracked but requests pass through. When the error rate exceeds the threshold, transition to OPEN.
- **OPEN**: Requests are blocked without attempting. A timer starts. After the recovery timeout, transition to HALF-OPEN.
- **HALF-OPEN**: One test request is allowed. If it succeeds, transition back to CLOSED. If it fails, transition back to OPEN.

This prevents the agent from wasting time and tokens on a system that is clearly failing.

## Build Something Real

Build a complete safety system combining all three layers.

```python
import anthropic
import time
import random

client = anthropic.Anthropic()


class SafetySystem:
    """Complete safety system with hard limits, retry, and circuit breaker."""

    def __init__(self):
        self.limits = HardLimits(max_turns=15, max_time_s=60, max_tokens=30000)
        self.breaker = CircuitBreaker(threshold=0.4, window_size=5, recovery_timeout=10.0)
        self.tools = [{
            "name": "lookup_data",
            "description": "Look up data from an external system",
            "input_schema": {
                "type": "object",
                "properties": {
                    "key": {"type": "string"}
                },
                "required": ["key"]
            }
        }]
        self.consecutive_timeouts = 0

    def safe_execute(self, tool_name: str, args: dict) -> str:
        """Execute a tool call with full safety checks."""
        if not self.breaker.allow_request():
            self.limits.record_error(True)
            return "ERROR: Circuit breaker is open. Try again later."

        try:
            result = retry_with_backoff(
                self._execute_tool,
                {"name": tool_name, "args": args},
                max_retries=2,
                base_delay=0.5,
            )
            self.consecutive_timeouts = 0
            self.breaker.record(True)
            self.limits.record_error(False)
            return result
        except Exception as e:
            self.consecutive_timeouts += 1
            self.breaker.record(False)
            self.limits.record_error(True)

            # Escalate if we have too many consecutive failures
            if self.consecutive_timeouts >= 3:
                return (
                    "ESCALATION: The external system is unavailable. "
                    "This task needs manual attention. "
                    "Please try again later or contact support."
                )
            return f"ERROR: {e}"

    def _execute_tool(self, name: str, args: dict) -> str:
        """Simulated tool execution that occasionally fails."""
        if random.random() < 0.3:  # 30% failure rate
            raise TimeoutError("Simulated timeout")
        return f"Data for '{args.get('key', 'unknown')}': [result data]"

    def run(self, user_input: str) -> str:
        """Run the agent with full safety protection."""
        messages = [{"role": "user", "content": user_input}]

        for turn in range(50):  # Unlimited loop — limits control termination
            ok, msg = self.limits.check()
            if not ok:
                return f"Agent stopped: {msg}"

            response = client.messages.create(
                model="claude-3-5-sonnet-20241022",
                max_tokens=500,
                temperature=0,
                messages=messages,
                tools=self.tools,
            )

            self.limits.record_tokens(response.usage.output_tokens)

            content = response.content[0]
            if content.type == "tool_use":
                result = self.safe_execute(content.name, content.input)
                messages.append({"role": "assistant", "content": content})
                messages.append({
                    "role": "user",
                    "content": [
                        {"type": "tool_result", "tool_use_id": content.id, "content": result}
                    ]
                })
            else:
                return content.text

        return "Agent stopped: turn limit reached."


safety = SafetySystem()
result = safety.run("Look up the data for keys: product_123, product_456, product_789.")
print(f"Result:\n{result}")
```

The safety system intercepts every tool call, tracks errors, retries with backoff, and escalates when failures persist. The agent never sees raw exceptions — only structured error messages it can act on.

## Performance Considerations

Safety systems add overhead: retries consume time, circuit breakers add state tracking, and hard limits require monitoring. Tune the thresholds to your system's normal operation. A circuit breaker threshold of 50% is reasonable for development; 10-20% is typical for production.
