# Shared Memory and State — Build It

*Reading time: 15 minutes*

## Breaking It Down

### 1. Centralized Shared Store

A central store accessed by all agents is the simplest shared-state pattern. Use a thread-safe dictionary or a lightweight database.

```python
import threading
from dataclasses import dataclass, field
from datetime import datetime, timezone


@dataclass
class SharedStore:
    """Thread-safe centralized state store for multi-agent systems."""
    _data: dict = field(default_factory=dict)
    _lock: threading.Lock = field(default_factory=threading.Lock)
    _version: int = 0

    def read(self, key: str, default=None):
        """Read a value from the store."""
        with self._lock:
            return self._data.get(key, default)

    def write(self, key: str, value, agent: str):
        """Write a value to the store with agent attribution."""
        with self._lock:
            self._version += 1
            self._data[key] = {
                "value": value,
                "updated_by": agent,
                "updated_at": datetime.now(timezone.utc).isoformat(),
                "version": self._version,
            }

    def read_all(self) -> dict:
        """Read the entire store state."""
        with self._lock:
            return dict(self._data)

    def compare_and_swap(self, key: str, expected_value,
                         new_value, agent: str) -> bool:
        """
        Atomic compare-and-swap. Only writes if the current value
        matches expected_value. Prevents lost updates.
        """
        with self._lock:
            current = self._data.get(key, {}).get("value")
            if current == expected_value:
                self.write(key, new_value, agent)
                return True
            return False


# Example: two agents writing to shared state
store = SharedStore()
store.write("revenue", {"q1": 420000, "q2": 510000}, "research_agent")

def agent_write(name: str, key: str, value):
    """Simulate an agent writing to shared state."""
    store.write(key, value, name)
    print(f"[{name}] Wrote {key} = {value}")

threading.Thread(target=agent_write, args=("agent_a", "status", "in_progress")).start()
threading.Thread(target=agent_write, args=("agent_b", "status", "reviewed")).start()

# Allow threads to complete
import time
time.sleep(0.1)
print(f"Final status: {store.read('status')}")
```

### 2. Event-Sourced State (Distributed Log)

Instead of storing the current state, store every event. Agents replay the event log to reconstruct the current state. This eliminates race conditions because events are append-only.

```python
@dataclass
class EventLog:
    """Append-only event log for event-sourced shared state."""
    events: list = field(default_factory=list)
    _lock: threading.Lock = field(default_factory=threading.Lock)

    def append(self, event_type: str, data: dict, agent: str):
        """Append an event to the log."""
        with self._lock:
            self.events.append({
                "event_type": event_type,
                "data": data,
                "agent": agent,
                "timestamp": datetime.now(timezone.utc).isoformat(),
                "sequence": len(self.events),
            })

    def replay(self) -> dict:
        """Replay all events to reconstruct current state."""
        state = {}
        with self._lock:
            for event in self.events:
                if event["event_type"] == "set":
                    state[event["data"]["key"]] = event["data"]["value"]
                elif event["event_type"] == "delete":
                    state.pop(event["data"]["key"], None)
                elif event["event_type"] == "merge":
                    key = event["data"]["key"]
                    if key not in state:
                        state[key] = {}
                    state[key].update(event["data"]["value"])
        return state


# Example: agents append events, state is reconstructed
log = EventLog()
log.append("set", {"key": "project_name", "value": "Atlas"}, "orchestrator")
log.append("set", {"key": "status", "value": "researching"}, "orchestrator")
log.append("set", {"key": "status", "value": "writing"}, "orchestrator")

state = log.replay()
print(f"Reconstructed state: {state}")
print(f"Number of events: {len(log.events)}")
```

### 3. Per-Agent Snapshots

Each agent gets a frozen snapshot of the shared state at the start of its task. It works against that snapshot and writes back only its results. This eliminates race conditions within a single task.

```python
@dataclass
class SnapshotStore:
    """Store that provides read-only snapshots to agents."""
    _master: dict = field(default_factory=dict)
    _lock: threading.Lock = field(default_factory=threading.Lock)

    def write_master(self, key: str, value, agent: str):
        """Write to the master state (only the orchestrator should do this)."""
        with self._lock:
            self._master[key] = {
                "value": value,
                "updated_by": agent,
            }

    def take_snapshot(self) -> dict:
        """Return a frozen copy of the current state."""
        with self._lock:
            return dict(self._master)

    def merge_result(self, agent: str, updates: dict):
        """Merge an agent's results back into master state."""
        with self._lock:
            for key, value in updates.items():
                self._master[key] = {
                    "value": value,
                    "updated_by": agent,
                }


# Example: agents receive snapshots, work, then merge results
store = SnapshotStore()
store.write_master("project", "Market Report", "orchestrator")

# Agent receives its snapshot
snapshot = store.take_snapshot()
print(f"Agent received snapshot: {snapshot}")

# Agent works and returns results
agent_results = {"total_revenue": 2045000, "data_quality": "high"}
store.merge_result("analyst_agent", agent_results)

# Another agent sees the merged result
next_snapshot = store.take_snapshot()
print(f"After merge: {next_snapshot}")
```

## How It Works Internally

The centralized store introduces a single point of contention — every read and write acquires a lock. Under high concurrency, this lock becomes the bottleneck. The event log avoids the write lock by appending only, but the replay operation is O(N) in the number of events, so state reconstruction grows slower over time. Snapshots trade off freshness for isolation: agents never see in-progress changes, which is safe but can cause agents to work with slightly stale data.

## Build Something Real

Build a collaborative document system where three agents share state through a snapshot store, each contributing to different sections of a business report.

```python
import json
from dataclasses import dataclass, field
from typing import Any


@dataclass
class CollaborativeDocument:
    """Multi-agent document with snapshot-based shared state."""
    store: SnapshotStore = field(default_factory=SnapshotStore)
    sections: dict = field(default_factory=dict)

    def assign_section(self, section: str, agent: str):
        """Assign a document section to an agent."""
        self.store.write_master(f"section:{section}:assigned_to", agent, "orchestrator")
        self.store.write_master(f"section:{section}:status", "assigned", "orchestrator")

    def work_on_section(self, section: str, content: str, agent: str) -> dict:
        """Agent works on an assigned section using its snapshot."""
        snapshot = self.store.take_snapshot()

        # Validate the agent is assigned to this section
        assignment = snapshot.get(f"section:{section}:assigned_to", {})
        if assignment.get("value") != agent:
            return {"error": f"Agent {agent} not assigned to section {section}"}

        result = {f"section:{section}:content": content,
                  f"section:{section}:status": "completed"}
        self.store.merge_result(agent, result)
        self.sections[section] = content
        return {"section": section, "status": "completed"}

    def get_report(self) -> dict:
        """Get the compiled report from shared state."""
        snapshot = self.store.take_snapshot()
        report = {}
        for key, entry in snapshot.items():
            if key.endswith(":content"):
                section = key.split(":")[1]
                report[section] = entry["value"]
        return report


# Run the collaborative workflow
doc = CollaborativeDocument()
doc.assign_section("executive_summary", "writer_agent")
doc.assign_section("financial_analysis", "analyst_agent")

# Agents work in parallel
doc.work_on_section(
    "executive_summary",
    "Revenue grew 22% year-over-year driven by enterprise adoption.",
    "writer_agent",
)
doc.work_on_section(
    "financial_analysis",
    "Gross margin: 68%. Operating margin: 22%. Revenue: $2.04M.",
    "analyst_agent",
)

report = doc.get_report()
print(json.dumps(report, indent=2))
```

This pattern scales to N agents without race conditions because each agent's work is isolated within its snapshot. The merge step is the only serialization point.

## Performance Considerations

The snapshot pattern eliminates read locks but introduces write serialization at merge time. For systems with fewer than 10 concurrent agents, this is rarely a bottleneck. If you have many agents writing results simultaneously, batch the merges or use a write-ahead log that the orchestrator processes asynchronously.

## Security Considerations

Shared state is a common vector for prompt injection. If one agent writes malformed data to the shared store and another agent reads it, the second agent may misinterpret the data as a task instruction. Validate every value written to shared state: ensure it matches the expected schema and does not contain raw text that could be interpreted as instructions.
