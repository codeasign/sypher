# Anatomy of an Agent — Build It

*Reading time: 14 minutes*

## Breaking It Down

### 1. Building the Orchestrator

The orchestrator is the loop that drives the agent. It builds prompts, calls the model, processes tool calls, and decides when to stop.

```python
import anthropic
from typing import Any, Callable

client = anthropic.Anthropic()


class Orchestrator:
    """The central loop that drives agent execution."""

    def __init__(self, system_prompt: str, tools: list[dict], tool_handlers: dict[str, Callable]):
        self.system_prompt = system_prompt
        self.tools = tools
        self.tool_handlers = tool_handlers
        self.messages = []
        self.turn_count = 0

    def add_message(self, role: str, content: Any):
        """Add a message to the conversation history."""
        self.messages.append({"role": role, "content": content})

    def step(self) -> tuple[bool, str | None]:
        """Execute one turn of the agent loop. Returns (is_final, output)."""
        response = client.messages.create(
            model="claude-3-5-sonnet-20241022",
            max_tokens=1000,
            temperature=0,
            system=self.system_prompt,
            messages=self.messages,
            tools=self.tools,
        )

        content = response.content[0]

        if content.type == "tool_use":
            # Execute the tool
            handler = self.tool_handlers.get(content.name)
            if handler:
                try:
                    result = handler(**content.input)
                except Exception as e:
                    result = f"Error: {e}"
            else:
                result = f"Unknown tool: {content.name}"

            # Record the tool call and result
            self.add_message("assistant", content)
            self.add_message("user", [
                {"type": "tool_result", "tool_use_id": content.id, "content": result}
            ])
            self.turn_count += 1
            return False, None  # Not final, no output yet

        else:
            # The model produced a text response — we are done
            return True, content.text

    def run(self, user_input: str, max_turns: int = 10) -> str:
        """Run the agent until it produces a final response."""
        self.add_message("user", user_input)

        for _ in range(max_turns):
            is_final, output = self.step()
            if is_final:
                return output

        return "Agent did not produce a final answer within the turn limit."
```

### 2. Wiring the Components

Now wire the model, tools, and memory through the orchestrator.

```python
# Define a handler function
def search_knowledge_base(query: str, max_results: int = 3) -> str:
    """Simulated knowledge base search."""
    knowledge = {
        "returns": "Returns are accepted within 30 days of purchase.",
        "shipping": "Free shipping on orders over $50. Standard delivery: 5-7 business days.",
        "warranty": "All products come with a 1-year limited warranty.",
    }
    results = []
    for key, value in knowledge.items():
        if query.lower() in key or any(word in key for word in query.lower().split()):
            results.append(f"[{key}] {value}")
    if results:
        return "\n".join(results[:max_results])
    return "No matching results found."


# Configure the agent
orchestrator = Orchestrator(
    system_prompt="You are a support agent with access to the knowledge base.",
    tools=[{
        "name": "search_knowledge_base",
        "description": "Search the knowledge base for support articles",
        "input_schema": {
            "type": "object",
            "properties": {
                "query": {"type": "string", "description": "Search query"},
                "max_results": {
                    "type": "integer",
                    "description": "Max results to return",
                    "default": 3
                }
            },
            "required": ["query"]
        }
    }],
    tool_handlers={"search_knowledge_base": search_knowledge_base},
)

# Run the agent
result = orchestrator.run("What is your return policy?")
print(result)
```

### 3. Adding a Safety Layer

The safety layer intercepts tool calls before they execute. It enforces constraints.

```python
class SafetyLayer:
    """Guards that prevent dangerous or wasteful actions."""

    def __init__(self, max_tool_calls: int = 20, rate_limit_window: int = 60):
        self.max_tool_calls = max_tool_calls
        self.tool_call_count = 0
        self.start_time = __import__('time').time()
        self.rate_limit_window = rate_limit_window

    def check(self, tool_name: str, tool_input: dict) -> tuple[bool, str | None]:
        """
        Check if a tool call is allowed.
        Returns (allowed, rejection_reason).
        """
        # Rate limiting
        self.tool_call_count += 1
        if self.tool_call_count > self.max_tool_calls:
            return False, "Tool call limit exceeded for this session."

        # Block dangerous parameter patterns (example: SQL injection-like inputs)
        for key, value in tool_input.items():
            if isinstance(value, str) and "--" in value and "drop" in value.lower():
                return False, "Input contains blocked patterns."

        # Block empty queries
        if "query" in tool_input and not tool_input["query"].strip():
            return False, "Empty query is not allowed."

        return True, None


class SafeOrchestrator(Orchestrator):
    """Orchestrator with a safety layer."""

    def __init__(self, system_prompt: str, tools: list[dict], tool_handlers: dict):
        super().__init__(system_prompt, tools, tool_handlers)
        self.safety = SafetyLayer()

    def step(self):
        response = client.messages.create(
            model="claude-3-5-sonnet-20241022",
            max_tokens=1000,
            temperature=0,
            system=self.system_prompt,
            messages=self.messages,
            tools=self.tools,
        )

        content = response.content[0]

        if content.type == "tool_use":
            # Check safety before executing
            allowed, reason = self.safety.check(content.name, content.input)
            if not allowed:
                result = f"SAFETY BLOCKED: {reason}"
            else:
                handler = self.tool_handlers.get(content.name)
                if handler:
                    try:
                        result = handler(**content.input)
                    except Exception as e:
                        result = f"Error: {e}"
                else:
                    result = f"Unknown tool: {content.name}"

            self.add_message("assistant", content)
            self.add_message("user", [
                {"type": "tool_result", "tool_use_id": content.id, "content": result}
            ])
            self.turn_count += 1
            return False, None

        return True, content.text
```

The safety layer operates before the tool executes. This means dangerous actions never reach the implementation function.

## How It Works Internally

The orchestrator manages a state machine with three states:

1. **Awaiting input** — waiting for the user or a tool result to arrive
2. **Awaiting model** — waiting for the LLM API response
3. **Processing tool** — executing a tool call returned by the model

Each transition updates the message history and triggers the next state. The orchestrator keeps the machine honest by ensuring tool results are always returned to the model in the correct format.

## Build Something Real

Build a complete agent with multiple components: model, tools, memory (short-term via messages), orchestrator, and safety.

```python
import anthropic
import json
import time
from typing import Any

client = anthropic.Anthropic()


class KnowledgeBase:
    """A simple knowledge base that stores and retrieves facts."""

    def __init__(self):
        self.facts = {
            "company": "Acme Corp",
            "founded": 2019,
            "employees": 250,
            "products": ["Widget Pro", "Widget Lite", "Widget Enterprise"],
        }

    def query(self, key: str) -> str:
        """Retrieve a fact by key."""
        result = self.facts.get(key)
        if result:
            return json.dumps({key: result})
        return f"Fact '{key}' not found."

    def store(self, key: str, value: Any) -> str:
        """Store a new fact."""
        self.facts[key] = value
        return f"Stored: {key} = {value}"


class Calculator:
    """Simple calculator tool."""

    @staticmethod
    def calculate(expression: str) -> str:
        """Evaluate a mathematical expression."""
        safe_chars = set("0123456789.+-*/() ")
        if not all(c in safe_chars for c in expression):
            return "Error: expression contains invalid characters."
        try:
            result = eval(expression, {"__builtins__": {}}, {})
            return f"{expression} = {result}"
        except Exception as e:
            return f"Error evaluating expression: {e}"


kb = KnowledgeBase()
calc = Calculator()

TOOLS = [
    {
        "name": "knowledge_query",
        "description": "Query a fact from the knowledge base by key",
        "input_schema": {
            "type": "object",
            "properties": {
                "key": {"type": "string", "description": "Fact key"}
            },
            "required": ["key"]
        }
    },
    {
        "name": "knowledge_store",
        "description": "Store a fact in the knowledge base",
        "input_schema": {
            "type": "object",
            "properties": {
                "key": {"type": "string", "description": "Fact key"},
                "value": {"description": "Fact value"}
            },
            "required": ["key", "value"]
        }
    },
    {
        "name": "calculate",
        "description": "Evaluate a mathematical expression",
        "input_schema": {
            "type": "object",
            "properties": {
                "expression": {"type": "string", "description": "Math expression"}
            },
            "required": ["expression"]
        }
    },
]


HANDLERS = {
    "knowledge_query": lambda **kw: kb.query(kw["key"]),
    "knowledge_store": lambda **kw: kb.store(kw["key"], kw["value"]),
    "calculate": lambda **kw: calc.calculate(kw["expression"]),
}

messages: list[dict] = []
turn_count = 0
max_turns = 15

user_input = (
    "What year was Acme Corp founded? "
    "Also calculate what year it will be 5 years after that. "
    "Then store 'milestone' as '10 year anniversary in 2029'."
)
messages.append({"role": "user", "content": user_input})

while turn_count < max_turns:
    response = client.messages.create(
        model="claude-3-5-sonnet-20241022",
        max_tokens=1000,
        temperature=0,
        system="You are an agent with knowledge base and calculation tools.",
        messages=messages,
        tools=TOOLS,
    )

    content = response.content[0]

    if content.type == "tool_use":
        handler = HANDLERS.get(content.name)
        if handler:
            result = handler(**content.input)
        else:
            result = f"Unknown tool: {content.name}"

        print(f"[Turn {turn_count + 1}] Tool: {content.name}{content.input} → {result}")
        messages.append({"role": "assistant", "content": content})
        messages.append({
            "role": "user",
            "content": [
                {"type": "tool_result", "tool_use_id": content.id, "content": result}
            ]
        })
        turn_count += 1
    else:
        print(f"\nFinal response:\n{content.text}")
        break
```

This agent chains multiple tools together. It queries knowledge, performs calculations, and stores results — all through the same orchestrator loop.
