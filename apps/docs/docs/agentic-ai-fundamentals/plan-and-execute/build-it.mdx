# Plan and Execute — Build It

*Reading time: 14 minutes*

## Breaking It Down

### 1. The Plan Document

Unlike ReAct, where the plan exists only in the model's context, Plan-and-Execute uses an explicit plan document.

```python
class Plan:
    """A mutable plan document maintained outside the conversation."""

    def __init__(self, task: str, steps: list[str]):
        self.task = task
        self.steps = [{"description": s, "status": "pending"} for s in steps]

    def complete_step(self, index: int, result: str):
        """Mark a step as completed."""
        if 0 <= index < len(self.steps):
            self.steps[index]["status"] = "completed"
            self.steps[index]["result"] = result

    def fail_step(self, index: int, reason: str):
        """Mark a step as failed."""
        if 0 <= index < len(self.steps):
            self.steps[index]["status"] = "failed"
            self.steps[index]["reason"] = reason

    def add_step(self, description: str, after_index: int = -1):
        """Insert a new step."""
        new_step = {"description": description, "status": "pending"}
        if after_index < 0:
            self.steps.append(new_step)
        else:
            self.steps.insert(after_index + 1, new_step)

    def next_pending(self) -> tuple[int, dict] | None:
        """Get the next pending step."""
        for i, step in enumerate(self.steps):
            if step["status"] == "pending":
                return i, step
        return None

    def is_complete(self) -> bool:
        """Check if all steps are done."""
        return all(s["status"] == "completed" for s in self.steps)

    def summary(self) -> str:
        """Get a human-readable plan summary."""
        lines = [f"Task: {self.task}"]
        for i, step in enumerate(self.steps):
            status_mark = {
                "pending": "[ ]", "completed": "[✓]", "failed": "[✗]"
            }.get(step["status"], "[?]")
            lines.append(f"  {status_mark} {step['description']}")
            if step["status"] == "completed":
                lines.append(f"       Result: {step.get('result', '')[:60]}")
            if step["status"] == "failed":
                lines.append(f"       Reason: {step.get('reason', '')}")
        return "\n".join(lines)


# Example
plan = Plan("Research cloud platforms", [
    "Search for AWS AI services",
    "Search for Azure AI services",
    "Search for GCP AI services",
    "Compile comparison table",
])
print(plan.summary())
plan.complete_step(0, "Found 12 AI services on AWS")
plan.complete_step(1, "Found 10 AI services on Azure")
plan.fail_step(2, "Search API returned timeout")
plan.add_step("Retry GCP search", after_index=2)
print("\nAfter updates:")
print(plan.summary())
```

### 2. Plan Phase

Generate the plan from the task description.

```python
import anthropic

client = anthropic.Anthropic()


def generate_plan(task: str) -> Plan:
    """Use the LLM to generate a plan from a task description."""
    response = client.messages.create(
        model="claude-3-5-sonnet-20241022",
        max_tokens=500,
        temperature=0,
        messages=[{
            "role": "user",
            "content": (
                f"Task: {task}\n\n"
                f"Break this task into 3-5 specific, actionable steps. "
                f"Each step should be something you can do with a search tool. "
                f"Number each step on a new line, starting with 'Step N:'."
            )
        }]
    )

    text = response.content[0].text
    steps = []
    for line in text.strip().split("\n"):
        line = line.strip()
        if line and (line.startswith("Step") or line[0].isdigit()):
            # Extract the step description after the "Step N:" prefix
            step_text = line.split(":", 1)[-1].strip() if ":" in line else line
            steps.append(step_text)

    return Plan(task, steps)
```

### 3. Execute Phase

Work through the plan, updating it as steps complete or fail.

```python
class PlanExecutor:
    """Executes a plan step by step with tool calls."""

    def __init__(self):
        self.tools = [{
            "name": "search_information",
            "description": "Search for information on a topic",
            "input_schema": {
                "type": "object",
                "properties": {
                    "query": {"type": "string", "description": "Search query"}
                },
                "required": ["query"]
            }
        }]

    def execute_step(self, step_description: str, context: str) -> str:
        """Execute one plan step using the agent loop."""
        messages = [
            {"role": "user", "content": f"Context:\n{context}\n\nStep: {step_description}"}
        ]

        for turn in range(5):
            response = client.messages.create(
                model="claude-3-5-sonnet-20241022",
                max_tokens=500,
                temperature=0,
                messages=messages,
                tools=self.tools,
            )

            content = response.content[0]
            if content.type == "tool_use":
                result = f"Results for '{content.input['query']}': [simulated data]"
                messages.append({"role": "assistant", "content": content})
                messages.append({
                    "role": "user",
                    "content": [
                        {"type": "tool_result", "tool_use_id": content.id, "content": result}
                    ]
                })
            else:
                return content.text

        return "Step did not complete within turn limit."


def plan_and_execute(task: str) -> str:
    """Full Plan-and-Execute cycle."""
    # PLAN
    print("=== PLAN PHASE ===")
    plan = generate_plan(task)
    print(plan.summary())

    # EXECUTE
    print("\n=== EXECUTE PHASE ===")
    executor = PlanExecutor()
    results = []

    while not plan.is_complete():
        next_step = plan.next_pending()
        if not next_step:
            break

        idx, step = next_step
        print(f"\nExecuting step {idx + 1}: {step['description'][:60]}...")

        context = f"Task: {plan.task}\n" + "\n".join(
            f"Step {i + 1} ({s['status']}): {s.get('result', s['description'])[:100]}"
            for i, s in enumerate(plan.steps)
        )

        result = executor.execute_step(step["description"], context)
        plan.complete_step(idx, result)
        results.append({"step": step["description"], "result": result})
        print(f"  Result: {result[:80]}...")

    # SYNTHESIZE
    print("\n=== SYNTHESIS ===")
    synthesis = client.messages.create(
        model="claude-3-5-sonnet-20241022",
        max_tokens=500,
        temperature=0,
        messages=[{
            "role": "user",
            "content": (
                f"Task: {task}\n\n"
                f"Results:\n" + "\n".join(
                    f"Step {i + 1}: {r['result']}" for i, r in enumerate(results)
                ) +
                "\n\nProvide a final answer that synthesizes these results."
            )
        }]
    )
    return synthesis.content[0].text


result = plan_and_execute("Research three programming languages: Python, Rust, and Go. "
                          "Compare their performance, learning curve, and primary use cases.")
print(f"\n=== FINAL ANSWER ===\n{result}")
```

## How It Works Internally

Plan-and-Execute maintains two separate data structures:

1. **The plan** — a mutable list of steps with status, stored outside the message list
2. **The conversation** — the message list containing tool calls and results

The separation prevents the model from confusing what it planned to do with what it actually did. When the model reads the plan, it sees completed steps marked `[✓]` and pending steps marked `[ ]`. This concrete signal reduces drift.

## Build Something Real

Build a research agent that uses Plan-and-Execute with replanning on failure.

```python
import anthropic
import json

client = anthropic.Anthropic()


class ResearchAgent:
    """Research agent with Plan-and-Execute and replanning."""

    def __init__(self):
        self.tools = [{
            "name": "search_topic",
            "description": "Search for information about a topic",
            "input_schema": {
                "type": "object",
                "properties": {
                    "topic": {"type": "string"},
                    "aspect": {"type": "string", "description": "What aspect to search"}
                },
                "required": ["topic", "aspect"]
            }
        }]
        self.plan = Plan("", [])

    def plan_task(self, task: str):
        """Generate a plan for the task."""
        response = client.messages.create(
            model="claude-3-5-sonnet-20241022",
            max_tokens=500,
            temperature=0,
            messages=[{
                "role": "user",
                "content": (
                    f"Task: {task}\n\n"
                    f"Create a numbered plan with specific, searchable steps. "
                    f"Each step should be a concrete search query or action."
                )
            }]
        )
        text = response.content[0].text
        steps = [
            line.split(":", 1)[-1].strip()
            for line in text.strip().split("\n")
            if line.strip() and (line[0].isdigit() or line.strip().startswith("-"))
        ]
        self.plan = Plan(task, steps)

    def execute_plan(self) -> list[dict]:
        """Execute the plan with replanning on failure."""
        results = []
        max_replans = 3
        replan_count = 0

        while not self.plan.is_complete() and replan_count <= max_replans:
            next_step = self.plan.next_pending()
            if not next_step:
                break

            idx, step = next_step
            context = f"Task: {self.plan.task}\nPlan:\n{self.plan.summary()}"
            messages = [
                {"role": "user", "content": f"{context}\n\nExecute step: {step['description']}"}
            ]

            step_succeeded = False
            for turn in range(5):
                response = client.messages.create(
                    model="claude-3-5-sonnet-20241022",
                    max_tokens=500,
                    temperature=0,
                    messages=messages,
                    tools=self.tools,
                )

                content = response.content[0]
                if content.type == "tool_use":
                    result = json.dumps({
                        "topic": content.input["topic"],
                        "aspect": content.input["aspect"],
                        "findings": f"Found information about {content.input['topic']} - {content.input['aspect']}"
                    })
                    messages.append({"role": "assistant", "content": content})
                    messages.append({
                        "role": "user",
                        "content": [
                            {"type": "tool_result", "tool_use_id": content.id, "content": result}
                        ]
                    })
                else:
                    self.plan.complete_step(idx, content.text)
                    results.append({"step": step["description"], "result": content.text})
                    step_succeeded = True
                    break

            if not step_succeeded:
                self.plan.fail_step(idx, "Step failed after max turns")
                replan_count += 1
                if replan_count <= max_replans:
                    new_plan_response = client.messages.create(
                        model="claude-3-5-sonnet-20241022",
                        max_tokens=300,
                        temperature=0,
                        messages=[{
                            "role": "user",
                            "content": (
                                f"Original task: {self.plan.task}\n"
                                f"Current plan:\n{self.plan.summary()}\n\n"
                                f"Step '{step['description']}' failed. "
                                f"Suggest an alternative approach or revised step."
                            )
                        }]
                    )
                    suggestion = new_plan_response.content[0].text
                    self.plan.add_step(f"Alternative: {suggestion[:100]}", after_index=idx)

        return results

    def synthesize(self, results: list[dict]) -> str:
        """Synthesize results into a final answer."""
        response = client.messages.create(
            model="claude-3-5-sonnet-20241022",
            max_tokens=500,
            temperature=0,
            messages=[{
                "role": "user",
                "content": (
                    f"Task: {self.plan.task}\n\n"
                    f"Results:\n{json.dumps(results, indent=2)}\n\n"
                    f"Synthesize into a clear, complete answer."
                )
            }]
        )
        return response.content[0].text

    def run(self, task: str) -> str:
        """Run the full Plan-and-Execute cycle."""
        print("=== PLANNING ===")
        self.plan_task(task)
        print(self.plan.summary())

        print("\n=== EXECUTING ===")
        results = self.execute_plan()

        print("\n=== SYNTHESIZING ===")
        return self.synthesize(results)


agent = ResearchAgent()
result = agent.run("Compare the use cases for SQL and NoSQL databases. "
                   "When would you choose one over the other?")
print(f"\n=== RESULT ===\n{result}")
```

## Performance Considerations

Plan generation costs one API call. For long-running agents, regenerate the plan every N steps to account for changed circumstances. The plan document itself has near-zero storage cost — the overhead is entirely in the plan generation and update API calls.
