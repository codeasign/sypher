## Step 2: Write the Code

Open `urlcheck.py` in your editor and replace the placeholder with the following:

```python
"""urlcheck -- CLI tool to check URL reachability.

Usage:
    python urlcheck.py <url>
    python urlcheck.py --batch <json-file>
"""

import sys
import json
import argparse

import requests


def check_url(url: str) -> tuple[str, int | None, str]:
    """Check if a URL is reachable and return (url, status_code, status_text)."""
    try:
        response = requests.head(url, timeout=5, allow_redirects=True)
        status = response.status_code
        if 200 <= status < 400:
            return (url, status, "OK")
        elif 400 <= status < 500:
            return (url, status, "Client Error")
        else:
            return (url, status, "Server Error")
    except requests.ConnectionError:
        return (url, None, "Unreachable")
    except requests.Timeout:
        return (url, None, "Timed Out")


def check_batch(file_path: str) -> list[tuple[str, int | None, str]]:
    """Check multiple URLs from a JSON file."""
    with open(file_path) as f:
        data = json.load(f)
    results = []
    for url in data["urls"]:
        results.append(check_url(url))
    return results


def format_result(url: str, status: int | None, text: str) -> str:
    """Format a check result for display."""
    if status is None:
        return f"{url} -- {text}"
    return f"{url} -- {status} {text}"


def main() -> None:
    parser = argparse.ArgumentParser(description="Check URL reachability")
    parser.add_argument("target", nargs="?", help="URL to check")
    parser.add_argument("--batch", "-b", metavar="FILE", help="JSON file with URLs to check")
    args = parser.parse_args()

    if args.batch:
        results = check_batch(args.batch)
        for url, status, text in results:
            flag = " (FAIL)" if status and status >= 400 else ""
            print(format_result(url, status, text) + flag)
    elif args.target:
        url, status, text = check_url(args.target)
        flag = " (FAIL)" if status and status >= 400 else ""
        print(format_result(url, status, text) + flag)
    else:
        parser.print_help()
        sys.exit(1)


if __name__ == "__main__":
    main()
```

```python
"""Tests for urlcheck."""

import json
import tempfile
from pathlib import Path

import pytest

from urlcheck import check_url, format_result, check_batch


class TestCheckUrl:
    def test_reachable_url(self) -> None:
        url, status, text = check_url("https://example.com")
        assert status == 200
        assert text == "OK"

    def test_format_result_ok(self) -> None:
        result = format_result("https://example.com", 200, "OK")
        assert result == "https://example.com -- 200 OK"

    def test_format_result_unreachable(self) -> None:
        result = format_result("https://nonexistent.example", None, "Unreachable")
        assert result == "https://nonexistent.example -- Unreachable"


class TestCheckBatch:
    def test_batch_returns_results(self) -> None:
        urls = {"urls": ["https://example.com", "https://httpstat.us/200"]}
        with tempfile.NamedTemporaryFile(mode="w", suffix=".json", delete=False) as f:
            json.dump(urls, f)
            f.flush()
            results = check_batch(f.name)
            assert len(results) == 2
            assert results[0][1] == 200
        Path(f.name).unlink(missing_ok=True)


class TestCli:
    def test_help_exits_nonzero(self) -> None:
        """Running with no arguments should exit with code 1."""
        with pytest.raises(SystemExit) as exc:
            from urlcheck import main
            import sys
            sys.argv = ["urlcheck.py"]
            main()
        assert exc.value.code == 1
```

## Step 3: Commit Strategy -- Atomic Commits

Good commits are small, self-contained, and have clear messages. You will make two commits on this feature branch:

| Commit | Files | Message |
|--------|-------|---------|
| 1 | `urlcheck.py` | `feat: add urlcheck CLI with single and batch modes` |
| 2 | `test_urlcheck.py`, `requirements.txt` | `test: add unit tests for urlcheck` |

### Commit 1: The CLI Code

```bash
git add urlcheck.py
git commit -m "feat: add urlcheck CLI with single and batch modes"
```

### Commit 2: The Tests

```bash
git add test_urlcheck.py requirements.txt
git commit -m "test: add unit tests for urlcheck"
```

### Why Separate Commits?

Separating code from tests lets someone reviewing the commit history see that the tests were added intentionally, not patched in later. It also means if a test is wrong, you can revert just the test commit without touching the code.
