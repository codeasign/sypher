# Control Flow — Hands On

⏱ 12 minutes

## Breaking It Down

### Step 1: The Basic if/elif/else

```python
# basic_if.py

def classify_sentiment(score):
    """Classify a sentiment score into a label."""
    if score is None:
        return "unknown"

    if score > 0.6:
        return "positive"
    elif score > 0.3:
        return "neutral"
    else:
        return "negative"

print(classify_sentiment(0.85))  # positive
print(classify_sentiment(0.45))  # neutral
print(classify_sentiment(0.15))  # negative
print(classify_sentiment(None))  # unknown
```

The order of conditions matters. If you put `score > 0.3` before `score > 0.6`, the `score > 0.3` branch catches everything above 0.3 and `score > 0.6` never executes.

### Step 2: Truthy and Falsy Values

```python
# truthy_demo.py
def is_valid_response(response):
    """Check if a response has meaningful content."""
    # None, empty string, empty list/dict all return False
    if not response:
        return False
    return True

print(is_valid_response(None))          # False
print(is_valid_response(""))            # False
print(is_valid_response([]))            # False
print(is_valid_response("Hello"))       # True
print(is_valid_response(["result"]))    # True
print(is_valid_response(0))             # False
print(is_valid_response(1))             # True
```

Falsy values in Python:

| Value | Falsy because |
|-------|--------------|
| `False` | The boolean false |
| `None` | No value |
| `0`, `0.0`, `0j` | Zero of any numeric type |
| `""` | Empty string |
| `[]`, `()`, `{}`, `set()` | Empty collections |

### Step 3: Combining Conditions with `and`, `or`, `not`

```python
# combining.py

def should_retry(status_code, retries_left, is_idempotent):
    """Determine if a failed API call should be retried."""
    if (status_code >= 500 or status_code == 429) and retries_left > 0:
        return True

    if status_code == 408 and retries_left > 0 and is_idempotent:
        return True

    return False

print(should_retry(503, 3, True))   # True — server error, retries available
print(should_retry(429, 0, True))   # False — no retries left
print(should_retry(408, 2, False))  # False — not idempotent
print(should_retry(400, 3, True))   # False — client error (4xx), don't retry

# Use parentheses for clarity — even when Python's precedence would work
# Without parens: `status_code >= 500 or status_code == 429 and retries_left > 0`
# `and` binds tighter than `or`, so the above means status_code >= 500 OR (status_code == 429 AND retries_left > 0)
# The parentheses here change the meaning: (status_code >= 500 OR status_code == 429) AND retries_left > 0
```

### Step 4: Ternary Expressions

For simple if/else that assigns a value, use the ternary expression:

```python
# ternary_demo.py

# Instead of:
if confidence > 0.5:
    label = "positive"
else:
    label = "negative"

# Write:
label = "positive" if confidence > 0.5 else "negative"

# Nesting ternaries — avoid for complex logic
# "high" if > 0.8, "medium" if > 0.4, "low" otherwise
level = "high" if confidence > 0.8 else ("medium" if confidence > 0.4 else "low")
```

Keep ternaries simple — one condition, two outcomes. If you need more, use a full if/elif/else.

### Step 5: match/case (Python 3.10+)

The `match`/`case` statement provides pattern matching, similar to switch in other languages:

```python
# match_demo.py

def route_model(model_name):
    """Route a model name to its provider API."""
    match model_name:
        case "gpt-4" | "gpt-3.5-turbo":
            return "https://api.openai.com/v1/chat/completions"
        case "claude-3-opus" | "claude-3-sonnet":
            return "https://api.anthropic.com/v1/messages"
        case "llama-3":
            return "https://api.groq.com/openai/v1/chat/completions"
        case _:
            return "unknown_provider"

print(route_model("gpt-4"))          # https://api.openai.com/v1/chat/completions
print(route_model("claude-3-opus"))   # https://api.anthropic.com/v1/messages
print(route_model("fictional-model")) # unknown_provider
```

`case _` is the default/wildcard branch, matching anything.

### Step 6: Build Something Real — API Response Classifier

```python
# response_classifier.py — Classify API responses by status and content

def classify_api_response(response):
    """
    Classify an API response dict into a summary status.

    Expected response format:
    {
        "status_code": int,
        "body": dict or None,
        "error": str or None,
    }
    """
    status = response.get("status_code", 0)
    body = response.get("body")
    error = response.get("error")

    # Determine the outcome
    if status >= 200 and status < 300:
        # Success — check body
        if body is None:
            return "success_empty_body"
        return "success"
    elif status == 429:
        return "rate_limited"
    elif status >= 500:
        if error and "timeout" in error.lower():
            return "timeout"
        return "server_error"
    elif status == 401 or status == 403:
        return "auth_error"
    elif error:
        return f"client_error: {error}"
    else:
        return "unknown"

responses = [
    {"status_code": 200, "body": {"result": "ok"}, "error": None},
    {"status_code": 429, "body": None, "error": "rate limit exceeded"},
    {"status_code": 503, "body": None, "error": "timeout upstream"},
    {"status_code": 401, "body": None, "error": "invalid API key"},
]

for resp in responses:
    status = classify_api_response(resp)
    print(f"Status {resp['status_code']} → {status}")
```
