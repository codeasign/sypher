# Functions — Hands On

## Breaking It Down

### Step 1: Defining and Calling

```python
# basic_function.py

def greet(name):
    """Return a personalised greeting string."""
    return f"Hello, {name}! Welcome to the AI course."

# Call the function
message = greet("Alice")
print(message)  # Hello, Alice! Welcome to the AI course.

# A function without return returns None
def log_progress(step):
    print(f"Completed step {step}")

result = log_progress(3)
print(result)  # None
```

### Step 2: Parameters and Arguments

```python
# parameters.py

# Positional arguments — order matters
def build_prompt(system_msg, user_msg, example=None):
    prompt = f"System: {system_msg}\n\n"
    if example:
        prompt += f"Example: {example}\n\n"
    prompt += f"User: {user_msg}"
    return prompt

print(build_prompt("You are helpful.", "Tell me a joke."))

# Keyword arguments — specify by name, order does not matter
print(build_prompt(
    user_msg="Tell me a joke.",
    system_msg="You are helpful.",
    example="Q: Why did the chicken cross the road? A: To get to the other side.",
))

# Default parameter values — example defaults to None
print(build_prompt("You are helpful.", "Tell me a joke."))
```

### Step 3: Variable-Length Arguments

```python
# variable_args.py

# *args — any number of positional arguments (collected as a tuple)
def calculate_total(*args):
    """Sum any number of values."""
    total = 0
    for value in args:
        total += value
    return total

print(calculate_total(1, 2, 3))           # 6
print(calculate_total(10, 20, 30, 40))    # 100

# **kwargs — any number of keyword arguments (collected as a dict)
def create_config(**kwargs):
    """Build a configuration dict from keyword arguments."""
    config = {
        "temperature": 0.7,
        "max_tokens": 1024,
        "top_p": 1.0,
    }
    config.update(kwargs)
    return config

config = create_config(model="gpt-4", temperature=0.3, max_tokens=4096)
print(config)
# {'temperature': 0.3, 'max_tokens': 4096, 'top_p': 1.0, 'model': 'gpt-4'}

# Combined
def wrapper_function(arg1, arg2, *args, **kwargs):
    """Demonstrate mixing positional, *args, and **kwargs."""
    print(f"arg1={arg1}, arg2={arg2}")
    print(f"extra positional: {args}")
    print(f"extra keyword: {kwargs}")

wrapper_function(1, 2, 3, 4, 5, key="value", debug=True)
# arg1=1, arg2=2
# extra positional: (3, 4, 5)
# extra keyword: {'key': 'value', 'debug': True}
```

### Step 4: Type Hints

Type hints document what types a function expects and returns. They are not enforced at runtime but are used by IDEs and linters:

```python
# type_hints.py

def classify_text(
    text: str,
    threshold: float = 0.5,
    labels: list[str] | None = None,
) -> dict[str, float]:
    """
    Classify text and return a dict mapping labels to scores.

    Args:
        text: Input text to classify.
        threshold: Minimum confidence to include a label.
        labels: Specific labels to consider. If None, use all.

    Returns:
        Dict of label -> confidence score.
    """
    # Simulated classification
    scores = {
        "positive": 0.92,
        "negative": 0.03,
        "neutral": 0.05,
    }

    if labels:
        scores = {k: v for k, v in scores.items() if k in labels}

    return {k: v for k, v in scores.items() if v >= threshold}

result = classify_text("Great product!", threshold=0.5, labels=["positive"])
print(result)  # {'positive': 0.92}

# This shows the type annotation in IDEs and when you hover over the function
# It does NOT raise an error if you pass wrong types — that is what mypy is for
```

### Step 5: Lambda Functions

Lambdas are small, anonymous functions defined in one expression:

```python
# lambda_demo.py

# Lambda: lambda arguments: expression
square = lambda x: x ** 2
print(square(5))  # 25

# Using lambda with sorted — sort models by length
models = ["gpt-4", "claude-3-opus", "llama-3", "gemini-pro"]
sorted_models = sorted(models, key=lambda m: len(m))
print(sorted_models)  # ['gpt-4', 'llama-3', 'gemini-pro', 'claude-3-opus']

# Using lambda with filter — keep only short prompts
prompts = ["Tell me a joke", "Hi", "Explain quantum computing in detail"]
short_prompts = list(filter(lambda p: len(p) < 20, prompts))
print(short_prompts)  # ['Tell me a joke', 'Hi']

# Using lambda with map — add prefix
with_prefix = list(map(lambda p: "Q: " + p, prompts))
print(with_prefix)
# ['Q: Tell me a joke', 'Q: Hi', 'Q: Explain quantum computing in detail']
```

### Step 6: Scope — How Python Finds Variables

```python
# scope_demo.py

model_name = "gpt-4"  # Global scope

def run_inference():
    model_name = "claude-3"  # Local scope — shadows the global
    print(f"Inside function: {model_name}")

run_inference()  # Inside function: claude-3
print(f"Outside function: {model_name}")  # Outside function: gpt-4

# To modify a global variable inside a function, use 'global' (rarely needed)
counter = 0

def increment():
    global counter
    counter += 1

increment()
print(counter)  # 1

# Better approach: pass and return
counter = 0

def increment(c):
    return c + 1

counter = increment(counter)
print(counter)  # 1
```

### Step 7: Build Something Real — Data Processing Pipeline

```python
# pipeline.py — A reusable data processing pipeline using functions

def load_texts(file_paths: list[str]) -> list[str]:
    """Load text content from multiple files."""
    texts = []
    for path in file_paths:
        try:
            with open(path, "r", encoding="utf-8") as f:
                texts.append(f.read())
        except FileNotFoundError:
            print(f"Warning: {path} not found, skipping.")
    return texts

def clean_text(text: str) -> str:
    """Basic text cleaning for AI input."""
    text = text.strip()
    text = text.replace("\n", " ")
    # Remove extra whitespace
    import re
    text = re.sub(r"\s+", " ", text)
    return text

def chunk_text(text: str, max_chars: int = 1000) -> list[str]:
    """Split text into overlapping chunks for model input."""
    chunks = []
    for i in range(0, len(text), max_chars):
        chunk = text[i:i + max_chars]
        chunks.append(chunk)
    return chunks

def run_pipeline(file_paths: list[str]) -> list[str]:
    """Run the full pipeline: load -> clean -> chunk."""
    texts = load_texts(file_paths)
    all_chunks = []

    for text in texts:
        text = clean_text(text)
        chunks = chunk_text(text, max_chars=1000)
        all_chunks.extend(chunks)

    return all_chunks

```

## Performance Considerations

- **Function call overhead** is small but non-zero — ~50 nanoseconds per call. For hot loops, consider inlining simple logic or using a comprehension with a lambda.
- **Default arguments are evaluated once** at definition time, not at call time. Use `None` for mutable defaults (see Avoid Mistakes).
- **Type hints have zero runtime cost** — they are ignored by the interpreter and used only by external tools.
