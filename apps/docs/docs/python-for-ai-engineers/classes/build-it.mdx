# Classes — Hands On

## Breaking It Down

### Step 1: Defining a Simple Class

```python
# simple_class.py

class Model:
    """A simple model class."""

    def __init__(self, name: str, version: int):
        """Constructor — called when you create a new Model instance."""
        self.name = name
        self.version = version
        self.is_loaded = False

    def load(self) -> None:
        """Simulate loading model weights."""
        print(f"Loading {self.name} v{self.version}...")
        self.is_loaded = True

    def predict(self, input_text: str) -> str:
        """Make a prediction."""
        if not self.is_loaded:
            raise RuntimeError("Model not loaded. Call .load() first.")
        return f"{self.name} prediction: '{input_text}' → positive"

# Create instances
gpt4 = Model("gpt-4", 4)
claude = Model("claude-3", 3)

gpt4.load()
result = gpt4.predict("Great product!")
print(result)
# Loading gpt-4 v4...
# gpt-4 prediction: 'Great product!' → positive
```

### Step 2: Instance, Class, and Static Methods

```python
# method_types.py

class Dataset:
    """A dataset with different method types."""

    # Class attribute — shared by all instances
    supported_formats = ["csv", "json", "parquet"]

    def __init__(self, path: str, format: str = "csv"):
        """Instance method — operates on one instance."""
        if format not in Dataset.supported_formats:
            raise ValueError(f"Unsupported format: {format}")
        self.path = path
        self.format = format
        self.size = 0

    def load(self) -> None:
        """Instance method — requires self."""
        print(f"Loading {self.format} data from {self.path}")

    @classmethod
    def from_json(cls, path: str) -> "Dataset":
        """Class method — alternative constructor.
        Receives the class (cls), not an instance (self)."""
        return cls(path, format="json")

    @staticmethod
    def validate_format(format: str) -> bool:
        """Static method — no self or cls.
        Like a regular function, but namespaced in the class."""
        return format in Dataset.supported_formats

# Instance method — works on an instance
ds = Dataset("data/train.csv")
ds.load()

# Class method — creates an instance via alternative constructor
ds2 = Dataset.from_json("data/test.json")
print(ds2.format)  # json

# Static method — no instance needed
print(Dataset.validate_format("csv"))   # True
print(Dataset.validate_format("xlsx"))  # False
```

### Step 3: Properties — Controlled Attribute Access

Properties let you add getter/setter logic without changing the external interface:

```python
# properties.py

class Config:
    """Configuration with validation via properties."""

    def __init__(self, learning_rate: float = 0.001):
        self._learning_rate = learning_rate

    @property
    def learning_rate(self) -> float:
        """Getter for learning_rate."""
        return self._learning_rate

    @learning_rate.setter
    def learning_rate(self, value: float) -> None:
        """Setter with validation."""
        if not 0.0 < value <= 1.0:
            raise ValueError(f"Learning rate must be in (0, 1], got {value}")
        self._learning_rate = value

    @property
    def summary(self) -> str:
        """Read-only property — no setter."""
        return f"Config(lr={self._learning_rate})"

cfg = Config()
print(cfg.learning_rate)  # 0.001 — calls getter
cfg.learning_rate = 0.01  # Calls setter
# cfg.learning_rate = 2.0  # ValueError!
print(cfg.summary)         # Config(lr=0.01)
```

### Step 4: Special Methods — `__str__` and `__repr__`

```python
# str_repr.py

class Prediction:
    """A prediction result with readable string representations."""

    def __init__(self, text: str, label: str, confidence: float):
        self.text = text
        self.label = label
        self.confidence = confidence

    def __repr__(self) -> str:
        """For developers — should be unambiguous. Used in logs/debugging."""
        return f"Prediction({self.text!r}, {self.label!r}, {self.confidence})"

    def __str__(self) -> str:
        """For users — should be readable. Used by print()."""
        return f"'{self.text}' → {self.label} ({self.confidence:.1%})"

pred = Prediction("Great product!", "positive", 0.95)
print(repr(pred))  # Prediction('Great product!', 'positive', 0.95)
print(str(pred))   # 'Great product!' → positive (95.0%)
print(pred)        # Same as str(pred) — 'Great product!' → positive (95.0%)
```

### Step 5: Build Something Real — Text Classifier Class

```python
# text_classifier.py

class TextClassifier:
    """A configurable text classifier that wraps an underlying model."""

    DEFAULT_CONFIG = {
        "temperature": 0.7,
        "max_tokens": 256,
        "top_p": 1.0,
    }

    def __init__(self, model_name: str = "gpt-4", config: dict | None = None):
        self.model_name = model_name
        self.config = {**self.DEFAULT_CONFIG, **(config or {})}
        self._call_count = 0

    @property
    def call_count(self) -> int:
        """Read-only: number of predictions made."""
        return self._call_count

    def classify(self, text: str) -> dict:
        """Classify text and return a result dict."""
        self._call_count += 1
        # Simulated classification
        score = min(max(len(text) % 100 / 100.0, 0.0), 1.0)
        return {
            "text": text,
            "label": "positive" if score > 0.5 else "negative",
            "confidence": score,
        }

    def classify_batch(self, texts: list[str]) -> list[dict]:
        """Classify multiple texts."""
        return [self.classify(t) for t in texts]

    @classmethod
    def from_config_file(cls, path: str) -> "TextClassifier":
        """Alternative constructor from a JSON config file."""
        import json
        with open(path, "r", encoding="utf-8") as f:
            config = json.load(f)
        return cls(
            model_name=config.get("model", "gpt-4"),
            config=config,
        )

    @staticmethod
    def validate_text(text: str) -> bool:
        """Check that text is valid for classification."""
        return bool(text) and len(text.strip()) > 0

    def __str__(self) -> str:
        return f"TextClassifier(model={self.model_name}, calls={self._call_count})"

# Usage
clf = TextClassifier("gpt-4", {"temperature": 0.3})
texts = ["Great product!", "Not good", "Okay", ""]
valid_texts = [t for t in texts if TextClassifier.validate_text(t)]

results = clf.classify_batch(valid_texts)
for r in results:
    print(f"{r['text']}: {r['label']} ({r['confidence']:.2f})")

print(clf)                     # TextClassifier(model=gpt-4, calls=3)
print(f"Calls: {clf.call_count}")  # 3
```
