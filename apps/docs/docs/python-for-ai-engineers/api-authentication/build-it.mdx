# API Authentication — Hands On

⏱ 18 minutes

## Breaking It Down

### Step 1: API Key Authentication

The simplest pattern — send a key in the header:

```python
# api_key_auth.py
import os
import requests

# Load from environment — never hardcode
API_KEY = os.environ.get("OPENAI_API_KEY", "sk-test-placeholder")

# API key in header (most common)
headers = {
    "Authorization": f"Bearer {API_KEY}",
}
response = requests.get(
    "https://httpbin.org/headers",
    headers=headers,
    timeout=10,
)
print(response.json()["headers"]["Authorization"])  # Bearer sk-test-...

# Or API key in a custom header
headers = {"X-API-Key": API_KEY}
response = requests.get(
    "https://httpbin.org/headers",
    headers=headers,
    timeout=10,
)
```

Always load API keys from environment variables, not from your code. Set them before running:

```bash
export OPENAI_API_KEY="sk-your-key-here"
python api_key_auth.py
```

### Step 2: Bearer Token with Session

Most AI providers use bearer token authentication. Combine it with a session for connection reuse:

```python
# bearer_auth.py
import os
import requests

class BearerAuthClient:
    """Simple bearer token client."""

    def __init__(self, api_key: str, base_url: str = "https://api.openai.com/v1"):
        self.session = requests.Session()
        self.session.headers.update({
            "Authorization": f"Bearer {api_key}",
            "Content-Type": "application/json",
        })
        self.base_url = base_url.rstrip("/")

    def complete(self, prompt: str, model: str = "gpt-4") -> dict:
        payload = {
            "model": model,
            "messages": [{"role": "user", "content": prompt}],
        }
        response = self.session.post(
            f"{self.base_url}/chat/completions",
            json=payload,
            timeout=30,
        )
        response.raise_for_status()
        return response.json()

# Usage
client = BearerAuthClient(api_key=os.environ.get("OPENAI_API_KEY", "test"))
# In a real scenario this would call the actual API.
# Here we use httpbin to echo the request headers.
echo = requests.post(
    "https://httpbin.org/post",
    headers={"Authorization": f"Bearer {os.environ.get('OPENAI_API_KEY', 'test')}"},
    json={"test": True},
)
print(echo.status_code)
```

### Step 3: Auth Header Injection with httpx Auth Transports

httpx supports custom authentication transports that inject headers automatically:

```python
# httpx_auth.py
import httpx

class BearerAuth(httpx.Auth):
    """Custom auth class that injects a bearer token."""

    def __init__(self, token: str):
        self.token = token

    def auth_flow(self, request: httpx.Request):
        """Called for each request — inject the auth header."""
        request.headers["Authorization"] = f"Bearer {self.token}"
        yield request

# Usage
auth = BearerAuth(token="sk-test-token")
with httpx.Client(auth=auth) as client:
    response = client.get("https://httpbin.org/headers")
    print(response.json()["headers"]["Authorization"])  # Bearer sk-test-...

# Async version
import asyncio

async def async_auth():
    auth = BearerAuth(token="sk-test-token")
    async with httpx.AsyncClient(auth=auth) as client:
        response = await client.get("https://httpbin.org/headers")
        print(response.json()["headers"]["Authorization"])

asyncio.run(async_auth())
```

The `auth_flow` method is a generator that yields the request. It can yield multiple requests (e.g., first a token refresh, then retry the original request) — this is how httpx supports automatic retry with refreshed tokens.

### Step 4: Automatic Token Refresh

Many APIs issue short-lived tokens (15–60 minutes) that must be refreshed periodically:

```python
# token_refresh.py
import httpx
import time

class RefreshingAuth(httpx.Auth):
    """Auth that automatically refreshes expired tokens."""

    def __init__(self, client_id: str, client_secret: str, auth_url: str):
        self.client_id = client_id
        self.client_secret = client_secret
        self.auth_url = auth_url
        self._token = None
        self._expires_at = 0

    def _fetch_token(self) -> tuple[str, int]:
        """Get a new token from the auth server."""
        response = httpx.post(
            self.auth_url,
            data={
                "client_id": self.client_id,
                "client_secret": self.client_secret,
                "grant_type": "client_credentials",
            },
            timeout=10,
        )
        response.raise_for_status()
        data = response.json()
        expires_in = data.get("expires_in", 3600)
        return data["access_token"], time.time() + expires_in - 60  # 60s buffer

    def _ensure_token(self):
        """Get a valid token, refreshing if needed."""
        if time.time() >= self._expires_at:
            self._token, self._expires_at = self._fetch_token()

    def auth_flow(self, request: httpx.Request):
        self._ensure_token()
        request.headers["Authorization"] = f"Bearer {self._token}"
        yield request

# Usage
auth = RefreshingAuth(
    client_id="your-client-id",
    client_secret="your-client-secret",
    auth_url="https://httpbin.org/post",
)

with httpx.Client(auth=auth) as client:
    response = client.get("https://httpbin.org/headers")
    print(f"Auth header present: {'Authorization' in response.request.headers}")
```

The token is fetched once, cached, and automatically refreshed before expiry. The 60-second buffer prevents edge cases where the token expires between the check and the API call.

### Step 5: Secure Credential Storage

Never store credentials in code. Use a dedicated credential loader:

```python
# credentials.py
import os
import json
from pathlib import Path

class CredentialManager:
    """Load credentials from environment variables or a secure file."""

    REQUIRED_KEYS = {
        "openai": ["api_key"],
        "anthropic": ["api_key"],
        "huggingface": ["api_token"],
    }

    def __init__(self, env_prefix: str = "AI_CREDS"):
        self.env_prefix = env_prefix

    def get(self, provider: str, key: str) -> str:
        """Get a credential from environment variables."""
        env_key = f"{self.env_prefix}_{provider.upper()}_{key.upper()}"
        value = os.environ.get(env_key)
        if not value:
            raise ValueError(
                f"Missing credential: set {env_key} environment variable"
            )
        return value

    def get_all(self, provider: str) -> dict:
        """Get all credentials for a provider."""
        creds = {}
        for key in self.REQUIRED_KEYS.get(provider, []):
            creds[key] = self.get(provider, key)
        return creds

# Usage
mgr = CredentialManager(env_prefix="AI_CREDS")

# Set before running:
# export AI_CREDS_OPENAI_API_KEY="sk-..."
# export AI_CREDS_ANTHROPIC_API_KEY="sk-ant-..."

try:
    openai_key = mgr.get("openai", "api_key")
    print(f"OpenAI key loaded: {openai_key[:8]}...")
except ValueError as e:
    print(f"Note: {e}")
```

Environment variables are the standard for production. For development, a `.env` file (loaded with `python-dotenv`) is acceptable — but never commit it to version control.

### Step 6: Handling 401 Errors with Retry

Sometimes tokens expire between the check and the request. Handle this with a retry that refreshes the token:

```python
# retry_401.py
import time
import requests

class ResilientClient:
    """API client that retries on 401 with a fresh token."""

    def __init__(self, api_key: str, base_url: str):
        self.api_key = api_key
        self.base_url = base_url.rstrip("/")
        self.session = requests.Session()

    def _make_request(self, payload: dict) -> requests.Response:
        """Send a request with the current token."""
        self.session.headers.update({
            "Authorization": f"Bearer {self.api_key}",
        })
        response = self.session.post(
            f"{self.base_url}/chat/completions",
            json=payload,
            timeout=30,
        )
        return response

    def complete(self, prompt: str, model: str = "gpt-4") -> dict:
        """Send a completion request with 401 retry."""
        payload = {
            "model": model,
            "messages": [{"role": "user", "content": prompt}],
        }

        # First attempt
        response = self._make_request(payload)

        if response.status_code == 401:
            # Token may be expired — in production, refresh it here
            # self.api_key = refresh_token()
            print("Got 401 — would refresh token and retry")
            response = self._make_request(payload)

        response.raise_for_status()
        return response.json()

# Test with httpbin — 401 from /status/401
client = ResilientClient(api_key="test", base_url="https://httpbin.org")
# This would normally call the real API
test_response = requests.get("https://httpbin.org/status/401")
print(f"Test 401 response: {test_response.status_code}")
```

## Build Something Real — Multi-Provider Authenticated Client

```python
# multi_provider.py
import os
import requests
from pathlib import Path

class MultiProviderClient:
    """Authenticated client for multiple AI providers."""

    PROVIDERS = {
        "openai": {
            "base_url": "https://api.openai.com/v1",
            "header_format": "Bearer {key}",
        },
        "anthropic": {
            "base_url": "https://api.anthropic.com/v1",
            "header_format": "Bearer {key}",
            "extra_headers": {"anthropic-version": "2023-06-01"},
        },
        "groq": {
            "base_url": "https://api.groq.com/openai/v1",
            "header_format": "Bearer {key}",
        },
    }

    def __init__(self):
        self.sessions = {}

    def _get_session(self, provider: str) -> requests.Session:
        """Get or create a session for a provider."""
        if provider not in self.sessions:
            config = self.PROVIDERS.get(provider)
            if not config:
                raise ValueError(f"Unknown provider: {provider}")

            api_key = os.environ.get(f"{provider.upper()}_API_KEY")
            if not api_key:
                raise ValueError(
                    f"Set {provider.upper()}_API_KEY environment variable"
                )

            session = requests.Session()
            auth_header = config["header_format"].format(key=api_key)
            session.headers.update({
                "Authorization": auth_header,
                "Content-Type": "application/json",
            })
            session.headers.update(config.get("extra_headers", {}))
            self.sessions[provider] = session

        return self.sessions[provider]

    def complete(self, provider: str, prompt: str, model: str = None) -> dict:
        """Send a completion request to the specified provider."""
        session = self._get_session(provider)
        config = self.PROVIDERS[provider]
        payload = {
            "model": model or f"{provider}/default",
            "messages": [{"role": "user", "content": prompt}],
        }

        response = session.post(
            f"{config['base_url']}/chat/completions",
            json=payload,
            timeout=30,
        )
        response.raise_for_status()
        return response.json()

    def close(self):
        for session in self.sessions.values():
            session.close()

# Usage
client = MultiProviderClient()

try:
    # Check which providers are configured
    configured = []
    for provider in ["openai", "anthropic", "groq"]:
        if os.environ.get(f"{provider.upper()}_API_KEY"):
            configured.append(provider)

    if configured:
        print(f"Configured providers: {', '.join(configured)}")
    else:
        print("No provider API keys found. Set OPENAI_API_KEY, ANTHROPIC_API_KEY, etc.")
except Exception as e:
    print(f"Error: {e}")
finally:
    client.close()
```

This client handles multiple providers with different auth patterns, header formats, and base URLs — all through a unified interface.

## Security Considerations

- **Never commit API keys** to version control. Add `.env` to `.gitignore`. Use pre-commit hooks to scan for credentials.
- **Rotate keys regularly.** If a key is compromised, revoke it immediately and issue a new one.
- **Use separate keys** for development, staging, and production.
- **Scope keys to minimum permissions.** Most providers let you create keys with restricted access.
- **Monitor usage.** Unexpected spikes may indicate a leaked key being used by an unauthorized party.
- **Use environment variables** for credentials. For teams, use a secrets manager like Vault, AWS Secrets Manager, or Doppler.
