## Breaking It Down

### Step 1: Creating Lists

```python
# list_creation.py

# Empty list
empty = []

# List with values
fruits = ["apple", "banana", "cherry"]

# Mixed types — lists can hold anything
mixed = [42, "hello", 3.14, True, None]

# Nested lists (lists within lists)
matrix = [
    [1, 2, 3],
    [4, 5, 6],
    [7, 8, 9],
]

# List from another iterable (e.g. a string)
chars = list("hello")
print(chars)  # ['h', 'e', 'l', 'l', 'o']
```

### Step 2: Accessing and Modifying Elements

```python
# access_modify.py
items = ["a", "b", "c", "d", "e"]

# Access by index
print(items[0])    # a
print(items[-1])   # e — last element
print(items[1:3])  # ['b', 'c'] — slice (stop is exclusive)

# Modify by index — lists are mutable
items[0] = "z"
print(items)       # ['z', 'b', 'c', 'd', 'e']

# Slice assignment — replace a range
items[1:3] = ["x", "y"]
print(items)       # ['z', 'x', 'y', 'd', 'e']

# Delete by index
del items[0]
print(items)       # ['x', 'y', 'd', 'e']
```

### Step 3: List Methods

```python
# list_methods.py
numbers = []

# Add to the end
numbers.append(10)
numbers.append(20)
numbers.append(30)
print(numbers)  # [10, 20, 30]

# Insert at a position
numbers.insert(1, 15)
print(numbers)  # [10, 15, 20, 30]

# Extend with another list
numbers.extend([40, 50])
print(numbers)  # [10, 15, 20, 30, 40, 50]

# Remove by value (first occurrence)
numbers.remove(20)
print(numbers)  # [10, 15, 30, 40, 50]

# Pop by index (default last)
last = numbers.pop()
print(last)      # 50
print(numbers)   # [10, 15, 30, 40]

first = numbers.pop(0)
print(first)     # 10
print(numbers)   # [15, 30, 40]

# Find index of a value
print(numbers.index(30))  # 1

# Check membership
print(30 in numbers)   # True
print(99 in numbers)   # False

# Count occurrences
nums = [1, 2, 2, 3, 2]
print(nums.count(2))   # 3

# Sort in-place
numbers.sort()
print(numbers)  # [15, 30, 40]

# Reverse in-place
numbers.reverse()
print(numbers)  # [40, 30, 15]
```

### Step 4: List Length and Iteration

```python
# iteration.py
models = ["gpt-4", "claude-3", "llama-3", "gemini-pro"]

# Length
print(len(models))  # 4

# Basic for loop
for model in models:
    print(model)

# Loop with index using enumerate
for index, model in enumerate(models):
    print(f"{index}: {model}")
# 0: gpt-4
# 1: claude-3
# 2: llama-3
# 3: gemini-pro
```

### Step 5: Copying Lists — Reference vs. Shallow Copy

This is one of the most important concepts in Python. Read carefully:

```python
# copy_demo.py

# Assignment creates a reference, NOT a copy
original = [1, 2, 3]
alias = original
alias.append(4)
print(original)  # [1, 2, 3, 4] — original changed too!

# Shallow copy with .copy() or list() or [:]
copy1 = original.copy()
copy2 = list(original)
copy3 = original[:]

copy1.append(5)
print(original)  # [1, 2, 3, 4] — unchanged
print(copy1)     # [1, 2, 3, 4, 5]

# Deep copy needed for nested lists
nested = [[1, 2], [3, 4]]
shallow = nested.copy()
shallow[0].append(99)
print(nested)    # [[1, 2, 99], [3, 4]] — inner list shared!

import copy
deep = copy.deepcopy(nested)
deep[0].append(100)
print(nested)    # [[1, 2, 99], [3, 4]] — unchanged
```

### Step 6: Build Something Real — Batch Prediction Collector

```python
# batch_predictor.py — Simulates collecting model predictions

def mock_predict(text):
    """Simulate a model prediction with a score."""
    # In reality, this would call an API
    score = hash(text) % 100 / 100.0  # deterministic "score"
    return {"text": text, "score": score, "label": "positive" if score > 0.5 else "negative"}

inputs = [
    "Great product, highly recommend",
    "Not what I expected",
    "Works well for the price",
    "Terrible customer support",
    "Would buy again",
]

# Collect predictions
predictions = []
for text in inputs:
    predictions.append(mock_predict(text))

# Filter high-confidence predictions
high_conf = [p for p in predictions if p["score"] > 0.6]
print(f"Total predictions: {len(predictions)}")
print(f"High confidence: {len(high_conf)}")

# Extract just the labels
labels = [p["label"] for p in predictions]
print(f"Labels: {labels}")

# Count positives
positive_count = labels.count("positive")
print(f"Positive: {positive_count} / {len(labels)}")
```

## Performance Considerations

- **Access by index** (`items[i]`) is O(1) — instant regardless of list size.
- **Append** at the end is amortized O(1) — usually instant.
- **Insert or pop at the beginning** is O(n) — slow for large lists because all other elements shift. Use `collections.deque` if you need fast front insertions.
- **Membership check** (`x in list`) is O(n) — Python scans from start to end. For large lists with frequent membership checks, use a set instead.
