# Sets — Hands On

⏱ 10 minutes

## Breaking It Down

### Step 1: Creating Sets

```python
# set_creation.py

# Empty set — must use set(), not {}
empty_set = set()
not_a_set = {}  # This is an empty dict, not a set!

# Set with values
animals = {"cat", "dog", "emu", "fox"}

# From a list (deduplicates automatically)
numbers = set([1, 2, 2, 3, 3, 3])
print(numbers)  # {1, 2, 3}

# From a string (gives unique characters)
chars = set("hello")
print(chars)    # {'h', 'e', 'l', 'o'} — order may vary
```

### Step 2: Adding and Removing Elements

```python
# set_mutate.py
items = {"a", "b", "c"}

# Add
items.add("d")
print(items)  # {'a', 'b', 'c', 'd'}

# Adding an existing element has no effect
items.add("a")
print(items)  # {'a', 'b', 'c', 'd'} — unchanged

# Remove — raises KeyError if not found
items.remove("b")
# items.remove("z")  # KeyError: 'z'

# Discard — does NOT raise an error if missing
items.discard("z")  # No error — silently does nothing

# Pop — remove and return an arbitrary element
arbitrary = items.pop()
print(arbitrary)  # Some element, e.g. 'a'

# Clear all
items.clear()
print(items)  # set()
```

### Step 3: Set Operations

```python
# set_operations.py
all_models = {"gpt-4", "claude-3", "llama-3", "gemini-pro"}
production_models = {"gpt-4", "claude-3"}

# Union — all elements from both sets
print(all_models | production_models)
# {'gpt-4', 'claude-3', 'llama-3', 'gemini-pro'}

# Intersection — elements in both sets
print(all_models & production_models)
# {'gpt-4', 'claude-3'}

# Difference — elements in first but not second
print(all_models - production_models)
# {'llama-3', 'gemini-pro'}

# Symmetric difference — elements in either but not both
a = {"cat", "dog"}
b = {"dog", "fox"}
print(a ^ b)  # {'cat', 'fox'}

# Subset / superset checks
small = {"gpt-4"}
print(small.issubset(all_models))        # True
print(all_models.issuperset(small))      # True
print(small.isdisjoint({"llama-3"}))     # True — no overlap
```

### Step 4: Frozen Sets (Immutable Sets)

```python
# frozenset_demo.py

# frozenset is an immutable set — hashable, can be a dict key
frozen = frozenset(["a", "b", "c"])
# frozen.add("d")  # AttributeError: 'frozenset' object has no attribute 'add'

# Useful when you need a set in a set
set_of_sets = {frozenset([1, 2]), frozenset([3, 4])}
print(set_of_sets)  # {frozenset({1, 2}), frozenset({3, 4})}

# Regular set cannot contain a set
# set_of_sets = { {1, 2} }  # TypeError: unhashable type: 'set'
```

### Step 5: Build Something Real — Duplicate Detection and Filtering

```python
# data_dedup.py — Clean and filter dataset entries

def deduplicate_predictions(predictions, key_func=None):
    """Remove duplicate predictions based on a key function."""
    seen = set()
    unique = []

    for pred in predictions:
        if key_func is None:
            key = pred
        else:
            key = key_func(pred)

        if key not in seen:
            seen.add(key)
            unique.append(pred)

    return unique

# Simulated duplicate predictions
raw_predictions = [
    {"id": "001", "label": "positive", "text": "Great product"},
    {"id": "002", "label": "negative", "text": "Not good"},
    {"id": "001", "label": "positive", "text": "Great product"},  # Duplicate
    {"id": "003", "label": "neutral", "text": "It is okay"},
    {"id": "002", "label": "negative", "text": "Not good"},        # Duplicate
]

# Deduplicate by "id"
clean = deduplicate_predictions(raw_predictions, key_func=lambda p: p["id"])
print(f"Before: {len(raw_predictions)}, After: {len(clean)}")  # Before: 5, After: 3

# ---

# Fast stopword filtering using set membership
STOPWORDS = {"the", "a", "an", "is", "was", "are", "were",
             "in", "on", "at", "to", "for", "of", "and", "or", "it"}

def filter_stopwords(tokens):
    """Remove stopwords from a token list — O(n) total."""
    return [t for t in tokens if t.lower() not in STOPWORDS]

tokens = ["The", "cat", "is", "on", "the", "mat"]
filtered = filter_stopwords(tokens)
print(filtered)  # ['cat', 'mat']

# Without sets, this would be O(n * m) where m is the number of stopwords
# With sets, it is O(n) — each check is O(1)
```

## Performance Considerations

- **Membership test** (`x in s`): O(1) — the main reason to use sets over lists for this purpose.
- **Insert / delete:** O(1) on average.
- **Union / intersection / difference:** O(n) where n is the size of the larger set — these must iterate over all elements.
- **Memory:** Sets use more memory than lists because each element stores a hash value and a pointer. For 1 million integers, a list uses ~8 MB, while a set uses ~45 MB.
