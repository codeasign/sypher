# Type Hints — Hands On

⏱ 12 minutes

## Breaking It Down

### Step 1: Basic Function Annotations

```python
# basic_hints.py

def greet(name: str) -> str:
    """Return a personalised greeting."""
    return f"Hello, {name}!"

def add(a: int, b: int) -> int:
    """Add two integers and return an integer."""
    return a + b

def process_data(data: list[str]) -> int:
    """Process a list of strings and return a count."""
    return len(data)

# Using the functions — no runtime enforcement
result: str = greet("Alice")
count: int = process_data(["a", "b", "c"])

# Wrong types pass at runtime but are flagged by mypy
# greet(42)           # mypy error: Argument 1 has incompatible type "int"; expected "str"
# add("a", "b")       # mypy error — but at runtime "a" + "b" = "ab" works!
```

### Step 2: Variable Annotations

```python
# variable_hints.py

# Simple variable annotation
name: str = "gpt-4"
version: int = 4
is_active: bool = True

# Complex types
tags: list[str] = ["production", "experimental"]
config: dict[str, float] = {"temperature": 0.7, "top_p": 1.0}

# Without annotation — mypy infers from assignment
inferred = "hello"  # mypy knows this is str
```

### Step 3: Optional, Union, and Any

```python
# typing_module.py
from typing import Optional, Union, Any

# Optional[X] means X or None
def find_model(name: str) -> Optional[str]:
    """Find a model by name. Returns None if not found."""
    models = {"gpt-4": "openai", "claude-3": "anthropic"}
    return models.get(name)  # May return None

# Union[X, Y] means X or Y
def parse_confidence(value: Union[str, float]) -> float:
    """Parse a confidence value from string or float."""
    if isinstance(value, str):
        return float(value)
    return value

# Any — opt out of type checking (use sparingly)
def debug_log(value: Any) -> None:
    """Log any value — type checking disabled."""
    print(value)

# Python 3.10+ shorthand: X | Y instead of Union[X, Y]
def parse(value: str | float) -> float:
    return float(value) if isinstance(value, str) else value
```

### Step 4: Collection Type Hints

```python
# collections.py
from typing import List, Dict, Tuple, Set

# List of strings
def get_tags() -> list[str]:
    return ["production", "stable"]

# Dictionary mapping string keys to float values
def get_metrics() -> dict[str, float]:
    return {"accuracy": 0.94, "f1": 0.92}

# Tuple with fixed types per position
def get_version() -> tuple[str, int, int]:
    """Return (name, major, minor)."""
    return ("gpt-4", 4, 1)

# Set of integers
def get_unique_ids() -> set[int]:
    return {1, 2, 3}
```

### Step 5: Type Aliases

For complex types, define an alias to avoid repetition:

```python
# aliases.py
from typing import TypeAlias  # Python 3.10+

# Alias for a prediction result
PredictionResult: TypeAlias = dict[str, str | float | list[str]]

# Or with older Python:
# PredictionResult = dict[str, Union[str, float, List[str]]]

def run_inference(text: str) -> PredictionResult:
    """Run inference. Returns a dict with text, label, and confidence."""
    return {
        "text": text,
        "label": "positive",
        "confidence": 0.95,
        "tags": ["important"],
    }

# Alias for a batch of results
BatchResult: TypeAlias = list[PredictionResult]

def batch_inference(texts: list[str]) -> BatchResult:
    return [run_inference(t) for t in texts]
```

### Step 6: Callable and TypeVar (Simple Generics)

```python
# generics.py
from typing import Callable, TypeVar

# Callable[[ArgType1, ArgType2], ReturnType]
def apply_transform(
    text: str,
    transform: Callable[[str], str],
) -> str:
    """Apply a transform function to text."""
    return transform(text)

result = apply_transform("hello", str.upper)
print(result)  # HELLO

# TypeVar for generic functions
T = TypeVar("T")

def first(items: list[T]) -> T | None:
    """Return the first item in a list, or None if empty."""
    return items[0] if items else None

print(first([1, 2, 3]))     # 1 — T inferred as int
print(first(["a", "b"]))    # a — T inferred as str
```

### Step 7: Running mypy

Install and run `mypy` to check your type hints:

```bash
pip install mypy
mypy your_script.py
```

Create `typed_code.py` and check it:

```python
# typed_code.py
def divide(a: float, b: float) -> float:
    """Divide a by b."""
    return a / b

result: str = divide(10.0, 2.0)  # Should fail: float cannot be assigned to str
```

Run:

```bash
mypy typed_code.py
# typed_code.py:6: error: Incompatible types in assignment (expression has type "float", variable has type "str")
```

### Step 8: Build Something Real — Type-Safe Data Pipeline

```python
# pipeline.py
from typing import Callable, TypeVar

T = TypeVar("T")
U = TypeVar("U")

def pipeline(
    data: list[T],
    *transforms: Callable[[T], U],
) -> list[U]:
    """Apply a chain of transforms to data."""
    for transform in transforms:
        data = [transform(item) for item in data]  # type: ignore
    return data

# Typed transform functions
def normalize(text: str) -> str:
    return text.strip().lower()

def tokenize(text: str) -> list[str]:
    return text.split()

def count_words(tokens: list[str]) -> int:
    return len(tokens)

# Pipeline is fully type-checked
result = pipeline(
    [" Hello ", "  World  "],
    normalize,
    tokenize,
    count_words,
)
print(result)  # [1, 1]
```
