# JSON — Hands On

⏱ 14 minutes

## Breaking It Down

### Step 1: Serializing and Deserializing

```python
# basic_json.py
import json

# Python dict → JSON string (serialize)
data = {
    "model": "gpt-4",
    "temperature": 0.7,
    "max_tokens": 2048,
    "stream": False,
    "stop": None,
}

json_str = json.dumps(data, indent=2)
print(json_str)
# {
#   "model": "gpt-4",
#   "temperature": 0.7,
#   "max_tokens": 2048,
#   "stream": false,
#   "stop": null
# }

# JSON string → Python dict (deserialize)
parsed = json.loads(json_str)
print(parsed["model"])        # gpt-4
print(type(parsed["stream"])) # <class 'bool'>
print(parsed["stop"])         # None
```

`indent=2` produces human-readable output. Without it, `dumps()` produces a compact single-line string — useful for saving space but hard to read.

### Step 2: Reading and Writing JSON Files

```python
# file_json.py
import json

config = {
    "model": "claude-3-opus",
    "api_key": "sk-...",
    "parameters": {
        "temperature": 0.5,
        "max_tokens": 4096,
        "top_p": 0.9,
    },
}

# Write to file
with open("config.json", "w", encoding="utf-8") as f:
    json.dump(config, f, indent=2)

# Read from file
with open("config.json", "r", encoding="utf-8") as f:
    loaded = json.load(f)

print(loaded["model"])                   # claude-3-opus
print(loaded["parameters"]["top_p"])     # 0.9

# Clean up
import os
os.remove("config.json")
```

Always specify `encoding="utf-8"` — JSON files should always be UTF-8, and explicit is safer than relying on the system default.

### Step 3: Nested JSON from API Responses

```python
# api_response.py
import json

# Simulated API response (what an LLM provider returns)
api_response = """
{
  "id": "chatcmpl-abc123",
  "object": "chat.completion",
  "created": 1718000000,
  "model": "gpt-4",
  "choices": [
    {
      "index": 0,
      "message": {
        "role": "assistant",
        "content": "The capital of France is Paris."
      },
      "finish_reason": "stop"
    }
  ],
  "usage": {
    "prompt_tokens": 14,
    "completion_tokens": 7,
    "total_tokens": 21
  }
}
"""

parsed = json.loads(api_response)

# Navigate nested structure
content = parsed["choices"][0]["message"]["content"]
print(content)  # The capital of France is Paris.

tokens = parsed["usage"]["total_tokens"]
print(f"Total tokens: {tokens}")  # Total tokens: 21

model = parsed["model"]
print(f"Model: {model}")  # Model: gpt-4
```

Navigate nested structures step by step: the response is a dict with a `"choices"` key whose value is a list, each element of which has a `"message"` dict with `"content"`.

### Step 4: Handling Missing Keys Safely

```python
# safe_access.py
import json

response = json.loads('{"id": "123", "choices": []}')

# Unsafe — KeyError if key is missing
# content = response["choices"][0]["message"]["content"]

# Safe — .get() with defaults
content = (
    response.get("choices", [{}])[0]
    .get("message", {})
    .get("content", "no content")
)
print(content)  # no content

# Or check the structure explicitly
if response.get("choices") and len(response["choices"]) > 0:
    first_choice = response["choices"][0]
    if "message" in first_choice and "content" in first_choice["message"]:
        content = first_choice["message"]["content"]
    else:
        content = "no content"
else:
    content = "no content"
```

API responses can vary — an error response might have an `"error"` key instead of `"choices"`. Always validate the structure you expect.

### Step 5: Custom Serialization

```python
# custom_json.py
import json
from datetime import datetime
import numpy as np

# Python types that json.dumps cannot handle by default
data = {
    "event": "training_start",
    "timestamp": datetime.now(),
    "learning_rate": 0.001,
    "scores": np.array([0.92, 0.88, 0.95]),
}

def custom_serializer(obj):
    """Convert non-serializable types to JSON-safe types."""
    if isinstance(obj, datetime):
        return obj.isoformat()
    if isinstance(obj, np.ndarray):
        return obj.tolist()
    raise TypeError(f"Object of type {type(obj)} is not JSON serializable")

json_str = json.dumps(data, default=custom_serializer, indent=2)
print(json_str)
# {
#   "event": "training_start",
#   "timestamp": "2026-06-30T12:00:00",
#   "learning_rate": 0.001,
#   "scores": [0.92, 0.88, 0.95]
# }
```

### Step 6: JSON Lines (NDJSON)

JSON Lines stores one valid JSON object per line. This is the standard format for streaming datasets, logs, and batch inference results.

```python
# json_lines.py
import json

# Write JSON Lines
records = [
    {"id": 1, "prompt": "Hello", "response": "Hi there!"},
    {"id": 2, "prompt": "How are you?", "response": "I am good."},
    {"id": 3, "prompt": "What is AI?", "response": "Artificial intelligence."},
]

with open("data.jsonl", "w", encoding="utf-8") as f:
    for record in records:
        f.write(json.dumps(record) + "\n")

# Read JSON Lines — memory-efficient for large files
with open("data.jsonl", "r", encoding="utf-8") as f:
    for line in f:
        record = json.loads(line.strip())
        print(f"ID {record['id']}: {record['prompt']}")

# Clean up
import os
os.remove("data.jsonl")
```

JSON Lines is ideal when you have millions of records — you process one line at a time instead of loading the entire file into memory.

## Build Something Real — Batch Inference Logger

```python
# batch_logger.py
import json
from pathlib import Path
from datetime import datetime

class BatchLogger:
    """Log batch inference results to JSON Lines."""

    def __init__(self, log_dir="batch_logs"):
        self.log_dir = Path(log_dir)
        self.log_dir.mkdir(exist_ok=True)
        self.run_id = datetime.now().strftime("%Y%m%d_%H%M%S")
        self.log_path = self.log_dir / f"run_{self.run_id}.jsonl"

    def log_result(self, prompt, response, tokens_used, latency_ms):
        """Log one inference result."""
        record = {
            "timestamp": datetime.now().isoformat(),
            "prompt": prompt,
            "response": response,
            "tokens_used": tokens_used,
            "latency_ms": latency_ms,
        }
        with open(self.log_path, "a", encoding="utf-8") as f:
            f.write(json.dumps(record) + "\n")

    def read_results(self):
        """Read all logged results."""
        results = []
        if not self.log_path.exists():
            return results
        with open(self.log_path, "r", encoding="utf-8") as f:
            for line in f:
                results.append(json.loads(line.strip()))
        return results

    def summary(self):
        """Print summary statistics."""
        results = self.read_results()
        if not results:
            print("No results logged.")
            return
        total_tokens = sum(r["tokens_used"] for r in results)
        avg_latency = sum(r["latency_ms"] for r in results) / len(results)
        print(f"Total inferences: {len(results)}")
        print(f"Total tokens: {total_tokens}")
        print(f"Average latency: {avg_latency:.0f} ms")

# Usage
logger = BatchLogger()
logger.log_result("What is Python?", "A programming language.", 6, 450)
logger.log_result("Explain JSON", "A data format.", 5, 380)
logger.log_result("What is an array?", "A collection of items.", 7, 420)

logger.summary()
# Total inferences: 3
# Total tokens: 18
# Average latency: 417 ms

# Clean up
import shutil
shutil.rmtree("batch_logs")
```

The token counting: 6 + 5 + 7 = 18. The latency averaging: (450 + 380 + 420) / 3 = 416.7 → 417 ms.
