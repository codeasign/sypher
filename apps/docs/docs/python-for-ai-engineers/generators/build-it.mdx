# Generators — Hands On

⏱ 14 minutes

## Breaking It Down

### Step 1: Generator Function Basics

```python
# basic_generator.py

def count_up_to(n):
    """Yield numbers from 1 to n, one at a time."""
    i = 1
    while i <= n:
        yield i
        i += 1

# Calling the function returns a generator object — no code runs yet
counter = count_up_to(3)
print(counter)  # <generator object count_up_to at 0x...>

# Get values one at a time with next()
print(next(counter))  # 1
print(next(counter))  # 2
print(next(counter))  # 3
# print(next(counter))  # StopIteration — no more values

# Or use a for loop (handles StopIteration automatically)
for num in count_up_to(5):
    print(num)  # 1, 2, 3, 4, 5
```

### Step 2: Generator Expressions

```python
# gen_expressions.py

# Generator expression — lazy version of list comprehension
squares_gen = (x ** 2 for x in range(1, 6))
print(squares_gen)  # <generator object <genexpr> at 0x...>

# Materialise into a list (defeats the purpose for large data)
squares = list(squares_gen)
print(squares)  # [1, 4, 9, 16, 25]

# Compare memory usage:
import sys
list_comp = [x for x in range(10000)]
gen_expr = (x for x in range(10000))
print(sys.getsizeof(list_comp))  # ~80,000 bytes
print(sys.getsizeof(gen_expr))   # ~200 bytes — constant!
```

### Step 3: Generator with `yield from`

`yield from` delegates to another iterable or generator:

```python
# yield_from.py

def flatten(nested):
    """Flatten a list of lists using yield from."""
    for sublist in nested:
        yield from sublist

matrix = [[1, 2], [3, 4], [5, 6]]
flat = list(flatten(matrix))
print(flat)  # [1, 2, 3, 4, 5, 6]

# Chaining generators
def numbers():
    yield from range(1, 4)
    yield from range(10, 13)

print(list(numbers()))  # [1, 2, 3, 10, 11, 12]
```

### Step 4: Infinite Generators

Generators can produce values indefinitely — useful for streams:

```python
# infinite_gen.py

def token_id_generator():
    """Yield incrementing token IDs forever."""
    i = 1
    while True:
        yield i
        i += 1

ids = token_id_generator()
print(next(ids))  # 1
print(next(ids))  # 2
print(next(ids))  # 3
# Can keep calling next() forever — no memory growth
```

### Step 5: Generator Pipeline

Chain multiple generators to build a processing pipeline:

```python
# pipeline.py

def read_lines(path):
    """Generator: yield lines from a file one at a time."""
    with open(path, "r", encoding="utf-8") as f:
        for line in f:
            yield line.strip()

def filter_comments(lines):
    """Generator: yield only lines that are not comments."""
    for line in lines:
        if line and not line.startswith("#"):
            yield line

def parse_csv_lines(lines, delimiter=","):
    """Generator: yield parsed CSV fields from lines."""
    for line in lines:
        yield line.split(delimiter)

def filter_by_threshold(rows, column_index, threshold):
    """Generator: yield rows where column value exceeds threshold."""
    for row in rows:
        try:
            value = float(row[column_index])
            if value > threshold:
                yield row
        except (ValueError, IndexError):
            continue

# Create a sample file to process
with open("sample_data.csv", "w", encoding="utf-8") as f:
    f.write("# model,accuracy,f1\n")
    f.write("gpt-4,0.94,0.92\n")
    f.write("claude-3,0.93,0.91\n")
    f.write("llama-3,0.89,0.87\n")
    f.write("gemini-pro,0.31,0.28\n")

# Chain: read -> filter_comments -> parse -> filter_by_threshold
pipeline = filter_by_threshold(
    parse_csv_lines(
        filter_comments(
            read_lines("sample_data.csv")
        )
    ),
    column_index=1,   # accuracy column
    threshold=0.9,
)

for row in pipeline:
    print(f"Model: {row[0]}, Accuracy: {row[1]}")
# Model: gpt-4, Accuracy: 0.94
# Model: claude-3, Accuracy: 0.93

import os
os.remove("sample_data.csv")
```

The pipeline never holds more than one row in memory at a time, no matter how large the file is.

### Step 6: Build Something Real — Batch Generator for Model Inference

```python
# batch_generator.py — Yield batches of data for model inference

def batch_generator(data, batch_size=32):
    """Yield successive batches from a dataset."""
    batch = []
    for item in data:
        batch.append(item)
        if len(batch) == batch_size:
            yield batch
            batch = []
    # Yield the final partial batch
    if batch:
        yield batch

# Simulated dataset of 1000 prompts
prompts = [f"Prompt {i}: What is AI?" for i in range(1000)]

# Process in batches
for batch_num, batch in enumerate(batch_generator(prompts, batch_size=64)):
    print(f"Batch {batch_num + 1}: {len(batch)} prompts")
    # In real code: results = model(batch)
    # Total iterations: ceil(1000/64) = 16

# ---

def stream_model_responses(prompts, batch_size=32):
    """Generator: process prompts in batches and yield results."""
    for batch in batch_generator(prompts, batch_size):
        # Simulate batch inference
        results = [f"Result for: {p}" for p in batch]
        yield from results

# Lazy — no list built, results are yielded as they are produced
for response in stream_model_responses(["Hi", "Hello", "Test"]):
    print(response)
```

## Performance Considerations

- **Memory:** A generator storing one value uses O(1) memory regardless of the total number of values produced.
- **Speed:** Generator expressions are slightly slower than list comprehensions per iteration (due to generator overhead), but they avoid the O(n) memory cost.
- **Single-use:** A generator can be iterated only once. After it is exhausted, create a new one.
- **CPU vs I/O:** Generators are ideal for I/O-bound pipelines (reading files, network streams). For CPU-bound work, the generator overhead may matter — benchmark if performance is critical.
