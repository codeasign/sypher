## Breaking It Down

### Step 1: Pipeline Stage Decorator

A pipeline stage is a function with metadata for retries, timeouts, and logging. A decorator keeps the definition clean:

```python
# stage.py
import time
import logging
from functools import wraps
from typing import Any, Callable

logger = logging.getLogger(__name__)


def pipeline_stage(
    name: str,
    max_retries: int = 2,
    timeout: float = 30.0,
):
    """Decorator that wraps a function as a pipeline stage.

    Args:
        name: Human-readable stage name for logging.
        max_retries: Number of retries on exception.
        timeout: Max seconds for one execution attempt.

    Usage:
        @pipeline_stage("classify", max_retries=3)
        def classify(email_body: str) -> str: ...
    """
    def decorator(func: Callable) -> Callable:
        @wraps(func)
        def wrapper(*args, **kwargs) -> Any:
            last_error = None
            for attempt in range(max_retries + 1):
                try:
                    start = time.perf_counter()
                    result = func(*args, **kwargs)
                    elapsed = time.perf_counter() - start
                    logger.info(
                        "Stage '%s' succeeded in %.2fs (attempt %d)",
                        name, elapsed, attempt + 1,
                    )
                    return result
                except Exception as e:
                    last_error = e
                    elapsed = time.perf_counter() - start
                    logger.warning(
                        "Stage '%s' failed in %.2fs (attempt %d/%d): %s",
                        name, elapsed, attempt + 1, max_retries + 1, e,
                    )
                    if attempt < max_retries:
                        time.sleep(2 ** attempt)

            raise RuntimeError(
                f"Stage '{name}' failed after {max_retries + 1} attempts"
            ) from last_error
        return wrapper
    return decorator
```

### Step 2: Building Individual Pipeline Stages

Each stage is a focused function. Here is a complete email-processing pipeline:

```python
# email_pipeline.py
import os
import json
import logging
from datetime import datetime

from anthropic import Anthropic
from pydantic import BaseModel

from stage import pipeline_stage

logger = logging.getLogger(__name__)
client = Anthropic(api_key=os.environ["ANTHROPIC_API_KEY"])


# --- Data models for typed intermediate results ---

class ClassifiedEmail(BaseModel):
    email_id: str
    body: str
    intent: str       # complaint, question, feedback, spam
    urgency: str      # low, medium, high
    confidence: float


class ExtractedInfo(BaseModel):
    email_id: str
    customer_name: str | None = None
    order_id: str | None = None
    priority: str | None = None
    key_issues: list[str] = []


class DraftResponse(BaseModel):
    email_id: str
    subject: str
    body: str
    tone: str


# --- Pipeline stages ---

@pipeline_stage("load_emails", max_retries=1)
def load_emails(filepath: str) -> list[dict]:
    """Load emails from a JSON file."""
    with open(filepath, "r") as f:
        emails = json.load(f)
    logger.info("Loaded %d emails from %s", len(emails), filepath)
    return emails


@pipeline_stage("classify_email", max_retries=2)
def classify_email(email: dict) -> ClassifiedEmail:
    """Classify an email's intent and urgency using an LLM."""
    response = client.messages.create(
        model="claude-sonnet-4-20250514",
        max_tokens=256,
        system=(
            "Classify the email into one of: complaint, question, feedback, spam. "
            "Rate urgency as: low, medium, high. "
            "Respond with valid JSON only: "
            '{"intent": "...", "urgency": "...", "confidence": 0.0-1.0}'
        ),
        messages=[{"role": "user", "content": email["body"][:2000]}],
    )
    result = json.loads(response.content[0].text)
    return ClassifiedEmail(
        email_id=email["id"],
        body=email["body"],
        intent=result["intent"],
        urgency=result["urgency"],
        confidence=result["confidence"],
    )


@pipeline_stage("extract_info", max_retries=1)
def extract_info(classified: ClassifiedEmail) -> ExtractedInfo:
    """Extract structured information from the email body."""
    if classified.intent == "spam":
        return ExtractedInfo(email_id=classified.email_id)

    response = client.messages.create(
        model="claude-sonnet-4-20250514",
        max_tokens=512,
        system=(
            "Extract customer information from this email. "
            "Respond with valid JSON: "
            '{"customer_name": "..." or null, "order_id": "..." or null, '
            '"priority": "low/medium/high" or null, '
            '"key_issues": ["...", "..."]}'
        ),
        messages=[{"role": "user", "content": classified.body[:2000]}],
    )
    result = json.loads(response.content[0].text)
    return ExtractedInfo(
        email_id=classified.email_id,
        **result,
    )


@pipeline_stage("generate_response", max_retries=2)
def generate_response(info: ExtractedInfo) -> DraftResponse | None:
    """Generate a draft response based on classification and extracted info."""
    if info.email_id == "skip":
        return None

    # Build structured prompt from extracted data
    context_parts = [f"Customer issue: {', '.join(info.key_issues)}"]
    if info.priority:
        context_parts.append(f"Priority: {info.priority}")

    response = client.messages.create(
        model="claude-sonnet-4-20250514",
        max_tokens=512,
        system=(
            "You are a customer support agent. Write a professional, empathetic "
            "response to the customer based on the extracted issues. "
            "Keep it under 150 words. "
            "Respond with JSON: "
            '{"subject": "...", "body": "...", "tone": "professional|apologetic|informative"}'
        ),
        messages=[{"role": "user", "content": "\n".join(context_parts)}],
    )
    result = json.loads(response.content[0].text)
    return DraftResponse(
        email_id=info.email_id,
        **result,
    )


@pipeline_stage("store_result", max_retries=2)
def store_result(draft: DraftResponse) -> dict:
    """Simulate storing in a database."""
    record = {
        "email_id": draft.email_id,
        "subject": draft.subject,
        "response_body": draft.body,
        "tone": draft.tone,
        "processed_at": datetime.utcnow().isoformat(),
    }
    # In production, write to PostgreSQL, MongoDB, etc.
    logger.info("Stored response for email %s", draft.email_id)
    return record
```

### Step 3: Pipeline Orchestrator

The orchestrator runs each item through all stages, collecting successes and failures:

```python
# orchestrator.py
import logging
from email_pipeline import (
    load_emails, classify_email, extract_info,
    generate_response, store_result,
)
from pydantic import BaseModel

logger = logging.getLogger(__name__)

class PipelineResult(BaseModel):
    email_id: str
    success: bool
    stages_completed: int
    error: str | None = None
    result: dict | None = None


def run_pipeline(filepath: str) -> list[PipelineResult]:
    """Run all emails through the pipeline, collecting results."""
    results = []

    try:
        raw_emails = load_emails(filepath)
    except Exception as e:
        logger.critical("Pipeline failed at load stage: %s", e)
        return [PipelineResult(email_id="unknown", success=False,
                                stages_completed=0, error=str(e))]

    for email in raw_emails:
        stages_completed = 0
        try:
            classified = classify_email(email)
            stages_completed += 1

            info = extract_info(classified)
            stages_completed += 1

            draft = generate_response(info)
            stages_completed += 1

            stored = store_result(draft)
            stages_completed += 1

            results.append(PipelineResult(
                email_id=email["id"],
                success=True,
                stages_completed=stages_completed,
                result=stored,
            ))
            logger.info("Email %s: pipeline complete", email["id"])

        except Exception as e:
            results.append(PipelineResult(
                email_id=email.get("id", "unknown"),
                success=False,
                stages_completed=stages_completed,
                error=str(e),
            ))
            logger.error("Email %s failed at stage %d: %s",
                          email.get("id", "unknown"), stages_completed + 1, e)

    return results
```

### Step 4: Running the Pipeline

Create a sample input file and run the pipeline:

```python
# main.py
import json
import logging

from orchestrator import run_pipeline

logging.basicConfig(
    level=logging.INFO,
    format="%(asctime)s [%(levelname)s] %(message)s",
)

# Create sample data
sample_emails = [
    {
        "id": "email_001",
        "body": "I ordered a laptop on order #12345 but it arrived damaged. "
                "The screen is cracked. Please help.",
    },
    {
        "id": "email_002",
        "body": "What are your business hours on weekends?",
    },
    {
        "id": "email_003",
        "body": "Congratulations! You won a free iPhone. Click here to claim.",
    },
]

with open("sample_emails.json", "w") as f:
    json.dump(sample_emails, f)

# Run pipeline
results = run_pipeline("sample_emails.json")

# Summary
successes = sum(1 for r in results if r.success)
failures = sum(1 for r in results if not r.success)
print(f"\nPipeline complete: {successes} succeeded, {failures} failed")
for r in results:
    status = "OK" if r.success else "FAIL"
    print(f"  {r.email_id}: {status} (stages: {r.stages_completed})")
```

Expected output:

```
Pipeline complete: 3 succeeded, 0 failed
  email_001: OK (stages: 4)
  email_002: OK (stages: 4)
  email_003: OK (stages: 4)
```

Spam emails get extracted and responded to with minimal processing — the pipeline handles them without crashing.

## Build Something Real — Concurrent Pipeline with Async

For large batches, run pipeline stages concurrently using async:

```python
# async_pipeline.py
import asyncio
import os
from anthropic import Anthropic

client = Anthropic(api_key=os.environ["ANTHROPIC_API_KEY"])
semaphore = asyncio.Semaphore(5)  # Max 5 concurrent LLM calls


async def classify_concurrent(email_body: str) -> dict:
    """Classify an email with concurrent limit."""
    async with semaphore:
        response = await client.messages.create(
            model="claude-sonnet-4-20250514",
            max_tokens=128,
            messages=[{"role": "user", "content": email_body[:2000]}],
        )
        return {"id": "classified", "intent": response.content[0].text}


async def run_batch(emails: list[dict]):
    """Run classification over all emails concurrently."""
    tasks = [classify_concurrent(e["body"]) for e in emails]
    return await asyncio.gather(*tasks, return_exceptions=True)


# Usage
emails = [{"body": f"Test email number {i}"} for i in range(20)]
results = asyncio.run(run_batch(emails))
print(f"Processed {len(results)} emails concurrently")
```

## Performance Considerations

- **Parallelism vs. concurrency.** For I/O-bound stages (LLM calls, database writes), use async concurrency. For CPU-bound stages (parsing, validation), use `concurrent.futures.ProcessPoolExecutor`.
- **Semaphore throttling.** Without a semaphore, concurrent LLM calls hit rate limits. Start with 5 concurrent calls and tune based on your provider's limit.
- **Batch sizing.** Process items in batches of 10-100 for checkpointing. If the pipeline crashes, you only re-process one batch, not the entire dataset.

## Security Considerations

- **Intermediate data persistence.** Pipeline stages may write intermediate results to disk or a queue. Ensure these intermediate stores have the same access controls as your final data store.
- **Failures leak data.** Error messages that include the failed input (e.g., `"Failed to parse: {raw_email}"`) may leak sensitive content. Sanitise error logs.
- **Stage boundaries for access control.** If one stage handles PII and another does not, ensure they run in separate processes or containers with different credentials.
