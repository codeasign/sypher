# Inheritance — Hands On

⏱ 14 minutes

## Breaking It Down

### Step 1: Basic Inheritance

```python
# basic_inheritance.py

class BaseModel:
    """Base class for all models."""

    def __init__(self, name: str, version: str):
        self.name = name
        self.version = version

    def describe(self) -> str:
        return f"{self.name} v{self.version}"

    def predict(self, text: str) -> str:
        """Base predict — subclasses should override."""
        raise NotImplementedError("Subclasses must implement predict()")


class GPTModel(BaseModel):
    """GPT model — inherits from BaseModel."""

    def predict(self, text: str) -> str:
        """Override with GPT-specific logic."""
        return f"[GPT-{self.version}] Prediction: '{text}' → positive"


class ClaudeModel(BaseModel):
    """Claude model — inherits from BaseModel."""

    def predict(self, text: str) -> str:
        """Override with Claude-specific logic."""
        return f"[Claude-{self.version}] Prediction: '{text}' → neutral"


# Usage
gpt = GPTModel("gpt-4", "4.1")
claude = ClaudeModel("claude-3-opus", "3.0")

print(gpt.describe())   # gpt-4 v4.1 — inherited from BaseModel
print(gpt.predict("Hi"))   # [GPT-4.1] Prediction: 'Hi' → positive
print(claude.predict("Hi"))  # [Claude-3.0] Prediction: 'Hi' → neutral
```

### Step 2: Using `super()` to Extend the Parent

```python
# super_demo.py

class BaseModel:
    def __init__(self, name: str, version: str):
        self.name = name
        self.version = version
        self.call_count = 0

    def predict(self, text: str) -> str:
        self.call_count += 1
        raise NotImplementedError

    def stats(self) -> str:
        return f"Calls: {self.call_count}"


class GPTModel(BaseModel):
    def __init__(self, name: str, version: str, api_key: str):
        # Call parent __init__ to set name, version, call_count
        super().__init__(name, version)
        self.api_key = api_key  # Additional attribute

    def predict(self, text: str) -> str:
        # Call parent's predict to increment call_count
        super().predict(text)  # This will raise NotImplementedError
        # Actually, let's do it properly:
        self.call_count += 1
        return f"[GPT] '{text}' → positive"


# Better pattern — let parent handle the counting
class BaseModelV2:
    def __init__(self, name: str, version: str):
        self.name = name
        self.version = version
        self.call_count = 0

    def predict(self, text: str) -> str:
        """Child classes override this, but call super().predict() to count."""
        self.call_count += 1
        # No return — child must return the result


class GPTModelV2(BaseModelV2):
    def predict(self, text: str) -> str:
        super().predict(text)  # Increments call_count
        return f"[GPT] '{text}' → positive (call #{self.call_count})"


gpt = GPTModelV2("gpt-4", "4.1")
print(gpt.predict("Hi"))   # [GPT] 'Hi' → positive (call #1)
print(gpt.predict("Hello"))  # [GPT] 'Hello' → positive (call #2)
print(gpt.stats())         # Calls: 2
```

### Step 3: Abstract Base Classes

Use `abc.ABC` and `@abstractmethod` to define interfaces that subclasses must implement:

```python
# abstract.py
from abc import ABC, abstractmethod

class ModelInterface(ABC):
    """Abstract base class — defines the interface all models must implement."""

    @abstractmethod
    def load(self) -> None:
        """Load the model. Must be implemented by subclasses."""
        pass

    @abstractmethod
    def predict(self, text: str) -> dict:
        """Run inference. Must be implemented by subclasses."""
        pass

    @abstractmethod
    def unload(self) -> None:
        """Free resources. Must be implemented by subclasses."""
        pass

    def describe(self) -> str:
        """Concrete method — inherited as-is."""
        return f"Model: {self.__class__.__name__}"


class GPTModel(ModelInterface):
    def load(self) -> None:
        print("Loading GPT model (simulated)...")

    def predict(self, text: str) -> dict:
        return {"text": text, "label": "positive", "confidence": 0.95}

    def unload(self) -> None:
        print("Unloading GPT model...")


# model = ModelInterface()  # TypeError — cannot instantiate abstract class!
gpt = GPTModel()
print(gpt.describe())  # Model: GPTModel
print(gpt.predict("Hi"))  # {'text': 'Hi', 'label': 'positive', 'confidence': 0.95}
```

### Step 4: Multiple Inheritance and MRO

```python
# mro_demo.py

class LoggingMixin:
    """Mixin that adds logging capability."""
    def log(self, message: str) -> None:
        print(f"[LOG] {message}")

class CacheMixin:
    """Mixin that adds caching capability."""
    def __init__(self):
        self._cache = {}

    def cached_predict(self, text: str) -> str | None:
        return self._cache.get(text)

    def store_cache(self, text: str, result: str) -> None:
        self._cache[text] = result

class SmartModel(LoggingMixin, CacheMixin, ModelInterface):
    """Combines logging, caching, and the model interface."""

    def load(self) -> None:
        CacheMixin.__init__(self)
        self.log("Model loaded")

    def predict(self, text: str) -> dict:
        cached = self.cached_predict(text)
        if cached:
            self.log(f"Cache hit for: {text}")
            return {"text": text, "label": cached, "from_cache": True}

        result = {"text": text, "label": "positive", "confidence": 0.9}
        self.store_cache(text, result["label"])
        self.log(f"Predicted: {text}")
        return result

    def unload(self) -> None:
        self.log("Model unloaded")

# Check Method Resolution Order
print(SmartModel.__mro__)
# (<class 'SmartModel'>, <class 'LoggingMixin'>, <class 'CacheMixin'>,
#  <class 'ModelInterface'>, <class 'ABC'>, <class 'object'>)

model = SmartModel()
model.load()
print(model.predict("Hello"))  # Predicted + cache miss
print(model.predict("Hello"))  # Cache hit!
model.unload()
```

### Step 5: Build Something Real — Model Zoo

```python
# model_zoo.py
from abc import ABC, abstractmethod
import time

class BaseInferenceModel(ABC):
    """Base class for all inference models."""

    def __init__(self, model_name: str):
        self.model_name = model_name
        self.total_latency = 0.0
        self._loaded = False

    @abstractmethod
    def _load_weights(self) -> None:
        """Load model-specific weights."""
        pass

    @abstractmethod
    def _run_inference(self, text: str) -> dict:
        """Model-specific inference logic."""
        pass

    def load(self) -> None:
        print(f"Loading {self.model_name}...")
        self._load_weights()
        self._loaded = True

    def predict(self, text: str) -> dict:
        if not self._loaded:
            raise RuntimeError("Model not loaded")
        start = time.perf_counter()
        result = self._run_inference(text)
        elapsed = time.perf_counter() - start
        self.total_latency += elapsed
        result["latency"] = round(elapsed, 4)
        return result

    def stats(self) -> dict:
        return {
            "model": self.model_name,
            "total_latency": round(self.total_latency, 4),
        }


class OpenAICompatibleModel(BaseInferenceModel):
    def _load_weights(self) -> None:
        time.sleep(0.1)  # Simulate loading

    def _run_inference(self, text: str) -> dict:
        score = min(max(len(text) % 100 / 100.0, 0.0), 1.0)
        return {
            "text": text,
            "label": "positive" if score > 0.5 else "negative",
            "confidence": score,
            "provider": "openai-compatible",
        }


class LocalModel(BaseInferenceModel):
    def _load_weights(self) -> None:
        time.sleep(0.2)  # Simulate slower local loading

    def _run_inference(self, text: str) -> dict:
        return {
            "text": text,
            "label": "positive",
            "confidence": 0.8,
            "provider": "local",
        }


# Unified interface for different model types
models: list[BaseInferenceModel] = [
    OpenAICompatibleModel("gpt-4"),
    LocalModel("llama-3"),
]

for model in models:
    model.load()
    result = model.predict("Hello, world!")
    print(f"{model.model_name}: {result['label']} ({result['confidence']})")
    print(f"   Stats: {model.stats()}")
```
