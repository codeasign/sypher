# Performance — Hands On

⏱ 20 minutes

## Breaking It Down

### Step 1: Profile with cProfile

Create a slow version of a text-processing pipeline, then profile it:

```python
# slow_pipeline.py
"""A deliberately slow text-processing pipeline for profiling."""

import time
import json
from pathlib import Path
from typing import List, Dict


def load_documents(data_dir: str) -> List[Dict]:
    """Load JSON documents from a directory."""
    docs = []
    data_path = Path(data_dir)
    for file_path in data_path.glob("*.json"):
        with open(file_path, "r") as f:
            doc = json.load(f)
            docs.append(doc)
    return docs


def tokenize(text: str) -> List[str]:
    """Split text into lowercase tokens (deliberately slow version)."""
    import re
    words = []
    current = ""
    for char in text.lower():
        if char.isalnum() or char == "'":
            current += char
        else:
            if current:
                words.append(current)
                current = ""
    if current:
        words.append(current)
    return words


def compute_tf(tokens: List[str]) -> Dict[str, float]:
    """Compute term frequency for a list of tokens."""
    tf = {}
    for token in tokens:
        if token in tf:
            tf[token] += 1.0
        else:
            tf[token] = 1.0
    total = len(tokens)
    for token in tf:
        tf[token] /= total
    return tf


def process_document(doc: Dict) -> Dict:
    """Process a single document: tokenize and compute TF."""
    tokens = tokenize(doc.get("text", ""))
    tf = compute_tf(tokens)
    return {"id": doc.get("id"), "tf": tf, "token_count": len(tokens)}


def main():
    docs = load_documents("data/sample")
    results = []
    for doc in docs:
        result = process_document(doc)
        results.append(result)
    print(f"Processed {len(results)} documents")


if __name__ == "__main__":
    main()
```

Run it with cProfile:

```bash
python -m cProfile -o profile_stats.prof slow_pipeline.py
```

Analyse the results with `pstats`:

```python
import pstats
from pstats import SortKey

p = pstats.Stats("profile_stats.prof")
p.sort_stats(SortKey.CUMULATIVE).print_stats(15)
```

Output will look like:

```

    1    0.002    0.002    3.245    3.245  slow_pipeline.py:1(<module>)
  100    2.100    0.021    2.800    0.028  slow_pipeline.py:29(tokenize)
  100    0.500    0.005    0.500    0.005  slow_pipeline.py:49(compute_tf)
  100    0.200    0.002    0.200    0.002  slow_pipeline.py:18(load_documents)
```

This tells you `tokenize` is the bottleneck — 2.8 seconds of cumulative time out of 3.25 total.

### Step 2: Optimise the Hot Loop (Vectorisation)

The `tokenize` function uses a character-by-character loop. Replace it with a vectorised version:

```python
import re
from typing import List


def tokenize_fast(text: str) -> List[str]:
    """Split text into lowercase tokens using regex (vectorised)."""
    return re.findall(r"[a-z0-9']+", text.lower())
```

Profile again. The `tokenize_fast` version should be 10x-20x faster because `re.findall` operates in C internally.

The same principle applies to NumPy operations:

```python
import numpy as np

# ❌ Python loop — slow
def compute_lengths_slow(arrays):
    result = []
    for arr in arrays:
        result.append(len(arr))
    return result

# ✅ Vectorised (if arrays are NumPy arrays)
lengths = np.array([len(a) for a in arrays])  # Still a Python comprehension
# Better: use NumPy operations on 2D arrays
```

For true vectorisation with NumPy, ensure your data is in arrays and use array operations:

```python
import numpy as np

# ❌ Loop over elements
def normalize_slow(embeddings: np.ndarray) -> np.ndarray:
    result = np.zeros_like(embeddings)
    for i in range(len(embeddings)):
        norm = np.linalg.norm(embeddings[i])
        result[i] = embeddings[i] / norm
    return result

# ✅ Vectorised — no Python loop
def normalize_fast(embeddings: np.ndarray) -> np.ndarray:
    norms = np.linalg.norm(embeddings, axis=1, keepdims=True)
    return embeddings / norms
```

### Step 3: Cache Expensive Operations

When your pipeline makes API calls or repeats the same computation, caching eliminates redundant work:

```python
from functools import lru_cache
from typing import List
import hashlib
import json


# ❌ No cache — same text embedded twice
def get_embedding(text: str) -> List[float]:
    """Call an embedding API (expensive)."""
    response = client.embeddings.create(
        model="text-embedding-3-small",
        input=text,
    )
    return response.data[0].embedding

# ✅ Cached — identical text returns instantly
@lru_cache(maxsize=1024)
def get_embedding_cached(text: str) -> List[float]:
    response = client.embeddings.create(
        model="text-embedding-3-small",
        input=text,
    )
    return response.data[0].embedding
```

For pipelines that process documents with repeated content (e.g., the same system prompt prefix, common stop words, or repeated queries), caching at the function level provides dramatic speedups.

For larger caches that must persist across runs, use `diskcache`:

```python
import diskcache as dc
cache = dc.Cache("embedding_cache")

def get_embedding_persistent(text: str) -> List[float]:
    key = hashlib.sha256(text.encode()).hexdigest()
    if key in cache:
        return cache[key]
    response = client.embeddings.create(
        model="text-embedding-3-small",
        input=text,
    )
    embedding = response.data[0].embedding
    cache[key] = embedding
    return embedding
```

### Step 4: Concurrent I/O with asyncio + httpx

When your pipeline makes many API calls, sequential execution wastes time waiting for network responses:

```python
import asyncio
import httpx
from typing import List, Dict


# ❌ Sequential — waits for each response before sending next request
def classify_sequential(texts: List[str]) -> List[str]:
    results = []
    for text in texts:
        response = client.chat.completions.create(
            model="gpt-4o-mini",
            messages=[{"role": "user", "content": text}],
        )
        results.append(response.choices[0].message.content)
    return results


# ✅ Concurrent — all requests in flight simultaneously
async def classify_concurrent(texts: List[str]) -> List[str]:
    async with httpx.AsyncClient(timeout=60.0) as client:
        tasks = []
        for text in texts:
            payload = {
                "model": "gpt-4o-mini",
                "messages": [{"role": "user", "content": text}],
            }
            tasks.append(
                client.post(
                    "https://api.openai.com/v1/chat/completions",
                    json=payload,
                    headers={"Authorization": f"Bearer {api_key}"},
                )
            )
        responses = await asyncio.gather(*tasks)
        results = []
        for resp in responses:
            data = resp.json()
            results.append(data["choices"][0]["message"]["content"])
        return results


results = asyncio.run(classify_concurrent(texts))
```

For 100 API calls each taking 800ms:
- Sequential: 80 seconds
- Concurrent (10 at a time): ~8 seconds

### Step 5: Build Something Real — Optimised Document Processing Pipeline

Here is the complete slow pipeline refactored with all three techniques:

```python
"""Optimised document processing pipeline with profiling support."""

import json
import re
import time
from functools import lru_cache
from pathlib import Path
from typing import List, Dict, Tuple


@lru_cache(maxsize=128)
def load_stopwords(path: str) -> set:
    """Cache the stopwords set so it is loaded from disk only once."""
    with open(path, "r") as f:
        return {line.strip().lower() for line in f}


def tokenize(text: str) -> Tuple[str, ...]:
    """Tokenise text into lowercase words using vectorised regex.

    Returns a tuple (hashable for caching) of tokens.
    """
    tokens = re.findall(r"[a-z0-9']+", text.lower())
    return tuple(tokens)


@lru_cache(maxsize=4096)
def compute_tf(tokens: Tuple[str, ...]) -> Dict[str, float]:
    """Compute term frequency from a tuple of tokens.

    Cached so that documents with identical text (e.g. retries)
    do not recompute TF.
    """
    tf: Dict[str, float] = {}
    for token in tokens:
        tf[token] = tf.get(token, 0.0) + 1.0
    total = len(tokens)
    for token in tf:
        tf[token] /= total
    return tf


def process_document(doc: Dict) -> Dict:
    """Process a single document: tokenize and compute TF."""
    text = doc.get("text", "")
    tokens = tokenize(text)
    tf = compute_tf(tokens)
    return {"id": doc.get("id"), "tf": tf, "token_count": len(tokens)}


def load_documents(data_dir: str) -> List[Dict]:
    """Load JSON documents from a directory."""
    data_path = Path(data_dir)
    docs = []
    for file_path in data_path.glob("*.json"):
        with open(file_path, "r") as f:
            docs.append(json.load(f))
    return docs


def main():
    """Run the optimised pipeline."""
    start = time.perf_counter()

    docs = load_documents("data/sample")
    results = [process_document(doc) for doc in docs]

    elapsed = time.perf_counter() - start
    print(f"Processed {len(results)} documents in {elapsed:.2f}s")
    print(f"Average: {elapsed / len(results) * 1000:.1f}ms per document")


if __name__ == "__main__":
    main()
```

Key optimisations applied:
- `re.findall` replaces the character-by-character tokeniser (vectorisation)
- `@lru_cache` on `load_stopwords` and `compute_tf` avoids redundant work (caching)
- List comprehension replaces an explicit for-loop with append (small gain)
- `time.perf_counter()` for accurate timing

## Performance Considerations

The techniques in this lesson address Python's most common performance bottlenecks. Here is when each applies:

| Bottleneck | Technique | Typical Speedup |
|------------|-----------|----------------|
| Python for-loop over arrays | NumPy vectorisation | 50x - 200x |
| Repeated function calls with same args | `@lru_cache` / `diskcache` | 1000x+ (or 0ms for hits) |
| Sequential I/O (API calls, file reads) | `asyncio` / `concurrent.futures` | Nx where N = concurrency |
| String processing in loops | `re` module (vectorised) | 10x - 30x |
| File I/O | Memory-mapping (`mmap`) | 2x - 5x |

For CPU-bound work (crunching numbers), use NumPy or Numba. For I/O-bound work (waiting on networks or disks), use asyncio or threading. Mixing them requires careful design — profile first, then apply the right tool.
