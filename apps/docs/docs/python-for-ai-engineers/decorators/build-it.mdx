# Decorators — Hands On

⏱ 14 minutes

## Breaking It Down

### Step 1: A Simple Decorator from Scratch

```python
# simple_decorator.py

def my_decorator(func):
    """Wrap func with a message before and after."""
    def wrapper():
        print("Before the function runs")
        func()
        print("After the function runs")
    return wrapper

@my_decorator
def say_hello():
    print("Hello!")

say_hello()
# Before the function runs
# Hello!
# After the function runs
```

### Step 2: Decorator with Arguments

When the wrapped function takes arguments, the wrapper must accept them:

```python
# argument_decorator.py

def log_call(func):
    """Log the arguments and return value of a function."""
    def wrapper(*args, **kwargs):
        print(f"Calling {func.__name__} with args={args} kwargs={kwargs}")
        result = func(*args, **kwargs)
        print(f"{func.__name__} returned {result}")
        return result
    return wrapper

@log_call
def add(a, b):
    return a + b

add(3, 5)
# Calling add with args=(3, 5) kwargs={}
# add returned 8
```

### Step 3: Preserving Metadata with `functools.wraps`

Without `@wraps`, the decorated function loses its name and docstring:

```python
# wraps_demo.py
import functools

def log_call(func):
    @functools.wraps(func)  # Preserves func.__name__, __doc__, etc.
    def wrapper(*args, **kwargs):
        print(f"Calling {func.__name__}")
        return func(*args, **kwargs)
    return wrapper

@log_call
def train(model, epochs=10):
    """Train a model for the specified number of epochs."""
    return f"Training {model} for {epochs} epochs"

print(train.__name__)  # 'train' (not 'wrapper')
print(train.__doc__)   # 'Train a model for the specified number of epochs.'
print(train("gpt-4", epochs=5))
# Calling train
# Training gpt-4 for 5 epochs
```

Always use `@functools.wraps` when writing a decorator — it prevents confusing debugging.

### Step 4: Timing Decorator

```python
# timing.py
import functools
import time

def timer(func):
    """Print how long the decorated function takes."""
    @functools.wraps(func)
    def wrapper(*args, **kwargs):
        start = time.perf_counter()
        result = func(*args, **kwargs)
        elapsed = time.perf_counter() - start
        print(f"{func.__name__} took {elapsed:.4f}s")
        return result
    return wrapper

@timer
def slow_function():
    """Simulate a slow operation."""
    total = 0
    for i in range(10_000_000):
        total += i
    return total

result = slow_function()
# slow_function took 0.XXXXs
print(f"Result: {result}")  # 49999995000000 = (9,999,999 * 10,000,000) / 2
```

### Step 5: Retry Decorator

```python
# retry.py
import functools
import time
import random

def retry(max_attempts=3, delay=1.0):
    """Retry a function if it raises an exception."""
    def decorator(func):
        @functools.wraps(func)
        def wrapper(*args, **kwargs):
            last_exception = None
            for attempt in range(1, max_attempts + 1):
                try:
                    return func(*args, **kwargs)
                except Exception as e:
                    last_exception = e
                    print(f"Attempt {attempt}/{max_attempts} failed: {e}")
                    if attempt < max_attempts:
                        time.sleep(delay)
            print(f"All {max_attempts} attempts failed")
            raise last_exception
        return wrapper
    return decorator

@retry(max_attempts=3, delay=0.5)
def unreliable_api_call():
    """Simulate an API that fails 60% of the time."""
    if random.random() < 0.6:
        raise ConnectionError("Network timeout")
    return "Success!"

# This will likely succeed within 3 attempts
try:
    result = unreliable_api_call()
    print(f"Got: {result}")
except ConnectionError:
    print("All attempts failed")
```

### Step 6: Chaining Multiple Decorators

```python
# chaining.py
import functools
import time

def log_call(func):
    @functools.wraps(func)
    def wrapper(*args, **kwargs):
        print(f"→ Calling {func.__name__}")
        result = func(*args, **kwargs)
        print(f"← {func.__name__} finished")
        return result
    return wrapper

def timer(func):
    @functools.wraps(func)
    def wrapper(*args, **kwargs):
        start = time.perf_counter()
        result = func(*args, **kwargs)
        elapsed = time.perf_counter() - start
        print(f"  [{func.__name__} took {elapsed:.3f}s]")
        return result
    return wrapper

# Decorators apply bottom-up: timer wraps first, then log_call wraps that
@log_call
@timer
def process(text):
    """Simulate text processing."""
    time.sleep(0.2)
    return text.upper()

process("hello")
# → Calling process
#   [process took 0.200s]
# ← process finished
```

### Step 7: Build Something Real — API Call Logger

```python
# api_logger.py
import functools
import time
import json

def log_api_call(log_file="api_calls.jsonl"):
    """Decorator: log every call to an API function to a JSONL file."""
    def decorator(func):
        @functools.wraps(func)
        def wrapper(*args, **kwargs):
            start = time.perf_counter()
            error = None
            result = None

            try:
                result = func(*args, **kwargs)
            except Exception as e:
                error = str(e)
                raise
            finally:
                elapsed = time.perf_counter() - start
                log_entry = {
                    "function": func.__name__,
                    "args": str(args),
                    "kwargs": {k: str(v) for k, v in kwargs.items()},
                    "duration_s": round(elapsed, 4),
                    "error": error,
                    "has_result": result is not None,
                }
                with open(log_file, "a", encoding="utf-8") as f:
                    f.write(json.dumps(log_entry) + "\n")
            return result
        return wrapper
    return decorator

@log_api_call()
def call_llm(prompt, model="gpt-4"):
    """Simulate an LLM API call."""
    time.sleep(0.1)  # Simulate network latency
    if "error" in prompt.lower():
        raise ValueError("Bad request")
    return f"Response to: {prompt}"

# Successful call
resp = call_llm("Hello, world!")
print(resp)

# Check the log file
with open("api_calls.jsonl", "r", encoding="utf-8") as f:
    for line in f:
        print(json.loads(line))

import os
os.remove("api_calls.jsonl")
```
