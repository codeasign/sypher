# Logging — Hands On

⏱ 14 minutes

## Breaking It Down

### Step 1: Basic Logging

```python
# basic_logging.py
import logging

# Simple configuration
logging.basicConfig(level=logging.INFO)

# Log messages at different levels
logging.debug("This is a debug message")   # Not shown — level is INFO
logging.info("Model loaded successfully")   # Shown
logging.warning("Low disk space: 2GB free")
logging.error("API call failed: timeout")
logging.critical("Out of memory — cannot continue")
```

Output:
```
INFO:root:Model loaded successfully
WARNING:root:Low disk space: 2GB free
ERROR:root:API call failed: timeout
CRITICAL:root:Out of memory — cannot continue
```

### Step 2: Configuring Format and Output

```python
# configure_logging.py
import logging

logging.basicConfig(
    level=logging.DEBUG,
    format="%(asctime)s [%(levelname)s] %(name)s: %(message)s",
    datefmt="%Y-%m-%d %H:%M:%S",
    filename="app.log",       # Log to a file instead of console
    filemode="a",             # Append mode
)

logging.info("Application started")
logging.debug("Loading configuration...")
```

Output in `app.log`:
```
2026-06-30 14:30:22 [INFO] root: Application started
2026-06-30 14:30:22 [DEBUG] root: Loading configuration...
```

### Step 3: Logger Objects (Not the Root Logger)

Instead of using `logging.info()` (which uses the root logger), create named loggers:

```python
# named_logger.py
import logging

# Create a logger for this module
logger = logging.getLogger(__name__)  # Uses the module name

# Set the level
logger.setLevel(logging.DEBUG)

# Create console handler
handler = logging.StreamHandler()
handler.setLevel(logging.DEBUG)

# Create formatter
formatter = logging.Formatter(
    "%(asctime)s [%(levelname)s] %(name)s: %(message)s"
)
handler.setFormatter(formatter)

# Add handler to logger
logger.addHandler(handler)

# Use it
logger.info("Data pipeline started")
logger.warning("Dataset has 5 missing values")
```

### Step 4: Multiple Handlers — Console and File

```python
# multi_handler.py
import logging

logger = logging.getLogger("pipeline")
logger.setLevel(logging.DEBUG)

# Console handler — INFO and above
console = logging.StreamHandler()
console.setLevel(logging.INFO)
console.setFormatter(logging.Formatter(
    "[%(levelname)s] %(message)s"
))

# File handler — DEBUG and above (more verbose)
file_handler = logging.FileHandler("pipeline.log", encoding="utf-8")
file_handler.setLevel(logging.DEBUG)
file_handler.setFormatter(logging.Formatter(
    "%(asctime)s [%(levelname)s] %(name)s: %(message)s"
))

logger.addHandler(console)
logger.addHandler(file_handler)

logger.info("Pipeline started")       # Both console and file
logger.debug("Batch 1: 32 items")     # File only (console level is INFO)
```

### Step 5: Logging Exceptions

Use `logger.exception()` inside an `except` block to include the full traceback:

```python
# exception_logging.py
import logging

logger = logging.getLogger(__name__)
logging.basicConfig(level=logging.ERROR)

try:
    result = 1 / 0
except ZeroDivisionError:
    logger.exception("Division failed")  # Includes traceback

# Output:
# ERROR:__main__:Division failed
# Traceback (most recent call last):
#   File "exception_logging.py", line 6, in <module>
#     result = 1 / 0
# ZeroDivisionError: division by zero
```

### Step 6: Structured Logging with Extra Context

```python
# structured_logging.py
import logging

logger = logging.getLogger("api")
logging.basicConfig(level=logging.INFO)

# Add extra fields to log records
logger.info(
    "API call completed",
    extra={
        "model": "gpt-4",
        "tokens": 150,
        "latency_ms": 2342,
        "status": "success",
    },
)

# To see extra fields, you need a custom formatter — or use a library like structlog
# For now, this adds them to the LogRecord (accessible in handlers)
```

### Step 7: Build Something Real — Inference Pipeline Logger

```python
# inference_logger.py
import logging
import time
from pathlib import Path

class PipelineLogger:
    """Configured logger for an AI inference pipeline."""

    def __init__(self, name: str, log_dir: str = "logs"):
        self.logger = logging.getLogger(name)
        self.logger.setLevel(logging.DEBUG)

        log_dir = Path(log_dir)
        log_dir.mkdir(exist_ok=True)

        # Console handler — INFO level
        console = logging.StreamHandler()
        console.setLevel(logging.INFO)
        console.setFormatter(logging.Formatter(
            "%(asctime)s | %(levelname)-8s | %(message)s",
            datefmt="%H:%M:%S",
        ))

        # File handler — DEBUG level with full details
        file_handler = logging.FileHandler(
            log_dir / f"{name}_{time.strftime('%Y%m%d')}.log",
            encoding="utf-8",
        )
        file_handler.setLevel(logging.DEBUG)
        file_handler.setFormatter(logging.Formatter(
            "%(asctime)s [%(levelname)s] %(name)s (%(filename)s:%(lineno)d): %(message)s"
        ))

        self.logger.addHandler(console)
        self.logger.addHandler(file_handler)

    def get_logger(self) -> logging.Logger:
        return self.logger

# Usage
pl = PipelineLogger("inference")
log = pl.get_logger()

log.info("Starting inference pipeline")
log.debug(f"Python version: 3.12")

# Simulate pipeline steps
for step in ["load_data", "preprocess", "inference", "evaluate"]:
    start = time.perf_counter()
    log.info(f"Step: {step}")
    time.sleep(0.05)
    elapsed = time.perf_counter() - start
    log.debug(f"  {step} took {elapsed:.3f}s")

log.info("Pipeline completed")

# Clean up log files
import shutil
shutil.rmtree("logs")
```
