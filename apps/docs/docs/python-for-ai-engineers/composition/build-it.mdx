# Composition — Hands On

⏱ 12 minutes

## Breaking It Down

### Step 1: Basic Composition — A Class That Has Another Class

```python
# basic_composition.py

class Tokeniser:
    """Splits text into tokens."""

    def tokenise(self, text: str) -> list[str]:
        return text.lower().split()


class Model:
    """A model that uses a tokeniser via composition."""

    def __init__(self, name: str):
        self.name = name
        self.tokeniser = Tokeniser()  # Composition: Model HAS a Tokeniser

    def predict(self, text: str) -> dict:
        tokens = self.tokeniser.tokenise(text)
        # Simplified inference
        return {
            "text": text,
            "tokens": tokens,
            "token_count": len(tokens),
            "model": self.name,
        }

model = Model("gpt-4")
result = model.predict("Hello world!")
print(f"Tokens: {result['tokens']}")  # ['hello', 'world!']
print(f"Count: {result['token_count']}")  # 2
```

### Step 2: Dependency Injection — Inject Dependencies via Constructor

```python
# dependency_injection.py

class Vectoriser:
    """Converts tokens to vectors."""

    def vectorise(self, tokens: list[str]) -> list[float]:
        """Simulate converting tokens to a vector."""
        return [hash(t) % 100 / 100.0 for t in tokens]


class Classifier:
    """Classifies text using injected dependencies."""

    def __init__(self, tokeniser: Tokeniser, vectoriser: Vectoriser):
        # Dependencies are injected — not created internally
        self.tokeniser = tokeniser
        self.vectoriser = vectoriser

    def classify(self, text: str) -> dict:
        tokens = self.tokeniser.tokenise(text)
        vector = self.vectoriser.vectorise(tokens)
        confidence = sum(vector) / len(vector) if vector else 0.0
        return {
            "label": "positive" if confidence > 0.5 else "negative",
            "confidence": round(confidence, 4),
            "token_count": len(tokens),
        }

# Inject dependencies
tokeniser = Tokeniser()
vectoriser = Vectoriser()
classifier = Classifier(tokeniser, vectoriser)

result = classifier.classify("Great service!")
print(result)
```

Dependency injection makes it easy to swap implementations:

```python
class MockVectoriser:
    """A mock vectoriser for testing."""
    def vectorise(self, tokens):
        return [0.9] * len(tokens)  # Always high confidence

test_classifier = Classifier(Tokeniser(), MockVectoriser())
print(test_classifier.classify("Bad"))  # Still 'positive' due to mock
```

### Step 3: Composable Pipeline

```python
# pipeline.py
from abc import ABC, abstractmethod

class Transformer(ABC):
    """Abstract step in a processing pipeline."""

    @abstractmethod
    def transform(self, text: str) -> str:
        pass


class LowercaseTransformer(Transformer):
    def transform(self, text: str) -> str:
        return text.lower()


class PunctuationRemover(Transformer):
    import string
    PUNCTUATION = string.punctuation

    def transform(self, text: str) -> str:
        return "".join(ch for ch in text if ch not in self.PUNCTUATION)


class WhitespaceNormaliser(Transformer):
    def transform(self, text: str) -> str:
        import re
        return re.sub(r"\s+", " ", text).strip()


class PreprocessingPipeline:
    """Composed pipeline — has a list of Transformers."""

    def __init__(self, transformers: list[Transformer] | None = None):
        self.transformers = transformers or []

    def add(self, transformer: Transformer) -> None:
        self.transformers.append(transformer)

    def process(self, text: str) -> str:
        for transformer in self.transformers:
            text = transformer.transform(text)
        return text

# Build a pipeline by composing transformers
pipeline = PreprocessingPipeline([
    LowercaseTransformer(),
    PunctuationRemover(),
    WhitespaceNormaliser(),
])

result = pipeline.process("  Hello, World!!  ")
print(repr(result))  # 'hello world'
```

### Step 4: Strategy Pattern via Composition

Instead of hard-coding behaviour, inject different strategies:

```python
# strategy.py

class ScoringStrategy(ABC):
    @abstractmethod
    def score(self, tokens: list[str]) -> float:
        pass


class LengthBasedScoring(ScoringStrategy):
    """Longer texts score higher (scaled)."""
    def score(self, tokens: list[str]) -> float:
        return min(len(" ".join(tokens)) / 100.0, 1.0)


class KeywordBasedScoring(ScoringStrategy):
    """Score based on presence of positive/negative keywords."""
    POSITIVE = {"good", "great", "excellent", "amazing"}
    NEGATIVE = {"bad", "terrible", "awful", "poor"}

    def score(self, tokens: list[str]) -> float:
        pos = sum(1 for t in tokens if t in self.POSITIVE)
        neg = sum(1 for t in tokens if t in self.NEGATIVE)
        total = pos + neg
        if total == 0:
            return 0.5
        return pos / total


class SentimentAnalyzer:
    """Analyzes sentiment — strategy injected via composition."""

    def __init__(self, strategy: ScoringStrategy):
        self.strategy = strategy

    def analyze(self, text: str) -> dict:
        tokens = text.lower().split()
        score = self.strategy.score(tokens)
        return {
            "text": text,
            "score": round(score, 3),
            "sentiment": "positive" if score > 0.5 else "negative",
            "strategy": self.strategy.__class__.__name__,
        }

# Use different strategies
length_analyzer = SentimentAnalyzer(LengthBasedScoring())
keyword_analyzer = SentimentAnalyzer(KeywordBasedScoring())

text = "This is a great and amazing product!"
print(length_analyzer.analyze(text)["sentiment"])   # positive
print(keyword_analyzer.analyze(text)["sentiment"])  # positive — "great" and "amazing" are keywords
```

### Step 5: Build Something Real — Composable Model Service

```python
# model_service.py

class Logger:
    def log(self, message: str) -> None:
        print(f"[{self.__class__.__name__}] {message}")

class ConsoleLogger(Logger):
    pass

class ModelBackend(ABC):
    @abstractmethod
    def generate(self, prompt: str) -> str:
        pass

class OpenAIBackend(ModelBackend):
    def generate(self, prompt: str) -> str:
        return f"[OpenAI] Response to: {prompt}"

class AnthropicBackend(ModelBackend):
    def generate(self, prompt: str) -> str:
        return f"[Anthropic] Response to: {prompt}"

class RetryHandler:
    def __init__(self, max_retries: int = 3):
        self.max_retries = max_retries

    def execute(self, fn, *args, **kwargs):
        last_error = None
        for attempt in range(self.max_retries):
            try:
                return fn(*args, **kwargs)
            except Exception as e:
                last_error = e
        raise RuntimeError(f"Failed after {self.max_retries} retries") from last_error

class ModelService:
    """Composed model service — has a backend, logger, and retry handler."""

    def __init__(
        self,
        backend: ModelBackend,
        logger: Logger | None = None,
        retry: RetryHandler | None = None,
    ):
        self.backend = backend
        self.logger = logger or ConsoleLogger()
        self.retry = retry or RetryHandler()

    def generate(self, prompt: str) -> str:
        self.logger.log(f"Generating for: {prompt[:50]}...")
        return self.retry.execute(self.backend.generate, prompt)

# Compose different configurations
service = ModelService(
    backend=OpenAIBackend(),
    logger=ConsoleLogger(),
    retry=RetryHandler(max_retries=5),
)
print(service.generate("Hello!"))
```
