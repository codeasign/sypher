# Text Justification — Solution

## Problem Overview

Format an array of words into fully justified lines of exactly `maxWidth` characters. This is a classic simulation problem testing precise rule-following.

## Brute Force Approach

The problem itself is already simulation-based. A naive approach might try all possible line breaks, but greedy packing per line is the required approach per the spec.

## Optimal Approach

Process words left to right, building one line at a time:

1. **Pack greedily**: Add words to the current line until adding the next word would exceed `maxWidth`
2. **Justify non-last lines**: Distribute spaces evenly between words. Left gaps get more spaces when uneven
3. **Handle last line**: Left-justify with single spaces between words, pad the rest with spaces

The simulation has three phases per line: pack, justify (if not last), and record.

## Deep Dive

The tricky parts of this problem are:
- **Space distribution**: For a line with `gaps = wordCount - 1` and `totalSpaces = maxWidth - sum(wordLengths)`, give `totalSpaces // gaps` to each gap, with the first `totalSpaces % gaps` gaps getting one extra
- **Single word on a line**: Left-justify (word + spaces)
- **Last line**: Left-justify with single spaces
- **Edge case**: A single word with maxWidth > word length — pad with spaces

The problem is marked "hard" because of the many edge cases and careful index management, not algorithmic complexity.

def fullJustify(words, maxWidth):
    result, line, line_len = [], [], 0
    for w in words:
        if line_len + len(line) + len(w) > maxWidth:
            if len(line) == 1:
                result.append(line[0] + ' ' * (maxWidth - line_len))
            else:
                spaces = maxWidth - line_len
                gaps = len(line) - 1
                for i in range(gaps):
                    line[i] += ' ' * (spaces // gaps + (1 if i < spaces % gaps else 0))
                result.append(''.join(line))
            line, line_len = [], 0
        line.append(w)
        line_len += len(w)
    result.append(' '.join(line) + ' ' * (maxWidth - line_len - len(line) + 1))
    return result

class Solution {
    public List<String> fullJustify(String[] words, int maxWidth) {
        List<String> result = new ArrayList<>();
        List<String> line = new ArrayList<>();
        int lineLen = 0;
        for (String w : words) {
            if (lineLen + line.size() + w.length() > maxWidth) {
                if (line.size() == 1) {
                    result.add(line.get(0) + " ".repeat(maxWidth - lineLen));
                } else {
                    int spaces = maxWidth - lineLen;
                    int gaps = line.size() - 1;
                    for (int i = 0; i < gaps; i++) {
                        line.set(i, line.get(i) + " ".repeat(
                            spaces / gaps + (i < spaces % gaps ? 1 : 0)));
                    }
                    result.add(String.join("", line));
                }
                line.clear(); lineLen = 0;
            }
            line.add(w);
            lineLen += w.length();
        }
        String last = String.join(" ", line);
        result.add(last + " ".repeat(maxWidth - last.length()));
        return result;
    }
}

class Solution {
public:
    vector<string> fullJustify(vector<string>& words, int maxWidth) {
        vector<string> result;
        vector<string> line;
        int lineLen = 0;
        for (const string& w : words) {
            if (lineLen + (int)line.size() + (int)w.length() > maxWidth) {
                if (line.size() == 1) {
                    result.push_back(line[0] + string(maxWidth - lineLen, ' '));
                } else {
                    int spaces = maxWidth - lineLen;
                    int gaps = line.size() - 1;
                    for (int i = 0; i < gaps; i++) {
                        line[i] += string(spaces / gaps + (i < spaces % gaps ? 1 : 0), ' ');
                    }
                    string joined;
                    for (const string& s : line) joined += s;
                    result.push_back(joined);
                }
                line.clear(); lineLen = 0;
            }
            line.push_back(w);
            lineLen += w.length();
        }
        string last;
        for (int i = 0; i < (int)line.size(); i++) {
            if (i > 0) last += " ";
            last += line[i];
        }
        last += string(maxWidth - last.length(), ' ');
        result.push_back(last);
        return result;
    }
};

fn full_justify(words: Vec<String>, max_width: i32) -> Vec<String> {
    let mut result: Vec<String> = Vec::new();
    let mut line: Vec<String> = Vec::new();
    let mut line_len = 0;
    let mw = max_width as usize;
    for w in words {
        if line_len + line.len() + w.len() > mw {
            if line.len() == 1 {
                result.push(format!("{}{}", line[0], " ".repeat(mw - line_len)));
            } else {
                let spaces = mw - line_len;
                let gaps = line.len() - 1;
                for i in 0..gaps {
                    let extra = spaces / gaps + if i < spaces % gaps { 1 } else { 0 };
                    line[i] = format!("{}{}", line[i], " ".repeat(extra));
                }
                result.push(line.concat());
            }
            line.clear(); line_len = 0;
        }
        line.push(w.clone());
        line_len += w.len();
    }
    let last = line.join(" ");
    result.push(format!("{}{}", last, " ".repeat(mw - last.len())));
    result
}

char** fullJustify(char** words, int wordsSize, int maxWidth, int* returnSize) {
    char** result = malloc(wordsSize * sizeof(char*));
    int ri = 0;
    for (int i = 0; i < wordsSize; ) {
        int lineLen = 0, lineCount = 0;
        int lineIdx[1000];
        while (i < wordsSize && lineLen + lineCount + (int)strlen(words[i]) <= maxWidth) {
            lineIdx[lineCount++] = i;
            lineLen += strlen(words[i]);
            i++;
        }
        char* row = calloc(maxWidth + 1, 1);
        if (lineCount == 1 || i == wordsSize) {
            int pos = 0;
            for (int j = 0; j < lineCount; j++) {
                if (j > 0) row[pos++] = ' ';
                strcpy(row + pos, words[lineIdx[j]]);
                pos += strlen(words[lineIdx[j]]);
            }
            memset(row + pos, ' ', maxWidth - pos);
        } else {
            int spaces = maxWidth - lineLen;
            int gaps = lineCount - 1, pos = 0;
            for (int j = 0; j < lineCount; j++) {
                strcpy(row + pos, words[lineIdx[j]]);
                pos += strlen(words[lineIdx[j]]);
                if (j < gaps) {
                    int n = spaces / gaps + (j < spaces % gaps ? 1 : 0);
                    memset(row + pos, ' ', n); pos += n;
                }
            }
        }
        row[maxWidth] = 0;
        result[ri++] = row;
    }
    *returnSize = ri;
    return result;
}

## Key Takeaways

- Greedy line packing is the required approach per the problem spec
- Space distribution math: `totalSpaces = maxWidth - sum(lengths)`, then allocate per gap
- Handle three cases: single word, multi-word non-last line, and last line
- The "hard" difficulty comes from edge case handling, not algorithmic complexity
