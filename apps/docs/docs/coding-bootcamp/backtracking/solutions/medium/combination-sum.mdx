# Combination Sum — Solution

## Problem Overview

Given distinct candidates and a target, find all combinations that sum to target, where each candidate may be reused unlimited times. Solved with **backtracking** — explore each candidate recursively, pruning when the remaining target goes negative.

## Approach: Backtracking with Pruning

**Idea:** Sort candidates. At each step, try adding candidate[i] and recurse with `remaining - candidates[i]`, starting from index `i` (not `i+1`) to allow reuse. When `remaining == 0`, record the combination. Since the array is sorted, once `candidates[i] > remaining` we can break early.

**Algorithm:**
1. Sort `candidates`.
2. `backtrack(remaining, start, path)`:
   - If `remaining == 0`: append copy of `path` to results.
   - For `i` from `start` to end: if `candidates[i] > remaining`, break. Else push, recurse with `remaining - candidates[i]` and index `i`, then pop.
3. Return results.

**Complexity:**
- **Time:** O(n^(t/m)) where t = target, m = min candidate value.
- **Space:** O(t/m) recursion depth.

def combination_sum(candidates: list[int], target: int) -> list[list[int]]:
    result = []
    candidates.sort()

    def backtrack(remaining: int, start: int, path: list[int]):
        if remaining == 0:
            result.append(path[:])
            return
        for i in range(start, len(candidates)):
            if candidates[i] > remaining:
                break
            path.append(candidates[i])
            backtrack(remaining - candidates[i], i, path)
            path.pop()

    backtrack(target, 0, [])
    return result

import java.util.*;

class Solution {
    public List<List<Integer>> combinationSum(int[] candidates, int target) {
        List<List<Integer>> result = new ArrayList<>();
        Arrays.sort(candidates);
        backtrack(candidates, target, 0, new ArrayList<>(), result);
        return result;
    }

    private void backtrack(int[] candidates, int remaining, int start,
                           List<Integer> path, List<List<Integer>> result) {
        if (remaining == 0) {
            result.add(new ArrayList<>(path));
            return;
        }
        for (int i = start; i < candidates.length; i++) {
            if (candidates[i] > remaining) break;
            path.add(candidates[i]);
            backtrack(candidates, remaining - candidates[i], i, path, result);
            path.remove(path.size() - 1);
        }
    }
}

pub fn combination_sum(mut candidates: Vec<i32>, target: i32) -> Vec<Vec<i32>> {

    let mut result = Vec::new();
    let mut path = Vec::new();

fn backtrack(candidates: &[i32], remaining: i32, start: usize,
             path: &mut Vec<i32>, result: &mut Vec<Vec<i32>>) {
    if remaining == 0 {
        result.push(path.clone());
        return;
    }
    for i in start..candidates.len() {
        if candidates[i] > remaining {
            break;
        }
        path.push(candidates[i]);
        backtrack(candidates, remaining - candidates[i], i, path, result);
        path.pop();
    }
}

#include <stdlib.h>
#include <string.h>

#define MAX_RESULTS 10000
#define MAX_DEPTH 50

static int results[MAX_RESULTS][MAX_DEPTH];
static int result_sizes[MAX_RESULTS];
static int result_count;

static int cmp(const void* a, const void* b) {
    return *(int*)a - *(int*)b;
}

static void backtrack(int* candidates, int candidatesSize, int remaining,
                      int start, int* path, int depth) {
    if (remaining == 0) {
        memcpy(results[result_count], path, depth * sizeof(int));
        result_sizes[result_count++] = depth;
        return;
    }
    for (int i = start; i < candidatesSize; i++) {
        if (candidates[i] > remaining) break;
        path[depth] = candidates[i];
        backtrack(candidates, candidatesSize, remaining - candidates[i],
                  i, path, depth + 1);
    }
}

int** combinationSum(int* candidates, int candidatesSize, int target,
                     int* returnSize, int** returnColumnSizes) {
    result_count = 0;
    qsort(candidates, candidatesSize, sizeof(int), cmp);
    int path[MAX_DEPTH];
    backtrack(candidates, candidatesSize, target, 0, path, 0);

    *returnSize = result_count;
    int** out = (int**)malloc(result_count * sizeof(int*));
    *returnColumnSizes = (int*)malloc(result_count * sizeof(int));
    for (int i = 0; i < result_count; i++) {
        (*returnColumnSizes)[i] = result_sizes[i];
        out[i] = (int*)malloc(result_sizes[i] * sizeof(int));
        memcpy(out[i], results[i], result_sizes[i] * sizeof(int));
    }
    return out;
}

import "sort"

func combinationSum(candidates []int, target int) [][]int {
    sort.Ints(candidates)
    result := [][]int{}
    var backtrack func(remaining, start int, path []int)
    backtrack = func(remaining, start int, path []int) {
        if remaining == 0 {

            copy(combo, path)
            result = append(result, combo)
            return
        }
        for i := start; i < len(candidates); i++ {
            if candidates[i] > remaining {
                break
            }
            backtrack(remaining-candidates[i], i, append(path, candidates[i]))
        }
    }
    backtrack(target, 0, []int{})
    return result
}

## Key Takeaways

- **Backtracking**: Try every candidate starting from index `i` (not `i+1`) to allow reuse of the same element.
- **Pruning**: Sort candidates first so you can break early when `candidates[i] > remaining`, avoiding unnecessary recursive calls.
- **Avoid duplicates**: By always starting from index `i` (not restarting from 0), each combination is generated exactly once in sorted order.
