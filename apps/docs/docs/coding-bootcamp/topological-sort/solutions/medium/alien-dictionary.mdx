# Alien Dictionary — Solution

## Problem Overview

Given a list of words sorted lexicographically according to the rules of an alien language, derive the order of characters in that language. Return the order as a string of unique characters. If the order is invalid or contains a cycle, return an empty string.

## Approach: Graph Construction + Topological Sort

**Idea:** Compare adjacent words to deduce character ordering relationships. Build a directed graph where an edge u -> v means u comes before  in the alien alphabet. Then perform topological sort to get the character order.

**Algorithm:**
1. Initialize a graph (adjacency list) and in-degree map for all unique characters.
2. For each pair of adjacent words (w1, w2):
   - Find the first position where characters differ.
   - If w1 is a prefix of w2, no ordering info is gained (continue).
   - If w2 is a prefix of w1 (longer word comes first), the input is invalid — return "".
   - Add an edge from w1[i] to w2[i] and increment in-degree of w2[i].
3. Perform Kahn's algorithm on the character graph.
4. If the result contains all unique characters, return it. Otherwise, return "".

**Why This Works:** The lexicographic ordering of words implies a partial order on characters. Adjacent word comparisons are sufficient because the ordering is transitive.

**Complexity Analysis:**
- **Time:** O(C) where C is the total number of characters across all words.
- **Space:** O(1) — at most 26 characters (English lowercase letters).

### Edge Cases

- **Single word**: All characters appear in any order, but the order must be consistent.
- **Prefix conflict**: If "abc" comes before "ab", the input is invalid (a longer word cannot be a prefix of a later word in sorted order).
- **Cycle**: If  < b and  < a from different pairs, the graph has a cycle.
- **Disconnected characters**: Characters with no ordering constraints can appear in any order.

```python
from collections import deque, defaultdict

def alien_order(words: list[str]) -> str:
    # Build graph for all unique characters
    adj = defaultdict(set)
    indegree = defaultdict(int)

    # Initialize all characters
    for word in words:
        for ch in word:
            indegree[ch] = 0

    # Compare adjacent words to find character order
    for i in range(len(words) - 1):
        w1, w2 = words[i], words[i + 1]
        min_len = min(len(w1), len(w2))

        # Find first differing character
        for j in range(min_len):
            if w1[j] != w2[j]:
                if w2[j] not in adj[w1[j]]:
                    adj[w1[j]].add(w2[j])
                    indegree[w2[j]] += 1
                break
        else:
            # Prefix rule: if w2 is a prefix of w1, invalid
            if len(w1) > len(w2):
                return ""

    # Kahn's algorithm
    q = deque([ch for ch in indegree if indegree[ch] == 0])
    result = []

    while q:
        ch = q.popleft()
        result.append(ch)
        for neighbor in adj[ch]:
            indegree[neighbor] -= 1
            if indegree[neighbor] == 0:
                q.append(neighbor)

    return "".join(result) if len(result) == len(indegree) else ""
```

```java
import java.util.*;

class Solution {
    public String alienOrder(String[] words) {
        // Build graph for all unique characters
        Map<Character, Set<Character>> adj = new HashMap<>();
        Map<Character, Integer> indegree = new HashMap<>();

        // Initialize all characters
        for (String word : words) {
            for (char ch : word.toCharArray()) {
                adj.putIfAbsent(ch, new HashSet<>());
                indegree.putIfAbsent(ch, 0);
            }
        }

        // Compare adjacent words to find character order
        for (int i = 0; i < words.length - 1; i++) {
            String w1 = words[i], w2 = words[i + 1];
            int minLen = Math.min(w1.length(), w2.length());
            int j = 0;

            while (j < minLen && w1.charAt(j) == w2.charAt(j)) {
                j++;
            }

            if (j < minLen) {
                char from = w1.charAt(j), to = w2.charAt(j);
                if (!adj.get(from).contains(to)) {
                    adj.get(from).add(to);
                    indegree.put(to, indegree.get(to) + 1);
                }
            } else if (w1.length() > w2.length()) {
                // Prefix rule: longer word comes first → invalid
                return "";
            }
        }

        // Kahn's algorithm
        Queue<Character> q = new LinkedList<>();
        for (char ch : indegree.keySet()) {
            if (indegree.get(ch) == 0) q.offer(ch);
        }

        StringBuilder result = new StringBuilder();
        while (!q.isEmpty()) {
            char ch = q.poll();
            result.append(ch);
            for (char neighbor : adj.get(ch)) {
                indegree.put(neighbor, indegree.get(neighbor) - 1);
                if (indegree.get(neighbor) == 0) q.offer(neighbor);
            }
        }

        return result.length() == indegree.size() ? result.toString() : "";
    }
}
```

```cpp
class Solution {
public:
    string alienOrder(vector<string>& words) {
        // Build graph for all unique characters
        unordered_map<char, unordered_set<char>> adj;
        unordered_map<char, int> indegree;

        // Initialize all characters
        for (const string& word : words) {
            for (char ch : word) {
                if (indegree.count(ch) == 0) indegree[ch] = 0;
            }
        }

        // Compare adjacent words to find character order
        for (int i = 0; i < (int)words.size() - 1; i++) {
            string& w1 = words[i], w2 = words[i + 1];
            int minLen = min((int)w1.length(), (int)w2.length());
            int j = 0;

            while (j < minLen && w1[j] == w2[j]) j++;

            if (j < minLen) {
                char from = w1[j], to = w2[j];
                if (adj[from].count(to) == 0) {
                    adj[from].insert(to);
                    indegree[to]++;
                }
            } else if (w1.length() > w2.length()) {
                return "";
            }
        }

        // Kahn's algorithm
        queue<char> q;
        for (auto& entry : indegree) {
            if (entry.second == 0) q.push(entry.first);
        }

        string result;
        while (!q.empty()) {
            char ch = q.front(); q.pop();
            result += ch;
            for (char neighbor : adj[ch]) {
                indegree[neighbor]--;
                if (indegree[neighbor] == 0) q.push(neighbor);
            }
        }

        return result.length() == indegree.size() ? result : "";
    }
};
```

```javascript
    const adj = new Map();
    const indegree = new Map();

    // Initialize all characters
    for (const word of words) {
        for (const ch of word) {
            if (!adj.has(ch)) adj.set(ch, new Set());
            if (!indegree.has(ch)) indegree.set(ch, 0);
        }
    }

    // Compare adjacent words to find character order
    for (let i = 0; i < words.length - 1; i++) {
        const w1 = words[i], w2 = words[i + 1];
        const minLen = Math.min(w1.length, w2.length);
        let j = 0;

        while (j < minLen && w1[j] === w2[j]) j++;

        if (j < minLen) {
            const from = w1[j], to = w2[j];
            if (!adj.get(from).has(to)) {
                adj.get(from).add(to);
                indegree.set(to, indegree.get(to) + 1);
            }
        } else if (w1.length > w2.length) {
            return "";
        }

    // Kahn's algorithm
    const q = [];
    for (const [ch, deg] of indegree) {
        if (deg === 0) q.push(ch);

            indegree.set(neighbor, indegree.get(neighbor) - 1);
            if (indegree.get(neighbor) === 0) q.push(neighbor);
```

```rust
use std::collections::{HashMap, HashSet, VecDeque};

fn alien_order(words: Vec<String>) -> String {
    // Build graph for all unique characters
    let mut adj: HashMap<char, Vec<char>> = HashMap::new();
    let mut indegree: HashMap<char, i32> = HashMap::new();

    // Initialize all characters
    for word in &words {
        for ch in word.chars() {
            adj.entry(ch).or_insert(vec![]);
            indegree.entry(ch).or_insert(0);
        }
    }

    // Compare adjacent words to find character order
    for i in 0..words.len() - 1 {
        let w1 = &words[i];
        let w2 = &words[i + 1];
        let min_len = w1.len().min(w2.len());
        let mut j = 0;

        let bytes1 = w1.as_bytes();
        let bytes2 = w2.as_bytes();

        while j < min_len && bytes1[j] == bytes2[j] {
            j += 1;
        }

        if j < min_len {
            let from = bytes1[j] as char;
            let to = bytes2[j] as char;
            if !adj[&from].contains(&to) {
                adj.get_mut(&from).unwrap().push(to);
                *indegree.get_mut(&to).unwrap() += 1;
            }
        } else if w1.len() > w2.len() {
            return String::new();
        }
    }

    // Kahn's algorithm
    let mut q = VecDeque::new();
    for (&ch, &deg) in &indegree {
        if deg == 0 {
            q.push_back(ch);
        }
    }

    let mut result = String::new();
    while let Some(ch) = q.pop_front() {
        result.push(ch);
        if let Some(neighbors) = adj.get(&ch) {
            for &neighbor in neighbors {
                *indegree.get_mut(&neighbor).unwrap() -= 1;
                if indegree[&neighbor] == 0 {
                    q.push_back(neighbor);
                }
            }
        }
    }

    if result.len() == indegree.len() { result } else { String::new() }
}
```

```c
#include <stdlib.h>
#include <string.h>
#include <stdbool.h>

#define MAX_CHARS 26

char* alienOrder(char** words, int wordsSize) {
    // Build graph for all unique characters
    bool adj[MAX_CHARS][MAX_CHARS] = {false};
    int indegree[MAX_CHARS];
    bool present[MAX_CHARS] = {false};
    int totalChars = 0;

    for (int i = 0; i < MAX_CHARS; i++) indegree[i] = 0;

    // Mark all characters that appear
    for (int i = 0; i < wordsSize; i++) {
        for (int j = 0; words[i][j]; j++) {
            int idx = words[i][j] - 'a';
            if (!present[idx]) {
                present[idx] = true;
                totalChars++;
            }
        }
    }

    // Compare adjacent words
    for (int i = 0; i < wordsSize - 1; i++) {
        char* w1 = words[i];
        char* w2 = words[i + 1];
        int minLen = strlen(w1) < strlen(w2) ? strlen(w1) : strlen(w2);
        int j = 0;

        while (j < minLen && w1[j] == w2[j]) j++;

        if (j < minLen) {
            int from = w1[j] - 'a', to = w2[j] - 'a';
            if (!adj[from][to]) {
                adj[from][to] = true;
                indegree[to]++;
            }
        } else if (strlen(w1) > strlen(w2)) {
            return NULL;
        }
    }

    // Kahn's algorithm
    int q[MAX_CHARS], front = 0, rear = 0;
    for (int i = 0; i < MAX_CHARS; i++) {
        if (present[i] && indegree[i] == 0) q[rear++] = i;
    }

    char* result = (char*)malloc((totalChars + 1) * sizeof(char));
    int idx = 0;

    while (front < rear) {
        int ch = q[front++];
        result[idx++] = ch + 'a';
        for (int neighbor = 0; neighbor < MAX_CHARS; neighbor++) {
            if (adj[ch][neighbor]) {
                indegree[neighbor]--;
                if (indegree[neighbor] == 0) q[rear++] = neighbor;
            }
        }
    }

    if (idx == totalChars) {
        result[idx] = '\0';
        return result;
    }

    free(result);
    return NULL;
}
```

```csharp
using System.Collections.Generic;

class Solution {
    public string AlienOrder(string[] words) {
        // Build graph for all unique characters
        Dictionary<char, HashSet<char>> adj = new Dictionary<char, HashSet<char>>();
        Dictionary<char, int> indegree = new Dictionary<char, int>();

        // Initialize all characters
        foreach (string word in words) {
            foreach (char ch in word) {
                if (!adj.ContainsKey(ch)) adj[ch] = new HashSet<char>();
                if (!indegree.ContainsKey(ch)) indegree[ch] = 0;
            }
        }

        // Compare adjacent words to find character order
        for (int i = 0; i < words.Length - 1; i++) {
            string w1 = words[i], w2 = words[i + 1];
            int minLen = Math.Min(w1.Length, w2.Length);
            int j = 0;

            while (j < minLen && w1[j] == w2[j]) j++;

            if (j < minLen) {
                char from = w1[j], to = w2[j];
                if (!adj[from].Contains(to)) {
                    adj[from].Add(to);
                    indegree[to]++;
                }
            } else if (w1.Length > w2.Length) {
                return "";
            }
        }

        // Kahn's algorithm
        Queue<char> q = new Queue<char>();
        foreach (var kvp in indegree) {
            if (kvp.Value == 0) q.Enqueue(kvp.Key);
        }

        List<char> result = new List<char>();
        while (q.Count > 0) {
            char ch = q.Dequeue();
            result.Add(ch);
            foreach (char neighbor in adj[ch]) {
                indegree[neighbor]--;
                if (indegree[neighbor] == 0) q.Enqueue(neighbor);
            }
        }

        return result.Count == indegree.Count ? new string(result.ToArray()) : "";
    }
}
```

```go
func alienOrder(words []string) string {
    // Build graph for all unique characters
    adj := make(map[byte][]byte)
    indegree := make(map[byte]int)

    // Initialize all characters
    for _, word := range words {
        for i := 0; i < len(word); i++ {
            ch := word[i]
            if _, ok := adj[ch]; !ok {
                adj[ch] = []byte{}
                indegree[ch] = 0
            }
        }
    }

    // Compare adjacent words to find character order
    for i := 0; i < len(words)-1; i++ {
        w1, w2 := words[i], words[i+1]
        minLen := len(w1)
        if len(w2) < minLen {
            minLen = len(w2)
        }
        j := 0
        for j < minLen && w1[j] == w2[j] {
            j++
        }

        if j < minLen {
            from, to := w1[j], w2[j]
            found := false
            for _, neighbor := range adj[from] {
                if neighbor == to {
                    found = true
                    break
                }
            }
            if !found {
                adj[from] = append(adj[from], to)
                indegree[to]++
            }
        } else if len(w1) > len(w2) {
            return ""
        }
    }

    // Kahn's algorithm
    q := make([]byte, 0)
    for ch, deg := range indegree {
        if deg == 0 {
            q = append(q, ch)
        }
    }

    result := make([]byte, 0)
    for len(q) > 0 {
        ch := q[0]
        q = q[1:]
        result = append(result, ch)
        for _, neighbor := range adj[ch] {
            indegree[neighbor]--
            if indegree[neighbor] == 0 {
                q = append(q, neighbor)
            }
        }
    }

    if len(result) == len(indegree) {
        return string(result)
    }
    return ""
}
```

## Key Takeaways

- **Adjacent word comparison** is sufficient to extract all ordering constraints — the transitive property ensures completeness.
- The **prefix rule** is a critical edge case: if word A is a prefix of word B, A must come before B in sorted order.
- The graph has at most **26 nodes** (lowercase English letters), making this problem efficient regardless of input size.
- Kahn's algorithm with character-level in-degree tracking cleanly handles the topological sort.
