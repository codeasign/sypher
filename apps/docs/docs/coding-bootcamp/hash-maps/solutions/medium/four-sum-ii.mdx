# 4Sum II — Solution

## Problem Overview

Given four integer arrays `nums1`, `nums2`, `nums3`, and `nums4`, count the number of tuples `(i, j, k, l)` such that `nums1[i] + nums2[j] + nums3[k] + nums4[l] == 0`.

## Approach: Hash Map of Pair Sums

**Idea:** Instead of checking all O(n⁴) combinations, break the problem into two halves. First, compute all sums of pairs from `nums1` and `nums2` and store their frequencies in a hash map. Then, for each pair from `nums3` and `nums4`, check if the complement (negative of their sum) exists in the hash map.

**Algorithm:**
1. Initialize a hash map `pair_sums` to store the frequency of each sum from `nums1` and `nums2`
2. For each `a` in `nums1` and `b` in `nums2`, increment `pair_sums[a + b]`
3. Initialize `count = 0`
4. For each `c` in `nums3` and `d` in `nums4`:
   - Let complement = `-(c + d)`
   - If complement exists in `pair_sums`, add its frequency to `count`
5. Return `count`

**Why This Works:** If `a + b + c + d = 0`, then `a + b = -(c + d)`. By pre-computing all sums of the first two arrays, we can find matching tuples from the last two arrays in O(1) time per pair.

**Complexity Analysis:**
- **Time:** O(n²) — we iterate through all pairs of nums1/nums2 and nums3/nums4
- **Space:** O(n²) — the hash map stores up to n² distinct sums

def four_sum_ii(nums1: list, nums2: list, nums3: list, nums4: list) -> int:
    # Count all sums of pairs from nums1 and nums2
    pair_sums = {}
    for a in nums1:
        for b in nums2:
            s = a + b
            pair_sums[s] = pair_sums.get(s, 0) + 1

    # Count tuples where complement exists
    count = 0
    for c in nums3:
        for d in nums4:
            complement = -(c + d)
            if complement in pair_sums:
                count += pair_sums[complement]

    return count

class Solution {
    public int fourSumCount(int[] nums1, int[] nums2, int[] nums3, int[] nums4) {
        Map<Integer, Integer> pairSums = new HashMap<>();

        // Count all sums of pairs from nums1 and nums2
        for (int a : nums1) {
            for (int b : nums2) {
                int sum = a + b;
                pairSums.put(sum, pairSums.getOrDefault(sum, 0) + 1);
            }
        }

        // Count tuples where complement exists
        int count = 0;
        for (int c : nums3) {
            for (int d : nums4) {
                int complement = -(c + d);
                count += pairSums.getOrDefault(complement, 0);
            }
        }

        return count;
    }
}

fn four_sum_count(nums1: &Vec<i32>, nums2: &Vec<i32>, nums3: &Vec<i32>, nums4: &Vec<i32>) -> i32 {
    let mut pair_sums: HashMap<i32, i32> = HashMap::new();

    // Count all sums of pairs from nums1 and nums2
    for a in nums1 {
        for b in nums2 {
            *pair_sums.entry(a + b).or_insert(0) += 1;

    for c in nums3 {
        for d in nums4 {
            let complement = -(c + d);
            count += pair_sums.get(&complement).unwrap_or(&0);
        }
    }

    count

#include <stdlib.h>

// Simple hash map for C using a table
#define TABLE_SIZE 1000003

typedef struct Entry {
    int key;
    int value;
    struct Entry* next;
} Entry;

int hash(int key) {
    return abs(key) % TABLE_SIZE;
}

int fourSumCount(int* nums1, int n1, int* nums2, int n2, int* nums3, int n3, int* nums4, int n4) {
    Entry* table[TABLE_SIZE] = {0};

    // Count all sums of pairs from nums1 and nums2
    for (int i = 0; i < n1; i++) {
        for (int j = 0; j < n2; j++) {
            int sum = nums1[i] + nums2[j];
            int h = hash(sum);
            Entry* entry = table[h];
            while (entry) {
                if (entry->key == sum) {
                    entry->value++;
                    break;
                }
                entry = entry->next;
            }
            if (!entry) {
                entry = malloc(sizeof(Entry));
                entry->key = sum;
                entry->value = 1;
                entry->next = table[h];
                table[h] = entry;
            }
        }
    }

    // Count tuples where complement exists
    int count = 0;
    for (int i = 0; i < n3; i++) {
        for (int j = 0; j < n4; j++) {
            int complement = -(nums3[i] + nums4[j]);
            int h = hash(complement);
            Entry* entry = table[h];
            while (entry) {
                if (entry->key == complement) {
                    count += entry->value;
                    break;
                }
                entry = entry->next;
            }
        }
    }

    // Free memory
    for (int i = 0; i < TABLE_SIZE; i++) {
        Entry* entry = table[i];
        while (entry) {
            Entry* next = entry->next;
            free(entry);
            entry = next;
        }
    }

    return count;
}

class Solution {
    public int FourSumCount(int[] nums1, int[] nums2, int[] nums3, int[] nums4) {
        var pairSums = new Dictionary<int, int>();

        // Count all sums of pairs from nums1 and nums2
        foreach (int a in nums1) {
            foreach (int b in nums2) {
                int sum = a + b;
                pairSums.TryGetValue(sum, out int val);
                pairSums[sum] = val + 1;
            }

        int count = 0;
        foreach (int c in nums3) {
            foreach (int d in nums4) {
                int complement = -(c + d);
                pairSums.TryGetValue(complement, out int val);
                count += val;
            }
        }

        return count;
    }
}

## Key Takeaways

- **Break the problem in half** — O(n²) / O(n²) beats O(n⁴) by a huge margin
- **Hash maps** provide O(1) frequency lookup for pair sums
- The same technique extends to k-sum problems by dividing into two groups
