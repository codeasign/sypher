## Tier 1 -- Beginner

### Exercise 1: Add a Third Model Tier

**Task:** Modify the smart router to use three tiers instead of two. Questions with complexity 1-3 go to the lightest model (GPT-4o-mini), 4-6 go to a mid-tier model (GPT-4o), and 7-10 go to the heaviest model (GPT-4o with longer output). Track the cost for each tier.

**Hint:** Create three answer functions and adjust the complexity threshold logic to use two cutoff values instead of one.

### Solution

```python
import os
import re
from openai import OpenAI
from dotenv import load_dotenv

load_dotenv()
client = OpenAI(api_key=os.environ["OPENAI_API_KEY"])

SYSTEM_PROMPT_CLASSIFIER = """
You are a query complexity classifier. Analyze the user's question and
return a single integer between 1 and 10 representing its complexity.

1-3: Simple factual questions, basic math, greetings
4-6: Moderate complexity, requires some explanation
7-10: Complex reasoning, analysis, creative work

Return ONLY the integer.
"""

LIGHT_MODEL = "gpt-4o-mini"
MID_MODEL = "gpt-4o"
HEAVY_MODEL = "gpt-4o"

def classify_complexity(question):
    response = client.chat.completions.create(
        model="gpt-4o-mini",
        messages=[
            {"role": "system", "content": SYSTEM_PROMPT_CLASSIFIER},
            {"role": "user", "content": question}
        ],
        temperature=0.0,
        max_tokens=5
    )
    text = response.choices[0].message.content.strip()
    match = re.search(r'\d+', text)
    return int(match.group()) if match else 5

def answer_light(question):
    response = client.chat.completions.create(
        model=LIGHT_MODEL,
        messages=[
            {"role": "system", "content": "Answer in one sentence."},
            {"role": "user", "content": question}
        ],
        max_tokens=100
    )
    return response.choices[0].message.content

def answer_mid(question):
    response = client.chat.completions.create(
        model=MID_MODEL,
        messages=[
            {"role": "system", "content": "Provide a concise but complete answer."},
            {"role": "user", "content": question}
        ],
        max_tokens=500
    )
    return response.choices[0].message.content

def answer_heavy(question):
    response = client.chat.completions.create(
        model=HEAVY_MODEL,
        messages=[
            {"role": "system", "content": "Provide a thorough, detailed answer with examples."},
            {"role": "user", "content": question}
        ],
        max_tokens=2000
    )
    return response.choices[0].message.content

def route_three_tier(question):
    score = classify_complexity(question)
    if score <= 3:
        return answer_light(question), LIGHT_MODEL, score
    elif score <= 6:
        return answer_mid(question), MID_MODEL, score
    else:
        return answer_heavy(question), HEAVY_MODEL, score

# Test
tests = [
    "What is the capital of Japan?",
    "Explain how a battery works.",
    "Design a fault-tolerant distributed database."
]
for t in tests:
    answer, model, score = route_three_tier(t)
    print(f"Score: {score} | Model: {model}")
    print(f"Answer: {answer[:80]}...")
    print()
```

Expected output:

```
Score: 1 | Model: gpt-4o-mini | Answer: The capital of Japan is Tokyo...
Score: 5 | Model: gpt-4o | Answer: A battery works by converting chemical energy...
Score: 9 | Model: gpt-4o | Answer: A fault-tolerant distributed database requires...
```

### Exercise 3: Swap Light Model Provider

**Task:** Replace GPT-4o-mini with a different light model provider. Use OpenRouter to route classification tasks to `meta-llama/llama-3.2-3b-instruct` instead of GPT-4o-mini. Compare the classification accuracy across 10 test questions.

**Hint:** You need an OpenRouter API key. Set `OPENROUTER_API_KEY` in your `.env` file. Use the OpenRouter API endpoint with the model string `meta-llama/llama-3.2-3b-instruct`.

### Solution

```python
import os
import requests
from dotenv import load_dotenv
import re

load_dotenv()

OPENROUTER_API_URL = "https://openrouter.ai/api/v1/chat/completions"
headers = {
    "Authorization": f"Bearer {os.environ['OPENROUTER_API_KEY']}",
    "Content-Type": "application/json"
}

def classify_with_llama(question):
    payload = {
        "model": "meta-llama/llama-3.2-3b-instruct",
        "messages": [
            {"role": "system", "content": "Classify query complexity 1-10. Return only the number."},
            {"role": "user", "content": question}
        ],
        "temperature": 0.0,
        "max_tokens": 5
    }
    response = requests.post(OPENROUTER_API_URL, headers=headers, json=payload)
    data = response.json()
    text = data["choices"][0]["message"]["content"].strip()
    match = re.search(r'\d+', text)
    return int(match.group()) if match else 5

# Test
questions = [
    "What is the capital of France?",
    "Write a recursive descent parser for JSON in Python.",
    "Explain neural networks."
]
for q in questions:
    score = classify_with_llama(q)
    print(f"Score: {score} | {q[:50]}")
```

Expected output:

```

Score: 8 | Write a recursive descent parser for JSON in Python.
Score: 6 | Explain neural networks.
```

## Tier 2 -- Intermediate

### Exercise 4: Combine with Streaming Responses

**Task:** Modify the router so that both the light and heavy models stream their responses instead of returning them all at once. The router should display each chunk as it arrives and track time-to-first-token (TTFT) for each model.

**Hint:** Use `stream=True` in the OpenAI API call. Capture the time before the first chunk and the time it arrives.

### Solution

```python
import os
import time
from openai import OpenAI
from dotenv import load_dotenv

load_dotenv()
client = OpenAI(api_key=os.environ["OPENAI_API_KEY"])

def stream_answer(model, question, system_prompt):
    start = time.time()
    first_token_time = None
    collected = []
    
    stream = client.chat.completions.create(
        model=model,
        messages=[
            {"role": "system", "content": system_prompt},
            {"role": "user", "content": question}
        ],
        stream=True, max_tokens=500
    )
    
    for chunk in stream:
        if first_token_time is None and chunk.choices[0].delta.content:
            first_token_time = time.time() - start
        content = chunk.choices[0].delta.content or ""
        collected.append(content)
        print(content, end="", flush=True)
    
    total_time = time.time() - start
    return {"answer": "".join(collected), "ttft": first_token_time, "total_time": total_time}

q = "Explain what a neural network is."
print("=== Light model (streaming) ===")
light_result = stream_answer("gpt-4o-mini", q, "Answer concisely.")
print(f"\nTTFT: {light_result['ttft']:.3f}s | Total: {light_result['total_time']:.3f}s\n")

print("=== Heavy model (streaming) ===")
heavy_result = stream_answer("gpt-4o", q, "Answer thoroughly.")
print(f"\nTTFT: {heavy_result['ttft']:.3f}s | Total: {heavy_result['total_time']:.3f}s")
```

Expected output (timings will vary):

```
=== Light model (streaming) ===
A neural network is a computing system inspired by the human brain...
TTFT: 0.210s | Total: 0.890s

=== Heavy model (streaming) ===
A neural network is a computational model inspired by the structure of biological neural networks...
TTFT: 0.480s | Total: 2.340s
```

The light model starts streaming faster (lower TTFT) and finishes sooner. The heavy model takes longer to start but produces more detailed content.

### Exercise 5: Add a Local Model Fallback

**Task:** Extend the router to use a local model as a fallback when the API is unreachable or returns an error. Install Ollama, pull the Llama 3.2 3B model, and modify your router so that if the OpenAI API call fails (timeout, connection error, rate limit), it falls back to the local Ollama model.

**Hint:** Install Ollama, then run `ollama pull llama3.2:3b`. Use the Ollama API at `http://localhost:11434/api/generate`.

### Solution

```python
import os
import requests
from openai import OpenAI
from dotenv import load_dotenv
import re

load_dotenv()
client = OpenAI(api_key=os.environ["OPENAI_API_KEY"])
OLLAMA_API = "http://localhost:11434/api/generate"

def answer_with_fallback(question, use_light=True):
    # Try OpenAI first
    try:
        model = "gpt-4o-mini" if use_light else "gpt-4o"
        response = client.chat.completions.create(
            model=model,
            messages=[
                {"role": "system", "content": "You are a helpful assistant."},
                {"role": "user", "content": question}
            ],
            max_tokens=500,
            timeout=10
        )
        return response.choices[0].message.content, f"openai/{model}"
    
    except Exception as e:
        print(f"OpenAI failed ({type(e).__name__}), falling back to local model...")
        # Fall back to Ollama
        payload = {
            "model": "llama3.2:3b",
            "prompt": f"Answer this question: {question}",
            "stream": False,
            "options": {"num_predict": 500}
        }
        response = requests.post(OLLAMA_API, json=payload, timeout=30)
        data = response.json()
        return data["response"], "ollama/llama3.2:3b"

# Test
q = "What is the capital of Australia?"
answer, source = answer_with_fallback(q)
print(f"Source: {source}")
print(f"Answer: {answer}")
```

To test the fallback, temporarily disconnect from the internet or use an invalid API key. The router should switch to the local model and still return a reasonable answer.

### Exercise 6: Keyword-Based Pre-Router

**Task:** Before calling the classifier, add a keyword-based pre-router that catches obvious simple queries without using any API credits. If the question starts with common patterns like "what is", "who is", "define", or consists of fewer than five words, send it directly to the light model without classification.

**Hint:** Use string methods like `startswith()` and `split()` to check the question. Track how many API calls you saved by comparing the count of pre-routed requests to total requests.

### Solution

```python
import os
from openai import OpenAI
from dotenv import load_dotenv

load_dotenv()
client = OpenAI(api_key=os.environ["OPENAI_API_KEY"])

SIMPLE_PREFIXES = ["what is", "who is", "what are", "who are", "define", "what does", "when is", "where is"]

def is_obviously_simple(question):
    """Check if a question is obviously simple without calling any API."""
    q = question.lower().strip()
    # Short questions are usually simple
    words = q.split()
    if len(words) <= 3:
        return True
    # Questions starting with known simple prefixes
    for prefix in SIMPLE_PREFIXES:
        if q.startswith(prefix):
            return True
    return False

def route_with_precheck(question):
    if is_obviously_simple(question):
        # Skip classifier, go straight to light model
        response = client.chat.completions.create(
            model="gpt-4o-mini",
            messages=[
                {"role": "system", "content": "Answer concisely."},
                {"role": "user", "content": question}
            ],
            max_tokens=100
        )
        return response.choices[0].message.content, "gpt-4o-mini (pre-routed)", 0
    
    # Fall through to normal routing
    from smart_router import route_and_answer  # reuse your existing router
    result = route_and_answer(question)
    return result["answer"], result["model_used"], result["complexity_score"]

# Test
tests = [
    "What is gravity?",
    "Design a distributed consensus algorithm.",
    "Who is Einstein?",
    "Hello",
    "Analyze the philosophical implications of determinism."
]
pre_routed = 0
for t in tests:
    answer, source, score = route_with_precheck(t)
    is_pre = "pre-routed" in source
    if is_pre:
        pre_routed += 1
    print(f"[{'PRE' if is_pre else 'CLASSIFIED'}] Score: {score} | Source: {source} | {t[:40]}")

print(f"\nPre-routed: {pre_routed}/{len(tests)} (saved {pre_routed} classifier API calls)")
```

Expected output:

```
[PRE] Score: 0 | Source: gpt-4o-mini (pre-routed) | What is gravity?
[CLASSIFIED] Score: 8 | Source: gpt-4o | Design a distributed consensus algorithm.
[PRE] Score: 0 | Source: gpt-4o-mini (pre-routed) | Who is Einstein?
[PRE] Score: 0 | Source: gpt-4o-mini (pre-routed) | Hello
[CLASSIFIED] Score: 7 | Source: gpt-4o | Analyze the philosophical implications...
Pre-routed: 3/5 (saved 3 classifier API calls)
```

## Tier 3 -- Advanced

### Exercise 7: Dynamic Threshold Tuning

**Task:** The fixed threshold of 5 in the practice exercise works for general questions, but different applications need different thresholds. Build a system that tunes the complexity threshold automatically. Run 100 sample queries through both the light and heavy models, compare the answer quality using an LLM-as-judge (use GPT-4o to rate each answer on a scale of 1 to 5), and find the optimal threshold that balances cost and quality for your specific use case.

There is no single correct answer for this exercise. The optimal threshold depends on your quality standards and budget.

### Solution Approach

```python
import os, json, re
from openai import OpenAI
from dotenv import load_dotenv

load_dotenv()
client = OpenAI(api_key=os.environ["OPENAI_API_KEY"])

def rate_quality(question, light_answer, heavy_answer):
    response = client.chat.completions.create(
        model="gpt-4o",
        messages=[{"role": "system", "content": "Rate each answer 1-5 for correctness, completeness, clarity. Return JSON: {\"light_score\": N, \"heavy_score\": N}"},
                  {"role": "user", "content": f"Question: {question}\n\nLight: {light_answer}\n\nHeavy: {heavy_answer}"}],
        temperature=0.0, response_format={"type": "json_object"}
    )
    result = json.loads(response.choices[0].message.content)
    return result["light_score"], result["heavy_score"]

def find_optimal_threshold(sample_questions):
    results = []
    for q in sample_questions:
        resp = client.chat.completions.create(
            model="gpt-4o-mini",
            messages=[{"role": "system", "content": "Return complexity 1-10 as integer only."}, {"role": "user", "content": q}],
            temperature=0.0, max_tokens=5
        )
        text = resp.choices[0].message.content.strip()
        match = re.search(r'\d+', text)
        score = int(match.group()) if match else 5
        
        light_resp = client.chat.completions.create(model="gpt-4o-mini", messages=[{"role": "user", "content": q}], max_tokens=200)
        heavy_resp = client.chat.completions.create(model="gpt-4o", messages=[{"role": "user", "content": q}], max_tokens=500)
        light_quality, heavy_quality = rate_quality(q, light_resp.choices[0].message.content, heavy_resp.choices[0].message.content)
        
        results.append({"complexity": score, "light_quality": light_quality, "heavy_quality": heavy_quality, "quality_gap": heavy_quality - light_quality})
        print(f"Complexity {score}: light={light_quality} heavy={heavy_quality} gap={heavy_quality - light_quality}")
    
    for threshold in range(2, 9):
        above = [r for r in results if r["complexity"] > threshold]
        below = [r for r in results if r["complexity"] <= threshold]
        if above and below:
            avg_gap_above = sum(r["quality_gap"] for r in above) / len(above)
            avg_gap_below = sum(r["quality_gap"] for r in below) / len(below)
            print(f"Threshold {threshold}: below={avg_gap_below:.2f} above={avg_gap_above:.2f}")
    return results
```

Run this with at least 20 diverse questions. The threshold where the average quality gap exceeds 1.0 is a good starting point for your production system.

### Exercise 8: Build a Confidence-Based Retry System

**Task:** Build a system where the light model provides a confidence score alongside its answer. If confidence is low, automatically retry with the heavy model. Ask the model to rate its confidence from 0.0 to 1.0 and set a minimum threshold of 0.7.

### Solution Approach

```python
import os, json
from openai import OpenAI
from dotenv import load_dotenv

load_dotenv()
client = OpenAI(api_key=os.environ["OPENAI_API_KEY"])

def answer_with_confidence(question, use_heavy=False):
    model = "gpt-4o" if use_heavy else "gpt-4o-mini"
    system = """Answer the question, then rate your confidence in this answer from 0.0 to 1.0.
Return JSON: {"answer": "...", "confidence": 0.0}"""
    response = client.chat.completions.create(
        model=model,
        messages=[{"role": "system", "content": system}, {"role": "user", "content": question}],
        temperature=0.0, response_format={"type": "json_object"}
    )
    return json.loads(response.choices[0].message.content)

def confident_router(question, min_confidence=0.7, max_retries=2):
    for attempt in range(max_retries + 1):
        use_heavy = attempt > 0
        result = answer_with_confidence(question, use_heavy)
        print(f"Attempt {attempt + 1} ({'heavy' if use_heavy else 'light'}): confidence={result['confidence']}")
        if result["confidence"] >= min_confidence:
            return result["answer"], use_heavy
        if attempt < max_retries:
            print("Confidence too low, retrying with heavy model...")
    return result["answer"], True

for q in ["What is the meaning of life?", "Solve for x: 2x + 5 = 13", "Is this grammatical: 'Me and him went to the store'?"]:
    print(f"\nQuestion: {q}")
    answer, used_heavy = confident_router(q)
    print(f"Final answer: {answer[:100]}...")
    print(f"Used heavy model: {used_heavy}")
    print("-" * 40)
```

### Exercise 9: Production Monitoring Dashboard

**Task:** Build a simple command-line dashboard that reads the routing log from the CSV file you created in Exercise 2 and displays real-time statistics: total requests, percentage routed to each model, average cost per request, total cost saved versus using only the heavy model, and average response time per model.

### Solution Approach

```python
import csv
import os
from collections import defaultdict

LOG_FILE = "routing_log.csv"

def show_dashboard():
    if not os.path.exists(LOG_FILE):
        print("No routing log found.")
        return
    total = 0
    model_counts = defaultdict(int)
    model_costs = defaultdict(float)
    model_times = defaultdict(list)
    total_cost = 0.0
    heavy_only_cost = 0.0
    
    with open(LOG_FILE, encoding="utf-8") as f:
        for row in csv.DictReader(f):
            total += 1
            model = row["model"]
            cost = float(row["cost_estimate"])
            elapsed = float(row["elapsed_seconds"])
            model_counts[model] += 1
            model_costs[model] += cost
            model_times[model].append(elapsed)
            total_cost += cost
            heavy_only_cost += cost * 15
    
    print("ROUTING DASHBOARD")
    for model in sorted(model_counts.keys()):
        count = model_counts[model]
        pct = (count / total) * 100
        avg_cost = model_costs[model] / count
        avg_time = sum(model_times[model]) / len(model_times[model])
        print(f"{model:20s} {count:4d} reqs ({pct:5.1f}%) cost=${avg_cost:.6f}/req avg_time={avg_time:.3f}s")
    print(f"Total cost (routed): ${total_cost:.4f}")
    print(f"Estimated cost (all heavy): ${heavy_only_cost:.4f}")
    print(f"Savings: ${heavy_only_cost - total_cost:.4f} ({((1 - total_cost/heavy_only_cost)*100):.1f}%)")

if __name__ == "__main__":
    show_dashboard()
```

Expected output:

```
ROUTING DASHBOARD
Total requests: 50
Model: gpt-4o           12 reqs (24.0%)  cost=$0.0085  avg_time=2.150s
Model: gpt-4o-mini      38 reqs (76.0%)  cost=$0.0018  avg_time=0.520s
Total cost (routed): $0.0103
Estimated cost (all heavy): $0.1545
Savings: $0.1442 (93.3%)
```

With 76 percent of requests handled by the light model, the system saves over 93 percent of the API cost. This is the real-world impact of smart model routing.
