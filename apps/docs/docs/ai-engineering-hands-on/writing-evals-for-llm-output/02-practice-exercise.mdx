### Step 2: Build a faithfulness eval using deepeval

Create a file called `faithfulness_eval.py`:

```python
"""
Faithfulness evaluation using deepeval's built-in metric.
"""

from deepeval.metrics import FaithfulnessMetric
from deepeval.test_case import LLMTestCase
from summaries import SOURCE_DOCUMENTS, GOOD_SUMMARIES, BAD_SUMMARIES, SOURCE_KEYS


def evaluate_faithfulness(
    summary: str,
    source_text: str,
    threshold: float = 0.7,
) -> dict:
    """
    Evaluate whether a summary is faithful to its source.

    Returns score and detailed results.
    """
    metric = FaithfulnessMetric(
        threshold=threshold,
        model="gpt-4o-mini",
        include_reason=True,
    )

    test_case = LLMTestCase(
        input="Summarize the following text",
        actual_output=summary,
        retrieval_context=[source_text],
    )

    metric.measure(test_case)
    return {
        "score": metric.score,
        "threshold": threshold,
        "passed": metric.is_successful(),
        "reason": metric.reason,
    }


def run_evaluation():
    """Run faithfulness eval on all summaries."""
    print("=== Faithfulness Evaluation ===\n")

    for i, key in enumerate(SOURCE_KEYS):
        source = SOURCE_DOCUMENTS[key]

        # Evaluate good summary
        good_result = evaluate_faithfulness(
            GOOD_SUMMARIES[i], source
        )
        print(f"Source: {key}")
        print(f"  Good summary score: {good_result['score']:.3f} "
              f"[{'PASS' if good_result['passed'] else 'FAIL'}]")
        print(f"  Reason: {good_result['reason'][:100]}...")

        # Evaluate bad summary
        bad_result = evaluate_faithfulness(
            BAD_SUMMARIES[i], source
        )
        print(f"  Bad summary score:  {bad_result['score']:.3f} "
              f"[{'PASS' if bad_result['passed'] else 'FAIL'}]")
        print(f"  Reason: {bad_result['reason'][:100]}...")
        print()


if __name__ == "__main__":
    run_evaluation()
```

### Step 3: Build a relevance eval

Create a file called `relevance_eval.py`:

```python
"""
Relevance evaluation using deepeval.
"""

from deepeval.metrics import AnswerRelevancyMetric
from deepeval.test_case import LLMTestCase
from summaries import SOURCE_DOCUMENTS, GOOD_SUMMARIES, BAD_SUMMARIES, SOURCE_KEYS


def evaluate_relevance(
    summary: str,
    prompt: str,
    threshold: float = 0.7,
) -> dict:
    """
    Evaluate whether a summary is relevant to the prompt.
    """
    metric = AnswerRelevancyMetric(
        threshold=threshold,
        model="gpt-4o-mini",
        include_reason=True,
    )

    test_case = LLMTestCase(
        input=prompt,
        actual_output=summary,
    )

    metric.measure(test_case)
    return {
        "score": metric.score,
        "threshold": threshold,
        "passed": metric.is_successful(),
        "reason": metric.reason,
    }


def run_relevance_eval():
    """Run relevance eval."""
    print("=== Relevance Evaluation ===\n")

    prompts = [
        "Summarize the Q3 earnings report for Acme Corp.",
        "Summarize the TechNova ShieldPro product launch.",
        "Summarize the Tau-12 protein research study.",

        good_result = evaluate_relevance(
            GOOD_SUMMARIES[i], prompt

        # Evaluate bad summary
        bad_result = evaluate_relevance(
            BAD_SUMMARIES[i], prompt
        )
        print(f"  Bad summary:  {bad_result['score']:.3f} "
              f"[{'PASS' if bad_result['passed'] else 'FAIL'}]")
        print()


if __name__ == "__main__":
```

### Step 4: Build a custom numeric-fact preservation eval

Create a file called `numeric_eval.py`:

```python
"""
Custom eval: check whether specific numerical facts from the source
are preserved in the summary.
"""

import re
from deepeval.metrics import BaseMetric
from deepeval.scorer import Scorer


class NumericFactPreservationMetric(BaseMetric):
    """
    Custom metric that checks whether numerical facts from the source
    are accurately preserved in the summary.
    """

    def __init__(self, threshold: float = 0.5):
        self.threshold = threshold
        self.score = 0.0

    def measure(self, test_case) -> float:
        source = test_case.retrieval_context[0]
        summary = test_case.actual_output

        # Extract numerical facts from source
        source_facts = self._extract_numeric_facts(source)
        if not source_facts:
            self.score = 1.0
            return self.score

        # Check each fact against the summary
        preserved = 0
        results = []
        for fact_type, value, unit in source_facts:
            if self._fact_in_summary(value, unit, summary):
                preserved += 1
                results.append(f"  ✓ {fact_type}: {value}{unit}")
            else:
                results.append(f"  ✗ {fact_type}: {value}{unit}")

        self.score = preserved / len(source_facts)

        if self.score < self.threshold:
            self.error = (
                f"Numeric preservation score {self.score:.2f} "
                f"below threshold {self.threshold}: "
                + "; ".join(r for r in results if "✗" in r)
            )

        return self.score

    def _extract_numeric_facts(self, text: str) -> list[tuple]:
        """
        Extract (description, numeric_value, unit) tuples from source.
        """
        facts = []
        patterns = [
            (r"(\d+\.?\d*)\s*%", "percentage"),
            (r"\$\s*(\d+\.?\d*)\s*(million|billion|M|B)", "dollar"),
            (r"(\d+)\s*percent", "percentage_text"),
            (r"(\d+)\s*per seat", "per_seat"),
        ]

        for pattern, fact_type in patterns:
            for match in re.finditer(pattern, text, re.IGNORECASE):
                value = match.group(1)
                facts.append((fact_type, float(value), match.group(0)))

        return facts

    def _fact_in_summary(
        self, value: float, fact_text: str, summary: str
    ) -> bool:
        """Check if a numerical fact appears in the summary."""
        # Check exact value
        if str(value) in summary:
            return True
        # Check the full fact text
        if fact_text in summary:
            return True
        return False

    def is_successful(self) -> bool:
        return self.score >= self.threshold

    @property
    def __name__(self):
        return "NumericFactPreservation"


def run_numeric_eval():
    """Run numeric fact preservation eval."""
    from deepeval.test_case import LLMTestCase
    from summaries import SOURCE_DOCUMENTS, GOOD_SUMMARIES, BAD_SUMMARIES, SOURCE_KEYS

    print("=== Numeric Fact Preservation Evaluation ===\n")
    metric = NumericFactPreservationMetric(threshold=0.5)

    for i, key in enumerate(SOURCE_KEYS):
        source = SOURCE_DOCUMENTS[key]

        # Evaluate good summary
        good_case = LLMTestCase(
            input="Summarize",
            actual_output=GOOD_SUMMARIES[i],
            retrieval_context=[source],
        )
        good_score = metric.measure(good_case)
        print(f"Source: {key}")
        print(f"  Good summary numeric preservation: {good_score:.3f} "
              f"[{'PASS' if good_score >= 0.5 else 'FAIL'}]")

        # Evaluate bad summary
        bad_case = LLMTestCase(
            input="Summarize",
            actual_output=BAD_SUMMARIES[i],
            retrieval_context=[source],
        )
        bad_score = metric.measure(bad_case)
        print(f"  Bad summary numeric preservation:  {bad_score:.3f} "
              f"[{'PASS' if bad_score >= 0.5 else 'FAIL'}]")
        print()


if __name__ == "__main__":
    run_numeric_eval()
```

### Step 5: Combine all evals into a single test suite

Create a file called `test_summary_quality.py`:

```python
"""
Comprehensive test suite combining all eval types.
"""

import pytest
from deepeval.metrics import FaithfulnessMetric, AnswerRelevancyMetric
from deepeval.test_case import LLMTestCase
from summaries import SOURCE_DOCUMENTS, GOOD_SUMMARIES, BAD_SUMMARIES, SOURCE_KEYS
from numeric_eval import NumericFactPreservationMetric

# Test data: (summary, source, expected_quality)
TEST_CASES = []
for i, key in enumerate(SOURCE_KEYS):
    source = SOURCE_DOCUMENTS[key]
    TEST_CASES.append(
        (GOOD_SUMMARIES[i], source, "good", key)
    )
    TEST_CASES.append(
        (BAD_SUMMARIES[i], source, "bad", key)
    )


@pytest.mark.parametrize(
    "summary, source, quality, key",
    TEST_CASES,
    ids=[f"{k}_{q}" for _, _, q, k in TEST_CASES],
)
def test_faithfulness(summary, source, quality, key):
    """All summaries must be faithful to source."""
    metric = FaithfulnessMetric(
        threshold=0.7, model="gpt-4o-mini"
    )
    test_case = LLMTestCase(
        input="Summarize",
        actual_output=summary,
        retrieval_context=[source],
    )
    metric.measure(test_case)

    if quality == "good":
        assert metric.is_successful(), (
            f"Good {key} summary failed faithfulness: "
            f"{metric.score:.2f}"
        )
    else:
        # Bad summaries should FAIL faithfulness
        assert not metric.is_successful(), (
            f"Bad {key} summary unexpectedly passed faithfulness: "
            f"{metric.score:.2f}"
        )


@pytest.mark.parametrize(
    "summary, source, quality, key",
    TEST_CASES,
    ids=[f"{k}_{q}" for _, _, q, k in TEST_CASES],
)
def test_numeric_facts(summary, source, quality, key):
    """Numeric facts must be preserved in summaries."""
    metric = NumericFactPreservationMetric(threshold=0.5)
    test_case = LLMTestCase(
        input="Summarize",
        actual_output=summary,
        retrieval_context=[source],
    )
    score = metric.measure(test_case)

    if quality == "good":
        assert score >= 0.5, (
            f"Good {key} summary lost numeric facts: {score:.2f}"
        )
    else:
        assert score < 0.5, (
            f"Bad {key} summary unexpectedly preserved facts: "
            f"{score:.2f}"
        )
```

## Common Errors at This Stage

### Error 1: deepeval not configured
```
deepeval.api: DeepEval is not configured. Set OPENAI_API_KEY environment variable.
```
**Cause:** deepeval needs an API key for its default evaluation model.
**Fix:** Set `OPENAI_API_KEY` environment variable, or configure a different model in deepeval settings.

### Error 2: LLMTestCase missing required field
```
TypeError: LLMTestCase.__init__() missing required argument 'retrieval_context'
```
**Cause:** The FaithfulnessMetric requires a retrieval_context with the source document.
**Fix:** Pass `retrieval_context=[source_text]` when creating the LLMTestCase.

### Error 3: Metric not imported
```
ModuleNotFoundError: No module named 'deepeval.metrics'
```
**Cause:** deepeval is not installed or the version is too old.
**Fix:** `pip install --upgrade deepeval`

### Error 4: Pytest passes everything including bad summaries
```
All 12 tests pass — including bad summaries
```
**Cause:** The threshold is too low or the eval metric is not sensitive enough.
**Fix:** Lower the threshold or switch to a more capable judge model (e.g., gpt-4o instead of gpt-4o-mini).
