## Tier 1 -- Beginner

### Exercise 1: Math Tool

**Task:** Define a tool called `calculate` that takes an `expression` parameter (string) and evaluates it using Python's `eval()`. Ask the LLM to compute `"What is 145 * 37 + 892?"` and print the result.

**Hint:** Your tool function should use `eval(expression)` and return the result as a string. Be aware that `eval` can be dangerous in production -- use `ast.literal_eval` or a safe math evaluator instead.

### Solution

```python
import os
from anthropic import Anthropic

client = Anthropic(api_key=os.environ["ANTHROPIC_API_KEY"])

TOOLS = [
    {
        "name": "calculate",
        "description": "Evaluate a mathematical expression and return the result. Accepts basic arithmetic: +, -, *, /",
        "input_schema": {
            "type": "object",
            "properties": {
                "expression": {
                    "type": "string",
                    "description": "The mathematical expression to evaluate, e.g. '145 * 37 + 892'"
                }
            },
            "required": ["expression"]
        }
    }
]

def calculate(expression: str) -> str:
    try:
        result = eval(expression)
        return f"{expression} = {result}"
    except Exception as e:
        return f"Error evaluating '{expression}': {str(e)}"

messages = [{"role": "user", "content": "What is 145 * 37 + 892?"}]

response = client.messages.create(
    model="claude-sonnet-4-20250514",
    max_tokens=1024,
    tools=TOOLS,
    messages=messages
)

for block in response.content:
    if block.type == "tool_use" and block.name == "calculate":
        result = calculate(**block.input)
        print(f"Computed: {result}")

        messages.append({"role": "assistant", "content": response.content})
        messages.append({"role": "user", "content": [
            {"type": "tool_result", "tool_use_id": block.id, "content": result}
        ]})

        final = client.messages.create(
            model="claude-sonnet-4-20250514",
            max_tokens=1024,
            tools=TOOLS,
            messages=messages
        )
        print(f"LLM: {final.content[0].text}")
```

### Exercise 2: Text Processing Tools

**Task:** Define two tools in a single request: `count_words` (takes a `text` string, returns the word count) and `reverse_text` (takes a `text` string, returns the reversed text). Ask the LLM to "Count the words in 'Hello world from my AI agent' and then reverse the text."

**Hint:** Pass both tool definitions in the `TOOLS` array. The LLM will call one or both depending on the request.

### Solution

```python
import os
from anthropic import Anthropic

client = Anthropic(api_key=os.environ["ANTHROPIC_API_KEY"])

TOOLS = [
    {
        "name": "count_words",
        "description": "Count the number of words in a given text string.",
        "input_schema": {
            "type": "object",
            "properties": {
                "text": {"type": "string", "description": "The text to count words in"}
            },
            "required": ["text"]
        }
    },
    {
        "name": "reverse_text",
        "description": "Reverse the characters of a given text string.",
        "input_schema": {
            "type": "object",
            "properties": {
                "text": {"type": "string", "description": "The text to reverse"}
            },
            "required": ["text"]
        }
    }
]

def count_words(text: str) -> str:
    return f"Word count: {len(text.split())}"

def reverse_text(text: str) -> str:
    return f"Reversed: {text[::-1]}"

messages = [{"role": "user", "content": "Count the words in 'Hello world from my AI agent' and then reverse the text."}]

response = client.messages.create(
    model="claude-sonnet-4-20250514",
    max_tokens=1024,
    tools=TOOLS,
    messages=messages
)

tool_results = []
for block in response.content:
    if block.type == "tool_use":
        if block.name == "count_words":
            result = count_words(**block.input)
        elif block.name == "reverse_text":
            result = reverse_text(**block.input)
        tool_results.append({"type": "tool_result", "tool_use_id": block.id, "content": result})
        print(f"  {block.name}: {result}")

messages.append({"role": "assistant", "content": response.content})
messages.append({"role": "user", "content": tool_results})

final = client.messages.create(
    model="claude-sonnet-4-20250514",
    max_tokens=1024,
    tools=TOOLS,
    messages=messages
)
print(f"LLM: {final.content[0].text}")
```

### Exercise 3: Data Lookup Tool

**Task:** Define a tool called `lookup_capital` that takes a `country` string and returns the capital city from a hardcoded dictionary. Ask the LLM for the capitals of France, Japan, and Brazil.

**Hint:** Your function can use a simple dictionary lookup. Handle missing countries with a friendly error message.

### Solution

```python
import os
from anthropic import Anthropic

client = Anthropic(api_key=os.environ["ANTHROPIC_API_KEY"])

CAPITALS = {
    "france": "Paris",
    "japan": "Tokyo",
    "brazil": "Brasilia",
    "germany": "Berlin",
    "india": "New Delhi",
    "australia": "Canberra",
    "canada": "Ottawa",
    "mexico": "Mexico City",
    "egypt": "Cairo",
    "south korea": "Seoul",
}

TOOLS = [
    {
        "name": "lookup_capital",
        "description": "Look up the capital city of a country. Returns the capital name or an error if the country is not found.",
        "input_schema": {
            "type": "object",
            "properties": {
                "country": {"type": "string", "description": "The name of the country, e.g. 'France' or 'Japan'"}
            },
            "required": ["country"]
        }
    }
]

def lookup_capital(country: str) -> str:
    normalized = country.lower().strip()
    capital = CAPITALS.get(normalized)
    if capital:
        return f"The capital of {country} is {capital}."
    return f"Sorry, I do not have the capital for '{country}' in my database."

messages = [{"role": "user", "content": "What are the capitals of France, Japan, and Brazil?"}]

response = client.messages.create(
    model="claude-sonnet-4-20250514",
    max_tokens=1024,
    tools=TOOLS,
    messages=messages
)

tool_results = []
for block in response.content:
    if block.type == "tool_use" and block.name == "lookup_capital":
        result = lookup_capital(**block.input)
        tool_results.append({"type": "tool_result", "tool_use_id": block.id, "content": result})

messages.append({"role": "assistant", "content": response.content})
messages.append({"role": "user", "content": tool_results})

final = client.messages.create(
    model="claude-sonnet-4-20250514",
    max_tokens=1024,
    tools=TOOLS,
    messages=messages
)
print(f"LLM: {final.content[0].text}")
```

### Exercise 4: Multi-Step Tool Flow

**Task:** Ask the LLM to "Find the capital of Germany, then count the number of characters in that capital name." This requires two sequential tool calls where the second depends on the first.

**Hint:** Your dispatch loop must handle multiple turns. The LLM will call `lookup_capital` first, then use the result to call `count_words` (or a new `count_characters` tool) in a second turn.

### Solution

```python
import os
from anthropic import Anthropic

client = Anthropic(api_key=os.environ["ANTHROPIC_API_KEY"])

CAPITALS = {"germany": "Berlin"}

TOOLS = [
    {
        "name": "lookup_capital",
        "description": "Look up the capital city of a country.",
        "input_schema": {
            "type": "object",
            "properties": {
                "country": {"type": "string"}
            },
            "required": ["country"]
        }
    },
    {
        "name": "count_characters",
        "description": "Count the number of characters in a text string.",
        "input_schema": {
            "type": "object",
            "properties": {
                "text": {"type": "string"}
            },
            "required": ["text"]
        }
    }
]

def lookup_capital(country: str) -> str:
    cap = CAPITALS.get(country.lower().strip())
    return f"The capital of {country} is {cap}." if cap else f"Unknown country: {country}"

def count_characters(text: str) -> str:
    return f"Character count: {len(text)}"

messages = [{"role": "user", "content": "Find the capital of Germany, then count the number of characters in that capital name."}]

MAX_TURNS = 5
for turn in range(MAX_TURNS):
    response = client.messages.create(
        model="claude-sonnet-4-20250514",
        max_tokens=1024,
        tools=TOOLS,
        messages=messages
    )

    if response.stop_reason != "tool_use":
        print(f"LLM: {response.content[0].text}")
        break

    tool_results = []
    for block in response.content:
        if block.type == "tool_use":
            if block.name == "lookup_capital":
                result = lookup_capital(**block.input)
            elif block.name == "count_characters":
                result = count_characters(**block.input)
            print(f"  Turn {turn+1}: {block.name} -> {result}")
            tool_results.append({"type": "tool_result", "tool_use_id": block.id, "content": result})

    messages.append({"role": "assistant", "content": response.content})
    messages.append({"role": "user", "content": tool_results})
```

## Tier 2 -- Intermediate

### Exercise 1: Tool Use with Streaming

**Task:** Modify the weather agent from the practice exercise to use streaming. Stream the LLM's response and accumulate tool use blocks as they arrive, then handle them after the stream ends.

**Hint:** Use `client.messages.stream()` and the `on_text`, `on_tool_use`, `on_message_done` event handlers.

### Solution

```python
import os
from anthropic import Anthropic

client = Anthropic(api_key=os.environ["ANTHROPIC_API_KEY"])

TOOLS = [
    {
        "name": "get_weather",
        "description": "Get the current weather for a city.",
        "input_schema": {
            "type": "object",
            "properties": {
                "city": {"type": "string", "description": "City name"}
            },
            "required": ["city"]
        }
    }
]

def get_weather(city: str) -> str:
    data = {"Tokyo": "22C, Sunny", "London": "14C, Rainy"}
    return data.get(city, f"20C, Unknown for {city}")

messages = [{"role": "user", "content": "What is the weather in Tokyo?"}]

collected_blocks = []
final_text = ""

with client.messages.stream(
    model="claude-sonnet-4-20250514",
    max_tokens=1024,
    tools=TOOLS,
    messages=messages
) as stream:
    for block in stream:
        if block.type == "tool_use":
            collected_blocks.append(block)
        elif block.type == "text":
            final_text += block.text

    # After stream ends, process tool calls
    message = stream.get_final_message()

if collected_blocks:
    tool_results = []
    for block in collected_blocks:
        if block.name == "get_weather":
            result = get_weather(**block.input)
            tool_results.append({"type": "tool_result", "tool_use_id": block.id, "content": result})

    messages.append({"role": "assistant", "content": message.content})
    messages.append({"role": "user", "content": tool_results})

    final = client.messages.create(
        model="claude-sonnet-4-20250514",
        max_tokens=1024,
        tools=TOOLS,
        messages=messages
    )
    print(f"LLM: {final.content[0].text}")
else:
    print(f"LLM: {final_text}")
```

### Exercise 2: Force Specific Tool Choice

**Task:** Use `tool_choice: {"type": "any"}` to force the LLM to always call a tool, even for simple questions that do not need one. Observe how the behavior differs from the default `auto` mode.

**Hint:** Compare what happens when you ask "What is the capital of France?" with `tool_choice: "any"` vs `tool_choice: "auto"`.

### Solution

```python
import os
from anthropic import Anthropic

client = Anthropic(api_key=os.environ["ANTHROPIC_API_KEY"])

TOOLS = [
    {
        "name": "get_fact",
        "description": "Get a fact about a topic. Use this for any factual question.",
        "input_schema": {
            "type": "object",
            "properties": {
                "topic": {"type": "string", "description": "The topic to look up"}
            },
            "required": ["topic"]
        }
    }
]

def get_fact(topic: str) -> str:
    facts = {
        "france": "The capital of France is Paris.",
        "python": "Python was created by Guido van Rossum in 1991.",
    }
    return facts.get(topic.lower(), f"I don't have a fact about '{topic}'.")

messages = [{"role": "user", "content": "What is the capital of France?"}]

# With tool_choice: "any", the LLM must call a tool
response = client.messages.create(
    model="claude-sonnet-4-20250514",
    max_tokens=1024,
    tools=TOOLS,
    tool_choice={"type": "any"},
    messages=messages
)

print(f"With 'any' mode, stop_reason is: {response.stop_reason}")
for block in response.content:
    if block.type == "tool_use":
        result = get_fact(**block.input)
        print(f"  Tool called: {block.name} with {block.input} -> {result}")

# Now compare with "auto" mode -- the LLM may answer from knowledge
response2 = client.messages.create(
    model="claude-sonnet-4-20250514",
    max_tokens=1024,
    tools=TOOLS,
    tool_choice={"type": "auto"},
    messages=messages
)

print(f"\nWith 'auto' mode, stop_reason is: {response2.stop_reason}")
print(f"  Text response: {response2.content[0].text if response2.content else 'No text'}")
```

### Exercise 3: Tool Result with Is Error Flag

**Task:** When a tool encounters an error, use the `is_error` flag in the `tool_result` block so the LLM knows the call failed. Build a tool that randomly fails and observe how the LLM handles the error.

**Hint:** Set `"is_error": True` in the `tool_result` dict to mark a failed execution.

### Solution

```python
import os
import random
from anthropic import Anthropic

client = Anthropic(api_key=os.environ["ANTHROPIC_API_KEY"])

TOOLS = [
    {
        "name": "risky_lookup",
        "description": "Look up information about a topic. May fail randomly.",
        "input_schema": {
            "type": "object",
            "properties": {
                "topic": {"type": "string"}
            },
            "required": ["topic"]
        }
    }
]

def risky_lookup(topic: str) -> str:
    if random.random() < 0.4:  # 40% chance of failure
        raise ValueError("Database timeout")
    return f"Found information about {topic}: It is very interesting."

messages = [{"role": "user", "content": "Tell me about machine learning and Python programming."}]

MAX_TURNS = 5
for turn in range(MAX_TURNS):
    response = client.messages.create(
        model="claude-sonnet-4-20250514",
        max_tokens=1024,
        tools=TOOLS,
        messages=messages
    )

    if response.stop_reason != "tool_use":
        print(f"LLM: {response.content[0].text}")
        break

    tool_results = []
    for block in response.content:
        if block.type == "tool_use" and block.name == "risky_lookup":
            try:
                result = risky_lookup(**block.input)
                tool_results.append({"type": "tool_result", "tool_use_id": block.id, "content": result})
                print(f"  Success: {result}")
            except Exception as e:
                error_msg = f"Error: {str(e)}"
                tool_results.append({
                    "type": "tool_result",
                    "tool_use_id": block.id,
                    "content": error_msg,
                    "is_error": True
                })
                print(f"  Failure: {error_msg}")

    messages.append({"role": "assistant", "content": response.content})
    messages.append({"role": "user", "content": tool_results})
```

### Exercise 1: External API Tool with Authentication

**Task:** Build a tool that calls the GitHub API to fetch user repository data. Use a personal access token stored in an environment variable. The tool should accept a GitHub username and return the names and descriptions of their repositories.

**Hint:** Use the `requests` library (install with `pip install requests`). Set your GitHub token as `GITHUB_TOKEN` in your environment.

### Solution

```python
import os
import requests
from anthropic import Anthropic

client = Anthropic(api_key=os.environ["ANTHROPIC_API_KEY"])

TOOLS = [
    {
        "name": "get_github_repos",
        "description": "Fetch public repository names and descriptions for a GitHub user.",
        "input_schema": {
            "type": "object",
            "properties": {
                "username": {
                    "type": "string",
                    "description": "GitHub username to look up"
                },
                "max_repos": {
                    "type": "integer",
                    "description": "Maximum number of repos to return (default 5)"
                }
            },
            "required": ["username"]
        }
    }
]

def get_github_repos(username: str, max_repos: int = 5) -> str:
    token = os.environ.get("GITHUB_TOKEN")
    headers = {"Authorization": f"Bearer {token}"} if token else {}
    url = f"https://api.github.com/users/{username}/repos?per_page={max_repos}&sort=updated"

    try:
        resp = requests.get(url, headers=headers, timeout=10)
        resp.raise_for_status()
        repos = resp.json()
    except requests.exceptions.RequestException as e:
        return f"Error fetching repos for '{username}': {str(e)}"
    except ValueError:
        return f"Error: Invalid response from GitHub API"

    if not repos:
        return f"User '{username}' has no public repositories."

    lines = [f"Repositories for {username}:"]
    for repo in repos:
        desc = repo.get("description") or "No description"
        lines.append(f"  - {repo['name']}: {desc}")
    return "\n".join(lines)

messages = [{"role": "user", "content": "Show me the public repos of 'torvalds'."}]

response = client.messages.create(
    model="claude-sonnet-4-20250514",
    max_tokens=1024,
    tools=TOOLS,
    messages=messages
)

tool_results = []
for block in response.content:
    if block.type == "tool_use" and block.name == "get_github_repos":
        result = get_github_repos(**block.input)
        print(result)
        tool_results.append({"type": "tool_result", "tool_use_id": block.id, "content": result})

messages.append({"role": "assistant", "content": response.content})
messages.append({"role": "user", "content": tool_results})

final = client.messages.create(
    model="claude-sonnet-4-20250514",
    max_tokens=1024,
    tools=TOOLS,
    messages=messages
)
print(f"\nLLM: {final.content[0].text}")
```

### Exercise 2: Graceful Tool Error Recovery

**Task:** Build an agent with three tools where one tool depends on the output of another. If the first tool fails, the agent should handle it gracefully and explain the issue to the user rather than crashing. Use the `is_error` flag.

**Hint:** The first tool fetches a user ID, and the second tool uses that ID to fetch details. If the first tool returns an error, the second tool should not be called.

### Solution

```python
import os
from anthropic import Anthropic

client = Anthropic(api_key=os.environ["ANTHROPIC_API_KEY"])

TOOLS = [
    {
        "name": "find_user_id",
        "description": "Find the user ID for a given username in the system.",
        "input_schema": {
            "type": "object",
            "properties": {
                "username": {"type": "string"}
            },
            "required": ["username"]
        }
    },
    {
        "name": "get_user_details",
        "description": "Get detailed information about a user by their user ID.",
        "input_schema": {
            "type": "object",
            "properties": {
                "user_id": {"type": "integer", "description": "The numeric user ID"}
            },
            "required": ["user_id"]
        }
    },
    {
        "name": "get_user_orders",
        "description": "Get the recent orders for a user by their user ID.",
        "input_schema": {
            "type": "object",
            "properties": {
                "user_id": {"type": "integer", "description": "The numeric user ID"}
            },
            "required": ["user_id"]
        }
    }
]

# Mock database
USER_DB = {"alice": 101, "bob": 102}
USER_DETAILS = {101: {"name": "Alice", "email": "alice@example.com", "member_since": "2024"},
                102: {"name": "Bob", "email": "bob@example.com", "member_since": "2025"}}
USER_ORDERS = {101: ["Order #5001: Laptop", "Order #5002: Mouse"],
               102: ["Order #6001: Keyboard"]}

def find_user_id(username: str) -> str:
    uid = USER_DB.get(username.lower().strip())
    if uid is None:
        return f"Error: User '{username}' not found in the system."
    return f"User ID for {username} is {uid}."

def get_user_details(user_id: int) -> str:
    details = USER_DETAILS.get(user_id)
    if details is None:
        return f"Error: No details found for user ID {user_id}."
    return f"Name: {details['name']}, Email: {details['email']}, Member since: {details['member_since']}"

def get_user_orders(user_id: int) -> str:
    orders = USER_ORDERS.get(user_id, [])
    if not orders:
        return f"User ID {user_id} has no orders."
    return "Orders:\n" + "\n".join(f"  {o}" for o in orders)

def dispatch_tool(block):
    if block.name == "find_user_id":
        return find_user_id(**block.input)
    elif block.name == "get_user_details":
        return get_user_details(**block.input)
    elif block.name == "get_user_orders":
        return get_user_orders(**block.input)
    return f"Error: Unknown tool '{block.name}'."

messages = [{"role": "user", "content": "Find user 'charlie' and show their details and orders."}]

MAX_TURNS = 6
for turn in range(MAX_TURNS):
    response = client.messages.create(
        model="claude-sonnet-4-20250514",
        max_tokens=1024,
        tools=TOOLS,
        messages=messages
    )

    if response.stop_reason != "tool_use":
        print(f"LLM: {response.content[0].text}")
        break

    tool_results = []
    for block in response.content:
        if block.type == "tool_use":
            result = dispatch_tool(block)
            is_error = result.startswith("Error:")
            tool_results.append({
                "type": "tool_result",
                "tool_use_id": block.id,
                "content": result,
                "is_error": is_error
            })
            status = "ERROR" if is_error else "OK"
            print(f"  [{status}] {block.name}: {result[:60]}...")

    messages.append({"role": "assistant", "content": response.content})
    messages.append({"role": "user", "content": tool_results})
```

### Exercise 3: Dynamic Tool Registration

**Task:** Build a system where tools are registered dynamically using a decorator pattern. Create a `@tool` decorator that registers a function as a tool, automatically generating the tool definition from the function's name, docstring, and type hints.

**Hint:** Use `inspect` module to extract function signatures and docstrings. Generate the `input_schema` dynamically from type hints.

### Solution

```python
import os
import inspect
from anthropic import Anthropic

client = Anthropic(api_key=os.environ["ANTHROPIC_API_KEY"])

# Dynamic tool registry
tool_registry = {}
tool_definitions = []

def tool(name: str = None, description: str = None):
    """Decorator that registers a function as an LLM tool."""
    def decorator(func):
        nonlocal name, description
        tool_name = name or func.__name__
        tool_desc = description or func.__doc__ or "No description provided."

        # Build input_schema from function signature
        sig = inspect.signature(func)
        properties = {}
        required = []
        for param_name, param in sig.parameters.items():
            param_type = param.annotation
            if param_type == str:
                json_type = "string"
            elif param_type == int:
                json_type = "integer"
            elif param_type == float:
                json_type = "number"
            elif param_type == bool:
                json_type = "boolean"
            else:
                json_type = "string"

            properties[param_name] = {
                "type": json_type,
                "description": f"Parameter '{param_name}'"
            }

            if param.default is inspect.Parameter.empty:
                required.append(param_name)

        tool_def = {
            "name": tool_name,
            "description": tool_desc,
            "input_schema": {
                "type": "object",
                "properties": properties,
                "required": required
            }
        }

        tool_definitions.append(tool_def)
        tool_registry[tool_name] = func

        def wrapper(**kwargs):
            return func(**kwargs)
        return wrapper
    return decorator

# Register tools using the decorator
@tool()
def multiply(a: int, b: int) -> str:
    """Multiply two integers together."""
    return f"{a} * {b} = {a * b}"

@tool()
def get_greeting(name: str) -> str:
    """Generate a personalized greeting for someone."""
    return f"Hello, {name}! Welcome to the AI agent workshop."

@tool()
def check_even(number: int) -> str:
    """Check if a number is even or odd."""
    return f"{number} is {'even' if number % 2 == 0 else 'odd'}."

# Use the dynamically registered tools
messages = [{"role": "user", "content": "Multiply 7 by 8, then check if 42 is even."}]

response = client.messages.create(
    model="claude-sonnet-4-20250514",
    max_tokens=1024,
    tools=tool_definitions,
    messages=messages
)

tool_results = []
for block in response.content:
    if block.type == "tool_use" and block.name in tool_registry:
        func = tool_registry[block.name]
        result = func(**block.input)
        print(f"  {block.name} -> {result}")
        tool_results.append({"type": "tool_result", "tool_use_id": block.id, "content": result})

messages.append({"role": "assistant", "content": response.content})
messages.append({"role": "user", "content": tool_results})

final = client.messages.create(
    model="claude-sonnet-4-20250514",
    max_tokens=1024,
    tools=tool_definitions,
    messages=messages
)
print(f"LLM: {final.content[0].text}")
```
