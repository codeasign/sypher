### Exercise A1: Adaptive Timeouts

Build a router that learns the optimal timeout for each provider based on
historical response times. Use the running average of the last N response
times plus two standard deviations as the timeout.

```python
import statistics

class AdaptiveTimeoutRouter:
    """Router that adjusts timeouts based on observed latency."""

    def __init__(self, providers: list, window_size=10):
        self.providers = providers
        self.window_size = window_size
        self.response_times = {}  # provider -> list of floats

    def record_response_time(self, provider: str, elapsed: float):
        """Record the response time for a provider."""
        if provider not in self.response_times:
            self.response_times[provider] = []
        self.response_times[provider].append(elapsed)
        # Keep only the last window_size entries
        if len(self.response_times[provider]) > self.window_size:
            self.response_times[provider].pop(0)

    def get_adaptive_timeout(self, provider: str, default=60.0):
        """Calculate timeout as mean + 2 * stdev of recent response times."""
        times = self.response_times.get(provider, [])
        if len(times) < 3:
            return default
        mean = statistics.mean(times)
        stdev = statistics.stdev(times) if len(times) > 1 else 0
        return mean + (2 * stdev)

    async def route(self, prompt: str):
        """Route with adaptive per-provider timeouts."""

### B3 Solution

```python
import signal

class TimeoutError(Exception):
    pass

def with_timeout(fn, max_seconds=2.0):
    def handler(signum, frame):
        raise TimeoutError(f"Function exceeded {max_seconds} seconds")

    signal.signal(signal.SIGALRM, handler)
    signal.alarm(int(max_seconds))
    try:
        result = fn()
        signal.alarm(0)
        return result
    except TimeoutError:
        raise
    finally:
        signal.alarm(0)
```

Note: On Windows, `signal.SIGALRM` is not available. An alternative is
using `threading.Timer` or `asyncio.wait_for`.

### I1 Solution

```python
class LoggingCircuitBreaker:
    def __init__(self, failure_threshold=3, cooldown_seconds=30):
        self.failure_threshold = failure_threshold
        self.cooldown_seconds = cooldown_seconds
        self.failure_count = 0
        self.state = "closed"
        self.last_failure_time = None
        self.transition_log = []

    def _transition(self, to_state, reason):
        self.transition_log.append(StateTransition(
            timestamp=datetime.now(),
            from_state=self.state,
            to_state=to_state,
            reason=reason,
        ))
        self.state = to_state

    def allow_request(self):
        if self.state == "closed":
            return True
        if self.state == "open":
            elapsed = (datetime.now() - self.last_failure_time).total_seconds()
            if elapsed >= self.cooldown_seconds:
                self._transition("half-open", "Cooldown expired")
                return True
            return False
        return True

    def record_success(self):
        self.failure_count = 0
        if self.state != "closed":
            self._transition("closed", "Successful request")

    def record_failure(self):
        self.failure_count += 1
        self.last_failure_time = datetime.now()
        if self.failure_count >= self.failure_threshold and self.state != "open":
            self._transition("open", f"{self.failure_count} consecutive failures")

    def get_transition_history(self):
        return list(self.transition_log)
```

### I2 Solution

```python
import asyncio
from collections import defaultdict

class PriorityRouter:
    def __init__(self, providers):
        self.providers = providers

    def get_sorted_providers(self):
        return sorted(
            self.providers.items(),
            key=lambda x: x[1]["priority"],
            reverse=True
        )

    async def route(self, prompt):
        # Group providers by priority
        tiers = defaultdict(list)
        for name, config in self.providers.items():
            tiers[config["priority"]].append((name, config))

        # Process tiers in descending priority order
        for priority in sorted(tiers.keys(), reverse=True):
            tier = tiers[priority]
            if len(tier) == 1:
                name, config = tier[0]
                try:
                    result = await asyncio.wait_for(
                        config["call"](prompt),
                        timeout=config.get("timeout", 30),
                    )
                    return result
                except Exception:
                    continue
            else:
                # Concurrent race for tied priorities
                tasks = []
                for name, config in tier:
                    task = asyncio.wait_for(
                        config["call"](prompt),
                        timeout=config.get("timeout", 30),
                    )
                    tasks.append(task)
                done, pending = await asyncio.wait(
                    tasks,
                    return_when=asyncio.FIRST_COMPLETED,
                )
                for task in done:
                    try:
                        result = task.result()
                        for t in pending:
                            t.cancel()
                        return result
                    except Exception:
                        continue
                for t in pending:
                    t.cancel()

        raise RuntimeError("All priority tiers failed")
```

### I3 Solution

```python
from datetime import datetime, timedelta

class StatusCache:
    def __init__(self, ttl_seconds=30):
        self.ttl = timedelta(seconds=ttl_seconds)
        self._cache = {}

    def set_status(self, provider_name, available):
        self._cache[provider_name] = (available, datetime.now())

    def get_status(self, provider_name):
        entry = self._cache.get(provider_name)
        if entry is None:
            return None
        status, timestamp = entry
        if datetime.now() - timestamp > self.ttl:
            del self._cache[provider_name]
            return None
        return status

    def is_available(self, provider_name):
        return self.get_status(provider_name) is True

    def clear(self):
        self._cache.clear()
```

class AdaptiveTimeoutRouter:
    def __init__(self, providers, window_size=10):
        self.providers = providers
        self.window_size = window_size
        self.response_times = {}

    def record_response_time(self, provider, elapsed):
        if provider not in self.response_times:
            self.response_times[provider] = []
        self.response_times[provider].append(elapsed)
        if len(self.response_times[provider]) > self.window_size:
            self.response_times[provider].pop(0)

    def get_adaptive_timeout(self, provider, default=60.0):
        times = self.response_times.get(provider, [])
        if len(times) < 3:
            return default
        mean = statistics.mean(times)
        stdev = statistics.stdev(times) if len(times) > 1 else 0
        return mean + (2 * stdev)

    async def route(self, prompt):
        import time
        for provider in self.providers:
            timeout = self.get_adaptive_timeout(provider["name"])
            start = time.monotonic()
            try:
                result = await asyncio.wait_for(
                    provider["call"](prompt),
                    timeout=timeout,
                )
                elapsed = time.monotonic() - start
                self.record_response_time(provider["name"], elapsed)
                return result
            except Exception:
                elapsed = time.monotonic() - start
                if elapsed < timeout:
                    self.record_response_time(provider["name"], elapsed)
                continue
        raise RuntimeError("All providers failed")
```

### A2 Solution

```python
class TieredRouter:
    def classify_complexity(self, prompt):
        if len(prompt) < 50 and "```" not in prompt:
            return "fast"
        if "```" in prompt or any(kw in prompt.lower() for kw in
            ["step by step", "multiple", "list", "explain", "analyze"]):
            return "capable"
        return "balanced"

    async def route(self, prompt):
        tier = self.classify_complexity(prompt)
        config = self.TIERS[tier]
        last_error = None
        for provider in config["providers"]:
            try:
                result = await provider["call"](prompt)
                return result
            except Exception as e:
                last_error = e
                continue
        raise last_error or RuntimeError("All providers in tier failed")
```
