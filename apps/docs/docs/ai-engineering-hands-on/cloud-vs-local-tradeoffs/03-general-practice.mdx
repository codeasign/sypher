### Exercise I1: Cost with Growth Projection

Extend the decision analyzer to produce a 12-month cost projection that accounts for monthly token growth. The projection should include a recommendation that may change over time — for example, cloud might be cheaper in month 1 but local might become cheaper by month 6.

```python
def growth_projection(scenario: Scenario) -> list[dict]:
    """Calculate monthly cloud and local costs for 12 months."""
    projections = []
    current_tokens = scenario.monthly_tokens
    for month in range(1, 13):
        # Calculate cloud cost for this month's volume
        input_tokens = int(current_tokens * 0.6)
        output_tokens = int(current_tokens * 0.4)
        cloud_cost = (input_tokens / 1000) * 0.00015 + (
            output_tokens / 1000
        ) * 0.0006

        # Local cost is fixed after hardware is purchased
        # Assume hardware is purchased in month 0
        if month == 1:
            local_cost = 1200 + 30  # hardware + first month electricity
        else:
            local_cost = 30  # electricity only

        projections.append({
            "month": month,
            "tokens": current_tokens,
            "cloud_cost": round(cloud_cost, 2),
            "local_cost": round(local_cost, 2),
            "cumulative_cloud": 0.0,
            "cumulative_local": 0.0,
        })
        current_tokens = int(
            current_tokens * (1 + scenario.expected_growth_rate)
        )

    # Calculate cumulative costs
    cum_cloud = 0.0
    cum_local = 0.0
    for entry in projections:
        cum_cloud += entry["cloud_cost"]
        cum_local += entry["local_cost"]
        entry["cumulative_cloud"] = round(cum_cloud, 2)
        entry["cumulative_local"] = round(cum_local, 2)

    return projections
```

### Exercise I2: Hybrid Cost Optimizer

Build a function that calculates the optimal hybrid split — what percentage of traffic should go to local vs cloud to minimize total cost for a given scenario.

The hybrid cost formula is:

```
total_cost = (local_percentage * local_monthly_cost)
           + (cloud_percentage * cloud_cost_for_volume)
```

Where:
- `local_monthly_cost` is the fixed cost of running local hardware.
- `cloud_percentage` is the fraction of tokens sent to the cloud API.
- `cloud_cost_for_volume` is the cost of sending 100% of tokens to the cloud.

```python
def optimize_hybrid_split(
    scenario: Scenario,
    cloud_per_token_cost: float,
    local_monthly_fixed_cost: float,
) -> dict:
    """Find the optimal percentage of traffic to route to local vs cloud."""
    total_monthly_tokens = scenario.monthly_tokens
    cloud_cost_per_token = cloud_per_token_cost
    local_fixed = local_monthly_fixed_cost

    results = []
    for split_pct in range(0, 101, 5):
        local_pct = split_pct / 100
        cloud_pct = 1 - local_pct

        local_tokens = int(total_monthly_tokens * local_pct)
        cloud_tokens = total_monthly_tokens - local_tokens

        local_cost = local_fixed  # Fixed regardless of local volume
        cloud_cost = cloud_tokens * cloud_cost_per_token

        total = local_cost + cloud_cost
        results.append({
            "local_pct": local_pct,
            "cloud_pct": cloud_pct,
            "local_tokens": local_tokens,
            "cloud_tokens": cloud_tokens,
            "local_cost": round(local_cost, 2),
            "cloud_cost": round(cloud_cost, 2),
            "total_cost": round(total, 2),
        })

    optimal = min(results, key=lambda r: r["total_cost"])
    return optimal
```

### Exercise I3: Multi-Factor Weighted Decision

The current decision analyzer uses a simple rule-based approach. Implement a weighted scoring system that produces a numerical score for each option across four dimensions.

```python
@dataclass
class WeightedScore:
    cost_score: float       # 0-100, higher = better for this option
    privacy_score: float    # 0-100, higher = better
    latency_score: float    # 0-100, higher = better
    capability_score: float # 0-100, higher = better
    total: float = 0.0

    def calculate_total(self, weights: dict[str, float]) -> float:
        self.total = (
            self.cost_score * weights.get("cost", 0.25)
            + self.privacy_score * weights.get("privacy", 0.25)
            + self.latency_score * weights.get("latency", 0.25)
            + self.capability_score * weights.get("capability", 0.25)
        )
        return self.total


def score_option(
    scenario: Scenario,
    option: str,  # "cloud" or "local"
    weights: dict[str, float],
) -> WeightedScore:
    """Score a deployment option across all dimensions."""
    # Cost score: higher monthly cost = lower score
    costs = calculate_costs(scenario)
    if option == "cloud":
        monthly = costs.monthly_cloud_cost
    else:
        monthly = costs.total_local_monthly

    # Cost score: $0 = 100, $500+ = 0
    cost_score = max(0, 100 - (monthly / 500) * 100)

    # Privacy score
    verdict, _ = assess_privacy(scenario)
    privacy_scores = {
        PrivacyVerdict.LOCAL_REQUIRED: {"cloud": 0, "local": 100},
        PrivacyVerdict.LOCAL_RECOMMENDED: {"cloud": 30, "local": 100},
        PrivacyVerdict.HYBRID_POSSIBLE: {"cloud": 60, "local": 80},
        PrivacyVerdict.CLOUD_ALLOWED: {"cloud": 100, "local": 60},
    }
    privacy_score = privacy_scores[verdict][option]

    # Latency score
    latency = assess_latency(scenario)
    if option == "local":
        est = latency.estimated_local_latency_ms
    else:
        est = latency.estimated_cloud_latency_ms
    latency_score = max(0, min(100, 100 - (est / 100)))

    # Capability score
    if option == "cloud":
        capability_score = 100  # Cloud has access to all models
    else:
        if scenario.model_tier == ModelTier.FRONTIER:
            capability_score = 0
        elif scenario.model_tier == ModelTier.LARGE:
            capability_score = 80
        else:
            capability_score = 90

    score = WeightedScore(
        cost_score=round(cost_score, 1),
        privacy_score=privacy_score,
        latency_score=round(latency_score, 1),
        capability_score=capability_score,
    )
    score.calculate_total(weights)
    return score
```

2. Test with default weights (0.25 each) for the five test scenarios from the decision analyzer. Which option scores higher for each?
3. Adjust the weights to prioritize privacy (0.50) over cost (0.10). How do the recommendations change?
4. Adjust the weights to prioritize latency (0.50) over capability (0.10). How do the recommendations change?
5. Write a function that finds the weight configuration that would change a recommendation from cloud to local for a given scenario.
