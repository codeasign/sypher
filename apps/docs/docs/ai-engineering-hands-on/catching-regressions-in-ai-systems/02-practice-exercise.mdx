### Step 2: Create the evaluation harness

Create a file called `evaluator.py`:

```python
"""
Evaluation harness for regression testing.
Scores responses using an LLM judge.
"""

import os
import json
from openai import OpenAI


def score_response(
    prompt: str,
    response: str,
    rubric: str,
    model: str = "gpt-4o-mini",
) -> float:
    """Score a response on 0-1 using an LLM judge."""
    client = OpenAI(api_key=os.environ["OPENAI_API_KEY"])

    scoring_prompt = f"""Score this response from 0.0 to 1.0.

Rubric: {rubric}

Prompt: {prompt}

Response: {response}

Return JSON with "score" and "reason"."""

    result = client.chat.completions.create(
        model=model,
        messages=[{"role": "user", "content": scoring_prompt}],
        temperature=0.0,
        response_format={"type": "json_object"},
    )

    data = json.loads(
        result.choices[0].message.content
    )
    return float(data["score"])


def run_eval_suite(
    system_prompt: str,
    model: str = "gpt-4o-mini",
    temperature: float = 0.5,
) -> dict:
    """
    Run the full eval suite on the golden dataset.

    Returns dict with scores per entry and aggregate stats.
    """
    from golden_dataset import GOLDEN_DATASET
    from openai import OpenAI

    client = OpenAI(api_key=os.environ["OPENAI_API_KEY"])

    scores = []
    entry_details = []

    for i, entry in enumerate(GOLDEN_DATASET):
        # Generate response
        response = client.chat.completions.create(
            model=model,
            messages=[
                {"role": "system", "content": system_prompt},
                {"role": "user", "content": entry.prompt},
            ],
            temperature=temperature,
        )
        text = response.choices[0].message.content

        # Score
        score = score_response(
            entry.prompt, text, entry.rubric
        )
        scores.append(score)
        entry_details.append({
            "index": i,
            "category": entry.category,
            "prompt": entry.prompt[:50],
            "score": score,
        })

        if (i + 1) % 5 == 0:
            print(f"  Evaluated {i+1}/{len(GOLDEN_DATASET)}")

    return {
        "scores": scores,
        "entries": entry_details,
        "system_prompt": system_prompt,
        "model": model,
        "temperature": temperature,
    }
```

### Step 3: Build the regression detector

Create a file called `regression_detector.py`:

```python
"""
Statistical regression detection for AI system changes.
"""

import numpy as np
from scipy import stats as scipy_stats


def analyze_scores(scores: list[float]) -> dict:
    """Compute statistical properties of a score distribution."""
    arr = np.array(scores)
    return {
        "n": len(scores),
        "mean": float(np.mean(arr)),
        "median": float(np.median(arr)),
        "std": float(np.std(arr, ddof=1)),
        "min": float(np.min(arr)),
        "max": float(np.max(arr)),
        "p5": float(np.percentile(arr, 5)),
        "p25": float(np.percentile(arr, 25)),
        "p75": float(np.percentile(arr, 75)),
        "p95": float(np.percentile(arr, 95)),
    }


def detect_regression(
    baseline_scores: list[float],
    candidate_scores: list[float],
    alpha: float = 0.05,
    min_effect_size: float = 0.2,
) -> dict:
    """
    Detect whether a candidate represents a regression.

    Tests for:
    1. Mean shift (t-test)
    2. Variance increase (Levene test)
    3. Distribution change (KS test)
    4. Tail degradation (P5 shift)

    Args:
        baseline_scores: Scores from the production system.
        candidate_scores: Scores from the candidate system.
        alpha: Significance threshold for statistical tests.
        min_effect_size: Minimum Cohen's d to flag.

    Returns:
        Dict with regression verdict and details.
    """
    baseline = analyze_scores(baseline_scores)
    candidate = analyze_scores(candidate_scores)

    results = {
        "baseline": baseline,
        "candidate": candidate,
        "regressions": [],
        "is_regression": False,
    }

    # 1. Mean shift
    t_stat, t_p = scipy_stats.ttest_ind(
        baseline_scores, candidate_scores, equal_var=False
    )
    mean_diff = candidate["mean"] - baseline["mean"]
    # Pooled standard deviation for Cohen's d
    pooled_std = np.sqrt(
        (baseline["std"]**2 + candidate["std"]**2) / 2
    )
    cohens_d = mean_diff / pooled_std if pooled_std > 0 else 0

    results["mean_test"] = {
        "t_statistic": float(t_stat),
        "p_value": float(t_p),
        "mean_difference": mean_diff,
        "cohens_d": float(cohens_d),
        "significant": t_p < alpha,
    }

    if t_p < alpha and mean_diff < 0 and abs(cohens_d) > min_effect_size:
        results["regressions"].append(
            f"Mean dropped from {baseline['mean']:.3f} to "
            f"{candidate['mean']:.3f} (d={cohens_d:.2f}, p={t_p:.4f})"
        )

    # 2. Variance increase
    levene_stat, levene_p = scipy_stats.levene(
        baseline_scores, candidate_scores
    )
    var_ratio = candidate["std"] / baseline["std"] if baseline["std"] > 0 else 1

    results["variance_test"] = {
        "levene_statistic": float(levene_stat),
        "p_value": float(levene_p),
        "std_ratio": float(var_ratio),
        "significant": levene_p < alpha,
    }

    if levene_p < alpha and var_ratio > 1.3:
        results["regressions"].append(
            f"Variance increased: std {baseline['std']:.3f} -> "
            f"{candidate['std']:.3f} (ratio={var_ratio:.2f})"
        )

    # 3. Tail degradation
    p5_diff = candidate["p5"] - baseline["p5"]

    results["tail_test"] = {
        "baseline_p5": baseline["p5"],
        "candidate_p5": candidate["p5"],
        "p5_difference": p5_diff,
        "degraded": p5_diff < -0.1,
    }

    if p5_diff < -0.1:
        results["regressions"].append(
            f"Tail degraded: P5 dropped from {baseline['p5']:.3f} "
            f"to {candidate['p5']:.3f}"
        )

    # 4. Full distribution change (KS test)
    ks_stat, ks_p = scipy_stats.ks_2samp(
        baseline_scores, candidate_scores
    )

    results["distribution_test"] = {
        "ks_statistic": float(ks_stat),
        "p_value": float(ks_p),
        "significant": ks_p < alpha,
    }

    results["is_regression"] = len(results["regressions"]) > 0
    return results


def print_regression_report(result: dict):
    """Print a human-readable regression report."""
    print("=" * 60)
    print("REGRESSION DETECTION REPORT")
    print("=" * 60)

    b = result["baseline"]
    c = result["candidate"]

    print(f"\nBaseline:   mean={b['mean']:.3f}, "
          f"std={b['std']:.3f}, p5={b['p5']:.3f}, n={b['n']}")
    print(f"Candidate:  mean={c['mean']:.3f}, "
          f"std={c['std']:.3f}, p5={c['p5']:.3f}, n={c['n']}")

    print(f"\nTests:")
    print(f"  Mean shift:  "
          f"d={result['mean_test']['cohens_d']:.2f}, "
          f"p={result['mean_test']['p_value']:.4f} "
          f"[{'SIGNIFICANT' if result['mean_test']['significant'] else 'none'}]")
    print(f"  Variance:    "
          f"ratio={result['variance_test']['std_ratio']:.2f}, "
          f"p={result['variance_test']['p_value']:.4f} "
          f"[{'SIGNIFICANT' if result['variance_test']['significant'] else 'none'}]")
    print(f"  Tail (P5):   {c['p5']:.3f} vs {b['p5']:.3f} "
          f"[{'DEGRADED' if result['tail_test']['degraded'] else 'stable'}]")

    if result["is_regression"]:
        print(f"\n⚠ REGRESSION DETECTED:")
        for r in result["regressions"]:
            print(f"  - {r}")
    else:
        print(f"\n✓ No regression detected.")
```

### Step 4: Run the baseline and candidate

Create a file called `run_regression_test.py`:

```python
"""
Run a full regression test comparing two system prompts.
"""

from evaluator import run_eval_suite
from regression_detector import detect_regression, print_regression_report

# System prompt baseline (current production)
BASELINE_PROMPT = (
    "You are a helpful assistant. Answer the user's question "
    "accurately and concisely."
)

# System prompt candidate (proposed change)
CANDIDATE_PROMPT = (
    "You are a helpful assistant. Be concise and avoid unnecessary details. "
    "Focus on answering the question directly."
)

print("Running baseline evaluation...")
baseline_results = run_eval_suite(
    system_prompt=BASELINE_PROMPT,
    model="gpt-4o-mini",
    temperature=0.5,
)

print("\nRunning candidate evaluation...")
candidate_results = run_eval_suite(
    system_prompt=CANDIDATE_PROMPT,
    model="gpt-4o-mini",
    temperature=0.5,
)

# Detect regression
result = detect_regression(
    baseline_results["scores"],
    candidate_results["scores"],
)

print_regression_report(result)
```

### Step 5: Add category-level regression analysis

Create a file called `category_analysis.py`:

```python
"""
Category-level regression analysis.
Shows which prompt categories are most affected by a change.
"""

import numpy as np
from golden_dataset import GOLDEN_DATASET, CATEGORIES


def analyze_by_category(
    baseline_results: dict,
    candidate_results: dict,
) -> dict:
    """
    Compare scores by category.
    """
    # Build category -> scores mapping
    baseline_by_cat = {cat: [] for cat in CATEGORIES}
    candidate_by_cat = {cat: [] for cat in CATEGORIES}

    for entry in baseline_results["entries"]:
        baseline_by_cat[entry["category"]].append(
            entry["score"]
        )

    for entry in candidate_results["entries"]:
        candidate_by_cat[entry["category"]].append(
            entry["score"]
        )

    print(f"{'Category':<15} {'Baseline':<10} {'Candidate':<10} "
          f"{'Change':<10} {'Status':<10}")
    print("-" * 55)

    regressions = []
    for cat in CATEGORIES:
        b_scores = baseline_by_cat[cat]
        c_scores = candidate_by_cat[cat]

        if not b_scores or not c_scores:
            continue

        b_mean = np.mean(b_scores)
        c_mean = np.mean(c_scores)
        change = c_mean - b_mean

        # Small effect: check if change is meaningful
        status = "OK" if change >= -0.05 else "REGRESSION"
        if status == "REGRESSION":
            regressions.append(cat)

        print(f"{cat:<15} {b_mean:<10.3f} {c_mean:<10.3f} "
              f"{change:<+10.3f} {status:<10}")

    return regressions


if __name__ == "__main__":
    # Load results from previous run
    from evaluator import run_eval_suite
    from regression_detector import detect_regression

    baseline = run_eval_suite(
        "You are a helpful assistant. Answer accurately and concisely.",
        temperature=0.5,
    )
    candidate = run_eval_suite(
        "You are a helpful assistant. Be concise. Avoid extra details.",
        temperature=0.5,
    )

    print("\nCategory Analysis:")
    regressed_cats = analyze_by_category(baseline, candidate)
    if regressed_cats:
        print(f"\nRegressions in: {', '.join(regressed_cats)}")
```

### Step 6: Create a CI-ready regression test

Create a file called `test_regression.py`:

```python
"""
CI-ready regression test that blocks deploys on regression.
"""

import pytest
from evaluator import run_eval_suite
from regression_detector import detect_regression

# Known-good baseline scores (from the last production run)
# In production, load these from a database
BASELINE_SCORES = [
    0.92, 0.88, 0.95, 0.78, 0.90, 0.85, 0.87, 0.96,
    0.84, 0.91, 0.88, 0.83, 0.86, 0.90, 0.89,
]

# Expected thresholds
EXPECTED_MEAN = 0.85
EXPECTED_MIN_P5 = 0.70
MAX_ALLOWED_STD = 0.15


@pytest.mark.regression
class TestNoRegression:
    """Tests that fail if a regression is detected."""

    @pytest.fixture(scope="class")
    def candidate_results(self):
        """Run the candidate evaluation once per test class."""
        results = run_eval_suite(
            system_prompt="You are a helpful assistant. "
                          "Answer concisely and accurately.",
            model="gpt-4o-mini",
            temperature=0.5,
        )
        return results

    def test_mean_not_degraded(self, candidate_results):
        """Mean score must not drop below threshold."""
        scores = candidate_results["scores"]
        mean = sum(scores) / len(scores)
        assert mean >= EXPECTED_MEAN, (
            f"Mean {mean:.3f} dropped below {EXPECTED_MEAN}"
        )

    def test_tail_not_degraded(self, candidate_results):
        """P5 percentile must not drop below threshold."""
        import numpy as np
        scores = candidate_results["scores"]
        p5 = np.percentile(scores, 5)
        assert p5 >= EXPECTED_MIN_P5, (
            f"P5 {p5:.3f} dropped below {EXPECTED_MIN_P5}"
        )

    def test_variance_not_increased(self, candidate_results):
        """Standard deviation must not exceed limit."""
        import numpy as np
        scores = candidate_results["scores"]
        std = np.std(scores, ddof=1)
        assert std <= MAX_ALLOWED_STD, (
            f"Std {std:.3f} exceeds {MAX_ALLOWED_STD}"
        )

    def test_no_statistical_regression(self, candidate_results):
        """No statistically significant regression vs baseline."""
        result = detect_regression(
            BASELINE_SCORES, candidate_results["scores"]
        )
        assert not result["is_regression"], (
            "Regression detected: " + "; ".join(
                result["regressions"]
            )
        )
```

## Common Errors at This Stage

### Error 1: Sample size too small
```
T-test p-value: 0.45 — No significant difference
```
**Cause:** With only 5-10 samples per version, the test lacks power to detect moderate regressions.
**Fix:** Use at least 15-20 samples per version. The golden dataset should have 50+ entries for reliable regression detection.

### Error 2: Ignoring the tail
```
Mean unchanged, P5 dropped from 0.72 to 0.45 — regression missed
```
**Cause:** Only checking the mean score. The mean can stay the same while the worst outputs get dramatically worse.
**Fix:** Always check P5, P25, and variance in addition to the mean.

### Error 3: Multiple comparison without correction
```
5 dimensions tested at alpha=0.05 — 25% chance of at least one false positive
```
**Cause:** Testing many dimensions inflates the false positive rate.
**Fix:** Apply Bonferroni correction: alpha / number_of_dimensions. Or use a multilevel model.

### Error 4: Over-relying on p-values
```
p=0.04, declares regression — but the effect size is d=0.05 (tiny)
```
**Cause:** With a large enough sample, even tiny differences become statistically significant.
**Fix:** Always check Cohen's d (effect size) alongside the p-value. Do not flag changes with d < 0.2 as regressions.
