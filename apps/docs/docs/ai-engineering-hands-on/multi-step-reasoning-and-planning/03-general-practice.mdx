## Tier 1 -- Beginner

### Exercise 1: Create a Simple Sequential Plan

**Task:** Write a script that takes a goal string and creates a sequential plan (each step depends on the previous one). The plan must break the goal into 3-5 steps. Print the steps in order.

**Hint:** Use the `create_plan` function from the practice exercise, but enforce that each step depends on the previous one.

### Solution

```python
from anthropic import Anthropic
import json

client = Anthropic()

def create_sequential_plan(goal: str):
    response = client.messages.create(
        model="claude-sonnet-4-20250514",
        max_tokens=1024,
        messages=[
            {
                "role": "user",
                "content": f"""
Goal: "{goal}"

Create a sequential plan of 3-5 steps where each step depends on the previous one.
Step 0 has no dependencies. Step 1 depends on step 0. Step 2 depends on step 1. And so on.

Return JSON array: [{{"step": int, "task": str, "depends_on": list[int]}}]
"""
            }
        ]
    )
    plan = json.loads(response.content[0].text)
    for step in plan:
        deps = step["depends_on"] if step["depends_on"] else "none"
        print(f"Step {step['step']}: {step['task']} (depends on {deps})")
    return plan

plan = create_sequential_plan("Plan a weekend trip to a national park")
```

### Exercise 2: Execute Steps and Track Progress

**Task:** Extend Exercise 1 to execute each step using a mock executor that simulates work (print "Executing step X..." and return a dummy result). Track completed steps in a set and print progress after each step.

### Solution

```python
from anthropic import Anthropic
import json
import time

client = Anthropic()

class ProgressTracker:
    def __init__(self, total_steps: int):
        self.total = total_steps
        self.completed = set()

    def mark_done(self, step: int):
        self.completed.add(step)
        pct = int(len(self.completed) / self.total * 100)
        print(f"Progress: {pct}% ({len(self.completed)}/{self.total} steps)")

    def is_done(self) -> bool:
        return len(self.completed) == self.total

def mock_execute(step: dict) -> str:
    print(f"  Executing step {step['step']}: {step['task']}...")
    time.sleep(0.5)
    return f"Result of step {step['step']}: completed successfully"

def run_plan(goal: str):
    # Create the plan
    response = client.messages.create(
        model="claude-sonnet-4-20250514",
        max_tokens=1024,
        messages=[
            {
                "role": "user",
                "content": f"""
Goal: "{goal}"
Create 3-5 sequential steps.
Return JSON array: [{{"step": int, "task": str, "depends_on": list[int]}}]
"""
            }
        ]
    )
    plan = json.loads(response.content[0].text)
    tracker = ProgressTracker(len(plan))

    print(f"Plan for: {goal}")
    for step in plan:
        deps_met = all(d in tracker.completed for d in step["depends_on"])
        print(f"  Step {step['step']}: deps met = {deps_met}")

    print("\nExecuting...")
    for step in plan:
        result = mock_execute(step)
        print(f"  {result}")
        tracker.mark_done(step["step"])

    print(f"\nPlan complete! All {tracker.total} steps executed.")

run_plan("Research and write a 3-paragraph summary of quantum computing")
```

### Exercise 3: Verify Step Results

**Task:** Add a verification step after each mock execution. The verifier checks if the result string contains certain keywords. If verification fails, retry once. Track how many retries were needed.

### Solution

```python
from anthropic import Anthropic
import json
import random

client = Anthropic()

def mock_execute(step: dict) -> str:
    """Simulate execution with occasional failures."""
    results = [
        "data retrieved successfully from source",
        "analysis complete with all metrics",
        "summary generated with 5 key points",
        "comparison table built with 3 models",
    ]
    return random.choice(results)

def verify_result(step: dict, result: str) -> bool:
    """Check if the result contains meaningful content."""
    keywords = ["data", "complete", "generated", "built", "success", "analysis", "summary"]
    found = any(kw in result.lower() for kw in keywords)
    print(f"  Verification: {'PASS' if found else 'FAIL'} for step {step['step']}")
    return found

def run_with_verification(goal: str):
    response = client.messages.create(
        model="claude-sonnet-4-20250514",
        max_tokens=1024,
        messages=[
            {
                "role": "user",
                "content": f"""
Goal: "{goal}"
Create 3-5 sequential steps.
Return JSON array: [{{"step": int, "task": str, "depends_on": list[int]}}]
"""
            }
        ]
    )
    plan = json.loads(response.content[0].text)
    retries = 0
    max_retries = 1

    for step in plan:
        success = False
        attempt = 0
        while not success and attempt <= max_retries:
            result = mock_execute(step)
            success = verify_result(step, result)
            if not success:
                attempt += 1
                retries += 1
                if attempt <= max_retries:
                    print(f"  Retrying step {step['step']}...")
                else:
                    print(f"  Step {step['step']} failed after {max_retries} retries")

    print(f"\nTotal retries needed: {retries}")

run_with_verification("Gather system metrics and generate a report")
```

### Exercise 2: Implement Parallel Execution

**Task:** Build a script that identifies independent tasks and runs them in parallel using Python's `concurrent.futures.ThreadPoolExecutor`. Compare the runtime of sequential vs parallel execution.

### Solution

```python
from anthropic import Anthropic
import json
import time
from concurrent.futures import ThreadPoolExecutor, as_completed

client = Anthropic()

def create_plan_with_parallel_tasks(goal: str) -> list[dict]:
    response = client.messages.create(
        model="claude-sonnet-4-20250514",
        max_tokens=1024,
        messages=[
            {
                "role": "user",
                "content": f"""
Goal: "{goal}"
Create 5-6 steps. Some steps should have no dependencies so they can run in parallel.
Return JSON array: [{{"step": int, "task": str, "depends_on": list[int]}}]
"""
            }
        ]
    )
    return json.loads(response.content[0].text)

def schedule(plan: list[dict]) -> list[list[int]]:
    batches = []
    completed = set()
    remaining = set(range(len(plan)))

    while remaining:
        batch = [i for i in remaining if all(d in completed for d in plan[i]["depends_on"])]
        if not batch:
            break
        batches.append(batch)
        completed.update(batch)
        remaining -= set(batch)

    return batches

def execute_task(task_str: str) -> str:
    """Simulate a task that takes some time."""
    time.sleep(1.0)
    return f"Completed: {task_str[:40]}..."

# Run the comparison
goal = "Research and compare 5 different AI code assistants"
plan = create_plan_with_parallel_tasks(goal)
batches = schedule(plan)

print(f"Plan has {len(plan)} steps in {len(batches)} batches")

# Sequential execution
print("\nSequential execution:")
start = time.time()
for batch in batches:
    for step_idx in batch:
        result = execute_task(plan[step_idx]["task"])
        print(f"  Step {step_idx}: {result}")
seq_time = time.time() - start
print(f"Sequential time: {seq_time:.2f}s")

# Parallel execution
print("\nParallel execution (within each batch):")
start = time.time()
for batch in batches:
    with ThreadPoolExecutor(max_workers=len(batch)) as executor:
        futures = {executor.submit(execute_task, plan[i]["task"]): i for i in batch}
        for future in as_completed(futures):
            idx = futures[future]
            result = future.result()
            print(f"  Step {idx}: {result}")
par_time = time.time() - start
print(f"Parallel time: {par_time:.2f}s")

print(f"\nSpeedup: {seq_time / par_time:.1f}x")
```

### Exercise 3: Dependency Validation

**Task:** Build a validator that checks a plan for circular dependencies and missing dependency references. The validator should report all issues found and suggest fixes.

### Solution

```python
from anthropic import Anthropic
import json

client = Anthropic()

def create_plan_with_potential_issues(goal: str) -> list[dict]:
    """Create a plan that might have dependency issues."""
    response = client.messages.create(
        model="claude-sonnet-4-20250514",
        max_tokens=1024,
        messages=[
            {
                "role": "user",
                "content": f"""
Goal: "{goal}"
Create 4-6 steps. Some steps should depend on others.
Return JSON array: [{{"step": int, "task": str, "depends_on": list[int]}}]
"""
            }
        ]
    )
    return json.loads(response.content[0].text)

def validate_plan(plan: list[dict]) -> list[str]:
    issues = []
    step_indices = {s["step"] for s in plan}

    # Check for missing dependency references
    for step in plan:
        for dep in step["depends_on"]:
            if dep not in step_indices:
                issues.append(
                    f"Step {step['step']} depends on non-existent step {dep}"
                )

    # Check for circular dependencies
    for start_step in plan:
        visited = set()
        current = start_step["step"]
        stack = [current]
        while stack:
            node = stack.pop()
            if node in visited:
                if node == current:
                    issues.append(
                        f"Circular dependency detected involving step {current}"
                    )
                break
            visited.add(node)
            step_obj = next(s for s in plan if s["step"] == node)
            stack.extend(step_obj["depends_on"])

    # Check for self-dependency
    for step in plan:
        if step["step"] in step["depends_on"]:
            issues.append(f"Step {step['step']} depends on itself")

    return issues

# Run validation
goal = "Analyze customer feedback data and generate insights"
plan = create_plan_with_potential_issues(goal)
print("Plan:")
for s in plan:
    print(f"  Step {s['step']}: {s['task']} (deps: {s['depends_on']})")

issues = validate_plan(plan)
if issues:
    print(f"\nFound {len(issues)} issue(s):")
    for issue in issues:
        print(f"  - {issue}")
else:
    print("\nNo dependency issues found")
```

### Exercise 1: Hierarchical Planner with Sub-Agents

**Task:** Build a hierarchical planner where a top-level agent decomposes the goal into high-level phases, then delegates each phase to a sub-agent that creates its own detailed plan. Each sub-agent executes its plan independently and reports back to the top-level agent.

**Hint:** Create a `TopLevelPlanner` class that spawns `SubAgent` instances, each with its own LLM call to create and execute a detailed sub-plan.

### Solution

```python
from anthropic import Anthropic
import json
import time

client = Anthropic()

class SubAgent:
    def __init__(self, name: str, phase: str, context: dict):
        self.name = name
        self.phase = phase
        self.context = context
        self.plan = []
        self.results = {}

    def create_detail_plan(self) -> list[dict]:
        response = client.messages.create(
            model="claude-sonnet-4-20250514",
            max_tokens=1024,
            messages=[
                {
                    "role": "user",
                    "content": f"""
Phase: {self.phase}
Context: {json.dumps(self.context)}

Create a detailed plan of 2-3 steps for this phase.
Return JSON array: [{{"step": int, "task": str, "depends_on": list[int]}}]
"""
                }
            ]
        )
        self.plan = json.loads(response.content[0].text)
        return self.plan

    def execute(self) -> dict:
        print(f"    Sub-agent '{self.name}' executing {len(self.plan)} steps...")
        for step in self.plan:
            time.sleep(0.3)
            self.results[step["step"]] = f"Executed: {step['task']}"
            print(f"      Step {step['step']}: {step['task']} DONE")
        return self.results


class TopLevelPlanner:
    def __init__(self, goal: str):
        self.goal = goal
        self.phases = []
        self.sub_agents = []

    def create_phases(self) -> list[str]:
        response = client.messages.create(
            model="claude-sonnet-4-20250514",
            max_tokens=1024,
            messages=[
                {
                    "role": "user",
                    "content": f"""
Goal: "{self.goal}"

Break this into 3 high-level phases. Each phase should be a distinct stage of work.
Return a JSON array of strings, each describing one phase.
"""
                }
            ]
        )
        self.phases = json.loads(response.content[0].text)
        return self.phases

    def run(self) -> dict:
        print(f"Top-Level Planner")
        print(f"Goal: {self.goal}\n")

        # Create phases
        self.create_phases()
        print(f"Phases ({len(self.phases)}):")
        for i, phase in enumerate(self.phases):
            print(f"  Phase {i}: {phase}")

        # Create sub-agents for each phase
        print("\nDelegating to sub-agents...")
        all_results = {}
        for i, phase in enumerate(self.phases):
            agent = SubAgent(f"agent-{i}", phase, {"phase_index": i})
            agent.create_detail_plan()
            print(f"\n  Phase {i} plan ({len(agent.plan)} steps):")
            for s in agent.plan:
                print(f"    Step {s['step']}: {s['task']}")
            results = agent.execute()
            all_results[f"phase_{i}"] = results

        # Synthesize final report
        print("\nSynthesizing final report...")
        response = client.messages.create(
            model="claude-sonnet-4-20250514",
            max_tokens=512,
            messages=[
                {
                    "role": "user",
                    "content": f"""
Goal: {self.goal}
Phase results: {json.dumps(all_results, indent=2)}

Generate a one-paragraph summary of all work completed.
"""
                }
            ]
        )
        print(f"\nFinal Report: {response.content[0].text}")

        return all_results

# Run the hierarchical planner
planner = TopLevelPlanner("Build a REST API for a todo application")
planner.run()
```

### Exercise 2: Plan Optimization -- Merge Redundant Steps

**Task:** Build an optimizer that takes a plan and identifies redundant steps that can be merged. For example, if step 2 says "fetch user data" and step 4 says "fetch user profile," the optimizer should merge them into one "fetch user data and profile" step. Report the token savings from optimization.

### Solution

```python
from anthropic import Anthropic
import json

client = Anthropic()

def create_unoptimized_plan(goal: str) -> list[dict]:
    response = client.messages.create(
        model="claude-sonnet-4-20250514",
        max_tokens=1024,
        messages=[
            {
                "role": "user",
                "content": f"""
Goal: "{goal}"
Create a detailed plan with 6-8 steps. Include some redundant or overlapping steps.
Return JSON array: [{{"step": int, "task": str, "depends_on": list[int]}}]
"""
            }
        ]
    )
    return json.loads(response.content[0].text)

def estimate_tokens(plan: list[dict]) -> int:
    """Rough token estimate for a plan."""
    text = json.dumps(plan)
    return len(text) // 4  # rough approximation

def optimize_plan(plan: list[dict]) -> tuple[list[dict], dict]:
    """Merge redundant steps and return (optimized_plan, stats)."""
    original_count = len(plan)
    original_tokens = estimate_tokens(plan)

    response = client.messages.create(
        model="claude-sonnet-4-20250514",
        max_tokens=1024,
        messages=[
            {
                "role": "user",
                "content": f"""
Original plan: {json.dumps(plan, indent=2)}

Optimize this plan by:
1. Merging any redundant or overlapping steps
2. Removing unnecessary dependency checks
3. Keeping at least 3 steps

Return JSON: {{"optimized_plan": array, "changes_made": [str]}}
"""
            }
        ]
    )
    result = json.loads(response.content[0].text)
    optimized = result["optimized_plan"]
    optimized_tokens = estimate_tokens(optimized)

    stats = {
        "original_steps": original_count,
        "optimized_steps": len(optimized),
        "original_tokens": original_tokens,
        "optimized_tokens": optimized_tokens,
        "token_savings": original_tokens - optimized_tokens,
        "changes": result["changes_made"],
    }
    return optimized, stats

# Run the optimization exercise
goal = "Analyze customer churn rates for a SaaS product"
plan = create_unoptimized_plan(goal)
print(f"Original plan ({len(plan)} steps, ~{estimate_tokens(plan)} tokens):")
for s in plan:
    print(f"  Step {s['step']}: {s['task']}")

optimized, stats = optimize_plan(plan)
print(f"\nOptimized plan ({stats['optimized_steps']} steps, ~{stats['optimized_tokens']} tokens):")
for s in optimized:
    print(f"  Step {s['step']}: {s['task']}")

print(f"\nOptimization stats:")
print(f"  Steps: {stats['original_steps']} -> {stats['optimized_steps']}")
print(f"  Tokens saved: {stats['token_savings']} ({stats['original_tokens'] - stats['optimized_tokens']} tokens)")
print(f"  Changes: {', '.join(stats['changes'])}")
```

**Task:** Build an optimizer that reorders plan steps to minimize the number of sequential batches while respecting dependencies. The optimizer should find the optimal execution ordering that maximizes parallelism.

### Solution

```python

import json

Reorder the steps to:

Return JSON: {{"reordered_plan": [{{"step": int, "task": str, "depends_on": list[int]}}], "reasoning": str}}

    result = json.loads(response.content[0].text)
    return result["reordered_plan"]
```
