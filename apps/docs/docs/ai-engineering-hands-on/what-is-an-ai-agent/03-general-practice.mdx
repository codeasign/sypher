## Tier 1: Beginner

These exercises involve adding a single new tool or modifying the agent loop behavior. No new architectural patterns.

### Exercise 1: Add a Calculator Tool

**Task:** Add a `calculate` tool to your agent that evaluates simple arithmetic expressions. The agent should answer questions like "What is 145 times 37?" by calling the calculator tool.

**Hint:** Use Python's `eval()` with restricted globals, or parse operators manually with the `operator` module. Restrict to mathematical operations only.

<details>
<summary>Show solution</summary>

```python
import os
from anthropic import Anthropic

client = Anthropic()

TOOLS = [
    {
        "name": "calculate",
        "description": "Evaluate a mathematical expression and return the result",
        "input_schema": {
            "type": "object",
            "properties": {
                "expression": {
                    "type": "string",
                    "description": "A mathematical expression (e.g., 145 * 37)"
                }
            },
            "required": ["expression"]
        }
    }
]

def execute_tool(name, input_data):
    if name == "calculate":
        expr = input_data.get("expression", "").strip()
        try:
            result = eval(expr, {"__builtins__": {}}, {})
            return str(result)
        except Exception as e:
            return f"Error: {str(e)}"
    raise ValueError(f"Unknown tool: {name}")

# Reuse agent_loop from the practice exercise
result = agent_loop("What is 145 times 37?")
print(result)
```
</details>

### Exercise 2: Add a Web Search Placeholder

**Task:** Add a `search_web` tool that returns mock results from a local dictionary. This teaches you how to build a mock tool that can later be replaced with a real API call. The agent should answer questions like "What is the capital of France?"

**Hint:** Use a dictionary mapping queries to responses. Include a fallback for unknown queries.

<details>
<summary>Show solution</summary>

```python
import os
from anthropic import Anthropic

client = Anthropic()

MOCK_DB = {
    "capital of france": "Paris is the capital of France.",
    "capital of japan": "Tokyo is the capital of Japan.",
    "population of earth": "Approximately 8 billion people."
}

TOOLS = [
    {
        "name": "search_web",
        "description": "Search the web for information on a topic",
        "input_schema": {
            "type": "object",
            "properties": {
                "query": {
                    "type": "string",
                    "description": "The search query string"
                }
            },
            "required": ["query"]
        }
    }
]

def execute_tool(name, input_data):
    if name == "search_web":
        query = input_data.get("query", "").lower().strip()
        for key, value in MOCK_DB.items():
            if key in query or query in key:
                return value
        return f"No results found for: {query}"
    raise ValueError(f"Unknown tool: {name}")

# Reuse agent_loop from the practice exercise
result = agent_loop("What is the capital of France?")
print(result)
```
</details>

### Exercise 4: Agent That Asks Clarifying Questions

**Task:** If the user says "What is the weather like?" without a city, the agent should ask "Which city?" instead of guessing.

**Hint:** Add a system prompt that instructs the agent to ask for missing information before calling tools.

<details>
<summary>Show solution</summary>

```python
import json
import os
from anthropic import Anthropic

client = Anthropic()

MOCK_WEATHER = {
    "Tokyo": {"temperature": 22, "conditions": "partly cloudy", "humidity": 65}
}

TOOLS = [
    {
        "name": "get_weather",
        "description": "Get the current weather for a city. Requires a city name.",
        "input_schema": {
            "type": "object",
            "properties": {
                "city": {"type": "string", "description": "City name"}
            },
            "required": ["city"]
        }
    }
]

SYSTEM_PROMPT = """You are a weather assistant. If the user asks for weather
but does not specify a city, ask them which city before calling the tool.
Never guess the city name."""

def execute_tool(name, input_data):
    if name == "get_weather":
        city = input_data.get("city", "")
        data = MOCK_WEATHER.get(city, {"error": f"No data for {city}"})
        return json.dumps(data)

# Pass system=SYSTEM_PROMPT in the agent_loop call
response = client.messages.create(
    model="claude-sonnet-4-20250514",
    max_tokens=1024,
    tools=TOOLS,
    messages=[{"role": "user", "content": "What is the weather like?"}],
    system=SYSTEM_PROMPT
)
print(response.content[0].text)
```
</details>

### Exercise 1: Streaming Agent Output

**Task:** Stream the LLM's text output token by token using `client.messages.stream()`. Tool calls should still be processed synchronously.

**Hint:** Use `with client.messages.stream(...) as stream:` and iterate `stream.text_stream`. Get the final message with `stream.get_final_message()`.

<details>
<summary>Show solution</summary>

```python
import os
import datetime
from anthropic import Anthropic

client = Anthropic()

TOOLS = [
    {
        "name": "get_current_time",
        "description": "Get the current date and time",
        "input_schema": {
            "type": "object",
            "properties": {
                "timezone": {"type": "string", "description": "Timezone"}
            },
            "required": ["timezone"]
        }
    }
]

def execute_tool(name, input_data):
    if name == "get_current_time":
        tz = input_data.get("timezone", "UTC")
        now = datetime.datetime.now(datetime.timezone.utc)
        return f"Current time in {tz}: {now.strftime('%Y-%m-%d %H:%M:%S')} UTC"

def agent_loop_streaming(user_message, max_iterations=10):
    messages = [{"role": "user", "content": user_message}]
    for i in range(1, max_iterations + 1):
        print(f"\n[Iteration {i}]", end=" ", flush=True)
        with client.messages.stream(
            model="claude-sonnet-4-20250514",
            max_tokens=1024,
            tools=TOOLS,
            messages=messages
        ) as stream:
            for text in stream.text_stream:
                print(text, end="", flush=True)
            print()
            response = stream.get_final_message()

        if response.stop_reason == "tool_use":
            for block in response.content:
                if block.type == "tool_use":
                    tool_result = execute_tool(block.name, block.input)
                    print(f"  [Tool: {block.name} => {tool_result}]")
                    messages.append({"role": "assistant", "content": response.content})
                    messages.append({"role": "user", "content": [{"type": "tool_result", "tool_use_id": block.id, "content": tool_result}]})
        elif response.stop_reason == "end_turn":
            print("[Done]")
            return
        else:
            return

agent_loop_streaming("What time is it in New York?")
```
</details>

### Exercise 2: Structured Output with Pydantic

**Task:** Make the agent return its final answer as a Pydantic-validated JSON object instead of free text.

**Hint:** Define a Pydantic `BaseModel`. Use a system prompt instructing JSON output. Parse and validate the LLM's text response with Pydantic.

<details>
<summary>Show solution</summary>

```python
import json
import os
from datetime import datetime, timezone
from pydantic import BaseModel
from anthropic import Anthropic

client = Anthropic()

class WeatherReport(BaseModel):
    city: str
    temperature: int
    conditions: str
    humidity: int
    timestamp: str

MOCK_WEATHER = {"Tokyo": {"temperature": 22, "conditions": "partly cloudy", "humidity": 65}}

TOOLS = [
    {
        "name": "get_weather",
        "description": "Get weather for a city",
        "input_schema": {
            "type": "object",
            "properties": {"city": {"type": "string", "description": "City name"}},
            "required": ["city"]
        }
    }
]

SYSTEM_PROMPT = """Output your final answer as JSON matching the schema:
{"city": str, "temperature": int, "conditions": str, "humidity": int, "timestamp": str}"""

def execute_tool(name, input_data):
    if name == "get_weather":
        city = input_data.get("city", "")
        data = MOCK_WEATHER.get(city, {})
        data["city"] = city
        data["timestamp"] = datetime.now(timezone.utc).isoformat()
        return json.dumps(data)

def agent_loop(user_message, max_iterations=10):
    messages = [{"role": "user", "content": user_message}]
    for i in range(1, max_iterations + 1):
        response = client.messages.create(
            model="claude-sonnet-4-20250514", max_tokens=1024,
            tools=TOOLS, messages=messages, system=SYSTEM_PROMPT
        )
        if response.stop_reason == "tool_use":
            for block in response.content:
                if block.type == "tool_use":
                    tool_result = execute_tool(block.name, block.input)
                    messages.append({"role": "assistant", "content": response.content})
                    messages.append({"role": "user", "content": [{"type": "tool_result", "tool_use_id": block.id, "content": tool_result}]})
        elif response.stop_reason == "end_turn":
            try:
                report = WeatherReport(**json.loads(response.content[0].text))
                return report.model_dump_json(indent=2)
            except (json.JSONDecodeError, ValueError):
                return response.content[0].text
        else:
            return response.content[0].text

result = agent_loop("What is the weather in Tokyo?")
print(result)
```
</details>

### Exercise 3: Multi-Tool Orchestration

**Task:** Create an agent with three tools — `get_weather`, `get_current_time`, and `send_email` — that work together. The user should be able to say "Check the weather in London and email it to admin@example.com."

**Hint:** The LLM naturally sequences tool calls. Use descriptive tool names and descriptions so the LLM understands the correct order. The `send_email` tool is a mock — only print the email, never actually send one.

<details>
<summary>Show solution</summary>

```python
import json, os, datetime
from anthropic import Anthropic

client = Anthropic()

MOCK_WEATHER = {"London": {"temperature": 15, "conditions": "rainy", "humidity": 80}}

TOOLS = [
    {"name": "get_weather", "description": "Get weather for a city",
     "input_schema": {"type": "object", "properties": {"city": {"type": "string"}}, "required": ["city"]}},
    {"name": "get_current_time", "description": "Get the current time",
     "input_schema": {"type": "object", "properties": {"timezone": {"type": "string"}}, "required": ["timezone"]}},
    {"name": "send_email", "description": "Send an email. Use AFTER gathering data.",
     "input_schema": {"type": "object", "properties": {
         "to": {"type": "string"}, "subject": {"type": "string"}, "body": {"type": "string"}
     }, "required": ["to", "subject", "body"]}}
]

def execute_tool(name, input_data):
    if name == "get_weather":
        return json.dumps(MOCK_WEATHER.get(input_data.get("city", ""), {}))
    elif name == "get_current_time":
        now = datetime.datetime.now(datetime.timezone.utc)
        return f"Time: {now.strftime('%Y-%m-%d %H:%M:%S')} UTC"
    elif name == "send_email":
        print(f"\n[MOCK EMAIL] To: {input_data['to']}, Subject: {input_data['subject']}")
        return "Email sent successfully."

def agent_loop(user_message, max_iterations=15):
    messages = [{"role": "user", "content": user_message}]
    for i in range(1, max_iterations + 1):
        response = client.messages.create(
            model="claude-sonnet-4-20250514", max_tokens=1024, tools=TOOLS, messages=messages
        )
        if response.stop_reason == "tool_use":
            for block in response.content:
                if block.type == "tool_use":
                    tool_result = execute_tool(block.name, block.input)
                    messages.append({"role": "assistant", "content": response.content})
                    messages.append({"role": "user", "content": [{"type": "tool_result", "tool_use_id": block.id, "content": tool_result}]})
        elif response.stop_reason == "end_turn":
            return response.content[0].text
        else:
            return response.content[0].text

result = agent_loop("Check the weather in London and email it to admin@example.com")
print(result)
```
</details>

### Exercise 4: Parallel Tool Calls

**Task:** Handle multiple `tool_use` blocks within a single response — the LLM may request two tools at once. Execute all requested tools before returning results.

**Hint:** Collect all `tool_use` blocks from `response.content`, execute them all, then append one assistant message and one user message containing all `tool_result` blocks.

<details>
<summary>Show solution</summary>

```python
import json, os, datetime
from anthropic import Anthropic

client = Anthropic()

MOCK_WEATHER = {"Tokyo": {"temperature": 22, "conditions": "partly cloudy", "humidity": 65}}

TOOLS = [
    {"name": "get_weather", "description": "Get weather",
     "input_schema": {"type": "object", "properties": {"city": {"type": "string"}}, "required": ["city"]}},
    {"name": "get_current_time", "description": "Get time",
     "input_schema": {"type": "object", "properties": {"timezone": {"type": "string"}}, "required": ["timezone"]}}
]

def execute_tool(name, input_data):
    if name == "get_weather":
        return json.dumps(MOCK_WEATHER.get(input_data.get("city", ""), {}))
    elif name == "get_current_time":
        now = datetime.datetime.now(datetime.timezone.utc)
        return f"{now.strftime('%Y-%m-%d %H:%M:%S')} UTC"

def agent_loop_parallel(user_message, max_iterations=10):
    messages = [{"role": "user", "content": user_message}]
    for i in range(1, max_iterations + 1):
        response = client.messages.create(
            model="claude-sonnet-4-20250514", max_tokens=1024, tools=TOOLS, messages=messages
        )
        if response.stop_reason == "tool_use":
            tool_blocks = [b for b in response.content if b.type == "tool_use"]
            if tool_blocks:
                tool_results = []
                for block in tool_blocks:
                    result = execute_tool(block.name, block.input)
                    tool_results.append({"type": "tool_result", "tool_use_id": block.id, "content": result})
                messages.append({"role": "assistant", "content": response.content})
                messages.append({"role": "user", "content": tool_results})
        elif response.stop_reason == "end_turn":
            return response.content[0].text
        else:
            return response.content[0].text

result = agent_loop_parallel("Weather in Tokyo and time in London?")
print(result)
```
</details>

### Exercise 1: Retry Logic with Exponential Backoff

**Task:** Wrap API calls in a retry function that catches `RateLimitError` and `APIError`, waits with exponential backoff (1s, 2s, 4s, 8s), and retries up to 5 times. Return a user-friendly error if all retries fail.

**Hint:** Use `time.sleep(2 ** attempt)` for backoff. Define a `retry_api_call()` wrapper that accepts a callable.

<details>
<summary>Show solution</summary>

```python
import os, time, datetime
from anthropic import Anthropic, RateLimitError, APIError

client = Anthropic()
TOOLS = [
    {"name": "get_current_time", "description": "Get the time",
     "input_schema": {"type": "object", "properties": {"timezone": {"type": "string"}}, "required": ["timezone"]}}
]

def retry_api_call(fn, max_retries=5):
    last_error = None
    for attempt in range(max_retries):
        try:
            return fn()
        except (RateLimitError, APIError) as e:
            last_error = e
            if attempt < max_retries - 1:
                time.sleep(2 ** attempt)
    raise last_error

def execute_tool(name, input_data):
    if name == "get_current_time":
        now = datetime.datetime.now(datetime.timezone.utc)
        return now.strftime('%Y-%m-%d %H:%M:%S') + " UTC"

def agent_loop(user_message, max_iterations=10):
    messages = [{"role": "user", "content": user_message}]
    for i in range(1, max_iterations + 1):
        try:
            response = retry_api_call(lambda: client.messages.create(
                model="claude-sonnet-4-20250514", max_tokens=1024, tools=TOOLS, messages=messages))
        except Exception as e:
            return f"Stopped after retries: {e}"
        if response.stop_reason == "tool_use":
            for block in response.content:
                if block.type == "tool_use":
                    tool_result = execute_tool(block.name, block.input)
                    messages.append({"role": "assistant", "content": response.content})
                    messages.append({"role": "user", "content": [{"type": "tool_result", "tool_use_id": block.id, "content": tool_result}]})
        elif response.stop_reason == "end_turn":
            return response.content[0].text

result = agent_loop("What time is it in UTC?")
print(result)
```
</details>

### Exercise 2: Fallback Behavior When Tools Fail

**Task:** When `get_weather` returns an error (city not found), the agent should try a fallback city or clearly explain the issue.

**Hint:** Return structured error information from the tool. Use a system prompt that encourages fallback behavior.

<details>
<summary>Show solution</summary>

```python
import json, os
from anthropic import Anthropic

client = Anthropic()

WEATHER = {"Tokyo": {"temperature": 22, "conditions": "partly cloudy", "humidity": 65}}
FALLBACK_MAP = {"tokyo": None, "nyc": "New York", "london": None}

TOOLS = [
    {"name": "get_weather", "description": "Get weather",
     "input_schema": {"type": "object", "properties": {"city": {"type": "string"}}, "required": ["city"]}}
]

SYSTEM_PROMPT = """If the weather tool returns an error, ask the user if they
meant a nearby city. Do not make up weather data."""

def execute_tool(name, input_data):
    if name == "get_weather":
        city = input_data.get("city", "").strip()
        data = WEATHER.get(city)
        if data is None:
            fb = FALLBACK_MAP.get(city.lower())
            if fb:
                return json.dumps({"status": "not_found", "suggested_fallback": fb})
            return json.dumps({"status": "not_found", "message": f"No data for {city}"})
        return json.dumps({"status": "ok", "city": city, **data})

def agent_loop(user_message, max_iterations=10):
    messages = [{"role": "user", "content": user_message}]
    for i in range(1, max_iterations + 1):
        response = client.messages.create(
            model="claude-sonnet-4-20250514", max_tokens=1024, tools=TOOLS,
            messages=messages, system=SYSTEM_PROMPT)
        if response.stop_reason == "tool_use":
            for block in response.content:
                if block.type == "tool_use":
                    tool_result = execute_tool(block.name, block.input)
                    messages.append({"role": "assistant", "content": response.content})
                    messages.append({"role": "user", "content": [{"type": "tool_result", "tool_use_id": block.id, "content": tool_result}]})
        elif response.stop_reason == "end_turn":
            return response.content[0].text

result = agent_loop("What is the weather in NYC?")
print(result)
```
</details>

### Exercise 3: Token Budget Per Session

**Task:** Track total input and output tokens across all iterations. Stop and report if the budget is exceeded (e.g., 10,000 input or 4,000 output tokens).

**Hint:** Access `response.usage.input_tokens` and `response.usage.output_tokens` on each response. Accumulate the totals and check after each iteration.

<details>
<summary>Show solution</summary>

```python
import os, datetime
from anthropic import Anthropic

client = Anthropic()
MAX_INPUT = 10000
MAX_OUTPUT = 4000

TOOLS = [
    {"name": "get_current_time", "description": "Get the time",
     "input_schema": {"type": "object", "properties": {"timezone": {"type": "string"}}, "required": ["timezone"]}}
]

def execute_tool(name, input_data):
    if name == "get_current_time":
        return datetime.datetime.now(datetime.timezone.utc).strftime('%Y-%m-%d %H:%M:%S') + " UTC"

def agent_loop_with_budget(user_message, max_iterations=15):
    messages = [{"role": "user", "content": user_message}]
    total_in = 0
    total_out = 0

    for i in range(1, max_iterations + 1):
        response = client.messages.create(
            model="claude-sonnet-4-20250514", max_tokens=1024, tools=TOOLS, messages=messages)
        total_in += response.usage.input_tokens
        total_out += response.usage.output_tokens

        if total_in > MAX_INPUT:
            return f"Stopped: input budget ({MAX_INPUT}) exceeded ({total_in})"
        if total_out > MAX_OUTPUT:
            return f"Stopped: output budget ({MAX_OUTPUT}) exceeded ({total_out})"

        if response.stop_reason == "tool_use":
            for block in response.content:
                if block.type == "tool_use":
                    tool_result = execute_tool(block.name, block.input)
                    messages.append({"role": "assistant", "content": response.content})
                    messages.append({"role": "user", "content": [{"type": "tool_result", "tool_use_id": block.id, "content": tool_result}]})
        elif response.stop_reason == "end_turn":
            return response.content[0].text

result = agent_loop_with_budget("Time in Tokyo? And New York? And London?")
print(result)
```
</details>
