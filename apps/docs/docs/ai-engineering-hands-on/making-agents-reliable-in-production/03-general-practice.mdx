## Tier 1 — Beginner

### Exercise 1.1: Add a circuit breaker to the tool caller

**Task:** Implement a circuit breaker that stops calling a failing tool after 3 consecutive failures and only retries after a 30-second cooldown. This prevents the agent from wasting retries on a service that is known to be down.

**Hint:** The circuit breaker has three states: CLOSED (normal), OPEN (stopped, fail-fast), and HALF_OPEN (testing after cooldown). Track the failure count per tool name.

### Solution

```python
import time


class CircuitBreaker:
    """Circuit breaker for external tool calls."""

    def __init__(
        self,
        failure_threshold: int = 3,
        cooldown_seconds: float = 30.0,
    ):
        self.failure_threshold = failure_threshold
        self.cooldown_seconds = cooldown_seconds
        self._failures: dict[str, int] = {}
        self._last_failure: dict[str, float] = {}
        self._state: dict[str, str] = {}

    def call(self, tool_name: str, func, *args, **kwargs):
        """
        Call a tool through the circuit breaker.

        Raises CircuitBreakerOpenError if the breaker is open.
        """
        state = self._state.get(tool_name, "closed")

        if state == "open":
            # Check if cooldown has elapsed
            last_fail = self._last_failure.get(tool_name, 0)
            if time.time() - last_fail >= self.cooldown_seconds:
                self._state[tool_name] = "half-open"
                print(f"  Circuit breaker HALF-OPEN for {tool_name}")
            else:
                raise CircuitBreakerOpenError(
                    f"Circuit breaker OPEN for {tool_name}. "
                    f"Try again in {self.cooldown_seconds - (time.time() - last_fail):.0f}s"
                )

        try:
            result = func(*args, **kwargs)
            # Success — reset
            self._failures[tool_name] = 0
            self._state[tool_name] = "closed"
            return result

        except Exception as e:
            self._failures[tool_name] = (
                self._failures.get(tool_name, 0) + 1
            )
            self._last_failure[tool_name] = time.time()

            if self._failures[tool_name] >= self.failure_threshold:
                self._state[tool_name] = "open"
                print(f"  Circuit breaker OPEN for {tool_name} "
                      f"(after {self._failures[tool_name]} failures)")

            raise e


class CircuitBreakerOpenError(Exception):
    """Raised when a circuit breaker is open."""
    pass


# Test
import random

breaker = CircuitBreaker(
    failure_threshold=3, cooldown_seconds=5
)


def flaky_search(query):
    """Simulate a flaky search API."""
    if random.random() < 0.6:
        raise ConnectionError("Search API unavailable")
    return f"Results for: {query}"


for i in range(10):
    try:
        result = breaker.call("search", flaky_search, "test query")
        print(f"Call {i+1}: SUCCESS — {result}")
    except CircuitBreakerOpenError as e:
        print(f"Call {i+1}: BLOCKED — {e}")
    except ConnectionError as e:
        print(f"Call {i+1}: FAILED — {e}")
    time.sleep(0.5)
```

### Exercise 1.2: Add a human-in-the-loop confirmation step

**Task:** Modify the agent to pause for human confirmation before executing any action that the agent classifies as "high stakes" (e.g., anything containing "delete", "purchase", "refund", "cancel"). The agent should return a "pending_approval" status and wait for explicit approval.

**Hint:** Check the agent's response for keywords and return a special status instead of executing. The caller can then call `agent.confirm(action_id)` to proceed.

### Solution

```python
import uuid
import time


class HumanInTheLoop:
    """Manages human-in-the-loop approvals."""

    def __init__(self):
        self.pending_actions: dict[str, dict] = {}

    HIGH_STAKE_KEYWORDS = [
        "delete", "remove", "cancel", "purchase", "buy",
        "refund", "credit", "charge", "suspend", "terminate",
    ]

    def check_action(self, action: dict) -> dict:
        """
        Check if an action needs human approval.

        Returns:
            {"status": "approved", "action_id": ...}
            or {"status": "pending_approval", "action_id": ..., "details": ...}
        """
        action_text = str(action).lower()
        needs_approval = any(
            keyword in action_text
            for keyword in self.HIGH_STAKE_KEYWORDS
        )

        if not needs_approval:
            return {"status": "approved", "action_id": None}

        action_id = str(uuid.uuid4())[:8]
        self.pending_actions[action_id] = {
            "action": action,
            "created_at": time.time(),
            "status": "pending",
        }
        return {
            "status": "pending_approval",
            "action_id": action_id,
            "details": action,
            "message": "This action requires human approval.",
        }

    def approve(self, action_id: str) -> dict:
        """Approve a pending action."""
        if action_id not in self.pending_actions:
            return {"status": "error", "message": "Action not found"}

        action = self.pending_actions[action_id]
        action["status"] = "approved"
        return {
            "status": "approved",
            "action_id": action_id,
            "action": action["action"],
        }

    def reject(self, action_id: str) -> dict:
        """Reject a pending action."""
        if action_id not in self.pending_actions:
            return {"status": "error", "message": "Action not found"}

        action = self.pending_actions[action_id]
        action["status"] = "rejected"
        return {
            "status": "rejected",
            "action_id": action_id,
            "message": "Action rejected by human.",
        }


# Example usage
hil = HumanInTheLoop()

# Test low-stakes action
result = hil.check_action(
    {"tool": "search", "query": "order status"}
)
print(f"Low-stakes: {result['status']}")

# Test high-stakes action
result = hil.check_action(
    {"tool": "update_order", "action": "cancel order #12345"}
)
print(f"High-stakes: {result['status']}")
if result['status'] == 'pending_approval':
    print(f"  Action ID: {result['action_id']}")
    print(f"  Message: {result['message']}")

    # Human approves
    approval = hil.approve(result['action_id'])
    print(f"  After approval: {approval['status']}")
```

### Exercise 1.3: Add token usage tracking to the logger

**Task:** Modify the `StructuredLogger` to write a summary report at the end of each agent session. The summary should include: total tokens used, total cost, steps taken, and a breakdown of tokens per step. This summary is useful for cost analysis and debugging.

**Hint:** After the agent finishes, call `logger.summarize(session_id)` to compute and log the summary.

### Solution

```python
import json
from collections import Counter


class EnhancedLogger:
    """Logger with session summary capability."""

    def __init__(self, log_path: str = "agent_log.jsonl"):
        self.log_path = log_path
        self._session_entries: dict[str, list] = {}

    def log(self, session_id: str, step: int, action: str, details: dict):
        """Write a structured log entry."""
        entry = {
            "session_id": session_id,
            "step": step,
            "action": action,
        }
        entry.update(details)

        with open(self.log_path, "a") as f:
            f.write(json.dumps(entry) + "\n")

        # Track in memory
        if session_id not in self._session_entries:
            self._session_entries[session_id] = []
        self._session_entries[session_id].append(entry)

    def summarize(self, session_id: str) -> dict:
        """Generate a summary for a session."""
        entries = self._session_entries.get(session_id, [])

        if not entries:
            return {"error": "No entries found"}

        total_tokens = sum(
            e.get("input_tokens", 0) + e.get("output_tokens", 0)
            for e in entries
        )
        total_cost = sum(
            e.get("cumulative_cost", 0)
            for e in entries
            if "cumulative_cost" in e
        )
        llm_calls = sum(
            1 for e in entries if e["action"] == "llm_call"
        )
        tool_calls = sum(
            1 for e in entries if e["action"] == "tool_call"
        )
        errors = sum(
            1 for e in entries if e["action"] == "llm_error"
        )

        # Token per step
        token_per_step = {}
        for e in entries:
            step = e["step"]
            if step not in token_per_step:
                token_per_step[step] = 0
            token_per_step[step] += (
                e.get("input_tokens", 0) + e.get("output_tokens", 0)
            )

        summary = {
            "session_id": session_id,
            "total_tokens": total_tokens,
            "total_cost": total_cost,
            "llm_calls": llm_calls,
            "tool_calls": tool_calls,
            "errors": errors,
            "steps_used": len(token_per_step),
            "tokens_per_step": token_per_step,
        }

        # Write summary to log
        self.log(session_id, "summary", "session_summary", summary)

        return summary


# Example
logger = EnhancedLogger()

# Simulate a session
logger.log("sess_001", 1, "llm_call",
           {"input_tokens": 50, "output_tokens": 30, "cumulative_cost": 0.0001})
logger.log("sess_001", 2, "tool_call", {"tool": "search", "latency_ms": 200})
logger.log("sess_001", 2, "llm_call",
           {"input_tokens": 80, "output_tokens": 40, "cumulative_cost": 0.0002})

summary = logger.summarize("sess_001")
for key, value in summary.items():
    print(f"  {key}: {value}")
```

### Exercise 2.1: Agent with tool-use fallback chain

**Task:** Build an agent that has three tools: `search_web` (primary), `search_cache` (secondary), and `fallback_response` (tertiary). If the primary tool fails, the agent automatically falls back to the secondary. If the secondary fails, it uses the fallback. The agent should never crash — it should always produce a response, even if it is "I could not find the information."

**Hint:** Use a chain-of-responsibility pattern. Each tool has a `can_handle()` and `execute()` method. If a tool fails, the next in the chain is tried.

### Solution

```python
import os
import json
from openai import OpenAI


class ToolChain:
    """Chain of tools with fallback."""

    def __init__(self):
        self.client = OpenAI(
            api_key=os.environ["OPENAI_API_KEY"]
        )
        self.tools = [
            self.search_web,
            self.search_cache,
            self.fallback_response,
        ]

    def search_web(self, query: str) -> dict:
        """Primary tool: web search."""
        print(f"  Trying: web search for '{query}'")
        # Simulate: could fail
        import random
        if random.random() < 0.4:
            raise ConnectionError("Web search API unavailable")
        return {"source": "web", "result": f"Web results for: {query}"}

    def search_cache(self, query: str) -> dict:
        """Secondary tool: cached results."""
        print(f"  Trying: cache search for '{query}'")
        # Cache is more reliable
        import random
        if random.random() < 0.2:
            raise ConnectionError("Cache unavailable")
        return {"source": "cache", "result": f"Cached result for: {query}"}

    def fallback_response(self, query: str) -> dict:
        """Tertiary tool: always works."""
        print(f"  Using: fallback response")
        return {
            "source": "fallback",
            "result": "I could not find the specific information you requested. "
                      "Please try rephrasing your question or contact support.",
        }

    def execute(self, query: str) -> dict:
        """Execute the tool chain."""
        for tool in self.tools:
            try:
                return tool(query)
            except Exception as e:
                print(f"    Tool failed: {e}")
                continue
        return {
            "source": "none",
            "result": "All tools failed. Please try again later.",
        }


# Test
chain = ToolChain()

for query in ["What is the weather?", "How do I reset my password?"]:
    print(f"\nQuery: {query}")
    result = chain.execute(query)
    print(f"  Source: {result['source']}")
    print(f"  Result: {result['result'][:80]}...")
```

### Exercise 2.2: Concurrent agent execution with pooled resources

**Task:** Build a system that runs multiple agent sessions concurrently, with a shared resource pool (API key rate limits, database connections). Each agent gets a token from the pool before making an API call. If the pool is empty, the agent waits.

**Hint:** Use `asyncio` and `asyncio.Semaphore` to limit concurrent API calls. The semaphore count should match your API's rate limit (e.g., 5 concurrent calls).

### Solution

```python
import asyncio
import os
import time
from openai import AsyncOpenAI


class RateLimitedAgentPool:
    """Manages concurrent agent execution with rate limiting."""

    def __init__(
        self,
        max_concurrent: int = 5,
        model: str = "gpt-4o-mini",
    ):
        self.semaphore = asyncio.Semaphore(max_concurrent)
        self.client = AsyncOpenAI(
            api_key=os.environ["OPENAI_API_KEY"]
        )
        self.model = model
        self.active_sessions = 0

    async def run_agent(self, session_id: str, prompt: str) -> dict:
        """Run a single agent session with rate limiting."""
        start = time.time()

        async with self.semaphore:
            self.active_sessions += 1
            print(f"[{session_id}] Started "
                  f"(active: {self.active_sessions})")

            try:
                response = await self.client.chat.completions.create(
                    model=self.model,
                    messages=[
                        {"role": "user", "content": prompt},
                    ],
                    max_tokens=100,
                )
                result = response.choices[0].message.content
                elapsed = time.time() - start
                print(f"[{session_id}] Completed in {elapsed:.1f}s")
                return {
                    "session_id": session_id,
                    "result": result,
                    "elapsed": elapsed,
                }

            except Exception as e:
                elapsed = time.time() - start
                print(f"[{session_id}] Failed after {elapsed:.1f}s: {e}")
                return {
                    "session_id": session_id,
                    "error": str(e),
                    "elapsed": elapsed,
                }

            finally:
                self.active_sessions -= 1

    async def run_batch(
        self, prompts: list[tuple[str, str]]
    ) -> list[dict]:
        """
        Run multiple agent sessions concurrently.

        Args:
            prompts: List of (session_id, prompt) tuples.
        """
        tasks = [
            self.run_agent(sid, prompt)
            for sid, prompt in prompts
        ]
        return await asyncio.gather(*tasks)


# Test
async def main():
    pool = RateLimitedAgentPool(max_concurrent=3)

    prompts = [
        (f"session_{i}", f"Tell me a fact about number {i}.")
        for i in range(10)
    ]

    print(f"Running {len(prompts)} sessions with "
          f"max {pool.semaphore._value} concurrent...")
    results = await pool.run_batch(prompts)

    succeeded = sum(1 for r in results if "result" in r)
    failed = sum(1 for r in results if "error" in r)
    total_time = max(r["elapsed"] for r in results)

    print(f"\nResults: {succeeded} succeeded, "
          f"{failed} failed, "
          f"total wall time: {total_time:.1f}s")

    # Compare with sequential (would be ~10x longer)
    print(f"Sequential would take approximately "
          f"{total_time * len(prompts) / pool.semaphore._value:.1f}s")


asyncio.run(main())
```

### Exercise 2.3: Agent performance benchmarking

**Task:** Create a benchmark that measures the reliability infrastructure overhead. Run the agent 10 times with all safeguards enabled, and 10 times with safeguards disabled. Compare: mean latency, failure rate, cost per run, and memory usage. This tells you the cost of reliability.

**Hint:** Create a `run_with_safeguards()` and `run_without_safeguards()` function. Use `time.perf_counter()` for latency and track all failures.

### Solution

```python
import time
import statistics


def benchmark_agent_infrastructure(n_runs: int = 10):
    """Benchmark the reliability infrastructure overhead."""
    from reliable_agent import ReliableAgent

    # Baseline: no safeguards
    results_baseline = []
    for i in range(n_runs):
        start = time.perf_counter()
        agent = ReliableAgent(max_steps=1, max_tokens=5000, max_cost=1.0)
        result = agent.run("What is 2+2?")
        elapsed = time.perf_counter() - start
        results_baseline.append({
            "latency": elapsed,
            "status": result["status"],
            "cost": result["stats"]["total_cost"],
        })

    # With full safeguards (retries, logging, state persistence)
    results_safeguarded = []
    for i in range(n_runs):
        start = time.perf_counter()
        agent = ReliableAgent(max_steps=1, max_tokens=100, max_cost=0.001)
        result = agent.run("Write a 1000 word essay about AI.")
        elapsed = time.perf_counter() - start
        results_safeguarded.append({
            "latency": elapsed,
            "status": result["status"],
            "cost": result["stats"]["total_cost"],
        })

    # Stats
    print(f"{'Metric':<35} {'Baseline':<15} {'Safeguarded':<15}")
    print("-" * 65)

    for metric, key in [
        ("Mean latency (s)", "latency"),
        ("Success rate", "status"),
        ("Mean cost ($)", "cost"),
    ]:
        if key == "latency":
            b = statistics.mean(r["latency"] for r in results_baseline)
            s = statistics.mean(r["latency"] for r in results_safeguarded)
        elif key == "status":
            b = sum(1 for r in results_baseline if r["status"] == "completed") / n_runs
            s = sum(1 for r in results_safeguarded if r["status"] != "completed") / n_runs
        elif key == "cost":
            b = statistics.mean(r["cost"] for r in results_baseline)
            s = statistics.mean(r["cost"] for r in results_safeguarded)

        if isinstance(b, float):
            print(f"{metric:<35} {b:<15.4f} {s:<15.4f}")
        else:
            print(f"{metric:<35} {b:<15} {s:<15}")

    overhead = (
        (statistics.mean(r["latency"] for r in results_safeguarded)
         / statistics.mean(r["latency"] for r in results_baseline) - 1) * 100
    )
    print(f"\nReliability infrastructure overhead: {overhead:.1f}% latency increase")


benchmark_agent_infrastructure(5)
```

### Exercise 3.1: Agent monitoring dashboard with real-time metrics

**Task:** Build a lightweight monitoring dashboard that shows real-time agent metrics: active sessions, running total cost, error rate, and average latency. Use a simple web interface (or print to console with updates every 2 seconds).

**Solution approach:**

```python
import time
import threading
import json
from collections import deque
from datetime import datetime


class AgentMonitor:
    """Real-time agent monitoring with metrics."""

    def __init__(self, window_minutes: int = 5):
        self.events = deque(maxlen=1000)
        self.window = window_minutes * 60
        self.active_sessions = 0
        self.total_sessions = 0
        self._lock = threading.Lock()

    def record_event(self, event: dict):
        """Record an agent event."""
        with self._lock:
            event["timestamp"] = time.time()
            self.events.append(event)
            if event.get("action") == "session_start":
                self.active_sessions += 1
                self.total_sessions += 1
            elif event.get("action") == "session_end":
                self.active_sessions -= 1

    def get_metrics(self) -> dict:
        """Get current metrics."""
        with self._lock:
            now = time.time()
            cutoff = now - self.window

            recent = [
                e for e in self.events
                if e["timestamp"] > cutoff
            ]

            if not recent:
                return {"status": "no_data"}

            total = len(recent)
            errors = sum(
                1 for e in recent
                if e.get("level") == "error"
            )
            latencies = [
                e.get("latency", 0) for e in recent
                if "latency" in e
            ]

            total_cost = sum(
                e.get("cost", 0) for e in recent
            )

            return {
                "timestamp": datetime.now().isoformat(),
                "active_sessions": self.active_sessions,
                "total_sessions": self.total_sessions,
                "recent_events": total,
                "error_rate": errors / total if total > 0 else 0,
                "avg_latency": (
                    sum(latencies) / len(latencies)
                    if latencies else 0
                ),
                "total_cost_window": total_cost,
                "events_per_second": total / self.window,
            }

    def print_dashboard(self):
        """Print a live dashboard (call every 2 seconds)."""
        metrics = self.get_metrics()
        if metrics.get("status") == "no_data":
            print("Waiting for data...")
            return

        print(f"\n[{metrics['timestamp']}]")
        print(f"  Active sessions: {metrics['active_sessions']}")
        print(f"  Total sessions:  {metrics['total_sessions']}")
        print(f"  Events (5min):   {metrics['recent_events']}")
        print(f"  Error rate:      {metrics['error_rate']:.1%}")
        print(f"  Avg latency:     {metrics['avg_latency']:.2f}s")
        print(f"  Cost (5min):     ${metrics['total_cost_window']:.4f}")
        print(f"  Events/sec:      {metrics['events_per_second']:.1f}")


# Simulate in a background thread
def simulate_agent_traffic(monitor):
    """Generate simulated agent events."""
    import random
    for i in range(50):
        monitor.record_event({
            "action": "session_start",
            "session_id": f"sim_{i}",
        })
        time.sleep(random.uniform(0.1, 0.5))

        # Simulate steps
        for step in range(random.randint(1, 5)):
            monitor.record_event({
                "action": "step",
                "session_id": f"sim_{i}",
                "step": step,
                "latency": random.uniform(0.5, 3.0),
                "cost": random.uniform(0.001, 0.01),
            })

        # Simulate occasional errors
        if random.random() < 0.1:
            monitor.record_event({
                "action": "session_end",
                "session_id": f"sim_{i}",
                "level": "error",
                "error": "Rate limit exceeded",
            })
        else:
            monitor.record_event({
                "action": "session_end",
                "session_id": f"sim_{i}",
                "level": "info",
            })


# Run
monitor = AgentMonitor()
t = threading.Thread(
    target=simulate_agent_traffic, args=(monitor,), daemon=True
)
t.start()

for _ in range(10):
    monitor.print_dashboard()
    time.sleep(2)
```

### Exercise 3.2: Agent with automatic recovery from crashes

**Task:** Build a system that detects when an agent has crashed (process killed, server restart) and automatically resumes it from the last state checkpoint. The resumed agent should continue from where it left off, not start over.

**Solution approach:**

```python
import json
import time
from state_store import StateStore, AgentState


class ResilientAgentRunner:
    """
    Agent runner that can recover from crashes using state checkpoints.
    """

    def __init__(self, state_store: StateStore = None):
        self.store = state_store or StateStore()

    def run_or_resume(self, session_id: str, user_input: str) -> dict:
        """
        Run an agent, resuming from checkpoint if available.

        If the session exists and is incomplete, resume it.
        """
        try:
            state = self.store.load(session_id)
            print(f"Resuming session {session_id} "
                  f"from step {state.step} "
                  f"(status: {state.status})")

            if state.status == "completed":
                return {
                    "session_id": session_id,
                    "status": "already_completed",
                    "response": state.conversation_history[-1]["content"],
                    "stats": {
                        "steps": state.step,
                        "total_tokens": state.total_tokens,
                        "total_cost": state.total_cost,
                    },
                }

            # Resume from last checkpoint
            return self._continue_from_state(state)

        except KeyError:
            # No existing session — start fresh
            print(f"Starting new session {session_id}")
            return self._start_new_session(session_id, user_input)

    def _start_new_session(
        self, session_id: str, user_input: str
    ) -> dict:
        """Start a brand new agent session."""
        from reliable_agent import ReliableAgent

        agent = ReliableAgent()
        agent.session_id = session_id
        return agent.run(user_input)

    def _continue_from_state(self, state: AgentState) -> dict:
        """Continue an agent from a saved state."""
        from reliable_agent import ReliableAgent

        agent = ReliableAgent()
        agent.session_id = state.session_id

        # Restore budget from state
        agent.budget.total_input_tokens = state.total_tokens // 2
        agent.budget.total_output_tokens = state.total_tokens // 2

        # Continue from last step
        for step in range(state.step + 1, agent.max_steps + 1):
            print(f"  Continuing at step {step}")
            # ... (simplified: in production, re-run the LLM
            # with the conversation history from the state)

        return {"status": "recovered", "session_id": state.session_id}

crash_state = AgentState(
    session_id="crash_recovery_test",
    step=2,
    conversation_history=[
        {"role": "user", "content": "Find information about topic X"},
        {"role": "assistant", "content": "I will search for topic X."},
    ],
    status="running",
    total_tokens=150,
)
runner.store.save(crash_state)

### Exercise 3.3: Cost-aware agent with dynamic model selection

**Task:** Build an agent that dynamically selects which model to use for each step based on the complexity of the task. Simple tasks (factual lookups, yes/no questions) use a cheap model ($0.15/1M tokens). Complex tasks (reasoning, code generation, multi-step planning) use an expensive model ($2.50/1M tokens). The agent tracks actual cost against a budget and downgrades models if running over budget.

**Solution approach:**

```python
import os
import time
from openai import OpenAI


class DynamicModelSelector:
    """Selects the appropriate model based on task complexity."""

    MODELS = {
        "cheap": {
            "name": "gpt-4o-mini",
            "cost_per_1k_input": 0.00015,
            "cost_per_1k_output": 0.00060,
        },
        "expensive": {
            "name": "gpt-4o",
            "cost_per_1k_input": 0.00250,
            "cost_per_1k_output": 0.01000,
        },
    }

    SIMPLE_TASK_INDICATORS = [
        "what is", "who is", "when did", "where is",
        "how many", "define", "list", "find", "search",
        "yes/no", "true/false", "capital of", "convert",
    ]

    def __init__(self, total_budget: float = 0.05):
        self.client = OpenAI(
            api_key=os.environ["OPENAI_API_KEY"]
        )
        self.total_budget = total_budget
        self.spent = 0.0
        self.cheap_calls = 0
        self.expensive_calls = 0

    def classify_task(self, task: str) -> str:
        """Classify task as 'simple' or 'complex'."""
        task_lower = task.lower()
        for indicator in self.SIMPLE_TASK_INDICATORS:
            if task_lower.startswith(indicator):
                return "cheap"
        return "expensive"

    def execute(self, task: str, system_prompt: str = None) -> dict:
        """Execute a task with dynamic model selection."""
        tier = self.classify_task(task)

        # If over budget, force cheap model
        if self.spent >= self.total_budget * 0.8:
            if tier == "expensive":
                print(f"  Budget at {self.spent:.4f}/{self.total_budget:.4f} "
                      f"→ downgrading to cheap model")
            tier = "cheap"

        model_info = self.MODELS[tier]
        model_name = model_info["name"]

        start = time.time()
        response = self.client.chat.completions.create(
            model=model_name,
            messages=[
                {"role": "system", "content": system_prompt or "You are helpful."},
                {"role": "user", "content": task},
            ],
        )
        elapsed = time.time() - start

        input_tokens = response.usage.prompt_tokens
        output_tokens = response.usage.completion_tokens

        cost = (
            input_tokens * model_info["cost_per_1k_input"] / 1000
            + output_tokens * model_info["cost_per_1k_output"] / 1000
        )
        self.spent += cost

        if tier == "cheap":
            self.cheap_calls += 1
        else:
            self.expensive_calls += 1

        return {
            "model": model_name,
            "tier": tier,
            "response": response.choices[0].message.content,
            "cost": cost,
            "total_spent": self.spent,
            "latency": elapsed,
            "input_tokens": input_tokens,
            "output_tokens": output_tokens,
        }


# Test
selector = DynamicModelSelector(total_budget=0.05)

tasks = [
    ("simple", "What is the capital of France?"),
    ("complex", "Write a Python function that implements "
                "a binary search tree with insert, delete, and search."),
    ("simple", "Convert 100 degrees Fahrenheit to Celsius."),
    ("complex", "Explain the difference between REST and GraphQL "
                "and provide examples of when to use each."),
    ("simple", "What is 2+2?"),
]

for difficulty, task in tasks:
    print(f"\n[{difficulty}] {task[:50]}...")
    result = selector.execute(task)
    print(f"  Model: {result['model']} "
          f"(cost=${result['cost']:.6f}, "
          f"total=${result['total_spent']:.4f})")

print(f"\nSummary:")
print(f"  Cheap calls: {selector.cheap_calls}")
print(f"  Expensive calls: {selector.expensive_calls}")
print(f"  Total spent: ${selector.spent:.4f}")
print(f"  Budget: ${selector.total_budget:.4f}")
print(f"  Budget remaining: ${selector.total_budget - selector.spent:.4f}")

The key insight is that most user requests are simple enough for a cheap model, and the expensive model is reserved for the minority of requests that genuinely need it.
