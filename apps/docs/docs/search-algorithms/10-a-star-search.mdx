def a_star_search(graph: dict[str, list[tuple[str, int]]], start: str, goal: str, heuristic: dict[str, int]) -> list[str]:
    if start not in graph or goal not in graph:
        return []
    if start == goal:
        return [start]

    g_score: dict[str, int] = {start: 0}
    parent: dict[str, str] = {}
    open_set: list[tuple[int, str]] = [(heuristic.get(start, 0), start)]

    while open_set:
        current_f, current = heapq.heappop(open_set)
        current_g = g_score.get(current, float("inf"))
        if current_f != current_g + heuristic.get(current, 0):
            continue  # stale entry, a better one was already processed

        for neighbor, weight in graph.get(current, []):
            tentative_g = current_g + weight
            if tentative_g < g_score.get(neighbor, float("inf")):
                g_score[neighbor] = tentative_g
                parent[neighbor] = current
                heapq.heappush(open_set, (tentative_g + heuristic.get(neighbor, 0), neighbor))

    if goal not in g_score:
        return []

    path: list[str] = []
    node: str | None = goal
    while node is not None:
        path.append(node)
        node = parent.get(node)
    path.reverse()
    return path`}</CodeBlock>

#include <string.h>
#include <limits.h>

#define MAX_NODES 50
#define MAX_EDGES 20
#define MAX_NAME 32
#define MAX_OPEN 400

typedef struct { char label[MAX_NAME]; int f; } OpenEntry;

typedef struct {
    char names[MAX_NODES][MAX_NAME];
    char edgeTo[MAX_NODES][MAX_EDGES][MAX_NAME];
    int edgeWeight[MAX_NODES][MAX_EDGES];
    int edgeCount[MAX_NODES];
    int nodeCount;
} Graph;

int findNode(Graph *g, const char *name) {
    for (int i = 0; i < g->nodeCount; i++) {
        if (strcmp(g->names[i], name) == 0) return i;
    }
    return -1;
}

int getHeuristic(char heurNames[][MAX_NAME], int heurValues[], int heurCount, const char *node) {
    for (int i = 0; i < heurCount; i++) {
        if (strcmp(heurNames[i], node) == 0) return heurValues[i];
    }
    return 0;
}

int aStarSearch(Graph *g, const char *start, const char *goal,
                char heurNames[][MAX_NAME], int heurValues[], int heurCount,
                char pathOut[][MAX_NAME]) {
    int startIdx = findNode(g, start);
    int goalIdx = findNode(g, goal);
    if (startIdx == -1 || goalIdx == -1) return 0;
    if (strcmp(start, goal) == 0) {
        strcpy(pathOut[0], start);
        return 1;
    }

    int gScore[MAX_NODES];
    int hasG[MAX_NODES];
    int parent[MAX_NODES];
    for (int i = 0; i < g->nodeCount; i++) { hasG[i] = 0; parent[i] = -1; }
    gScore[startIdx] = 0;
    hasG[startIdx] = 1;

    OpenEntry open[MAX_OPEN];
    int openSize = 0;

    strcpy(open[openSize].label, start);
    open[openSize].f = gScore[startIdx] + getHeuristic(heurNames, heurValues, heurCount, start);
    openSize++;

    while (openSize > 0) {
        int minIdx = 0;
        for (int i = 1; i < openSize; i++) {
            if (open[i].f < open[minIdx].f) minIdx = i;
        }
        OpenEntry current = open[minIdx];
        open[minIdx] = open[openSize - 1];
        openSize--;

        int curIdx = findNode(g, current.label);
        int curG = hasG[curIdx] ? gScore[curIdx] : INT_MAX;
        int curH = getHeuristic(heurNames, heurValues, heurCount, current.label);
        if (current.f != curG + curH) continue; // stale entry, a better one was already processed

        for (int e = 0; e < g->edgeCount[curIdx]; e++) {
            const char *neighbor = g->edgeTo[curIdx][e];
            int weight = g->edgeWeight[curIdx][e];
            int neighborIdx = findNode(g, neighbor);
            int tentativeG = curG + weight;
            int existingG = hasG[neighborIdx] ? gScore[neighborIdx] : INT_MAX;
            if (tentativeG < existingG) {
                gScore[neighborIdx] = tentativeG;
                hasG[neighborIdx] = 1;
                parent[neighborIdx] = curIdx;
                strcpy(open[openSize].label, neighbor);
                open[openSize].f = tentativeG + getHeuristic(heurNames, heurValues, heurCount, neighbor);
                openSize++;
            }
        }
    }

    if (!hasG[goalIdx]) return 0;

    char tempPath[MAX_NODES][MAX_NAME];
    int pathLen = 0;
    int node = goalIdx;
    while (node != -1) {
        strcpy(tempPath[pathLen], g->names[node]);
        pathLen++;

    }
    for (int i = 0; i < pathLen; i++) {
        strcpy(pathOut[i], tempPath[pathLen - 1 - i]);
    }

#include <vector>
#include <map>
#include <queue>
#include <limits>
#include <algorithm>

std::vector<std::string> aStarSearch(
    std::map<std::string, std::vector<std::pair<std::string, int>>> &graph,
    const std::string &start, const std::string &goal,
    std::map<std::string, int> &heuristic) {

    std::vector<std::string> result;
    if (graph.find(start) == graph.end() || graph.find(goal) == graph.end()) {
        return result;
    }
    if (start == goal) {
        result.push_back(start);
        return result;
    }

    auto getH = [&](const std::string &node) {
        auto it = heuristic.find(node);
        return it != heuristic.end() ? it->second : 0;
    };

    std::map<std::string, int> gScore;
    std::map<std::string, std::string> parent;
    gScore[start] = 0;

    typedef std::pair<int, std::string> Entry; // (f, label)
    std::priority_queue<Entry, std::vector<Entry>, std::greater<Entry>> openSet;
    openSet.push({gScore[start] + getH(start), start});

    while (!openSet.empty()) {
        Entry top = openSet.top();
        openSet.pop();
        int currentF = top.first;
        std::string current = top.second;
        int currentG = gScore.count(current) ? gScore[current] : std::numeric_limits<int>::max();
        if (currentF != currentG + getH(current)) {
            continue; // stale entry, a better one was already processed
        }
        for (auto &edge : graph[current]) {
            const std::string &neighbor = edge.first;
            int weight = edge.second;
            int tentativeG = currentG + weight;
            int existingG = gScore.count(neighbor) ? gScore[neighbor] : std::numeric_limits<int>::max();
            if (tentativeG < existingG) {
                gScore[neighbor] = tentativeG;
                parent[neighbor] = current;
                openSet.push({tentativeG + getH(neighbor), neighbor});
            }
        }
    }

    if (gScore.find(goal) == gScore.end()) {
        return result;
    }

    std::vector<std::string> path;
    std::string node = goal;
    while (true) {
        path.push_back(node);
        auto it = parent.find(node);
        if (it == parent.end()) break;
        node = it->second;
    }
    std::reverse(path.begin(), path.end());
    return path;
}`}</CodeBlock>

## Summary

A* Search finds the cheapest route between two points in a map of connected places, using an educated guess to avoid wasting time exploring in the wrong direction. Think of a GPS choosing which street to try next: it considers both "how far have I already driven" and "roughly how much further does it look to the destination," and always tries the option that looks most promising overall. That educated guess is called a heuristic. With a good heuristic, A* can find the cheapest path while barely glancing at most of the map; with a poor one, it ends up checking almost everything.

## Pseudocode

```text
1. If the start or the goal isn't in the graph, report an empty path.
2. Keep a running tally of the cheapest cost found so far to reach each place, starting with 0 for the start.
3. Keep a "to explore" pile, ordered by: cost so far to reach this place, plus a guess at how much further to the goal. Put the start in it.
4. While the pile isn't empty, take out the place with the smallest "cost so far plus guess" total.
5. If this entry is out of date (a cheaper way to reach it was found after it was added), skip it and go back to step 4.
6. For each neighbor of the current place, work out the cost to reach that neighbor through the current place.
7. If that's cheaper than any way found before to reach the neighbor, remember this new cheaper cost, remember the current place as how we got there, and add the neighbor to the pile with its updated total.
8. Once the pile is empty, if the goal was ever reached, follow the "how we got there" trail backwards from the goal to the start, then reverse it to get the path in order. Otherwise report an empty path.
```

## How It Works

- In the diagram, we want the cheapest way from `A` to `D`. Straight to `C` and then `D` looks tempting — it's only two hops — but it costs `4 + 1 = 5`.
- A* doesn't just look at hop count, it looks at total cost so far, plus a guess ("heuristic") of how much more it'll cost to reach the goal from here. That guess is written as `h` in the diagram — for example, `h=2` at `C` means "I'd estimate about 2 more to reach the goal from here."
- Going `A -> B` costs `1`. From `B`, going on to `C` costs `1 + 2 = 3` total so far — cheaper than reaching `C` directly from `A` (which costs `4`). So the route through `B` wins out for reaching `C`.
- From `C`, the last hop to `D` costs `1` more, for a total of `4`. That beats the direct `A -> C -> D` route's total of `5`.
- Final path: `A -> B -> C -> D`, at a total cost of `4`.
- This only works correctly if the heuristic never *overestimates* the real remaining distance — it's allowed to guess low, just never guess high. An overly optimistic guess is fine and even helpful; an overly confident (too-high) guess can trick the algorithm into picking a worse path without realizing it. It also assumes no connection has a negative cost.
- How fast is it? With a great heuristic, A* can go almost straight to the goal, barely glancing at any dead ends (close to the best case). With a weak or missing heuristic — for example, guessing 0 everywhere — A* ends up exploring about as much as if it weren't guessing at all, similar to checking every place in order of total cost so far (this is called `O(E log V)` time on average, and can be as slow as exploring nearly everything in the worst case).
