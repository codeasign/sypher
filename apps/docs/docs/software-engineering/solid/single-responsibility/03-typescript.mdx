## Implementation

A strict TypeScript implementation of SRP using an invoice processing system with typed interfaces for each concern.

```typescript
// SRP — TypeScript (strict mode) implementation
// Domain: Invoice processing

// ============================================================
// VIOLATION: Single class handling calculation, persistence,
// printing, and email — multiple reasons to change
// ============================================================
//
// interface OrderItem {
//   name: string;
//   price: number;
//   qty: number;
// }
//
// class InvoiceProcessor {
//   private invoices: Array<{ id: number; items: OrderItem[]; total: number }> = [];
//
//   process(items: OrderItem[], customerEmail: string): void {
//     // Calculation
//     const total = items.reduce((sum, item) => sum + item.price * item.qty, 0);
//     const invoice = { id: Date.now(), items, total, customerEmail };
//
//     // Persistence
//     this.invoices.push(invoice);
//     console.log(`Saved invoice ${invoice.id}`);
//
//     // Printing
//     console.log(`Invoice #${invoice.id}: $${total.toFixed(2)}`);
//
//     // Email
//     console.log(`Emailed to ${customerEmail}`);
//   }
// }

// ============================================================
// CORRECT IMPLEMENTATION — One interface, one class per concern
// ============================================================

interface OrderItem {
  readonly name: string;
  readonly price: number;
  readonly qty: number;
}

interface Invoice {
  readonly id: number;
  readonly items: readonly OrderItem[];
  readonly subtotal: number;
  readonly tax: number;
  readonly discount: number;
  readonly total: number;
  readonly customerEmail: string;
  readonly createdAt: string;
}

// Responsibility 1: Calculate invoice totals
interface IInvoiceCalculator {
  calculate(items: readonly OrderItem[]): {
    subtotal: number;
    tax: number;
    discount: number;
    total: number;
  };
}

class InvoiceCalculator implements IInvoiceCalculator {
  public calculate(items: readonly OrderItem[]): {
    subtotal: number;
    tax: number;
    discount: number;
    total: number;
  } {
    const subtotal = items.reduce((sum, item) => sum + item.price * item.qty, 0);
    const tax = subtotal * 0.08;
    const discount = subtotal > 100 ? subtotal * 0.1 : 0;
    const total = subtotal + tax - discount;
    return { subtotal, tax, discount, total };
  }
}

// Responsibility 2: Persist invoices
interface IInvoiceRepository {
  save(invoice: Invoice): Invoice;
  findById(id: number): Invoice | undefined;
}

class InvoiceRepository implements IInvoiceRepository {
  private readonly invoices: Map<number, Invoice> = new Map();

  public save(invoice: Invoice): Invoice {
    this.invoices.set(invoice.id, invoice);
    console.log(`Saved invoice ${invoice.id} to database`);
    return invoice;
  }

  public findById(id: number): Invoice | undefined {
    return this.invoices.get(id);
  }
}

// Responsibility 3: Print invoices
interface IInvoicePrinter {
  print(invoice: Invoice): void;
}

class InvoicePrinter implements IInvoicePrinter {
  public print(invoice: Invoice): void {
    console.log("=== INVOICE ===");
    console.log(`Invoice #${invoice.id}`);
    for (const item of invoice.items) {
      const lineTotal = item.price * item.qty;
      console.log(`${item.name.padEnd(20)} x${item.qty}  $${lineTotal.toFixed(2)}`);
    }
    console.log("-".repeat(40));
    console.log(`Subtotal:$${invoice.subtotal.toFixed(10)}`.padStart(38));
    console.log(`Discount: -$${invoice.discount.toFixed(2)}`.padStart(38));
    console.log(`Tax:      $${invoice.tax.toFixed(2)}`.padStart(38));
    console.log(`Total:    $${invoice.total.toFixed(2)}`.padStart(38));
    console.log("=".repeat(40));
  }
}

// Responsibility 4: Send invoice emails
interface IInvoiceEmailer {
  sendConfirmation(invoice: Invoice, to: string): void;
}

class InvoiceEmailer implements IInvoiceEmailer {
  public sendConfirmation(invoice: Invoice, to: string): void {
    console.log(`Sending invoice #${invoice.id} to ${to}...`);
    console.log(`Email sent to ${to}`);
  }
}

// --- Main driver ---
function main(): void {
  const orderItems: readonly OrderItem[] = [
    { name: "Widget A", price: 29.99, qty: 2 },
    { name: "Gadget B", price: 49.99, qty: 1 },
    { name: "Accessory C", price: 9.99, qty: 3 }
  ];

  // Compose SRP components
  const calculator: IInvoiceCalculator = new InvoiceCalculator();
  const repository: IInvoiceRepository = new InvoiceRepository();
  const printer: IInvoicePrinter = new InvoicePrinter();
  const emailer: IInvoiceEmailer = new InvoiceEmailer();

  // Process
  const { subtotal, tax, discount, total } = calculator.calculate(orderItems);

  const invoice: Invoice = {
    id: Date.now(),
    items: orderItems,
    subtotal,
    tax,
    discount,
    total,
    customerEmail: "customer@example.com",
    createdAt: new Date().toISOString()
  };

  repository.save(invoice);
  printer.print(invoice);
  emailer.sendConfirmation(invoice, invoice.customerEmail);
}

main();
```

## Key Language-Specific Notes

- **Interfaces per concern** — Each SRP class has a corresponding `I*` interface (`IInvoiceCalculator`, `IInvoiceRepository`), making it possible to mock or swap implementations without touching consumers.
- **`readonly` arrays and properties** — TypeScript's `readonly` modifier prevents accidental mutation of shared data structures, a common source of cross-concern coupling in SRP violations.
- **`strict` mode catches coupling** — With `strictNullChecks` and `noImplicitAny`, the compiler helps enforce clean boundaries: a validator returning `undefined` where a number is expected forces you to address the seam.
- **Type aliases for return types** — Inline object types for calculation results avoid creating a new class per return shape while still being type-safe, balancing SRP with convenience.
- **Dependency injection via constructor** — Each class receives its dependencies explicitly, making the component graph visible and testable at the type level.
