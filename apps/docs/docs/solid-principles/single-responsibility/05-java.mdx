## Implementation

import java.security.MessageDigest;
import java.security.NoSuchAlgorithmException;
import java.security.SecureRandom;
import java.util.HexFormat;
import java.util.Map;
import java.util.concurrent.ConcurrentHashMap;

// --- Record for user data (immutable DTO) ---

record User(String email, String hashedPassword, boolean verified) {}

// --- Responsibility 1: Password hashing ---

class PasswordHasher {
    public String hash(String password) {
        try {
            byte[] salt = new byte[16];
            SecureRandom.getInstanceStrong().nextBytes(salt);
            var md = MessageDigest.getInstance("SHA-256");
            md.update(salt);
            byte[] hash = md.digest(password.getBytes());
            return HexFormat.of().formatHex(salt) + ":" + HexFormat.of().formatHex(hash);
        } catch (NoSuchAlgorithmException e) {
            throw new RuntimeException("SHA-256 not available", e);
        }
    }

    public boolean verify(String password, String storedHash) {
        var parts = storedHash.split(":");
        if (parts.length != 2) return false;
        try {
            byte[] salt = HexFormat.of().parseHex(parts[0]);
            var md = MessageDigest.getInstance("SHA-256");
            md.update(salt);
            byte[] hash = md.digest(password.getBytes());
            String expectedHash = HexFormat.of().formatHex(hash);
            return parts[1].equals(expectedHash);
        } catch (NoSuchAlgorithmException e) {
            throw new RuntimeException("SHA-256 not available", e);
        }
    }
}

// --- Responsibility 2: User persistence ---

interface UserRepository {
    boolean exists(String email);
    void save(User user);
    User findByEmail(String email);
}

class InMemoryUserRepository implements UserRepository {
    private final Map<String, User> users = new ConcurrentHashMap<>();

    @Override
    public boolean exists(String email) {
        return users.containsKey(email);
    }

    @Override
    public void save(User user) {
        users.put(user.email(), user);
    }

    @Override
    public User findByEmail(String email) {
        return users.get(email);
    }
}

// --- Responsibility 3: Email verification ---

interface VerificationSender {
    void sendVerification(String email, String code);
}

class ConsoleVerificationSender implements VerificationSender {
    @Override
    public void sendVerification(String email, String code) {
        System.out.println("Sending verification code " + code + " to " + email);
    }
}

// --- Responsibility 4: Verification code generator ---

class VerificationCodeGenerator {
    private static final String CHARS = "0123456789ABCDEFGHIJKLMNOPQRSTUVWXYZ";

    public String generate(int length) {
        var random = new SecureRandom();
        var sb = new StringBuilder(length);
        for (int i = 0; i < length; i++) {
            sb.append(CHARS.charAt(random.nextInt(CHARS.length())));
        }
        return sb.toString();
    }
}

// --- Orchestrator ---

class UserRegistrar {
    private final PasswordHasher hasher;
    private final UserRepository repository;
    private final VerificationSender sender;
    private final VerificationCodeGenerator codeGen;

    public UserRegistrar(
            PasswordHasher hasher,
            UserRepository repository,
            VerificationSender sender,
            VerificationCodeGenerator codeGen) {
        this.hasher = hasher;
        this.repository = repository;
        this.sender = sender;
        this.codeGen = codeGen;
    }

    public void register(String email, String password) {
        if (repository.exists(email)) {
            throw new IllegalArgumentException("User already exists: " + email);
        }
        String hashedPassword = hasher.hash(password);
        repository.save(new User(email, hashedPassword, false));
        String code = codeGen.generate(6);
        sender.sendVerification(email, code);
    }
}

// --- Main driver ---

public class SingleResponsibilityDemo {
    public static void main(String[] args) {
        var registrar = new UserRegistrar(
            new PasswordHasher(),
            new InMemoryUserRepository(),
            new ConsoleVerificationSender(),
            new VerificationCodeGenerator()
        );

        registrar.register("alice@example.com", "secureP@ss1");
        registrar.register("bob@example.com", "anotherP@ss2");

        System.out.println("\nAll registrations completed successfully.");
    }
}

## Key Language-Specific Notes

- **`record` classes** (Java 16+) provide immutable DTOs with `equals()`, `hashCode()`, and `toString()` automatically — perfect for cross-boundary data that should not be mutable across SRP boundaries.
- **Interface per concern** — `UserRepository` and `VerificationSender` are interfaces, not base classes. Any class with those responsibilities implements the interface, keeping coupling to abstractions.
- **Constructor injection** — The `UserRegistrar` receives all dependencies through its constructor, making its dependencies explicit and its SRP boundaries testable.
- **`ConcurrentHashMap`** — A simple thread-safe map keeps the in-memory repository safe without adding external dependencies, demonstrating that SRP does not require complex infrastructure.
- **`HexFormat`** (Java 17+) — No external libraries needed for hex encoding; the standard library provides clean formatting.
