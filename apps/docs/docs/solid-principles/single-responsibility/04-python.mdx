## Implementation

A Python 3.11+ implementation of SRP using a report generation system. The violation shows a `Report` class that fetches data, formats HTML, writes to file, and sends by email. The correct version splits each concern into its own single-purpose class.

```python
# SRP — Python 3.11+ implementation
# Domain: Report generation

from abc import ABC, abstractmethod
from dataclasses import dataclass, field
from typing import Optional

# ============================================================
# VIOLATION: Report class handles data fetching, HTML formatting,
# file writing, and email sending — 4 reasons to change
# ============================================================
#
# import csv, smtplib
#
# class Report:
#     def __init__(self, filename: str):
#         self.filename = filename
#
#     def generate(self) -> None:
#         # Data fetching
#         data = []
#         with open(self.filename, "r") as f:
#             reader = csv.DictReader(f)
#             for row in reader:
#                 data.append(row)
#
#         # HTML formatting
#         html = "<html><body><table>"
#         for row in data:
#             html += "<tr>"
#             for _, value in row.items():
#                 html += f"<td>{value}</td>"
#             html += "</tr>"
#         html += "</table></body></html>"
#
#         # File writing
#         with open("report.html", "w") as f:
#             f.write(html)
#
#         # Email sending
#         server = smtplib.SMTP("smtp.example.com", 587)
#         server.sendmail(
#             "reports@example.com",
#             ["manager@example.com"],
#             f"Subject: Report\n\n{html}"
#         )
#         server.quit()
#
# Problem: Changing the data source, output format, file location,
# or email recipient all require editing this one class.

# ============================================================
# CORRECT IMPLEMENTATION
# Each class has exactly one responsibility
# ============================================================

@dataclass
class SalesRecord:
    product: str
    quantity: int
    price: float
    total: float = 0.0

    def __post_init__(self) -> None:
        self.total = self.quantity * self.price


class DataFetcher(ABC):
    """Responsibility: fetch data from a source."""
    @abstractmethod
    def fetch(self) -> list[dict[str, str]]:
        ...


class CsvDataFetcher(DataFetcher):
    """Concrete implementation: read CSV files."""
    def __init__(self, filename: str) -> None:
        self.filename = filename

    def fetch(self) -> list[dict[str, str]]:
        import csv
        with open(self.filename, "r", newline="") as f:
            return list(csv.DictReader(f))


class ReportFormatter(ABC):
    """Responsibility: format data for output."""
    @abstractmethod
    def format(self, data: list[SalesRecord]) -> str:
        ...


class HtmlFormatter(ReportFormatter):
    """Concrete implementation: HTML table format."""
    def format(self, data: list[SalesRecord]) -> str:
        rows = "\n".join(
            f"<tr><td>{r.product}</td><td>{r.quantity}</td>"
            f"<td>${r.price:.2f}</td><td>${r.total:.2f}</td></tr>"
            for r in data
        )
        return (
            "<html><body>\n"
            "<h1>Sales Report</h1>\n"
            '<table border="1">\n'
            "<tr><th>Product</th><th>Qty</th><th>Price</th><th>Total</th></tr>\n"
            f"{rows}\n"
            "</table>\n"
            "</body></html>"
        )


class FileWriter(ABC):
    """Responsibility: write content to a destination."""
    @abstractmethod
    def write(self, content: str, path: str) -> None:
        ...


class LocalFileWriter(FileWriter):
    """Concrete implementation: write to local filesystem."""
    def write(self, content: str, path: str) -> None:
        with open(path, "w", encoding="utf-8") as f:
            f.write(content)
        print(f"Written to {path}")


class EmailSender(ABC):
    """Responsibility: send email notifications."""
    @abstractmethod
    def send(self, to: str, subject: str, body: str) -> None:
        ...


class SmtpEmailSender(EmailSender):
    """Concrete implementation: send via SMTP."""
    def __init__(self, host: str = "localhost", port: int = 1025) -> None:
        self.host = host
        self.port = port

    def send(self, to: str, subject: str, body: str) -> None:
        print(f"Sending email to {to}")
        print(f"Subject: {subject}")
        print(f"Body length: {len(body)} chars")
        print(f"Email sent via {self.host}:{self.port}")


# --- Main driver ---
def main() -> None:
    # Create sample CSV
    import tempfile
    import os

    csv_content = (
        "product,quantity,price\n"
        "Laptop,10,999.99\n"
        "Mouse,50,29.99\n"
        "Keyboard,30,89.99\n"
        "Monitor,15,299.99\n"
    )
    tmp = tempfile.NamedTemporaryFile(mode="w", suffix=".csv", delete=False)
    tmp.write(csv_content)
    tmp.close()

    # Compose SRP components
    fetcher: DataFetcher = CsvDataFetcher(tmp.name)
    formatter: ReportFormatter = HtmlFormatter()
    writer: FileWriter = LocalFileWriter()
    emailer: EmailSender = SmtpEmailSender()

    # Process
    raw_data = fetcher.fetch()
    records = [SalesRecord(**row) for row in raw_data]
    html = formatter.format(records)

    output_path = "sales_report.html"
    writer.write(html, output_path)
    emailer.send("manager@example.com", "Monthly Sales Report", html)

    # Cleanup
    os.unlink(tmp.name)
    # Remove generated file to keep workspace clean
    if os.path.exists(output_path):
        os.unlink(output_path)


if __name__ == "__main__":
    main()
```

## Key Language-Specific Notes

- **ABC (Abstract Base Classes)** from the `abc` module formalize interface contracts — each `DataFetcher`, `ReportFormatter`, `FileWriter`, and `EmailSender` ABC enforces the single-responsibility boundary at the type level.
- **`@dataclass`** reduces boilerplate for data transfer objects like `SalesRecord`, keeping SRP-focused classes lean and free of infrastructure code.
- **Type hints** (`list[dict[str, str]]`, `-> None`) document the data flow between single-responsibility components without requiring a separate type system.
- **Duck typing remains idiomatic** — you don't need to inherit from ABCs in pure Python if you use `Protocol`; the ABC approach here makes the SRP boundaries explicit for learning purposes.
- **The `if __name__ == "__main__":` guard** ensures the driver code doesn't run on import, keeping each module's single responsibility clean.
