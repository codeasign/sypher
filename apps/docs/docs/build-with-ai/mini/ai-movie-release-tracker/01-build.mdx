You love movies, but keeping track of what is coming out is a full-time job. Streaming services release new titles every week, theaters have schedules that vary by region, and your watchlist on Letterboxd or IMDb has grown to over 200 titles — half of which you are not even sure have been released yet. You hear about a great movie from a friend, look it up, and find out it came out three months ago on a platform you do not subscribe to.

The problem is fragmentation. Trailers drop on YouTube, release dates change, streaming rights shift between platforms, and the one movie you actually want to see gets buried under fifty algorithmically recommended titles you do not care about. You need a personal movie scout — something that watches upcoming releases, compares them against your taste, and tells you only what you would actually enjoy watching.

### What You Will Build

A Python CLI tool that uses the free TMDB (The Movie Database) API to fetch upcoming movie releases, filters them with Claude Haiku based on your personal taste profile, and outputs a ranked watchlist you can actually use.

- Fetches upcoming movie releases from TMDB for the next 30-60 days
- Takes your taste profile — favorite genres, actors, directors, and mood
- Claude Haiku scores each movie for personal relevance with a reason
- Outputs a formatted table of ranked recommendations with scores
- Saves results to a file for later reference
- Runs on a schedule so you get fresh recommendations weekly

### Concept in 5 Minutes

Think of an LLM as your movie-buff friend who has seen everything and knows your taste perfectly. You tell it "I love Christopher Nolan, sci-fi, and anything with Cillian Murphy" — and it does not just search for those keywords. It understands that you might enjoy a Korean sci-fi thriller you have never heard of, because it shares themes and craftsmanship with the movies you already love. TMDB provides the raw data (titles, overviews, genres, cast, crew). Claude provides the taste filter. Your Python script is the glue.

#### Step 3: Build the TMDB Client

Create `movie_tracker.py` in `~/movie-tracker/` with the TMDB API integration:

```python
#!/usr/bin/env python3
"""Fetch upcoming movies from TMDB and rank them with Claude Haiku."""

import os
import sys
from datetime import datetime, timedelta
from dotenv import load_dotenv

load_dotenv()

MAX_ITERATIONS = int(os.environ.get("MAX_ITERATIONS", "10"))
MAX_TOKENS_PER_RUN = int(os.environ.get("MAX_TOKENS_PER_RUN", "10000"))

ANTHROPIC_API_KEY = os.environ.get("ANTHROPIC_API_KEY")
TMDB_API_KEY = os.environ.get("TMDB_API_KEY")

if not ANTHROPIC_API_KEY:
    print("Error: ANTHROPIC_API_KEY not found in .env file.")
    sys.exit(1)
if not TMDB_API_KEY:
    print("Error: TMDB_API_KEY not found in .env file.")
    print("Get a free key at themoviedb.org/settings/api")
    sys.exit(1)

TMDB_BASE = "https://api.themoviedb.org/3"


def fetch_upcoming(days_ahead: int = 60) -> list[dict]:
    """Fetch upcoming movie releases from TMDB."""
    import requests
    today = datetime.now().strftime("%Y-%m-%d")
    end = (datetime.now() + timedelta(days=days_ahead)).strftime("%Y-%m-%d")

    url = f"{TMDB_BASE}/discover/movie"
    params = {
        "api_key": TMDB_API_KEY,
        "primary_release_date.gte": today,
        "primary_release_date.lte": end,
        "sort_by": "popularity.desc",
        "page": 1,
    }
    print(f"Fetching upcoming movies from TMDB...")
    response = requests.get(url, params=params)
    response.raise_for_status()
    data = response.json()
    movies = []
    for m in data.get("results", [])[:20]:
        movies.append({
            "id": m["id"],
            "title": m["title"],
            "overview": m.get("overview", ""),
            "release_date": m.get("release_date", ""),
            "vote_average": m.get("vote_average", 0),
            "genre_ids": m.get("genre_ids", []),
        })
    print(f"Found {len(movies)} upcoming movies.")
    return movies


def main() -> None:
    movies = fetch_upcoming(60)
    for m in movies[:5]:
        print(f"  {m['release_date']} — {m['title']}")


if __name__ == "__main__":
    main()
```

#### Step 5: Add Claude Haiku Taste Filter

Add the Anthropic integration that scores each movie against your profile:

```python
from anthropic import Anthropic

client = Anthropic(api_key=ANTHROPIC_API_KEY)


def score_movie(movie: dict, profile: dict) -> dict:
    """Ask Claude Haiku to score a movie based on the taste profile."""
    profile_text = f"""
Favorite genres: {profile['favorite_genres']}
Favorite directors: {profile['favorite_directors']}
Favorite actors: {profile['favorite_actors']}
Preferred moods: {profile['moods']}
Disliked genres: {profile['disliked_genres']}
"""
    prompt = f"""Given this taste profile:{profile_text}
Score this upcoming movie from 1-10 for how much the user would enjoy it.
Return JSON: {{"score": <int>, "reason": "<one-sentence reason>"}}

Movie: {movie['title']}
Overview: {movie['overview'][:300]}
Release: {movie['release_date']}
Rating: {movie['vote_average']}/10"""

    response = client.messages.create(
        model="claude-3-5-haiku-latest",
        max_tokens=300,
        temperature=0.3,
        messages=[{"role": "user", "content": prompt}],
    )
    text = response.content[0].text.strip()
    if "{" in text:
        text = text[text.index("{"):text.rindex("}") + 1]
    try:
        result = json.loads(text)
    except json.JSONDecodeError:
        result = {"score": 5, "reason": "Could not parse rating."}
    return result


def filter_and_rank(movies: list[dict], profile: dict) -> list[dict]:
    """Score all movies and return them sorted by relevance."""
    scored = []
    total_tokens = 0
    for i, movie in enumerate(movies):
        if total_tokens > MAX_TOKENS_PER_RUN:
            print(f"  Token budget reached after {i} movies.")
            break
        result = score_movie(movie, profile)
        total_tokens += 500
        movie["recommendation_score"] = result["score"]
        movie["recommendation_reason"] = result["reason"]
        scored.append(movie)
        print(f"  [{result['score']}/10] {movie['title']}")
    scored.sort(key=lambda m: m["recommendation_score"], reverse=True)
    return scored


def display_watchlist(scored: list[dict]) -> None:
    """Print a formatted watchlist table."""
    print(f"\n{'=' * 70}")
    print("  YOUR PERSONALIZED WATCHLIST")
    print(f"{'=' * 70}\n")
    print(f"{'Score':<7} {'Title':<40} {'Release':<15}")
    print("-" * 70)
    for m in scored:
        bar = "★" * (m["recommendation_score"] // 2) + "☆" * (5 - m["recommendation_score"] // 2)
        print(f"{bar:<7} {m['title'][:38]:<40} {m['release_date']:<15}")
    print(f"\nTop Pick: {scored[0]['title']} ({scored[0]['recommendation_score']}/10)")
    print(f"  {scored[0]['recommendation_reason']}")
```

Update `main()`:

    display_watchlist(scored)

#### Step 6: Full Pipeline — Save, Refresh, and Schedule

Add file saving and the final entry point. Replace the `main()` function:

```python
def save_watchlist(scored: list[dict]) -> str:
    """Save the ranked watchlist to a markdown file."""
    safe_date = datetime.now().strftime("%Y%m%d")
    filename = f"watchlist_{safe_date}.md"
    lines = [
        f"# Movie Watchlist — {datetime.now().strftime('%B %d, %Y')}",
        "",
        f"_{len(scored)} upcoming movies ranked for your taste._",
        "",
        "| Score | Title | Release | Reason |",
        "|-------|-------|---------|--------|",
    ]
    for m in scored:
        stars = "★" * (m["recommendation_score"] // 2) + "☆" * (5 - m["recommendation_score"] // 2)
        lines.append(
            f"| {stars} | {m['title']} | {m['release_date']} "
            f"| {m['recommendation_reason']} |"
        )
    content = "\n".join(lines)
    with open(filename, "w", encoding="utf-8") as f:
        f.write(content)
    print(f"\nWatchlist saved to: {filename}")
    return filename


def main() -> None:
    profile = build_or_load_profile()
    movies = fetch_upcoming(60)

    print("\nScoring movies with Claude Haiku...\n")
    scored = filter_and_rank(movies, profile)

    display_watchlist(scored)

    save = input("\nSave watchlist to file? (y/n): ").strip().lower()
    if save.startswith("y"):
        save_watchlist(scored)

    print("\nDone! Check back next week for new releases.")
    print("Tip: Schedule this script with cron or Task Scheduler for weekly updates.")
```

### Make It Yours

Three customizations to tailor the tracker:

**1. Add streaming service filter.** TMDB provides watch provider data. Extend the fetch to include streaming availability:

```python
def get_watch_providers(movie_id: int) -> list[str]:
    import requests
    url = f"{TMDB_BASE}/movie/{movie_id}/watch/providers"
    res = requests.get(url, params={"api_key": TMDB_API_KEY})
    data = res.json()
    results = data.get("results", {})
    us = results.get("US", {})
    flatrate = [p["provider_name"] for p in us.get("flatrate", [])]
    return flatrate
```

Then pass `watch_providers` into the Claude prompt so it can note where each movie streams.
