## Improve It

### 1. Add batch processing for multiple transcripts

Process an entire directory of transcript files at once:

```python
from pathlib import Path
from typing import List, Tuple

def batch_process(directory: str) -> List[Tuple[str, str, str]]:
    """Process all .txt files in a directory. Returns (filename, type, summary)."""
    results = []
    for file_path in Path(directory).glob("*.txt"):
        print(f"Processing: {file_path.name}")
        transcript = file_path.read_text(encoding="utf-8")
        meeting_type = detect_meeting_type(transcript)
        summary = summarize_transcript(transcript, meeting_type)
        file_summary = save_summary(summary)
        results.append((file_path.name, meeting_type, file_summary))
        print(f"  → {meeting_type} → {file_summary}\n")
    return results

def main():
    import argparse
    parser = argparse.ArgumentParser(description="Summarize meeting transcripts")
    parser.add_argument("--dir", help="Directory of transcript files to batch process")
    parser.add_argument("--file", help="Single transcript file to process")
    args = parser.parse_args()

    if args.dir:
        batch_process(args.dir)
    elif args.file:
        transcript = load_transcript(args.file)
        meeting_type = detect_meeting_type(transcript)
        summary = summarize_transcript(transcript, meeting_type)
        print(summary)
        save_summary(summary)
    else:
        # Interactive mode
        main()
```

**Why it matters in production:** Teams generate dozens of transcripts per week. Batch processing lets you run the summariser across all of them in one command, creating a searchable archive of meeting summaries.

### 2. Add sentiment and team mood tracking

Extract sentiment trends over time:

```python
import json

def analyze_sentiment(transcript: str) -> dict:
    """Analyze the overall sentiment of a meeting transcript."""
    response = client.messages.create(
        model="claude-3-5-haiku-20241022",
        max_tokens=300,
        system=(
            "Analyze the sentiment of this meeting transcript. "
            "Return a JSON object with these fields: "
            "overall_sentiment (positive/neutral/negative), "
            "energy_level (1-10), "
            "tension_indicators (list of phrases that suggest tension), "
            "positive_indicators (list of phrases that suggest positivity)."
        ),
        messages=[{"role": "user", "content": transcript}]
    )
    return json.loads(response.content[0].text)

def track_sentiment_over_time(summary_dir: str = "summaries"):
    """Analyze sentiment trends across all saved summaries."""
    summaries = sorted(Path(summary_dir).glob("*.md"))
    if not summaries:
        print("No summaries found")
        return
    sentiment_history = []
    for summary_file in summaries:
        text = summary_file.read_text(encoding="utf-8")
        sentiment = analyze_sentiment(text)
        sentiment_history.append({
            "date": summary_file.stem.replace("meeting_summary_", ""),
            "sentiment": sentiment
        })
    print(f"Sentiment analysis across {len(sentiment_history)} meetings:")
    for entry in sentiment_history:
        s = entry["sentiment"]
        print(f"  {entry['date']}: {s.get('overall_sentiment')} "
              f"(energy: {s.get('energy_level')}/10)")
```

**Why it matters in production:** Team sentiment over time is a leading indicator of burnout, disengagement, or conflict. Tracking it automatically across meetings gives managers data instead of hunches.

### 3. Add action item tracking across meetings

Extract and track action items to see what's actually getting done:

```python
def extract_action_items(summary: str) -> list:
    """Parse action items from a structured summary."""
    items = []
    lines = summary.split("\n")
    in_action_section = False
    for line in lines:
        if line.strip().startswith("## Action Items"):
            in_action_section = True
            continue
        if in_action_section:
            if line.strip().startswith("##"):
                break
            if line.strip().startswith("- **") or line.strip().startswith("* **"):
                # Extract owner and task
                parts = line.strip().split("**")
                if len(parts) >= 3:
                    owner = parts[1]
                    task = parts[2].strip(": ").strip()
                    items.append({"owner": owner, "task": task})
    return items

def generate_action_item_report(summary_dir: str = "summaries"):
    """Generate a report of all action items across meetings."""
    all_items = []
    for summary_file in Path(summary_dir).glob("*.md"):
        text = summary_file.read_text(encoding="utf-8")
        items = extract_action_items(text)
        all_items.extend(items)
    # Group by owner
    from collections import Counter
    owner_counts = Counter(item["owner"] for item in all_items)
    print(f"\nAction Items by Owner ({len(all_items)} total):")
    for owner, count in owner_counts.most_common():
        print(f"  {owner}: {count} items")
    print(f"\nAll Action Items:")
    for item in all_items:
        print(f"  - [{item['owner']}] {item['task']}")
```

**Why it matters in production:** Action items are the primary output of meetings that actually drive work. Tracking them across meetings reveals who is over-committed and which decisions are being followed through.

### 4. Add a confirmation step before expensive operations

For batch processing large numbers of transcripts:

```python
def main():
    import argparse
    parser = argparse.ArgumentParser(description="Summarize meeting transcripts")
    parser.add_argument("--dir", help="Directory of transcript files")
    parser.add_argument("--file", help="Single transcript file")
    parser.add_argument("--no-confirm", action="store_true", help="Skip confirmation")
    args = parser.parse_args()

    if args.dir:
        files = list(Path(args.dir).glob("*.txt"))
        total_chars = sum(f.read_text(encoding="utf-8").__len__() for f in files)
        estimated_cost = (total_chars / 4) * 3e-6  # rough estimate
        print(f"Found {len(files)} transcript files, ~{total_chars} total characters")
        print(f"Estimated cost: ~${estimated_cost:.3f}")
        if not args.no_confirm:
            confirm = input("Continue? (y/n): ").strip().lower()
            if confirm != "y":
                print("Aborted.")
                sys.exit(0)
        batch_process(args.dir)
```

**Why it matters in production:** Batch processing 50 transcripts could cost $0.50–$1.00. Always show an estimate and ask for confirmation before running expensive operations. This prevents surprise bills from accidental large runs.

## Deploy It

Deploy as a Slack bot using Railway so team members can forward meeting transcripts and get summaries back in the channel.

### Step 1: Create a Slack bot endpoint

Create `slack_bot.py`:

```python
import os
import sys
import json
from pathlib import Path
from dotenv import load_dotenv
from fastapi import FastAPI, Request, HTTPException
from fastapi.responses import JSONResponse

load_dotenv()

ANTHROPIC_API_KEY = os.environ.get("ANTHROPIC_API_KEY")
SLACK_SIGNING_SECRET = os.environ.get("SLACK_SIGNING_SECRET")
if not ANTHROPIC_API_KEY:
    print("Error: ANTHROPIC_API_KEY not set")
    sys.exit(1)

import anthropic
client = anthropic.Anthropic(api_key=ANTHROPIC_API_KEY)

# Import your summarizer functions
from summarizer import detect_meeting_type, summarize_transcript

app = FastAPI(title="Meeting Notes Slack Bot")

@app.post("/slack/events")
async def slack_events(request: Request):
    body = await request.json()
    # URL verification
    if body.get("type") == "url_verification":
        return JSONResponse(content={"challenge": body["challenge"]})
    # Handle message with transcript
    if "event" in body:
        event = body["event"]
        if event.get("type") == "message" and "text" in event:
            transcript = event["text"]
            meeting_type = detect_meeting_type(transcript)
            summary = summarize_transcript(transcript, meeting_type)
            return JSONResponse(content={"summary": summary})
    return JSONResponse(content={"ok": True})

@app.get("/")
async def index():
    return {"status": "Meeting Notes Bot is running"}
```

### Step 2: Create `requirements.txt`

```text
anthropic==0.49.0
fastapi==0.115.6
uvicorn==0.34.0
python-dotenv==1.0.1
python-multipart==0.0.19
```

### Step 3: Create a `Dockerfile`

```dockerfile
FROM python:3.12-slim
WORKDIR /app
COPY requirements.txt .
RUN pip install --no-cache-dir -r requirements.txt
COPY . .
EXPOSE 8000
CMD ["uvicorn", "slack_bot:app", "--host", "0.0.0.0", "--port", "8000"]
```

### Step 4: Deploy to Railway

```bash
# Install Railway CLI (if not already installed)
npm install -g @railway/cli

# Login and deploy
railway login
railway init
railway up

# Set environment variables
railway variables set ANTHROPIC_API_KEY=your-key-here
railway variables set SLACK_SIGNING_SECRET=your-slack-signing-secret

# Open the app URL
railway open
```

**Verification:** Send a direct message to your Slack bot with a meeting transcript. The bot should reply with a structured summary within 3–5 seconds.
