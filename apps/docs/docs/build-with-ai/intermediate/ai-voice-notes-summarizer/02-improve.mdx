## Improve It

### 1. Add audio preprocessing for better transcription

Clean up noisy recordings before sending them to Whisper:

```python
from pydub import AudioSegment
import numpy as np

def preprocess_audio(input_path: str, output_path: str = None) -> str:
    """Normalize volume and reduce noise in an audio file."""
    if output_path is None:
        p = Path(input_path)
        output_path = str(p.parent / f"{p.stem}_cleaned{p.suffix}")

    audio = AudioSegment.from_file(input_path)

    # Normalize volume to -3 dB
    normalized = audio.apply_gain(-audio.max_dBFS + 3)

    # Reduce silence (trim leading/trailing silence)
    trimmed = normalized.strip_silence(
        silence_len=500,
        silence_thresh=-40
    )

    # High-pass filter to reduce low-frequency rumble
    filtered = trimmed.high_pass_filter(80)

    # Export
    filtered.export(output_path, format="mp3", bitrate="128k")
    print(f"Preprocessed audio saved to: {output_path}")
    return output_path

def transcribe_audio(audio_path: str, model_size: str = "small",
                     preprocess: bool = True) -> dict:
    if preprocess:
        cleaned_path = preprocess_audio(audio_path)
        audio_path = cleaned_path
    # ... rest of transcription
```

**Why it matters in production:** Voice memos recorded in noisy environments (cafes, cars, outdoors) have background noise that degrades Whisper accuracy. Preprocessing — normalising volume, filtering rumble, trimming silence — can improve Word Error Rate (WER) by 10–20% on real-world recordings.

### 2. Add batch processing for multiple files

Process an entire directory of voice memos in one command:

```python
from concurrent.futures import ThreadPoolExecutor, as_completed

def batch_process(
    directory: str,
    pattern: str = "*.mp3",
    whisper_model_size: str = "small"
) -> int:
    """Process all audio files in a directory."""
    files = sorted(Path(directory).glob(pattern))
    # Also look for other formats
    for ext in ["*.wav", "*.m4a", "*.ogg", "*.flac"]:
        files.extend(Path(directory).glob(ext))
    files = sorted(set(files))

    if not files:
        print(f"No audio files found in {directory}")
        return 0

    total_cost = 0
    print(f"Found {len(files)} audio files to process\n")

    for audio_file in files:
        print(f"[{files.index(audio_file) + 1}/{len(files)}] {audio_file.name}")
        result = process_audio(str(audio_file), whisper_model_size)
        archive = Archive()
        note_id = archive.add(result)
        total_cost += 0.002  # rough estimate
        print(f"  → Saved: {note_id}\n")

    print(f"Batch complete: {len(files)} files processed")
    print(f"Estimated total cost: ~${total_cost:.3f}")
    return len(files)
```

**Why it matters in production:** Most users have a backlog of voice memos to process. Batch processing lets them point the tool at their entire voice notes directory and walk away. Processing 20 files takes ~20–40 minutes of transcription time but requires no manual intervention.

### 3. Add speaker diarisation for multi-person recordings

Detect different speakers in meetings or interviews:

```python
def identify_speakers(transcript_segments: list) -> str:
    """Add speaker labels to transcript segments using LLM analysis."""
    # Group segments into a timeline
    timeline = []
    for seg in transcript_segments:
        start = seg.get("start", 0)
        text = seg.get("text", "").strip()
        if text:
            timeline.append(f"[{start:.1f}s] {text}")

    timeline_text = "\n".join(timeline)

    response = client.messages.create(
        model="claude-3-5-haiku-20241022",
        max_tokens=2000,
        system=(
            "Analyse this meeting transcript timeline and identify distinct speakers. "
            "Add speaker labels (Speaker A, Speaker B, etc.) to each segment. "
            "Return the full transcript with speaker labels. "
            "Base speaker changes on shifts in topic, question-answer patterns, "
            "and conversational dynamics."
        ),
        messages=[{"role": "user", "content": timeline_text}]
    )

    return response.content[0].text
```

**Why it matters in production:** A meeting recording with three people is much more useful when you can see who said what. Speaker-labelled transcripts make it clear who owns each action item and who raised each concern.

### 4. Add cost estimation before batch processing

```python
def estimate_batch_cost(directory: str) -> dict:
    """Estimate the cost of processing an entire directory."""
    files = []
    for ext in ["*.mp3", "*.wav", "*.m4a", "*.ogg"]:
        files.extend(Path(directory).glob(ext))

    total_duration = 0
    for f in files:
        try:
            audio = AudioSegment.from_file(str(f))
            total_duration += len(audio) / 1000  # ms to seconds
        except Exception:
            pass

    estimated_minutes = total_duration / 60
    # Rough: Whisper is free, LLM analysis ~$0.002 per minute of audio
    estimated_cost = estimated_minutes * 0.002

    return {
        "files": len(files),
        "total_duration_minutes": round(estimated_minutes, 1),
        "estimated_cost": round(estimated_cost, 3),
        "recommendation": "Proceed" if estimated_cost < 2.0 else "Too expensive"
    }

def main():
    # ... in the CLI
    if action == "estimate":
        if len(cmd) < 2:
            print("Usage: estimate <directory>")
            continue
        est = estimate_batch_cost(cmd[1])
        print(f"Files: {est['files']}")
        print(f"Total duration: {est['total_duration_minutes']} min")
        print(f"Estimated cost: ${est['estimated_cost']}")
        confirm = input("Process? (y/n): ")
        if confirm.lower() == "y":
            batch_process(cmd[1])
```

**Why it matters in production:** A user with 100 voice memos totalling 5 hours of audio could unknowingly spend $0.60 in API costs. Showing the estimate upfront and asking for confirmation prevents surprise bills and builds trust.

### 5. Add keyword extraction and tagging

Make notes more discoverable with automatic tags:

```python
def extract_tags(text: str, max_tags: int = 5) -> list:
    """Extract key tags/keywords from a transcript using the LLM."""
    response = client.messages.create(
        model="claude-3-5-haiku-20241022",
        max_tokens=100,
        system=(
            f"Extract up to {max_tags} keyword tags from this text. "
            "Tags should be single words or short phrases. "
            "Return as a comma-separated list."
        ),
        messages=[{"role": "user", "content": text}]
    )
    tags = [t.strip() for t in response.content[0].text.split(",") if t.strip()]
```

## Deploy It

Deploy as a FastAPI web application on Railway with a drag-and-drop upload interface.

### Step 1: Create the web application

Create `app.py`:

```python
import os
import sys
import json
from pathlib import Path
from dotenv import load_dotenv
from fastapi import FastAPI, UploadFile, File, Form, HTTPException
from fastapi.responses import HTMLResponse, JSONResponse
import tempfile
import shutil

load_dotenv()
ANTHROPIC_API_KEY = os.environ.get("ANTHROPIC_API_KEY")
if not ANTHROPIC_API_KEY:
    print("Error: ANTHROPIC_API_KEY not set")
    sys.exit(1)

import anthropic
client = anthropic.Anthropic(api_key=ANTHROPIC_API_KEY)

from voice_note import process_audio
from archive import Archive

app = FastAPI(title="Voice Notes Summarizer")

ALLOWED_EXTENSIONS = {".mp3", ".wav", ".m4a", ".ogg", ".flac"}
MAX_FILE_SIZE = 100 * 1024 * 1024  # 100 MB

@app.get("/", response_class=HTMLResponse)
async def index():
    return """
    <html>
    <head><title>Voice Notes Summarizer</title>
    <style>
      body { font-family: system-ui, sans-serif; max-width: 800px; margin: 0 auto; padding: 2rem; }
      .upload-zone { border: 2px dashed #ccc; padding: 2rem; text-align: center; border-radius: 8px; margin: 1rem 0; }
      .upload-zone.dragover { border-color: #007bff; background: #f0f7ff; }
      #result { margin-top: 1rem; }
      .note { background: #f9f9f9; padding: 1rem; border-radius: 8px; margin: 1rem 0; }
      .note h3 { margin-top: 0; }
      button { padding: 0.5rem 1rem; background: #007bff; color: #fff; border: none; border-radius: 4px; cursor: pointer; }
      .loading { display: none; }
    </style></head>
    <body>
      <h1>Voice Notes Summarizer</h1>
      <div class="upload-zone" id="dropZone">
        <p>Drop an audio file here or click to upload</p>
        <input type="file" id="fileInput" accept=".mp3,.wav,.m4a,.ogg,.flac" hidden />
        <button onclick="document.getElementById('fileInput').click()">Choose File</button>
      </div>
      <div class="loading" id="loading">Processing... (this may take a few minutes for long recordings)</div>
      <div id="result"></div>
      <script>
        const dropZone = document.getElementById('dropZone');
        const fileInput = document.getElementById('fileInput');

        dropZone.addEventListener('dragover', (e) => {
          e.preventDefault();
          dropZone.classList.add('dragover');
        });
        dropZone.addEventListener('dragleave', () => {
          dropZone.classList.remove('dragover');
        });
        dropZone.addEventListener('drop', (e) => {
          e.preventDefault();
          dropZone.classList.remove('dragover');
          if (e.dataTransfer.files.length) uploadFile(e.dataTransfer.files[0]);
        });
        fileInput.addEventListener('change', () => {
          if (fileInput.files.length) uploadFile(fileInput.files[0]);
        });

        async function uploadFile(file) {
          document.getElementById('loading').style.display = 'block';
          document.getElementById('result').innerHTML = '';
          const form = new FormData();
          form.append('file', file);
          const res = await fetch('/upload', { method: 'POST', body: form });
          const data = await res.json();
          document.getElementById('loading').style.display = 'none';
          document.getElementById('result').innerHTML = '<div class="note">' +
            '<h3>' + file.name + '</h3>' +
            '<pre style="white-space: pre-wrap">' + data.summary + '</pre></div>';
        }
      </script>
    </body>
    </html>
    """

@app.post("/upload")
async def upload_audio(file: UploadFile = File(...)):
    ext = Path(file.filename).suffix.lower()
    if ext not in ALLOWED_EXTENSIONS:
        raise HTTPException(400, f"Unsupported file type: {ext}")
    content = await file.read()
    if len(content) > MAX_FILE_SIZE:
        raise HTTPException(400, "File too large (max 100 MB)")
    with tempfile.NamedTemporaryFile(delete=False, suffix=ext) as tmp:
        tmp.write(content)
        tmp_path = tmp.name
    try:
        result = process_audio(tmp_path)
        archive = Archive()
        archive.add(result)
        return JSONResponse({
            "summary": result["summary"],
            "note_id": result.get("id", "unknown"),
            "duration": result.get("duration", 0)
        })
    finally:
        os.unlink(tmp_path)

@app.get("/notes")
async def list_notes():
    archive = Archive()
    notes = archive.list_notes()
    return JSONResponse(notes[:20])
```
