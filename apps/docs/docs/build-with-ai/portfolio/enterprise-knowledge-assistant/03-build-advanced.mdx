## Advanced Feature 1: Slack Integration

```python
# app/slack.py
from slack_bolt import App
from slack_bolt.adapter.socket_mode import SocketModeHandler
from app.api import query as query_handler
from app.models import QueryRequest
import os

slack_app = App(token=os.environ["SLACK_BOT_TOKEN"])

@slack_app.event("app_mention")
def handle_mention(event, say):
    text = event["text"]
    # Strip bot mention
    query = text.split(">")[-1].strip()

    if not query:
        say("How can I help?")
        return

    result = query_handler(QueryRequest(
        tenant_id=os.environ.get("SLACK_TENANT_ID", "default"),
        user_id=event["user"],
        query=query
    ))

    say(result["answer"])

def start_slack():
    handler = SocketModeHandler(slack_app, os.environ["SLACK_APP_TOKEN"])
    handler.start()
```

## Advanced Feature 2: Admin Dashboard

```python
# app/admin.py
from fastapi import APIRouter
from app.database import get_conn
from datetime import datetime, timedelta

router = APIRouter(prefix="/api/v1/admin")

@router.get("/usage")
def get_usage(hours: int = 24):
    cutoff = (datetime.utcnow() - timedelta(hours=hours)).isoformat()
    conn = get_conn()

    total = conn.execute("SELECT COUNT(*) FROM query_logs WHERE timestamp > ?", (cutoff,)).fetchone()[0]
    top_queries = conn.execute("SELECT query, COUNT(*) as c FROM query_logs WHERE timestamp > ? GROUP BY query ORDER BY c DESC LIMIT 10", (cutoff,)).fetchall()
    top_users = conn.execute("SELECT user_id, COUNT(*) as c FROM query_logs WHERE timestamp > ? GROUP BY user_id ORDER BY c DESC LIMIT 10", (cutoff,)).fetchall()

    conn.close()
    return {
        "total_queries": total,
        "top_queries": [dict(r) for r in top_queries],
        "top_users": [dict(r) for r in top_users]
    }

@router.get("/tenants")
def list_tenants():
    conn = get_conn()
    tenants = conn.execute("SELECT * FROM tenants").fetchall()
    conn.close()
    return [dict(t) for t in tenants]
```

## Advanced Feature 3: Notion Connector (API Stub)

```python
# app/ingestion/notion.py
import httpx, os
from app.vector_store import vector_store

async def ingest_notion(tenant_id: str, database_id: str) -> int:
    api_key = os.environ.get("NOTION_API_KEY", "")
    if not api_key:
        print("NOTION_API_KEY not set. Skipping.")
        return 0

    headers = {
        "Authorization": f"Bearer {api_key}",
        "Notion-Version": "2022-06-28"
    }

    documents = []
    async with httpx.AsyncClient() as client:
        response = await client.post(
            f"https://api.notion.com/v1/databases/{database_id}/query",
            headers=headers
        )
        if response.status_code == 200:
            data = response.json()
            for page in data.get("results", []):
                title = page.get("properties", {}).get("title", {}).get("title", [{}])[0].get("text", {}).get("content", "Untitled")
                content = page.get("properties", {}).get("content", {}).get("rich_text", [{}])[0].get("text", {}).get("content", "")
                documents.append({
                    "id": page["id"],
                    "content": content,
                    "source": f"notion://{database_id}/{page['id']}",
                    "title": title
                })

    if documents:
        vector_store.add_documents(tenant_id, documents)
    return len(documents)
```

# tests/test_knowbot.py
from app.vector_store import vector_store
from app.database import init_db, get_conn
import pytest

def test_vector_store_tenant_isolation():
    col_a = vector_store.get_collection("tenant-a")

def test_query_cache():
    init_db()
    conn = get_conn()
    conn.execute("INSERT INTO query_cache VALUES (?, ?, ?, ?)",
