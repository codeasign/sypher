## Build It — Core

### Step 1: Multi-Tenant Database

```python
# app/database.py
import sqlite3, os, json
from datetime import datetime

DB_PATH = os.environ.get("DB_PATH", "./data/knowbot.db")

def get_conn():
    os.makedirs(os.path.dirname(DB_PATH) or ".", exist_ok=True)
    conn = sqlite3.connect(DB_PATH)
    conn.row_factory = sqlite3.Row
    return conn

def init_db():
    conn = get_conn()
    conn.executescript("""
        CREATE TABLE IF NOT EXISTS tenants (
            id TEXT PRIMARY KEY,
            name TEXT NOT NULL,
            created_at TEXT NOT NULL
        );
        CREATE TABLE IF NOT EXISTS knowledge_sources (
            id TEXT PRIMARY KEY,
            tenant_id TEXT NOT NULL,
            source_type TEXT NOT NULL,
            config TEXT DEFAULT '{}',
            last_indexed TEXT,
            document_count INTEGER DEFAULT 0,
            FOREIGN KEY (tenant_id) REFERENCES tenants(id)
        );
        CREATE TABLE IF NOT EXISTS query_logs (
            id TEXT PRIMARY KEY,
            tenant_id TEXT NOT NULL,
            user_id TEXT NOT NULL,
            query TEXT NOT NULL,
            response_preview TEXT,
            tokens_used INTEGER DEFAULT 0,
            latency_ms INTEGER DEFAULT 0,
            timestamp TEXT NOT NULL
        );
        CREATE TABLE IF NOT EXISTS query_cache (
            query_hash TEXT PRIMARY KEY,
            tenant_id TEXT NOT NULL,
            response TEXT NOT NULL,
            created_at TEXT NOT NULL
        );
    """)
    conn.commit()
    conn.close()
```

### Step 2: Tenant Vector Store

```python
# app/vector_store.py
import chromadb
from chromadb.config import Settings
from app.config import settings
import os

class TenantVectorStore:
    def __init__(self):
        self.client = chromadb.PersistentClient(
            path=settings.CHROMA_PATH,
            settings=Settings(anonymized_telemetry=False)
        )

    def get_collection(self, tenant_id: str):
        collection_name = f"kb_{tenant_id}"
        try:
            return self.client.get_collection(collection_name)
        except:
            return self.client.create_collection(collection_name)

    def add_documents(self, tenant_id: str, documents: list[dict]):
        col = self.get_collection(tenant_id)
        col.add(
            ids=[d["id"] for d in documents],
            documents=[d["content"] for d in documents],
            metadatas=[{"source": d.get("source", ""), "title": d.get("title", "")} for d in documents]
        )

    def search(self, tenant_id: str, query: str, top_k: int = 5) -> list[dict]:
        col = self.get_collection(tenant_id)
        results = col.query(query_texts=[query], n_results=top_k)
        return [{
            "id": results["ids"][0][i],
            "content": results["documents"][0][i],
            "source": results["metadatas"][0][i].get("source", ""),
            "title": results["metadatas"][0][i].get("title", ""),
            "score": 1.0 - (i / top_k)
        } for i in range(len(results["ids"][0]))]

vector_store = TenantVectorStore()
```

### Step 3: Ingestion — Markdown Files

```python
# app/ingestion/markdown.py
from pathlib import Path
from app.vector_store import vector_store
import uuid, hashlib

def ingest_markdown(tenant_id: str, dir_path: str) -> int:
    base = Path(dir_path)
    documents = []

    for md_file in base.rglob("*.md"):
        content = md_file.read_text(encoding="utf-8")
        doc_id = hashlib.md5(content.encode()).hexdigest()[:12]
        documents.append({
            "id": doc_id,
            "content": content,
            "source": str(md_file),
            "title": md_file.stem
        })

    if documents:
        vector_store.add_documents(tenant_id, documents)
    return len(documents)
```

### Step 4: Query Endpoint

```python
# app/api.py
from fastapi import FastAPI, HTTPException
from pydantic import BaseModel
from anthropic import Anthropic
from app.config import settings
from app.vector_store import vector_store
from app.database import get_conn, init_db
from datetime import datetime
import uuid, hashlib, json

app = FastAPI(title="KnowBot API")
client = Anthropic(api_key=settings.ANTHROPIC_API_KEY)

class QueryRequest(BaseModel):
    tenant_id: str
    user_id: str = "anonymous"
    query: str

class IngestRequest(BaseModel):
    tenant_id: str
    source_type: str = "markdown"
    source_path: str

@app.on_event("startup")
async def startup():
    init_db()

@app.post("/api/v1/query")
def query(req: QueryRequest):
    # Check cache
    import time
    start = time.time()
    q_hash = hashlib.md5(f"{req.tenant_id}:{req.query}".encode()).hexdigest()
    conn = get_conn()
    cached = conn.execute("SELECT response FROM query_cache WHERE query_hash = ? AND tenant_id = ?", (q_hash, req.tenant_id)).fetchone()
    if cached:
        conn.close()
        return {"answer": json.loads(cached["response"]), "cached": True}

    # Search
    results = vector_store.search(req.tenant_id, req.query)
    if not results:
        return {"answer": "No relevant documents found.", "cached": False}

    context = "\n\n".join([f"[{r['title']}]({r['source']})\n{r['content'][:1000]}" for r in results])

    response = client.messages.create(
        model=settings.LLM_MODEL,
        max_tokens=1000,
        system="You are an enterprise knowledge assistant. Answer based ONLY on the provided context. Cite sources using [Source Name].",
        messages=[{"role": "user", "content": f"Context:\n{context}\n\nQuestion: {req.query}"}]
    )

    answer = response.content[0].text
    latency = int((time.time() - start) * 1000)

    # Cache
    conn.execute("INSERT OR REPLACE INTO query_cache VALUES (?, ?, ?, ?)",
                 (q_hash, req.tenant_id, json.dumps(answer), datetime.utcnow().isoformat()))
    conn.execute("INSERT INTO query_logs VALUES (?, ?, ?, ?, ?, ?, ?, ?)",
                 (uuid.uuid4().hex[:12], req.tenant_id, req.user_id, req.query, answer[:100],
                  response.usage.input_tokens + response.usage.output_tokens, latency, datetime.utcnow().isoformat()))
    conn.commit()
    conn.close()

    return {"answer": answer, "cached": False, "sources": [r["title"] for r in results]}

@app.post("/api/v1/ingest")
def ingest(req: IngestRequest):
    from app.ingestion.markdown import ingest_markdown
    count = ingest_markdown(req.tenant_id, req.source_path)
    return {"documents_indexed": count, "status": "success"}
```

## Understand the Core

- **Collection-per-tenant** in ChromaDB provides natural data isolation — no risk of cross-tenant data leaks
- **Query caching** uses MD5 hash of tenant + query for a 24-hour TTL (not shown but easy to add)
- **Token counting** tracks per-query usage for cost allocation per tenant
- **Ingestion** is modular — add a new source type by creating a new module in `app/ingestion/`
