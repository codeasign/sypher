### Step 1: Agent State

```python
# app/state.py
from typing import TypedDict, List, Optional, Dict, Any

class AgentMessage(TypedDict):
    agent: str
    content: str
    type: str  # "finding", "status", "error"

class ResearchState(TypedDict):
    session_id: str
    query: str
    sub_questions: List[str]
    findings: List[Dict[str, Any]]
    report: Optional[str]
    iteration: int
    max_iterations: int
    status: str
    messages: List[AgentMessage]
```

### Step 2: Orchestrator — Query Decomposition

```python
# app/orchestrator.py
from anthropic import Anthropic
from app.config import settings
import json

client = Anthropic(api_key=settings.ANTHROPIC_API_KEY)

def decompose_query(query: str) -> list[str]:
    prompt = f"""Decompose this research query into 3-6 sub-questions that can be researched independently:

Query: {query}

Return a JSON array of strings. Each sub-question should be specific, researchable, and non-overlapping."""

    response = client.messages.create(
        model="claude-sonnet-4-20250514",
        max_tokens=1000,
        messages=[{"role": "user", "content": prompt}]
    )

    try:
        return json.loads(response.content[0].text)
    except json.JSONDecodeError:
        return [f"Research: {query}"]
```

# =>   - ...

### Step 3: Web Researcher Agent

```python
# app/researchers/web.py
from anthropic import Anthropic
from app.config import settings
import httpx
from bs4 import BeautifulSoup
import json

client = Anthropic(api_key=settings.ANTHROPIC_API_KEY)

async def search_web(query: str) -> list[dict]:
    """Search DuckDuckGo and return top results."""
    async with httpx.AsyncClient() as http:
        response = await http.get(
            "https://api.duckduckgo.com/",
            params={"q": query, "format": "json"}
        )
        data = response.json()
        results = []

        # Extract from Abstract and RelatedTopics
        if data.get("Abstract"):
            results.append({
                "title": data.get("Heading", "Result"),
                "snippet": data["Abstract"],
                "url": data.get("AbstractURL", ""),
                "source": "abstract"
            })

        return results[:5]

async def fetch_page(url: str) -> str:
    """Fetch and extract text from a URL."""
    try:
        async with httpx.AsyncClient(timeout=15) as http:
            response = await http.get(url, headers={"User-Agent": "ResearchNet/1.0"})
            soup = BeautifulSoup(response.text, "html.parser")
            # Remove script and style elements
            for tag in soup(["script", "style", "nav", "footer"]):
                tag.decompose()
            text = soup.get_text(separator="\n", strip=True)
            return text[:5000]  # Limit content length
    except Exception as e:
        return f"Error fetching page: {e}"

async def research_sub_question(question: str) -> dict:
    search_results = await search_web(question)

    page_contents = []
    for result in search_results[:3]:
        if result.get("url"):
            content = await fetch_page(result["url"])
            page_contents.append({"url": result["url"], "content": content[:2000]})

    context = "\n\n".join([
        f"Source: {r['url']}\n{r['content']}"
        for r in page_contents
    ])

    response = client.messages.create(
        model=settings.LLM_MODEL,
        max_tokens=1500,
        system="You are a research assistant. Synthesize findings from the search results into a concise answer. Cite sources.",
        messages=[{"role": "user", "content": f"Research question: {question}\n\nSearch results:\n{context}\n\nProvide findings:"}]
    )

    return {
        "sub_question": question,
        "content": response.content[0].text,
        "sources": [r.get("url", "") for r in search_results if r.get("url")],
        "agent": "web_researcher"
    }
```

### Step 4: Synthesis Agent

```python
# app/synthesis.py
from anthropic import Anthropic
from app.config import settings
import json

client = Anthropic(api_key=settings.ANTHROPIC_API_KEY)

def synthesize(query: str, findings: list[dict]) -> str:
    findings_text = "\n\n".join([
        f"[{f['agent']}] {f['sub_question']}\n{f['content']}"
        for f in findings
    ])

    response = client.messages.create(
        model="claude-sonnet-4-20250514",
        max_tokens=4000,
        system="You are a research synthesis expert. Create a comprehensive, well-structured report from the research findings. Include: executive summary, key findings with evidence, analysis, and conclusions.",
        messages=[{"role": "user", "content": f"Research Query: {query}\n\nFindings:\n{findings_text}\n\nProduce a structured report:"}]
    )

    return response.content[0].text
```

### Step 5: Research Graph

```python
# app/graph.py
from langgraph.graph import StateGraph, END
from app.state import ResearchState
from app.orchestrator import decompose_query
from app.researchers.web import research_sub_question
from app.synthesis import synthesize
from app.database import save_session, save_finding
import asyncio, uuid
from datetime import datetime

async def decompose_node(state: ResearchState) -> dict:
    questions = decompose_query(state["query"])
    return {"sub_questions": questions, "messages": [{"agent": "orchestrator", "content": f"Decomposed into {len(questions)} questions", "type": "status"}]}

async def research_node(state: ResearchState) -> dict:
    findings = []
    for question in state["sub_questions"]:
        finding = await research_sub_question(question)
        findings.append(finding)
        save_finding(state["session_id"], finding)
    return {"findings": findings, "messages": [{"agent": "web_researcher", "content": f"Completed {len(findings)} research tasks", "type": "status"}]}

async def synthesize_node(state: ResearchState) -> dict:
    report = synthesize(state["query"], state["findings"])
    return {"report": report, "status": "completed", "messages": [{"agent": "synthesis", "content": "Report generated", "type": "status"}]}

def build_research_graph():
    workflow = StateGraph(ResearchState)
    workflow.add_node("decompose", decompose_node)
    workflow.add_node("research", research_node)
    workflow.add_node("synthesize", synthesize_node)
    workflow.set_entry_point("decompose")
    workflow.add_edge("decompose", "research")
    workflow.add_edge("research", "synthesize")
    workflow.add_edge("synthesize", END)
    return workflow.compile()
```

### Step 6: Run the Research Pipeline

```python
# app/runner.py
from app.graph import build_research_graph
from app.state import ResearchState
from app.config import settings
from app.database import init_db, save_session
import uuid

def run_research(query: str) -> dict:
    init_db()
    session_id = uuid.uuid4().hex[:12]
    save_session(session_id, query)

    graph = build_research_graph()
    initial = ResearchState(
        session_id=session_id,
        query=query,
        sub_questions=[],
        findings=[],
        report=None,
        iteration=0,
        max_iterations=settings.MAX_ITERATIONS,
        status="in_progress",
        messages=[]
    )

    import asyncio
    result = asyncio.run(graph.ainvoke(initial))
    return result
```

### Step 7: API with SSE Streaming

```python
# app/api.py
from fastapi import FastAPI
from sse_starlette.sse import EventSourceResponse
from pydantic import BaseModel
from app.runner import run_research
from app.database import get_session
import json, asyncio

app = FastAPI(title="ResearchNet API")

class ResearchRequest(BaseModel):
    query: str

@app.post("/api/research")
async def start_research(req: ResearchRequest):
    """Start research and stream progress."""
    async def event_generator():
        yield {"event": "status", "data": json.dumps({"message": "Starting research..."})}
        result = run_research(req.query)
        if result["report"]:
            yield {"event": "complete", "data": json.dumps({"report": result["report"], "findings": len(result["findings"])})}
        else:
            yield {"event": "error", "data": json.dumps({"message": "Research failed"})}

    return EventSourceResponse(event_generator())

@app.get("/api/sessions/{session_id}")
def get_session_data(session_id: str):
    return get_session(session_id)
```

## Understand the Core

- **Query decomposition** is the most important step — better sub-questions mean better research. The orchestrator uses Sonnet 4 specifically for this
- **Web researcher** uses DuckDuckGo's free API (no key needed) and falls back to fetch+extract for content
- **Synthesis** compiles all findings into a structured report with executive summary, findings, and conclusions
- **Each research session** costs roughly $0.02-0.10 depending on the number of sub-questions and pages fetched
