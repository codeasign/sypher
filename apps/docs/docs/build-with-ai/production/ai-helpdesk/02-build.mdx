# app/database.py
import sqlite3, os
from app.config import settings

def get_conn():
    os.makedirs(os.path.dirname(settings.DB_PATH), exist_ok=True)
    conn = sqlite3.connect(settings.DB_PATH)
    conn.row_factory = sqlite3.Row
    conn.execute("PRAGMA journal_mode=WAL")
    return conn

def init_db():
    conn = get_conn()
    conn.executescript("""
        CREATE TABLE IF NOT EXISTS tickets (
            id TEXT PRIMARY KEY,
            subject TEXT NOT NULL,
            description TEXT NOT NULL,
            customer_email TEXT NOT NULL,
            status TEXT NOT NULL DEFAULT 'open',
            priority TEXT NOT NULL DEFAULT 'medium',
            category TEXT,
            assigned_team TEXT,
            confidence_score REAL DEFAULT 0.0,
            resolution TEXT,
            created_at TEXT NOT NULL,
            resolved_at TEXT
        );
        CREATE TABLE IF NOT EXISTS kb_articles (
            id TEXT PRIMARY KEY,
            title TEXT NOT NULL,
            content TEXT NOT NULL,
            category TEXT NOT NULL,
            tags TEXT
        );
    """)
    conn.commit()
    conn.close()
    print("Database initialized.")

from pydantic import BaseModel, Field
from typing import Optional, Literal
from datetime import datetime
import uuid

class TicketCreate(BaseModel):
    subject: str = Field(..., min_length=1, max_length=200)
    description: str = Field(..., min_length=1, max_length=5000)
    customer_email: str = Field(..., pattern=r"^[^@]+@[^@]+\.[^@]+$")

class TicketResponse(BaseModel):
    id: str
    subject: str
    description: str
    customer_email: str
    status: str
    priority: str
    category: Optional[str] = None
    assigned_team: Optional[str] = None
    confidence_score: float = 0.0
    resolution: Optional[str] = None
    created_at: str
    resolved_at: Optional[str] = None

class KBArticleCreate(BaseModel):
    title: str = Field(..., min_length=1, max_length=200)
    content: str = Field(..., min_length=10)
    category: str = Field(..., max_length=50)
    tags: list[str] = []

class ClassificationResult(BaseModel):
    category: Literal["billing", "technical", "account", "general", "feature_request"]
    priority: Literal["low", "medium", "high", "critical"]
    sentiment: Literal["positive", "neutral", "frustrated", "angry"]
    required_team: Literal["support", "engineering", "billing", "product"]
    confidence: float = Field(..., ge=0.0, le=1.0)
    requires_escalation: bool
    escalation_reason: Optional[str] = None

### Step 3: Ticket Classifier

```python
# app/classify/classifier.py
from anthropic import Anthropic
from app.config import settings
from app.models import ClassificationResult
import json

classifier_client = Anthropic(api_key=settings.ANTHROPIC_API_KEY)

def classify_ticket(subject: str, description: str) -> ClassificationResult:
    prompt = f"""Classify this support ticket into JSON:

Subject: {subject}
Description: {description}

Return valid JSON only:
{{
    "category": "billing|technical|account|general|feature_request",
    "priority": "low|medium|high|critical",
    "sentiment": "positive|neutral|frustrated|angry",
    "required_team": "support|engineering|billing|product",
    "confidence": 0.0-1.0,
    "requires_escalation": true/false,
    "escalation_reason": null or string
}}"""

    response = classifier_client.messages.create(
        model=settings.LLM_MODEL,
        max_tokens=500,
        system="You are a helpdesk ticket classifier. Return only valid JSON.",
        messages=[{"role": "user", "content": prompt}]
    )

    try:
        return ClassificationResult(**json.loads(response.content[0].text))
    except (json.JSONDecodeError, Exception) as e:
        return ClassificationResult(
            category="general",
            priority="medium",
            sentiment="neutral",
            required_team="support",
            confidence=0.5,
            requires_escalation=True,
            escalation_reason=f"Classification failed: {e}"
        )
```

### Step 4: Knowledge Base Management

```python
# app/resolution/kb.py
import chromadb
from chromadb.config import Settings as ChromaSettings
from app.config import settings
from app.database import get_conn
from app.models import KBArticleCreate
import json, uuid

def get_kb_collection():
    client = chromadb.PersistentClient(
        path=settings.KB_CHROMA_PATH,
        settings=ChromaSettings(anonymized_telemetry=False)
    )
    return client.get_or_create_collection(name="kb_articles")

def add_article(article: KBArticleCreate) -> str:
    article_id = uuid.uuid4().hex[:12]
    col = get_kb_collection()
    col.add(
        ids=[article_id],
        documents=[article.content],
        metadatas=[{
            "title": article.title,
            "category": article.category,
            "tags": ",".join(article.tags)
        }]
    )
    conn = get_conn()
    conn.execute(
        "INSERT INTO kb_articles (id, title, content, category, tags) VALUES (?, ?, ?, ?, ?)",
        (article_id, article.title, article.content, article.category, json.dumps(article.tags))
    )
    conn.commit()
    conn.close()
    return article_id

def search_kb(query: str, top_k: int = 3) -> list[dict]:
    col = get_kb_collection()
    results = col.query(query_texts=[query], n_results=top_k)
    articles = []
    for i, doc_id in enumerate(results["ids"][0]):
        articles.append({
            "id": doc_id,
            "title": results["metadatas"][0][i]["title"],
            "content": results["documents"][0][i],
            "score": 1.0 - (i / top_k)
        })
    return articles
```

### Step 5: Auto-Resolution Engine

```python
# app/resolution/resolver.py
from anthropic import Anthropic
from app.config import settings
from app.resolution.kb import search_kb
from app.models import ClassificationResult

resolver_client = Anthropic(api_key=settings.ANTHROPIC_API_KEY)

def auto_resolve(
    ticket_subject: str,
    ticket_description: str,
    classification: ClassificationResult
) -> tuple[str, float]:
    """Generate an auto-resolution and return (resolution_text, confidence)."""
    kb_results = search_kb(f"{ticket_subject} {ticket_description}")

    if not kb_results:
        return None, 0.0

    kb_context = "\n\n".join([
        f"[Article: {r['title']}]\n{r['content']}"
        for r in kb_results
    ])

    response = resolver_client.messages.create(
        model=settings.LLM_MODEL,
        max_tokens=1000,
        system="You are a helpful support agent. Answer the customer's question using ONLY the knowledge base articles provided. If the articles don't fully answer the question, say so and recommend escalation.",
        messages=[{
            "role": "user",
            "content": f"KB Articles:\n{kb_context}\n\nCustomer Issue:\nSubject: {ticket_subject}\nDescription: {ticket_description}\n\nCategory: {classification.category}\nPriority: {classification.priority}\n\nProvide a helpful response to the customer:"
        }]
    )

    resolution = response.content[0].text
    confidence = classification.confidence * 0.9 + 0.1

    return resolution, confidence
```

### Step 6: Ticket API

```python
# app/api/tickets.py
from fastapi import APIRouter, HTTPException
from app.models import TicketCreate, TicketResponse, ClassificationResult
from app.classify.classifier import classify_ticket
from app.resolution.resolver import auto_resolve
from app.database import get_conn
from app.config import settings
from datetime import datetime
import uuid

router = APIRouter(prefix="/api/v1")

@router.post("/tickets", response_model=TicketResponse)
def create_ticket(ticket: TicketCreate):
    ticket_id = uuid.uuid4().hex[:12]
    now = datetime.utcnow().isoformat()

    # Classify
    classification = classify_ticket(ticket.subject, ticket.description)

    # Try auto-resolution
    resolution_text, confidence = None, classification.confidence
    if not classification.requires_escalation and confidence >= settings.AUTO_RESOLVE_THRESHOLD:
        resolution_text, confidence = auto_resolve(
            ticket.subject, ticket.description, classification
        )

    if resolution_text and confidence >= settings.AUTO_RESOLVE_THRESHOLD:
        status = "resolved"
        resolved_at = now
    else:
        status = "open"
        resolved_at = None

    conn = get_conn()
    conn.execute(
        """INSERT INTO tickets (id, subject, description, customer_email, status, priority,
           category, assigned_team, confidence_score, resolution, created_at, resolved_at)
           VALUES (?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?)""",
        (ticket_id, ticket.subject, ticket.description, ticket.customer_email,
         status, classification.priority, classification.category,
         classification.required_team, confidence, resolution_text, now, resolved_at)
    )
    conn.commit()
    conn.close()

    return TicketResponse(
        id=ticket_id, subject=ticket.subject, description=ticket.description,
        customer_email=ticket.customer_email, status=status,
        priority=classification.priority, category=classification.category,
        assigned_team=classification.required_team, confidence_score=confidence,
        resolution=resolution_text, created_at=now, resolved_at=resolved_at
    )

@router.get("/tickets/{ticket_id}", response_model=TicketResponse)
def get_ticket(ticket_id: str):
    conn = get_conn()
    row = conn.execute("SELECT * FROM tickets WHERE id = ?", (ticket_id,)).fetchone()
    conn.close()
    if not row:
        raise HTTPException(404, "Ticket not found")
    return TicketResponse(**dict(row))
```

# app/cli/dashboard.py
import click
import sqlite3
from app.config import settings
from datetime import datetime, timedelta

@click.group()
def cli():
    """HelpBot Dashboard"""
    pass

@cli.command()
def summary():
    """Show ticket summary."""
    conn = sqlite3.connect(settings.DB_PATH)
    conn.row_factory = sqlite3.Row

    total = conn.execute("SELECT COUNT(*) as c FROM tickets").fetchone()["c"]
    resolved = conn.execute("SELECT COUNT(*) as c FROM tickets WHERE status='resolved'").fetchone()["c"]
    escalated = conn.execute("SELECT COUNT(*) as c FROM tickets WHERE status='open'").fetchone()["c"]
    by_category = conn.execute(
        "SELECT category, COUNT(*) as c FROM tickets GROUP BY category ORDER BY c DESC"
    ).fetchall()

    conn.close()
    resolution_rate = (resolved / total * 100) if total > 0 else 0

    click.echo(f"Total Tickets: {total}")
    click.echo(f"Resolved: {resolved} ({resolution_rate:.0f}%)")
    click.echo(f"Open/Escalated: {escalated}")
    click.echo("\nBy Category:")
    for row in by_category:
        click.echo(f"  {row['category']}: {row['c']}")

@cli.command()
def recent(hours: int = 24):
    """Show recent tickets."""
    conn = sqlite3.connect(settings.DB_PATH)
    conn.row_factory = sqlite3.Row
    cutoff = (datetime.utcnow() - timedelta(hours=hours)).isoformat()
    rows = conn.execute(
        "SELECT subject, status, priority, confidence_score FROM tickets WHERE created_at > ? ORDER BY created_at DESC LIMIT 10",
        (cutoff,)
    ).fetchall()
    conn.close()
    for row in rows:
        emoji = "✓" if row["status"] == "resolved" else "○"
        click.echo(f"{emoji} [{row['priority']}] {row['subject']} (confidence: {row['confidence_score']:.2f})")

if __name__ == "__main__":
    cli()

### Step 12: Analytics Query

```python
# app/analytics.py
from app.database import get_conn
from datetime import datetime, timedelta

def get_resolution_rate(days: int = 7) -> dict:
    conn = get_conn()
    cutoff = (datetime.utcnow() - timedelta(days=days)).isoformat()
    total = conn.execute(
        "SELECT COUNT(*) FROM tickets WHERE created_at > ?", (cutoff,)
    ).fetchone()[0]
    resolved = conn.execute(
        "SELECT COUNT(*) FROM tickets WHERE status='resolved' AND created_at > ?", (cutoff,)
    ).fetchone()[0]
    conn.close()
    return {
        "total": total,
        "resolved": resolved,
        "rate": round(resolved / total * 100, 1) if total > 0 else 0.0
    }

def get_common_issues(limit: int = 5) -> list:
    conn = get_conn()
    rows = conn.execute(
        "SELECT category, COUNT(*) as c FROM tickets GROUP BY category ORDER BY c DESC LIMIT ?",
        (limit,)
    ).fetchall()
    conn.close()
    return [{"category": r["category"], "count": r["c"]} for r in rows]
```

## Understand What Happened

### Component Internals

- **Classifier:** Uses structured output prompting to extract category, priority, sentiment, and team from the ticket text. The LLM's confidence score is a self-assessment, not calibrated against ground truth
- **KB Search:** ChromaDB vector search finds semantically similar articles. The top-3 are concatenated as context for the resolver
- **Resolver:** The LLM generates a customer-facing response using only the KB articles. The response is audited — you can see exactly what the customer saw
- **Confidence:** Combined from classifier confidence (90%) and a small bonus (10%) for having any KB matches. Tune the threshold in settings

### Token / Cost Breakdown

| Operation | Tokens | Cost (Haiku) |
|---|---|---|
| Classify a ticket | ~300 in / ~100 out | $0.0004 |
| Search KB (no LLM) | ~500 (embedding) | $0.00001 |
| Auto-resolve | ~800 in / ~400 out | $0.0012 |
| **Total per ticket** | **~2,100** | **~$0.0016** |

### Where the Bottlenecks Are

- **Classifier latency:** ~1-2 seconds per ticket. Not a problem at low volume but becomes noticeable at 50+ tickets/minute
- **KB search:** ChromaDB queries are fast (under 50ms) but degrade with 10K+ articles
- **Resolver latency:** The LLM call takes 2-5 seconds. Streaming helps the user experience but not throughput

## Common Errors

### Error 1: LLM Returns Invalid JSON

```
json.JSONDecodeError: Expecting value: line 1 column 1
```

**Root cause:** The LLM sometimes returns markdown-wrapped JSON or extra text.

**Fix:** The classifier already handles this with a try/except fallback. If you see it frequently, add a more robust parser:
```python
import re
def extract_json(text: str) -> str:
    match = re.search(r'\{.*\}', text, re.DOTALL)
    return match.group(0) if match else text
```

### Error 2: ChromaDB Not Available

```
chromadb.errors.NoIndexException: No index found
```

**Root cause:** The KB collection was never created or the data directory was deleted.

**Fix:**
```bash
rm -rf data/kb_chroma
python -c "from app.resolution.kb import get_kb_collection; get_kb_collection()"
```

### Error 3: All Tickets Get Escalated

**Root cause:** The `AUTO_RESOLVE_THRESHOLD` in settings is too high (default 0.8).

**Fix:** Lower it to 0.6 and monitor:
```python

```
