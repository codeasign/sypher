### Step 1: Agent State and Types

```python
# app/state.py
from typing import TypedDict, List, Optional, Literal

class SpecialistResponse(TypedDict):
    specialist: str
    response: str
    confidence: float
    resolution: Optional[str]

class RoutingState(TypedDict):
    ticket_id: str
    subject: str
    description: str
    customer_email: str
    routing_decision: Optional[str]  # billing, technical, account, product, multi
    specialist_responses: List[SpecialistResponse]
    final_resolution: Optional[str]
    status: str
    messages: List[str]
```

### Step 3: Supervisor Agent

```python
# app/supervisor.py
from anthropic import Anthropic
from app.config import settings
import json

client = Anthropic(api_key=settings.ANTHROPIC_API_KEY)

def supervisor_route(subject: str, description: str) -> dict:
    prompt = f"""Analyze this support ticket and determine which specialist should handle it.

Subject: {subject}
Description: {description}

Choose the best routing:
- "billing": Payment, invoice, refund, subscription issues
- "technical": Bugs, errors, API, configuration, performance
- "account": Access, permissions, SSO, team management
- "product": Feature requests, roadmap, feedback
- "multi": Requires multiple specialists (rare)

Return JSON: {{"decision": "billing|technical|account|product|multi", "reasoning": "brief explanation", "confidence": 0.0-1.0}}"""

    response = client.messages.create(
        model="claude-sonnet-4-20250514",
        max_tokens=500,
        messages=[{"role": "user", "content": prompt}]
    )

    return json.loads(response.content[0].text)
```

```bash
python -c "
from app.supervisor import supervisor_route
result = supervisor_route('Refund for last month', 'I was charged twice for my subscription')
print(f'Decision: {result[\"decision\"]} (confidence: {result[\"confidence\"]})')
# => Decision: billing (confidence: 0.95)
"
```

**Milestone 1**: supervisor correctly identifies ticket type and routes to the right specialist.

### Step 4: Specialist Agents

```python
# app/specialists.py
from anthropic import Anthropic
from app.config import settings
from app.prompts import BILLING_SPECIALIST, TECHNICAL_SPECIALIST, ACCOUNT_SPECIALIST, PRODUCT_SPECIALIST

client = Anthropic(api_key=settings.ANTHROPIC_API_KEY)

SPECIALIST_PROMPTS = {
    "billing": BILLING_SPECIALIST,
    "technical": TECHNICAL_SPECIALIST,
    "account": ACCOUNT_SPECIALIST,
    "product": PRODUCT_SPECIALIST,
}

def call_specialist(specialist_type: str, subject: str, description: str) -> dict:
    system_prompt = SPECIALIST_PROMPTS.get(specialist_type, "You are a helpful support agent.")

    response = client.messages.create(
        model=settings.LLM_MODEL,
        max_tokens=1000,
        system=system_prompt,
        messages=[{
            "role": "user",
            "content": f"Ticket: {subject}\n\n{description}\n\nPlease handle this ticket:"
        }]
    )

    return {
        "specialist": specialist_type,
        "response": response.content[0].text,
        "confidence": 0.9,
        "resolution": response.content[0].text
    }
```

```bash
python -c "
from app.specialists import call_specialist
result = call_specialist('billing', 'Double charge', 'I was charged twice for my subscription')
print(f'Response: {result[\"response\"][:100]}...')
# => Response: I understand you were charged twice. Let me help resolve this...
"
```

**Milestone 2**: specialist agents provide domain-specific responses.

### Step 5: Multi-Agent Coordination

```python
# app/coordinator.py
from anthropic import Anthropic
from app.config import settings
from app.state import SpecialistResponse
import json

client = Anthropic(api_key=settings.ANTHROPIC_API_KEY)

def coordinate_responses(responses: list[SpecialistResponse], subject: str) -> str:
    context = "\n\n".join([
        f"[{r['specialist'].upper()}]\n{r['response']}"
        for r in responses
    ])

    response = client.messages.create(
        model="claude-sonnet-4-20250514",
        max_tokens=1000,
        system="You are a support coordinator. Combine the specialist responses into a single coherent answer for the customer.",
        messages=[{
            "role": "user",
            "content": f"Subject: {subject}\n\nSpecialist responses:\n{context}\n\nCombine into one response:"
        }]
    )
    return response.content[0].text
```

### Step 6: LangGraph Workflow

```python
# app/graph.py
from langgraph.graph import StateGraph, END
from app.state import RoutingState
from app.supervisor import supervisor_route
from app.specialists import call_specialist
from app.coordinator import coordinate_responses

def routing_node(state: RoutingState) -> dict:
    result = supervisor_route(state["subject"], state["description"])
    return {"routing_decision": result["decision"], "messages": [f"Routed: {result['decision']}"]}

def specialist_node(state: RoutingState) -> dict:
    decision = state["routing_decision"]
    if decision == "multi":
        responses = []
        for spec in ["billing", "technical", "account", "product"]:
            resp = call_specialist(spec, state["subject"], state["description"])
            responses.append(resp)
        resolution = coordinate_responses(responses, state["subject"])
        return {"specialist_responses": responses, "final_resolution": resolution, "status": "resolved"}
    else:
        resp = call_specialist(decision, state["subject"], state["description"])
        return {"specialist_responses": [resp], "final_resolution": resp["response"], "status": "resolved"}

def build_routing_graph():
    workflow = StateGraph(RoutingState)
    workflow.add_node("route", routing_node)
    workflow.add_node("handle", specialist_node)
    workflow.set_entry_point("route")
    workflow.add_edge("route", "handle")
    workflow.add_edge("handle", END)
    return workflow.compile()
```

### Step 7: Run the System

```python
# app/runner.py
from app.graph import build_routing_graph
from app.state import RoutingState
import uuid

def route_ticket(subject: str, description: str, email: str) -> dict:
    graph = build_routing_graph()
    initial = RoutingState(
        ticket_id=uuid.uuid4().hex[:12],
        subject=subject,
        description=description,
        customer_email=email,
        specialist_responses=[],
        status="pending",
        messages=[]
    )
    return graph.invoke(initial)
```

```bash
python -c "
from app.runner import route_ticket
result = route_ticket('API timeout error', 'Our API calls timeout after 30 seconds', 'dev@company.com')
print(f'Routed to: {result[\"routing_decision\"]}')
print(f'Resolution: {result[\"final_resolution\"][:100]}')
# => Routed to: technical
# => Resolution: I can help you troubleshoot the API timeout issue...
"
```

**Milestone 3**: multi-agent system routes and resolves tickets correctly.

## Understand What Happened

- **Supervisor agent** (Sonnet 4) determines the routing category. It's the most expensive call but only runs once per ticket
- **Specialist agents** (Haiku 3.5) handle the actual work. Each is a focused expert, not a generalist
- **Multi-specialty coordination** handles rare cases where a ticket spans billing AND technical issues
- **Structured state** passes information between agents without a shared database

## Common Errors

### Error 1: All Tickets Route to "Multi"

**Root cause:** The supervisor prompt is too vague or the confidence threshold is too low.

**Fix:** Tighten the routing criteria in the prompt. Add explicit negative examples.

### Error 2: Specialist Gives Wrong Advice

**Root cause:** The specialist prompt doesn't include enough domain-specific rules.

**Fix:** Add specific policies and constraints to the specialist system prompt.
