## Build It

### Step 1: Graph State Definition

```python
# app/state.py
from typing import TypedDict, List, Optional, Literal, Annotated
from typing_extensions import NotRequired

class Article(TypedDict):
    title: str
    content: str
    score: float

class Message(TypedDict):
    role: Literal["system", "agent", "human"]
    content: str
    timestamp: str

class SupportState(TypedDict):
    ticket_id: str
    subject: str
    description: str
    customer_email: str
    category: NotRequired[Optional[str]]
    priority: NotRequired[Optional[str]]
    sentiment: NotRequired[Optional[str]]
    assigned_team: NotRequired[Optional[str]]
    confidence: float
    relevant_articles: List[Article]
    resolution: NotRequired[Optional[str]]
    status: str
    messages: List[Message]
    error_count: int
```

### Step 2: Graph Builder

```python
# app/graph.py
from langgraph.graph import StateGraph, END
from app.state import SupportState
from app.nodes import classify_node, kb_search_node, resolve_node, escalate_node, check_followup_node
from datetime import datetime

def build_support_graph() -> StateGraph:
    workflow = StateGraph(SupportState)

    # Add nodes
    workflow.add_node("classify", classify_node)
    workflow.add_node("kb_search", kb_search_node)
    workflow.add_node("resolve", resolve_node)
    workflow.add_node("escalate", escalate_node)
    workflow.add_node("follow_up", check_followup_node)

    # Set entry point
    workflow.set_entry_point("classify")

    # Add edges
    workflow.add_conditional_edges(
        "classify",
        lambda state: "kb_search" if state.get("category") else "escalate"
    )

    workflow.add_conditional_edges(
        "kb_search",
        lambda state: "resolve" if state.get("relevant_articles") else "escalate"
    )

    workflow.add_conditional_edges(
        "resolve",
        lambda state: "follow_up" if state["confidence"] > 0.8 else "escalate"
    )

    workflow.add_edge("escalate", "follow_up")
    workflow.add_edge("follow_up", END)

    return workflow.compile()
```

### Step 3: Node — Classification

```python
# app/nodes/classify.py
from anthropic import Anthropic
from app.config import settings
from app.state import SupportState, Message
from datetime import datetime
import json

client = Anthropic(api_key=settings.ANTHROPIC_API_KEY)

def classify_node(state: SupportState) -> dict:
    prompt = f"""Classify this support ticket:

Subject: {state['subject']}
Description: {state['description']}

Return JSON with: category (technical/billing/account/general/feature_request), priority (low/medium/high/critical), sentiment (positive/neutral/frustrated/angry), assigned_team (support/engineering/billing/product), confidence (0-1)"""

    response = client.messages.create(
        model=settings.LLM_MODEL,
        max_tokens=500,
        messages=[{"role": "user", "content": prompt}]
    )

    try:
        result = json.loads(response.content[0].text)
    except json.JSONDecodeError:
        return {
            "status": "error",
            "messages": [{"role": "system", "content": "Classification failed", "timestamp": datetime.utcnow().isoformat()}],
            "error_count": state.get("error_count", 0) + 1
        }

    return {
        "category": result["category"],
        "priority": result["priority"],
        "sentiment": result["sentiment"],
        "assigned_team": result["assigned_team"],
        "confidence": result["confidence"],
        "status": "classified",
        "messages": state.get("messages", []) + [
            Message(role="system", content=f"Classified as {result['category']} with {result['priority']} priority", timestamp=datetime.utcnow().isoformat())
        ]
    }
```

### Step 4: Node — KB Search

```python
# app/nodes/kb_search.py
from app.state import SupportState, Article
from app.kb import search_kb
from datetime import datetime

def kb_search_node(state: SupportState) -> dict:
    query = f"{state['subject']} {state['description']}"
    results = search_kb(query, top_k=3)

    articles = [
        Article(title=r["title"], content=r["content"], score=r["score"])
        for r in results
    ]

    return {
        "relevant_articles": articles,
        "messages": state.get("messages", []) + [
            {"role": "system", "content": f"Found {len(articles)} relevant KB articles", "timestamp": datetime.utcnow().isoformat()}
        ]
    }
```

### Step 5: Node — Resolution

```python
# app/nodes/resolve.py
from anthropic import Anthropic
from app.config import settings
from app.state import SupportState
from datetime import datetime

client = Anthropic(api_key=settings.ANTHROPIC_API_KEY)

def resolve_node(state: SupportState) -> dict:
    kb_context = "\n\n".join([
        f"[{a['title']}]\n{a['content']}" for a in state.get("relevant_articles", [])
    ])

    response = client.messages.create(
        model=settings.LLM_MODEL,
        max_tokens=1000,
        messages=[{
            "role": "user",
            "content": f"KB Articles:\n{kb_context}\n\nCustomer Issue: {state['subject']}\n{state['description']}\n\nProvide a helpful resolution:"
        }]
    )

    return {
        "resolution": response.content[0].text,
        "status": "resolved",
        "messages": state.get("messages", []) + [
            {"role": "agent", "content": "Resolution generated", "timestamp": datetime.utcnow().isoformat()}
        ]
    }
```

### Step 7: Run the Graph

```python
# app/runner.py
from app.graph import build_support_graph
from app.state import SupportState
from app.config import settings
import uuid

def process_ticket(subject: str, description: str, email: str) -> dict:
    graph = build_support_graph()

    initial_state: SupportState = {
        "ticket_id": uuid.uuid4().hex[:12],
        "subject": subject,
        "description": description,
        "customer_email": email,
        "confidence": 0.0,
        "relevant_articles": [],
        "messages": [],
        "error_count": 0,
        "status": "new"
    }

    result = graph.invoke(initial_state)
    return result
```

## Understand What Happened

- **StateGraph** manages the entire ticket lifecycle. Each node reads from and writes to the shared state
- **Conditional edges** determine the path — classification quality, KB article availability, and confidence scores all influence routing
- **Cycle detection** prevents infinite loops. If a node errors 3 times, the ticket escalates to a human
- **Serialization** of graph state between nodes means the workflow can be paused and resumed

## Common Errors

### Error 1: Recursion Limit Exceeded

```
RecursionError: maximum recursion depth exceeded
```

**Root cause:** A cycle in the graph has no termination condition.

**Fix:** Add a `max_steps` check:
```python
if len(state["messages"]) > 20:
    return END
```

### Error 2: State Key Missing

```
KeyError: 'category'
```

**Root cause:** A node expects a state key that hasn't been set yet.

**Fix:** Use `.get()` with defaults in all node functions.
