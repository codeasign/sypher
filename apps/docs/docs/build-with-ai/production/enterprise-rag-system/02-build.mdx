### Step 2: Database Schema

Set up SQLite with FTS5 for full-text search and audit logging.

```python
# app/database.py
import sqlite3
import os
from app.config import settings

def get_connection() -> sqlite3.Connection:
    os.makedirs(os.path.dirname(settings.SQLITE_PATH), exist_ok=True)
    conn = sqlite3.connect(settings.SQLITE_PATH)
    conn.row_factory = sqlite3.Row
    conn.execute("PRAGMA journal_mode=WAL")
    conn.execute("PRAGMA foreign_keys=ON")
    return conn

def init_db():
    conn = get_connection()
    conn.executescript("""
        CREATE TABLE IF NOT EXISTS documents (
            id TEXT PRIMARY KEY,
            filename TEXT NOT NULL,
            source_path TEXT,
            content_type TEXT NOT NULL,
            access_group TEXT NOT NULL DEFAULT 'all',
            created_at TEXT NOT NULL DEFAULT (datetime('now')),
            updated_at TEXT NOT NULL DEFAULT (datetime('now')),
            chunk_count INTEGER DEFAULT 0,
            checksum TEXT NOT NULL
        );

        CREATE TABLE IF NOT EXISTS api_keys (
            key_hash TEXT PRIMARY KEY,
            user TEXT NOT NULL,
            access_group TEXT NOT NULL DEFAULT 'all',
            is_active INTEGER NOT NULL DEFAULT 1,
            created_at TEXT NOT NULL DEFAULT (datetime('now'))
        );

        CREATE TABLE IF NOT EXISTS query_log (
            id TEXT PRIMARY KEY,
            user TEXT NOT NULL,
            query TEXT NOT NULL,
            access_group TEXT NOT NULL,
            documents_retrieved TEXT,
            tokens_used INTEGER DEFAULT 0,
            latency_ms INTEGER DEFAULT 0,
            timestamp TEXT NOT NULL DEFAULT (datetime('now'))

        CREATE VIRTUAL TABLE IF NOT EXISTS docs_fts USING fts5(
            content, document_id, tokenize='porter unicode61'
```

### Step 4: Ingestion — Document Loader

Build the file reader that handles PDFs, Markdown, and plain text.

```python
# app/ingest/loader.py
from pathlib import Path
import hashlib

def load_file(filepath: Path) -> tuple[str, str]:
    """Load a file and return (content, content_type)."""
    content = filepath.read_bytes()
    ext = filepath.suffix.lower()

    if ext == ".pdf":
        from PyPDF2 import PdfReader
        import io
        reader = PdfReader(io.BytesIO(content))
        text = "\n".join(page.extract_text() for page in reader.pages)
        return text, "pdf"

    elif ext in (".md", ".mdx"):
        return content.decode("utf-8", errors="replace"), "md"

    elif ext in (".txt", ".py", ".js", ".ts", ".jsx", ".tsx", ".json", ".yaml", ".yml"):
        return content.decode("utf-8", errors="replace"), "text"

    else:
        raise ValueError(f"Unsupported file type: {ext}")

def compute_checksum(filepath: Path) -> str:
    return hashlib.sha256(filepath.read_bytes()).hexdigest()
```

### Step 5: Chunking Pipeline

Split documents into overlapping chunks with metadata.

```python
# app/ingest/chunker.py
from typing import List, Dict
import tiktoken

def chunk_text(
    text: str,
    chunk_size: int = 500,
    chunk_overlap: int = 50,
    document_id: str = "",
    filename: str = ""
) -> List[Dict]:
    """Split text into overlapping chunks using token-aware splitting."""
    enc = tiktoken.get_encoding("cl100k_base")
    tokens = enc.encode(text)
    chunks = []

    start = 0
    while start < len(tokens):
        end = min(start + chunk_size, len(tokens))
        chunk_tokens = tokens[start:end]
        chunk_text = enc.decode(chunk_tokens)

        chunks.append({
            "document_id": document_id,
            "filename": filename,
            "chunk_index": len(chunks),
            "content": chunk_text,
            "token_count": len(chunk_tokens),
        })

        if end >= len(tokens):
            break
        start += chunk_size - chunk_overlap

    return chunks
```

```bash
python -c "

print(f'First chunk: {chunks[0][\"content\"][:40]}...')

# => Generated 5 chunks
# => First chunk: Hello world. Hello world. Hello world....

```

Token-aware chunking ensures each chunk fits within the LLM's context window. The 50-token overlap prevents concepts from being split across chunk boundaries.

### Step 6: Vector Store Setup

Initialize ChromaDB with the embedding function.

```python
# app/search/vector.py
import chromadb
from chromadb.config import Settings as ChromaSettings
from app.config import settings
from typing import List, Dict
import openai

class EmbeddingFunction:
    def __init__(self):
        self.client = openai.OpenAI(api_key=settings.OPENAI_API_KEY)
        self.model = settings.EMBEDDING_MODEL

    def __call__(self, input: List[str]) -> List[List[float]]:
        response = self.client.embeddings.create(
            model=self.model,
            input=input
        )
        return [r.embedding for r in response.data]

def get_vector_store():
    chroma_client = chromadb.PersistentClient(
        path=settings.CHROMA_DB_PATH,
        settings=ChromaSettings(anonymized_telemetry=False)
    )
    collection = chroma_client.get_or_create_collection(
        name="enterprise_rag",
        embedding_function=EmbeddingFunction()
    )
    return chroma_client, collection
```

### Step 7: Ingestion Pipeline — Wire It Together

Connect loading, chunking, embedding, and storage.

```python
# app/ingest/pipeline.py
from pathlib import Path
import json
from app.ingest.loader import load_file, compute_checksum
from app.ingest.chunker import chunk_text
from app.search.vector import get_vector_store
from app.database import get_connection
from app.config import settings
from app.models import IngestResponse
import uuid

def ingest_document(
    filepath: Path,
    access_group: str = "all",
    filename: str = None
) -> IngestResponse:
    content, content_type = load_file(filepath)
    checksum = compute_checksum(filepath)
    doc_id = uuid.uuid4().hex[:12]
    display_name = filename or filepath.name

    # Check for duplicates
    conn = get_connection()
    existing = conn.execute(
        "SELECT id FROM documents WHERE checksum = ?", (checksum,)
    ).fetchone()
    if existing:
        conn.close()
        return IngestResponse(
            document_id=existing["id"],
            chunks=0,
            status="skipped"
        )

    # Chunk
    chunks = chunk_text(
        content,
        document_id=doc_id,
        filename=display_name
    )

    # Store in ChromaDB
    chroma_client, collection = get_vector_store()
    collection.add(
        ids=[f"{doc_id}_{c['chunk_index']}" for c in chunks],
        documents=[c["content"] for c in chunks],
        metadatas=[{
            "document_id": doc_id,
            "filename": display_name,
            "chunk_index": c["chunk_index"],
            "access_group": access_group
        } for c in chunks]
    )

    # Store in SQLite
    conn.execute(
        """INSERT INTO documents (id, filename, content_type, access_group, chunk_count, checksum)
           VALUES (?, ?, ?, ?, ?, ?)""",
        (doc_id, display_name, content_type, access_group, len(chunks), checksum)
    )

    # Index in FTS
    for c in chunks:
        conn.execute(
            "INSERT INTO docs_fts (content, document_id) VALUES (?, ?)",
            (c["content"], c["document_id"])
        )

    conn.commit()
    conn.close()

    return IngestResponse(
        document_id=doc_id,
        chunks=len(chunks),
        status="ingested"
    )
```

### Step 8: Hybrid Search

Combine vector similarity with keyword search using result fusion.

```python
# app/search/hybrid.py
from typing import List, Dict
from app.search.vector import get_vector_store
from app.database import get_connection
from app.config import settings

def search(
    query: str,
    access_group: str = "all",
    top_k: int = None
) -> List[Dict]:
    top_k = top_k or settings.VECTOR_TOP_K

    # 1. Vector search
    _, collection = get_vector_store()
    vector_results = collection.query(
        query_texts=[query],
        n_results=top_k * 2,
        where={"access_group": access_group} if access_group != "all" else None
    )

    vector_scores = {}
    for i, doc_id in enumerate(vector_results["ids"][0]):
        vector_scores[doc_id] = 1.0 - (i / (top_k * 2))

    # 2. Keyword search (BM25 via FTS5)
    conn = get_connection()
    # Sanitize query for FTS5
    safe_query = " OR ".join(
        word for word in query.split()
        if len(word) > 2
    )
    fts_results = conn.execute(
        """SELECT rowid, content, document_id, rank
           FROM docs_fts
           WHERE docs_fts MATCH ?
           ORDER BY rank
           LIMIT ?""",
        (safe_query, top_k * 2)
    ).fetchall()

    keyword_scores = {}
    for i, row in enumerate(fts_results):
        keyword_scores[row["document_id"]] = 1.0 - (i / (top_k * 2))

    conn.close()

    # 3. Fusion — reciprocal rank fusion
    all_ids = set(list(vector_scores.keys()) + list(keyword_scores.keys()))
    fused = []
    for doc_id in all_ids:
        v_score = vector_scores.get(doc_id, 0.0)
        k_score = keyword_scores.get(doc_id, 0.0)
        fused_score = 0.6 * v_score + 0.4 * k_score
        fused.append({"id": doc_id, "score": fused_score})

    fused.sort(key=lambda x: x["score"], reverse=True)
    return fused[:top_k]
```

### Step 9: LLM Query with Context

Build the query endpoint that retrieves context and generates an answer.

```python
# app/llm/client.py
from anthropic import Anthropic
from app.config import settings
from typing import List, Dict
import time

class LLMClient:
    def __init__(self):
        self.client = Anthropic(api_key=settings.ANTHROPIC_API_KEY)
        self.model = settings.LLM_MODEL
        self.total_tokens = 0

    def query(self, query: str, context: List[Dict], stream: bool = False):
        if self.total_tokens > settings.MAX_TOKENS_PER_RUN:
            return {"answer": "Token budget reached. Set MAX_TOKENS_PER_RUN to increase.", "tokens": 0}

        context_text = "\n\n".join([
            f"[Source: {c.get('filename', 'unknown')}, Chunk {c.get('chunk_index', 0)}]\n{c.get('content', '')}"
            for c in context
        ])

        system_prompt = (
            "You are an enterprise RAG assistant. Answer the user's question based ONLY on the provided context. "
            "If the context doesn't contain enough information, say so. "
            "Cite your sources using the [Source: filename] tags. "
            "Be concise and accurate."
        )

        response = self.client.messages.create(
            model=self.model,
            max_tokens=1024,
            system=system_prompt,
            messages=[{
                "role": "user",
                "content": f"Context:\n{context_text}\n\nQuestion: {query}"
            }],
            stream=stream
        )

        if stream:
            return response

        answer = response.content[0].text
        self.total_tokens += response.usage.input_tokens + response.usage.output_tokens
        return {
            "answer": answer,
            "tokens": response.usage.input_tokens + response.usage.output_tokens
        }
```

### Step 10: API Routes

Wire everything into FastAPI endpoints.

```python
# app/api/query_routes.py
from fastapi import APIRouter, Depends, HTTPException
from app.models import QueryRequest, QueryResponse, Source
from app.search.hybrid import search
from app.llm.client import LLMClient
from app.database import get_connection
from app.auth import get_current_user
from app.config import settings
import time
import json

router = APIRouter(prefix="/api/v1")
llm = LLMClient()

@router.post("/query", response_model=QueryResponse)
async def query_endpoint(
    req: QueryRequest,
    user: dict = Depends(get_current_user)
):
    start = time.time()

    # Search
    search_results = search(req.query, access_group=user["access_group"])
    if not search_results:
        return QueryResponse(
            answer="No relevant documents found in your accessible groups.",
            sources=[],
            tokens_used=0,
            latency_ms=0
        )

    # Get full chunk content
    conn = get_connection()
    sources = []
    for sr in search_results:
        doc_id, chunk_idx = sr["id"].rsplit("_", 1)
        row = conn.execute(
            "SELECT filename, content_type FROM documents WHERE id = ?", (doc_id,)
        ).fetchone()
        if row:
            sources.append(Source(
                document_id=doc_id,
                filename=row["filename"],
                chunk_index=int(chunk_idx),
                relevance_score=sr["score"],
                excerpt=""
            ))
    conn.close()

    # Generate answer
    llm_result = llm.query(req.query, [s.model_dump() for s in sources], stream=req.stream)

    if req.stream:
        # Handle streaming
        full_text = ""
        for event in llm_result:
            if event.type == "content_block_delta":
                full_text += event.delta.text
        answer = full_text
        tokens = 0
    else:
        answer = llm_result["answer"]
        tokens = llm_result["tokens"]

    # Log
    conn = get_connection()
    conn.execute(
        """INSERT INTO query_log (id, user, query, access_group, documents_retrieved, tokens_used, latency_ms)
           VALUES (?, ?, ?, ?, ?, ?, ?)""",
        (sources[0].document_id if sources else "none",
         user["user"], req.query, user["access_group"],
         json.dumps([s.document_id for s in sources]),
         tokens, int((time.time() - start) * 1000))
    )
    conn.commit()
    conn.close()

    return QueryResponse(
        answer=answer,
        sources=sources,
        tokens_used=tokens,
        latency_ms=int((time.time() - start) * 1000)
    )
```

```python
# app/auth.py
from fastapi import Header, HTTPException, Depends
from app.database import get_connection
from app.models import hash_key
from typing import Optional

def get_current_user(authorization: Optional[str] = Header(None)):
    if not authorization:
        raise HTTPException(status_code=401, detail="Missing API key")
    key = authorization.replace("Bearer ", "")
    key_hash = hash_key(key)

    conn = get_connection()
    row = conn.execute(
        "SELECT user, access_group, is_active FROM api_keys WHERE key_hash = ?",
        (key_hash,)
    ).fetchone()
    conn.close()

    if not row or not row["is_active"]:
        raise HTTPException(status_code=401, detail="Invalid or deactivated API key")

    return {"user": row["user"], "access_group": row["access_group"]}
```

### Step 11: CLI Client

Build a command-line interface for quick queries.

```python
# cli/commands.py
import click
import httpx
import os
from pathlib import Path

API_BASE = os.environ.get("RAGFLOW_API", "http://localhost:8000")
API_KEY = os.environ.get("RAGFLOW_API_KEY", "")

@click.group()
def cli():
    """RAGFlow — Enterprise RAG CLI"""
    pass

@cli.command()
@click.argument("query")
def ask(query: str):
    """Ask a question to your document collection."""
    headers = {"Authorization": f"Bearer {API_KEY}"}
    resp = httpx.post(
        f"{API_BASE}/api/v1/query",
        json={"query": query, "stream": False},
        headers=headers,
        timeout=30
    )
    if resp.status_code == 401:
        click.echo("Error: Invalid API key. Set RAGFLOW_API_KEY.", err=True)
        return
    resp.raise_for_status()
    data = resp.json()

    click.echo(f"\n{data['answer']}\n")
    click.echo("── Sources ──")
    for s in data["sources"]:
        click.echo(f"  {s['filename']} (score: {s['relevance_score']:.2f})")
    click.echo(f"\nTokens: {data['tokens_used']} | Latency: {data['latency_ms']}ms")

@cli.command()
@click.argument("filepath", type=click.Path(exists=True))
@click.option("--group", default="all", help="Access group")
def ingest(filepath: str, group: str):
    """Ingest a document."""
    headers = {"Authorization": f"Bearer {API_KEY}"}
    with open(filepath, "rb") as f:
        files = {"file": (Path(filepath).name, f)}
        data = {"access_group": group}
        resp = httpx.post(
            f"{API_BASE}/api/v1/ingest",
            files=files,
            data=data,
            headers=headers,
            timeout=60
        )
    resp.raise_for_status()
    result = resp.json()
    click.echo(f"Ingested: {result['document_id']} ({result['chunks']} chunks)")

if __name__ == "__main__":
    cli()
```

```python
    file: UploadFile = File(...),
    access_group: str = Form("all"),
    user: dict = Depends(get_current_user)
):
    suffix = Path(file.filename).suffix
    with tempfile.NamedTemporaryFile(suffix=suffix, delete=False) as tmp:
        content = await file.read()
        tmp.write(content)
        tmp_path = Path(tmp.name)

    try:
        result = ingest_document(tmp_path, access_group=access_group, filename=file.filename)
        return result
    except ValueError as e:
        raise HTTPException(status_code=400, detail=str(e))
    finally:
        tmp_path.unlink(missing_ok=True)
```

## Understand What Happened

### What Each Major Component Does

- **Ingestion pipeline:** Reads files from disk, extracts text (PDF or plain), splits into token-aware chunks with overlap, embeds each chunk, and stores both the vector (ChromaDB) and the text (SQLite FTS5)
- **Hybrid search:** Runs two independent searches — vector similarity for meaning, BM25 for exact keywords — then fuses results using weighted RRF (0.6 vector, 0.4 keyword)
- **LLM layer:** Constructs a prompt from the retrieved context, calls Claude Haiku, tracks token usage, and enforces the budget cap
- **Auth layer:** Validates API keys on every request, enforces document-level access groups

### Token / Cost Breakdown

| Operation | Input Tokens | Output Tokens | Cost (Haiku) | Cost (Sonnet) |
|---|---|---|---|---|
| Simple query (1-2 chunks) | ~500 | ~200 | $0.0007 | $0.0035 |
| Complex query (5 chunks) | ~2,500 | ~500 | $0.0025 | $0.0125 |
| Document ingestion (10 pages) | ~3,000 (embedding) | — | $0.00006 | — |

### Where the Bottlenecks Are

- **Embedding generation:** The slowest part of ingestion. A 100-page PDF takes 5-10 seconds just to embed
- **ChromaDB metadata filtering:** Filtering by `access_group` on every query adds 30-100ms. At scale, you'd want a separate collection per group
- **LLM generation:** The dominant cost and latency factor. Streaming helps perceived latency but not total cost

### What Would Break First Under Load

1. **SQLite write contention:** Concurrent ingestion + queries will hit `SQLITE_BUSY`. WAL mode helps but doesn't eliminate it
2. **ChromaDB memory:** With over 100K chunks, ChromaDB loads everything into memory. You'd hit OOM on a small server
3. **Embedding API rate limits:** OpenAI's embedding API has tiered rate limits. At 500+ documents/minute, you'd need to batch

## Common Errors

### Error 1: Missing API Key

```
anthropic.AnthropicError: API key not found
```

**Root cause:** `ANTHROPIC_API_KEY` is not set in `.env` or the environment.

**Fix:**
```bash
echo "ANTHROPIC_API_KEY=sk-ant-..." >> .env
source .env  # or restart your shell
```

### Error 2: ChromaDB Collection Already Exists

```
chromadb.errors.DuplicateError: Collection enterprise_rag already exists
```

**Root cause:** Running the script twice without clearing data.

**Fix:**
```bash
rm -rf data/chroma
python -c "from app.search.vector import get_vector_store; get_vector_store()"
```

### Error 3: FTS5 Syntax Error on Query

```
sqlite3.OperationalError: fts5: syntax error near "the"
```

**Root cause:** FTS5 doesn't accept short stop words (under 3 chars) in MATCH queries.

**Fix:** The hybrid search already filters words shorter than 3 characters. If you see this error, check that your `safe_query` generation is working.

### Error 4: PDF Extraction Returns Empty Text

```
PyPDF2.errors.PdfReadError: Could not read malformed PDF
```

**Root cause:** Scanned PDFs (image-based) have no extractable text.

**Fix:** Use a PDF with actual text, or add OCR support with `pytesseract`:
```python
# Optional OCR fallback
import subprocess
def extract_with_ocr(filepath):
    result = subprocess.run(["tesseract", str(filepath), "stdout"], capture_output=True, text=True)
    return result.stdout
```

### Error 5: OpenAPI Rate Limit

```
openai.RateLimitError: 429 Too Many Requests
```

**Root cause:** You've exceeded your embedding API rate limit (typically 3,000 RPM for free tier).

**Fix:** Add exponential backoff to the embedding function:
```python
import time
import random

def embed_with_retry(client, input, max_retries=3):
    for attempt in range(max_retries):
        try:
            return client.embeddings.create(model="text-embedding-3-small", input=input)
        except openai.RateLimitError:
            if attempt == max_retries - 1:
                raise
            time.sleep(2 ** attempt + random.uniform(0, 1))
```
